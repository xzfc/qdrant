1//! Hierarchical progress tracker.
2//!
3//! # Example
4//!
5//! ```text
6//!                                                             now
7//! ─────────────────────────── Time ────────────────────────────┴╶╶╶╶╶╶▶
8//!
9//! ├───────────────────── Segment Indexing ──────────────────────╶╶╶╶╶╶╶
10//!  ├─────── Quantization ───────┤├──── HNSW Index Building ─────╶╶╶╶╶╶╶
11//!   ├─ Vector A ─┤├─ Vector B ─┤  ├─ Vector A ─┤├─ Vector B ────╶╶╶╶╶╶╶
12//! ```
13//!
14//! # Errors and Panic Safety
15//!
16//! Most of methods are infallible (not returning `Result`/`Option`).
17//! On debug builds they might panic, on release builds they will fallback to
18//! some placeholder behavior.
19//!
20//! Why? Because progress tracking is a non-critical feature, and we don't want
21//! to abort segment building just because of a bug in progress tracking code.
22
23use std::sync::Arc;
24use std::sync::atomic::{AtomicU64, Ordering};
25use std::time::{Duration, Instant};
26
27use chrono::{DateTime, Utc};
28use parking_lot::Mutex;
29use schemars::JsonSchema;
30use serde::{Deserialize, Serialize};

32/// Read-only view of a root progress node.
33///
34/// Keep it around to observe the progress from another thread.
35#[derive(Clone, Debug)]
36pub struct ProgressView {
37    root: Arc<Mutex<ProgressNode>>,
38    /// This field is redundant, but kept outside of the lock for faster access.
39    started_at: DateTime<Utc>,
40}
41
42#[derive(Clone, Debug, Serialize, Deserialize, JsonSchema)]
43pub struct ProgressTree {
44    /// Name of the operation.
45    pub name: String,
46
47    /// When the operation started.
48    #[serde(default, skip_serializing_if = "Option::is_none")]
49    pub started_at: Option<DateTime<Utc>>,
50
51    /// When the operation finished.
52    #[serde(default, skip_serializing_if = "Option::is_none")]
53    pub finished_at: Option<DateTime<Utc>>,
54
55    /// For finished operations, how long they took, in seconds.
56    #[serde(default, skip_serializing_if = "Option::is_none")]
57    pub duration_sec: Option<f64>,
58
59    /// Number of completed units of work, if applicable.
60    #[serde(default, skip_serializing_if = "Option::is_none")]
61    pub done: Option<u64>,
62
63    /// Total number of units of work, if applicable and known.
64    #[serde(default, skip_serializing_if = "Option::is_none")]
65    pub total: Option<u64>,
66
67    /// Child operations.
68    #[serde(default, skip_serializing_if = "Vec::is_empty")]
69    pub children: Vec<ProgressTree>,
70}
71
72impl ProgressView {
73    pub fn snapshot(&self, root: impl Into<String>) -> ProgressTree {
74        self.root.lock().render(root.into())
75    }
76
77    /// The same as `self.snapshot("").started_at().unwrap()`.
78    pub fn started_at(&self) -> DateTime<Utc> {
79        self.started_at
80    }
81}

83/// Write-only handle to report the progress of operation.
84///
85/// Might be root node or a sub-task.
86pub struct ProgressTracker {
87    root: Arc<Mutex<ProgressNode>>,
88    path: Vec<usize>,
89}
90
91#[derive(Debug)]
92pub struct ProgressNode {
93    /// Sub-tasks of this task, with their names, in order of creation.
94    children: Vec<(String, ProgressNode)>,
95    progress: Option<NodeProgress>,
96    state: ProgressState,
97}
98
99#[derive(Debug)]
100struct NodeProgress {
101    current: Arc<AtomicU64>,
102    total: Option<u64>,
103}
104
105#[derive(Debug)]
106enum ProgressState {
107    Pending,
108    InProgress {
109        started_at: DateTime<Utc>,
110        started_instant: Instant,
111    },
112    Finished {
113        started_at: DateTime<Utc>,
114        finished_at: DateTime<Utc>,
115        duration: Duration,
116    },
117}
118
119/// Create a new root progress tracker.
120///
121/// Returns a read-only [`ProgressView`] to observe the progress,
122/// and a write-only [`ProgressTracker`] to signal progress updates.
123pub fn new_progress_tracker() -> (ProgressView, ProgressTracker) {
124    let started_at = Utc::now();
125    let started_instant = Instant::now();
126    let root = Arc::new(Mutex::new(ProgressNode {
127        children: Vec::new(),
128        progress: None,
129        state: ProgressState::InProgress {
130            started_at,
131            started_instant,
132        },
133    }));
134    (
135        ProgressView {
136            root: root.clone(),
137            started_at,
138        },
139        ProgressTracker {
140            root,
141            path: Vec::new(),
142        },
143    )
144}
145
146impl ProgressTracker {
147    #[cfg(any(test, feature = "testing"))]
148    pub fn new_for_test() -> Self {
149        new_progress_tracker().1
150    }
151
152    /// Create a pending subtask.
153    pub fn subtask(&self, name: impl Into<String>) -> ProgressTracker {
154        self.subtask_impl(name.into(), true)
155    }
156
157    /// Similar to creating a [`Self::subtask()`], then immediately calling
158    /// [`Self::start()`] on it.
159    pub fn running_subtask(&self, name: impl Into<String>) -> ProgressTracker {
160        self.subtask_impl(name.into(), false)
161    }
162
163    fn subtask_impl(&self, name: String, pending: bool) -> ProgressTracker {
164        let mut root = self.root.lock();
165        if let Some(parent) = root.get_mut(&self.path) {
166            let mut path = Vec::with_capacity(self.path.len() + 1);
167            path.extend_from_slice(&self.path);
168            path.push(parent.children.len());
169
170            parent.children.push((
171                name,
172                ProgressNode {
173                    children: Vec::new(),
174                    progress: None,
175                    state: if pending {
176                        ProgressState::Pending
177                    } else {
178                        ProgressState::InProgress {
179                            started_at: Utc::now(),
180                            started_instant: Instant::now(),
181                        }
182                    },
183                },
184            ));
185
186            ProgressTracker {
187                root: self.root.clone(),
188                path,
189            }
190        } else {
191            // Should never happen. But if it does, return an obviously invalid
192            // path to avoid a panic.
193            debug_assert!(false, "bug: invalid path when creating subtask");
194            ProgressTracker {
195                root: self.root.clone(),
196                path: vec![usize::MAX, usize::MAX],
197            }
198        }
199    }
200
201    /// Enable progress tracking for this task.
202    ///
203    /// Accepts the total number of units of work, if known.
204    /// Returns a counter that the caller should increment to report progress.
205    /// Before entering hot loops, don't forget to call `Arc::deref` on it.
206    pub fn track_progress(&self, total: Option<u64>) -> Arc<AtomicU64> {
207        let progress = Arc::new(AtomicU64::new(0));
208        let mut root = self.root.lock();
209        if let Some(node) = root.get_mut(&self.path) {
210            debug_assert!(
211                node.progress.is_none(),
212                "usage error: track_progress called multiple times on the same node",
213            );
214            node.progress = Some(NodeProgress {
215                current: progress.clone(),
216                total,
217            });
218        } else {
219            debug_assert!(
220                false,
221                "bug: invalid path when adding adding progress tracking",
222            );
223        }
224        progress
225    }
226
227    /// For tasks created using [`Self::subtask`], mark them as in-progress.
228    pub fn start(&self) {
229        let mut root = self.root.lock();
230        if let Some(node) = root.get_mut(&self.path) {
231            match node.state {
232                ProgressState::Pending => {
233                    node.state = ProgressState::InProgress {
234                        started_at: Utc::now(),
235                        started_instant: Instant::now(),
236                    };
237                }
238                ProgressState::InProgress { .. } | ProgressState::Finished { .. } => (),
239            }
240        } else {
241            debug_assert!(false, "bug: invalid path when starting a task");
242        }
243    }
244}
245
246impl Drop for ProgressTracker {
247    fn drop(&mut self) {
248        ProgressNode::finish(&self.root, &self.path);
249    }
250}
251
252impl ProgressNode {
253    fn get_mut(&mut self, path: &[usize]) -> Option<&mut ProgressNode> {
254        let mut current = &mut *self;
255        for &idx in path {
256            current = &mut current.children.get_mut(idx)?.1;
257        }
258        Some(current)
259    }
260
261    fn render(&self, name: String) -> ProgressTree {
262        let Self {
263            children,
264            progress,
265            state,
266        } = self;
267        let (done, total) = match progress {
268            Some(NodeProgress { current, total }) => {
269                (Some(current.load(Ordering::Relaxed)), *total)
270            }
271            None => (None, None),
272        };
273        let (started_at, finished_at, duration_sec) = match state {
274            ProgressState::Pending => (None, None, None),
275            ProgressState::InProgress { started_at, .. } => (Some(*started_at), None, None),
276            ProgressState::Finished {
277                started_at,
278                finished_at,
279                duration,
280            } => (
281                Some(*started_at),
282                Some(*finished_at),
283                Some(duration.as_secs_f64()),
284            ),
285        };
286        ProgressTree {
287            name,
288            started_at,
289            finished_at,
290            duration_sec,
291            done,
292            total,
293            children: children
294                .iter()
295                .map(|(child_name, child_node)| child_node.render(child_name.clone()))
296                .collect(),
297        }
298    }
299
300    fn finish(root: &Arc<Mutex<ProgressNode>>, path: &[usize]) {
301        let mut root = root.lock();
302        if let Some(node) = root.get_mut(path) {
303            match &node.state {
304                ProgressState::Pending => {
305                    let now = Utc::now();
306                    node.state = ProgressState::Finished {
307                        started_at: now,
308                        finished_at: now,
309                        duration: Duration::from_secs(0),
310                    };
311                }
312                ProgressState::InProgress {
313                    started_at,
314                    started_instant,
315                } => {
316                    let finished_instant = Instant::now();
317                    node.state = ProgressState::Finished {
318                        started_at: *started_at,
319                        finished_at: Utc::now(),
320                        duration: finished_instant.duration_since(*started_instant),
321                    };
322                }
323                ProgressState::Finished { .. } => (),
324            }
325        } else {
326            // Should never happen.
327            debug_assert!(false, "bug: invalid path when finishing a task");
328        }
329    }
330}

339    #[test]
340    #[expect(unused_variables, reason = "testing drop behavior")]
341    fn test_progress_tracker() {
342        let (view, p) = new_progress_tracker();
343
344        let p_foo = p.subtask("foo");
345        let p_bar = p.subtask("bar");
346        let p_baz = p.subtask("baz");
347
348        let p_foo_x = p_foo.subtask("x");
349        let p_foo_y = p_foo.subtask("y");
350        let p_foo_z = p_foo.subtask("z");
351
352        p_foo.start();
353        p_foo_x.start();
354        {
355            let p_foo_x_a = p_foo_x.subtask("a");
356            let p_foo_x_b = p_foo_x.subtask("b");
357            let p_foo_x_c = p_foo_x.subtask("c");
358
359            p_foo_x_a.start();
360            p_foo_x_b.start();
361            // c is not started explicitly, so it becomes finished on drop
362
363            p_foo_x_a
364                .track_progress(Some(7))
365                .store(5, Ordering::Relaxed);
366        }
367        drop(p_foo_x);
368
369        p_foo_y.start();
370        {
371            let p_foo_y_a = p_foo_y.subtask("a");
372            let p_foo_y_b = p_foo_y.subtask("b");
373            let p_foo_y_c = p_foo_y.subtask("c");
374
375            p_foo_y_a.start();
376
377            p_foo_y_a.track_progress(None).store(3, Ordering::Relaxed);
378
379            check_state(
380                &view,
381                "
382                    :in-progress {
383                        foo:in-progress {
384                            x:finished {
385                                a:finished[5/7] {}
386                                b:finished {}
387                                c:finished {}
388                            }
389                            y:in-progress {
390                                a:in-progress[3/?] {}
391                                b:pending {}
392                                c:pending {}
393                            }
394                            z:pending {}
395                        }
396                        bar:pending {}
397                        baz:pending {}
398                    }
399                ",
400            );
401        }
402    }

404    fn test_render(node: &ProgressNode, output: &mut String) {
405        output.push(':');
406        match &node.state {
407            ProgressState::Pending => output.push_str("pending"),
408            ProgressState::InProgress { .. } => output.push_str("in-progress"),
409            ProgressState::Finished { .. } => output.push_str("finished"),
410        }
411        if let Some(progress) = &node.progress {
412            write!(output, "[{}/", progress.current.load(Ordering::Relaxed)).unwrap();
413            if let Some(total) = progress.total {
414                write!(output, "{total}]").unwrap();
415            } else {
416                output.push_str("?]");
417            }
418        }
419        output.push('{');
420        for (child_name, child_node) in &node.children {
421            output.push_str(child_name);
422            test_render(child_node, output);
423        }
424        output.push('}');
425    }
426
427    fn check_state(view: &ProgressView, expected: &str) {
428        let mut rendered = String::new();
429        test_render(&view.root.lock(), &mut rendered);
430        assert_eq!(rendered, expected.replace(&[' ', '\n'][..], ""));
431    }

45#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord, EnumIter)]
46pub enum DetailsLevel {
47    /// Minimal information level
48    /// - app info
49    /// - minimal telemetry by endpoint
50    /// - cluster status
51    Level0,
52    /// Detailed common info level
53    /// - app info details
54    /// - system info
55    ///   - hardware flags
56    ///   - hardware usage per collection
57    ///   - RAM usage
58    /// - cluster basic details
59    /// - collections basic info
60    Level1,
61    /// Detailed consensus info - peers info
62    /// Collections:
63    ///  - detailed config
64    ///  - Shards - basic config
65    Level2,
66    /// Shards:
67    ///  - detailed config
68    ///  - Optimizers info
69    Level3,
70    /// Segment level telemetry
71    Level4,
72}

76        TelemetryDetail {

96/// Tweaks the filtering of deferred points.
97/// Can be used in places where we sometimes must access all points, including deferred ones.
98#[derive(Clone, Copy)]
99pub enum DeferredBehavior {
100    /// Deferred points are not affected nor visible.
101    Exclude,
102
103    /// Deferred points are affected and visible.
104    IncludeAll,
105}
106
107impl DeferredBehavior {
108    /// Apply the behavior to a given `deferred_internal_id`.
109    pub fn apply(&self, deferred_internal_id: Option<PointOffsetType>) -> Option<PointOffsetType> {
110        match self {
111            // Excluding deferred points from the result, if `deferred_internal_id` is set.
112            DeferredBehavior::Exclude => deferred_internal_id,
113
114            // Setting `deferred_internal_id` to `None` results in no points being left out
115            // at deferred-point filtering.
116            DeferredBehavior::IncludeAll => None,
117        }
118    }
119
120    /// Returns `true` if filtering deferred points should be disabled
121    /// and *all* points should be included in the result.
122    pub fn include_all_points(&self) -> bool {
123        matches!(self, DeferredBehavior::IncludeAll)
124    }
125}

127#[cfg(test)]
128mod test {

131    #[test]
132    fn test_deferred_overwrite() {
133        assert_eq!(DeferredBehavior::Exclude.apply(Some(32)), Some(32));
134        assert_eq!(DeferredBehavior::Exclude.apply(None), None);
135
136        // `IncludeAll` always returns `None` because then the filtering of deferred points is disabled.
137        assert_eq!(DeferredBehavior::IncludeAll.apply(Some(32)), None);
138        assert_eq!(DeferredBehavior::IncludeAll.apply(None), None);

1use std::io;
2use std::path::{Path, PathBuf};
3
4use fs_err as fs;
5use fs_extra::dir::CopyOptions;
6
7/// Move directory from one location to another
8///
9/// Handles the case when the source and destination are on different filesystems.
10/// If destination directory exists, the contents of the source directory are moved
11/// into the destination directory, preserving existing files in the destination.
12pub fn move_dir(from: impl Into<PathBuf>, to: impl Into<PathBuf>) -> io::Result<()> {
13    let from = from.into();
14    let to = to.into();
15
16    let Err(_err) = fs::rename(&from, &to) else {
17        return Ok(());
18    };
19
20    if !to.exists() {
21        fs::create_dir(&to)?;
22    }
23
24    let opts = CopyOptions::new().content_only(true).overwrite(true);
25
26    fs_extra::dir::move_dir(&from, &to, &opts).map_err(|err| {
27        io::Error::other(format!(
28            "failed to move directory {} to {}: {err}",
29            from.display(),
30            to.display(),
31        ))
32    })?;
33    Ok(())
34}
35
36/// Move file from one location to another.
37/// Handles the case when the source and destination are on different filesystems.
38pub fn move_file(from: impl AsRef<Path>, to: impl AsRef<Path>) -> io::Result<()> {
39    let from = from.as_ref();
40    let to = to.as_ref();
41
42    // Try to rename first and fallback to copy to prevent TOCTOU.
43    if let Ok(()) = fs::rename(from, to) {
44        return Ok(());
45    }
46
47    // If rename failed, try to copy.
48    // It is possible that the source and destination are on different filesystems.
49    if let Err(err) = fs::copy(from, to) {
50        let _res = cleanup_file(to);
51        return Err(err);
52    }
53
54    if let Err(err) = fs::remove_file(from) {
55        let _res = cleanup_file(to);
56        return Err(err);
57    }
58
59    Ok(())
60}
61
62/// Remove the file if it exists. Print a warning if the file can't be removed.
63fn cleanup_file(path: &Path) -> io::Result<()> {
64    if let Err(err) = fs::remove_file(path)
65        && err.kind() != std::io::ErrorKind::NotFound
66    {
67        return Err(err);
68    }
69    Ok(())
70}

72#[cfg(test)]
73mod tests {
74    use tempfile::tempdir;
75
76    use super::*;
77
78    #[test]
79    fn test_move_dir_preserves_existing_destination_files() {
80        let temp = tempdir().unwrap();
81
82        // Create source directory with a file
83        let source_dir = temp.path().join("source");
84        fs::create_dir(&source_dir).unwrap();
85        fs::write(source_dir.join("source_file.txt"), "source content").unwrap();
86
87        // Create destination directory with a different file
88        let dest_dir = temp.path().join("dest");
89        fs::create_dir(&dest_dir).unwrap();
90        fs::write(dest_dir.join("existing_file.txt"), "existing content").unwrap();
91
92        // Move source to destination
93        move_dir(&source_dir, &dest_dir).unwrap();
94
95        // Verify source file was moved
96        assert!(dest_dir.join("source_file.txt").exists());
97        assert_eq!(
98            fs::read_to_string(dest_dir.join("source_file.txt")).unwrap(),
99            "source content"
100        );
101
102        // Verify existing destination file is preserved
103        assert!(dest_dir.join("existing_file.txt").exists());
104        assert_eq!(
105            fs::read_to_string(dest_dir.join("existing_file.txt")).unwrap(),
106            "existing content"
107        );
108
109        // Verify source directory is removed
110        assert!(!source_dir.exists());
111    }

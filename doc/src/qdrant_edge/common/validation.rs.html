14#[allow(clippy::manual_try_fold)] // `try_fold` can't be used because it shortcuts on Err
15pub fn validate_iter<T: Validate>(iter: impl Iterator<Item = T>) -> Result<(), ValidationErrors> {
16    let errors = iter
17        .filter_map(|v| v.validate().err())
18        .fold(Err(ValidationErrors::new()), |bag, err| {
19            ValidationErrors::merge(bag, "?", Err(err))
20        })
21        .unwrap_err();
22    errors.errors().is_empty().then_some(()).ok_or(errors)
23}

25/// Validate the value is in `[min, max]`
26#[inline]
27pub fn validate_range_generic<N>(
28    value: N,
29    min: Option<N>,
30    max: Option<N>,
31) -> Result<(), ValidationError>
32where
33    N: PartialOrd + Serialize,
34{
35    // If value is within bounds we're good
36    if min.as_ref().map(|min| &value >= min).unwrap_or(true)
37        && max.as_ref().map(|max| &value <= max).unwrap_or(true)
38    {
39        return Ok(());
40    }
41
42    let mut err = ValidationError::new("range");
43    if let Some(min) = min {
44        err.add_param(Cow::from("min"), &min);
45    }
46    if let Some(max) = max {
47        err.add_param(Cow::from("max"), &max);
48    }
49    Err(err)
50}

54    if value.is_empty() {
55        Err(ValidationError::new("not_empty"))
56    } else {
57        Ok(())
58    }

61/// Validate the collection name contains no illegal characters
62///
63/// This does not check the length of the name.
64pub fn validate_collection_name(value: &str) -> Result<(), ValidationError> {
65    const INVALID_CHARS: [char; 11] =
66        ['<', '>', ':', '"', '/', '\\', '|', '?', '*', '\0', '\u{1F}'];
67
68    match INVALID_CHARS.into_iter().find(|c| value.contains(*c)) {
69        Some(c) => {
70            let mut err = ValidationError::new("does_not_contain");
71            err.add_param(Cow::from("pattern"), &c);
72            err.message
73                .replace(format!("collection name cannot contain \"{c}\" char").into());
74            Err(err)
75        }
76        None => Ok(()),
77    }
78}

80/// Validate the collection name contains no illegal characters, legacy edition
81///
82/// Similar to [`validate_collection_name`], but this still allows some special characters that
83/// were supported pre Qdrant 1.5. More specifically, this only disallows characters that could
84/// never have been used on both Linux and Windows filesystems.
85///
86/// This does not check the length of the name.
87pub fn validate_collection_name_legacy(value: &str) -> Result<(), ValidationError> {
88    // Disallowed characters on on both Linux/Windows, sourced from: <https://stackoverflow.com/a/31976060/1000145>
89    const INVALID_CHARS: [char; 2] = ['/', '\0'];
90
91    match INVALID_CHARS.into_iter().find(|c| value.contains(*c)) {
92        Some(c) => {
93            let mut err = ValidationError::new("does_not_contain");
94            err.add_param(Cow::from("pattern"), &c);
95            err.message
96                .replace(format!("collection name cannot contain \"{c}\" char").into());
97            Err(err)
98        }
99        None => Ok(()),
100    }
101}

103/// Validate a polygon has at least 4 points and is closed.
104pub fn validate_geo_polygon<T>(points: &[T]) -> Result<(), ValidationError>
105where
106    T: PartialEq,
107{
108    let min_length = 4;
109    if points.len() < min_length {
110        let mut err = ValidationError::new("min_polygon_length");
111        err.add_param(Cow::from("length"), &points.len());
112        err.add_param(Cow::from("min_length"), &min_length);
113        return Err(err);
114    }
115
116    let first_point = &points[0];
117    let last_point = &points[points.len() - 1];
118    if first_point != last_point {
119        return Err(ValidationError::new("closed_polygon"));
120    }
121
122    Ok(())
123}

125/// Validate that shard request has two different peers
126///
127/// We do allow transferring from/to the same peer if the source and target shard are different.
128/// This may be used during resharding shard transfers.
129pub fn validate_shard_different_peers(
130    from_peer_id: u64,
131    to_peer_id: u64,
132    shard_id: u32,
133    to_shard_id: Option<u32>,
134) -> Result<(), ValidationErrors> {
135    if to_peer_id != from_peer_id {
136        return Ok(());
137    }
138
139    // If source and target shard is different, we do allow transferring from/to the same peer
140    if to_shard_id.is_some_and(|to_shard_id| to_shard_id != shard_id) {
141        return Ok(());
142    }
143
144    let mut errors = ValidationErrors::new();
145    errors.add("to_peer_id", {
146        let mut error = ValidationError::new("must_not_match");
147        error.add_param(Cow::from("value"), &to_peer_id.to_string());
148        error.add_param(Cow::from("other_field"), &"from_peer_id");
149        error.add_param(Cow::from("other_value"), &from_peer_id.to_string());
150        error.add_param(
151            Cow::from("message"),
152            &format!("cannot transfer shard to itself, \"to_peer_id\" must be different than {from_peer_id} in \"from_peer_id\""),
153        );
154        error
155    });
156    Err(errors)
157}

159/// Validate optional lowercase hexadecimal sha256 hash string.
160pub fn validate_sha256_hash(value: &str) -> Result<(), ValidationError> {
161    if value.len() != 64 {
162        let mut err = ValidationError::new("invalid_sha256_hash");
163        err.add_param(Cow::from("length"), &value.len());
164        err.add_param(Cow::from("expected_length"), &64);
165        return Err(err);
166    }
167
168    if !value.chars().all(|c| c.is_ascii_hexdigit()) {
169        let mut err = ValidationError::new("invalid_sha256_hash");
170        err.add_param(
171            Cow::from("message"),
172            &"invalid characters, expected 0-9, a-f, A-F",
173        );
174        return Err(err);
175    }
176
177    Ok(())
178}

180pub fn validate_multi_vector_by_length(multivec_length: &[usize]) -> Result<(), ValidationErrors> {
181    // non_empty
182    if multivec_length.is_empty() {
183        let mut errors = ValidationErrors::default();
184        let mut err = ValidationError::new("empty_multi_vector");
185        err.add_param(Cow::from("message"), &"multi vector must not be empty");
186        errors.add("data", err);
187        return Err(errors);
188    }
189
190    // check all individual vectors non-empty
191    if multivec_length.contains(&0) {
192        let mut errors = ValidationErrors::default();
193        let mut err = ValidationError::new("empty_vector");
194        err.add_param(Cow::from("message"), &"all vectors must be non-empty");
195        errors.add("data", err);
196        return Err(errors);
197    }
198
199    // total size of all vectors must be less than MAX_MULTIVECTOR_FLATTENED_LEN
200    let flattened_len = multivec_length.iter().sum::<usize>();
201    if flattened_len >= MAX_MULTIVECTOR_FLATTENED_LEN {
202        let mut errors = ValidationErrors::default();
203        let mut err = ValidationError::new("multi_vector_too_large");
204        err.add_param(Cow::from("message"), &format!("Total size of all vectors ({flattened_len}) must be less than {MAX_MULTIVECTOR_FLATTENED_LEN}"));
205        errors.add("data", err);
206        return Err(errors);
207    }
208
209    // all vectors must have the same length
210    let dim = multivec_length[0];
211    if let Some(bad_vec) = multivec_length.iter().find(|v| **v != dim) {
212        let mut errors = ValidationErrors::default();
213        let mut err = ValidationError::new("inconsistent_multi_vector");
214        err.add_param(
215            Cow::from("message"),
216            &format!(
217                "all vectors must have the same dimension, found vector with dimension {bad_vec}",
218            ),
219        );
220        errors.add("data", err);
221        return Err(errors);
222    }
223
224    Ok(())
225}

227pub fn validate_multi_vector<T>(multivec: &[Vec<T>]) -> Result<(), ValidationErrors> {
228    let multivec_length: Vec<_> = multivec.iter().map(|v| v.len()).collect();
229    validate_multi_vector_by_length(&multivec_length)
230}

232pub fn validate_multi_vector_len(
233    vectors_count: u32,
234    flatten_dense_vector: &[f32],
235) -> Result<(), ValidationErrors> {
236    if vectors_count == 0 {
237        let mut errors = ValidationErrors::default();
238        let mut err = ValidationError::new("invalid_vector_count");
239        err.add_param(
240            Cow::from("vectors_count"),
241            &"vectors count must be greater than 0",
242        );
243        errors.add("data", err);
244        return Err(errors);
245    }
246
247    let dense_vector_len = flatten_dense_vector.len();
248    if dense_vector_len >= MAX_MULTIVECTOR_FLATTENED_LEN {
249        let mut errors = ValidationErrors::default();
250        let mut err = ValidationError::new("Vector size is too large");
251        err.add_param(Cow::from("vector_len"), &dense_vector_len);
252        err.add_param(Cow::from("vectors_count"), &vectors_count);
253        errors.add("data", err);
254        return Err(errors);
255    }
256
257    if !dense_vector_len.is_multiple_of(vectors_count as usize) {
258        let mut errors = ValidationErrors::default();
259        let mut err = ValidationError::new("invalid dense vector length for vectors count");
260        err.add_param(Cow::from("vector_len"), &dense_vector_len);
261        err.add_param(Cow::from("vectors_count"), &vectors_count);
262        errors.add("data", err);
263        Err(errors)
264    } else {
265        Ok(())
266    }
267}

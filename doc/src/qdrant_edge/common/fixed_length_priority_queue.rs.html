1use std::cmp::Reverse;
2use std::collections::BinaryHeap;
3use std::num::NonZeroUsize;
4use std::vec::IntoIter as VecIntoIter;
5
6use bytemuck::{TransparentWrapper as _, TransparentWrapperAlloc as _};
7use serde::{Deserialize, Serialize};
8
9/// To avoid excessive memory allocation, FixedLengthPriorityQueue
10/// imposes a reasonable limit on the allocation size. If the limit
11/// is extremely large, we treat it as if no limit was set and
12/// delay allocation, assuming that the results will fit within a
13/// predefined threshold.
14const LARGEST_REASONABLE_ALLOCATION_SIZE: usize = 1_048_576;
15
16/// A container that forgets all but the top N elements
17///
18/// This is a MinHeap by default - it will keep the largest elements, pop smallest
19#[derive(Deserialize, Serialize, Clone, Debug)]
20pub struct FixedLengthPriorityQueue<T: Ord> {
21    heap: BinaryHeap<Reverse<T>>,
22    length: NonZeroUsize,
23}
24
25impl<T: Ord> Default for FixedLengthPriorityQueue<T> {
26    fn default() -> Self {
27        Self::new(1)
28    }
29}
30
31impl<T: Ord> FixedLengthPriorityQueue<T> {
32    /// Creates a new queue with the given length
33    /// Panics if length is 0
34    pub fn new(length: usize) -> Self {
35        let heap = BinaryHeap::with_capacity(
36            length
37                .saturating_add(1)
38                .min(LARGEST_REASONABLE_ALLOCATION_SIZE),
39        );
40        let length = NonZeroUsize::new(length).expect("length must be greater than zero");
41        FixedLengthPriorityQueue::<T> { heap, length }
42    }
43
44    /// Pushes a value into the priority queue.
45    ///
46    /// If the queue if full, replaces the smallest value and returns it.
47    pub fn push(&mut self, value: T) -> Option<T> {
48        if !self.is_full() {
49            self.heap.push(Reverse(value));
50            return None;
51        }
52
53        let mut x = self.heap.peek_mut().unwrap();
54        let mut value = Reverse(value);
55        if x.0 < value.0 {
56            std::mem::swap(&mut *x, &mut value);
57        }
58        Some(value.0)
59    }
60
61    /// Consumes the [`FixedLengthPriorityQueue`] and returns a vector
62    /// in sorted (descending) order.
63    pub fn into_sorted_vec(self) -> Vec<T> {
64        Reverse::peel_vec(self.heap.into_sorted_vec())
65    }
66
67    /// Returns an iterator over the elements in the queue, in arbitrary order.
68    pub fn iter_unsorted(&self) -> std::slice::Iter<'_, T> {
69        Reverse::peel_slice(self.heap.as_slice()).iter()
70    }
71
72    /// Returns an iterator over the elements in the queue
73    /// in sorted (descending) order.
74    pub fn into_iter_sorted(self) -> VecIntoIter<T> {
75        self.into_sorted_vec().into_iter()
76    }
77
78    /// Returns the smallest element of the queue,
79    /// if there is any.
80    pub fn top(&self) -> Option<&T> {
81        self.heap.peek().map(|x| &x.0)
82    }
83
84    /// Returns actual length of the queue
85    pub fn len(&self) -> usize {
86        self.heap.len()
87    }
88
89    /// Checks if the queue is empty
90    pub fn is_empty(&self) -> bool {
91        self.heap.is_empty()
92    }
93
94    /// Checks if the queue is full
95    pub fn is_full(&self) -> bool {
96        self.heap.len() >= self.length.into()
97    }
98
99    pub fn retain<F>(&mut self, mut f: F)
100    where
101        F: FnMut(&T) -> bool,
102    {
103        self.heap.retain(|x| f(&x.0));
104    }
105}

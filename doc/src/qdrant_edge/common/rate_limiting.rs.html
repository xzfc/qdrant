1use std::time::{Duration, Instant};
2
3/// A rate limiter based on the token bucket algorithm.
4/// Designed to limit the number of requests per minute.
5/// The bucket is refilled at a constant rate of `tokens_per_sec` tokens per second.
6/// The bucket has a maximum capacity of `capacity_per_minute` tokens to allow for bursts.
7#[derive(Debug)]
8pub struct RateLimiter {
9    // Maximum tokens the bucket can hold.
10    capacity_per_minute: u64,
11    // Tokens added per second.
12    tokens_per_sec: f64,
13    // Current tokens in the bucket.
14    tokens: f64,
15    // Last time tokens were updated.
16    last_check: Instant,
17}
18
19impl RateLimiter {
20    /// Create a new rate limiter for `requests_num` requests per minute.
21    pub fn new_per_minute(requests_num: usize) -> Self {
22        let tokens_per_sec = requests_num as f64 / 60.0;
23        RateLimiter {
24            capacity_per_minute: requests_num as u64,
25            tokens_per_sec,
26            tokens: requests_num as f64, // Start with a full bucket to allow burst at the beginning.
27            last_check: Instant::now(),
28        }
29    }
30
31    /// Attempt to consume a given number of tokens.
32    ///
33    /// Returns:
34    /// - `Ok(())` if allowed and consumes the tokens.
35    /// - `Err(RateLimitError)` if denied.
36    pub fn try_consume(&mut self, tokens: f64) -> Result<(), RateLimitError> {
37        // Consumer wants more than maximum capacity, that's impossible
38        if tokens > self.capacity_per_minute as f64 {
39            return Err(RateLimitError::AlwaysOverBudget(
40                "request larger than rate limiter capacity, please try to split your request",
41            ));
42        }
43
44        let now = Instant::now();
45        let elapsed = now.duration_since(self.last_check);
46        self.last_check = now;
47
48        // Refill tokens based on elapsed time.
49        self.tokens += self.tokens_per_sec * elapsed.as_secs_f64();
50        if self.tokens > self.capacity_per_minute as f64 {
51            self.tokens = self.capacity_per_minute as f64;
52        }
53
54        if self.tokens >= tokens {
55            self.tokens -= tokens; // Consume `cost` tokens.
56            Ok(()) // Request allowed.
57        } else {
58            let missing_tokens = tokens - self.tokens;
59            let retry_after = Duration::from_secs_f64(missing_tokens / self.tokens_per_sec);
60            debug_assert!(retry_after > Duration::from_secs(0));
61            let retry_error = RetryError {
62                tokens_available: self.tokens,
63                retry_after,
64            };
65            Err(RateLimitError::Retry(retry_error))
66        }
67    }
68}
69
70#[derive(Clone, Copy, Debug, PartialEq)]
71pub struct RetryError {
72    /// Number of tokens that were available at the time of the request but didn't suffice.
73    pub tokens_available: f64,
74    /// Estimated time to wait before retrying request
75    pub retry_after: Duration,
76}
77
78/// Error when too many tokens have been tried to consume from the rate limiter.
79#[derive(Clone, Copy, Debug, PartialEq)]
80pub enum RateLimitError {
81    /// Operation that will always be over budget.
82    AlwaysOverBudget(&'static str),
83
84    /// Operation that can be retried later.
85    Retry(RetryError),
86}

88#[cfg(test)]
89mod tests {
90    use super::*;
91
92    fn assert_eq_floats(a: f64, b: f64, tolerance: f64) {
93        assert!(
94            (a - b).abs() < tolerance,
95            "assertion failed: `(left == right)` (left: `{a}`, right: `{b}`, tolerance: `{tolerance}`)",
96        );
97    }
98
99    #[test]
100    fn test_rate_one_per_minute() {
101        let mut limiter = RateLimiter::new_per_minute(1);
102        assert_eq!(limiter.capacity_per_minute, 1);
103        assert_eq_floats(limiter.tokens_per_sec, 0.016, 0.001);
104        assert_eq!(limiter.tokens, 1.0);
105
106        assert_eq!(limiter.try_consume(1.0), Ok(()));
107        assert_eq!(limiter.tokens, 0.0);
108
109        // rate limit reached
110        assert!(limiter.try_consume(1.0).is_err());
111    }
112
113    #[test]
114    fn test_rate_more_per_minute() {
115        let mut limiter = RateLimiter::new_per_minute(600);
116        assert_eq!(limiter.capacity_per_minute, 600);
117        assert_eq!(limiter.tokens_per_sec, 10.0);
118        assert_eq!(limiter.tokens, 600.0);
119
120        assert_eq!(limiter.try_consume(1.0), Ok(()));
121        assert_eq!(limiter.tokens, 599.0);
122
123        assert_eq!(limiter.try_consume(10.0), Ok(()));
124        assert!((589.0..=590.0).contains(&limiter.tokens));
125    }
126
127    #[test]
128    fn test_rate_huge_request() {
129        let mut limiter = RateLimiter::new_per_minute(100);
130
131        // request too large to ever pass the rate limiter
132        assert!(limiter.try_consume(99999.0).is_err());
133    }

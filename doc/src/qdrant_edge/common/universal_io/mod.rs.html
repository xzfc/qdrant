54#[derive(thiserror::Error, Debug)]
55pub enum UniversalIoError {
56    #[error(transparent)]
57    Io(#[from] std::io::Error),
58    #[error(transparent)]
59    Mmap(#[from] crate::common::mmap::Error),
60    #[error("Data range {start}..{end} is out of bounds (data size: {data_length} elements)")]
61    OutOfBounds {
62        start: u64,
63        end: u64,
64        data_length: usize,
65    },
66    #[error(transparent)]
67    SerdeJson(#[from] serde_json::Error),
68    /// Path does not exist or is not accessible; backends may use this instead of
69    /// `Io(NotFound)` so callers can match without relying on a specific io::ErrorKind.
70    #[error("Not found: {path:?}")]
71    NotFound { path: PathBuf },
72    /// Source id is not valid for this multi-source storage.
73    #[error("Invalid source id {file_index} (num sources: {num_files})")]
74    InvalidFileIndex { file_index: usize, num_files: usize },
75}

77/// Open a file via universal io, read it as a whole, and deserialize as JSON.
78///
79/// Uses a single logical read when the backend overrides [`UniversalRead::read_whole`].
80pub fn read_json_via<S, T>(path: impl AsRef<Path>) -> Result<T>
81where
82    S: UniversalRead<u8> + Sized,
83    T: DeserializeOwned,
84{
85    let options = OpenOptions {
86        need_sequential: false,
87        disk_parallel: None,
88        populate: Some(false),
89        advice: Some(AdviceSetting::Advice(Advice::Sequential)),
90    };
91
92    let storage = S::open(path, options)?;
93    let bytes = storage.read_whole()?;
94    serde_json::from_slice(&bytes).map_err(UniversalIoError::from)
95}

15impl SegmentHolder {
16    /// Flushes all segments and returns maximum version to persist
17    ///
18    /// Before flushing, this read-locks all segments. It prevents writes to all not-yet-flushed
19    /// segments during flushing. All locked segments are flushed and released one by one.
20    ///
21    /// If there are unsaved changes after flush - detects lowest unsaved change version.
22    /// If all changes are saved - returns max version.
23    pub fn flush_all(&self, sync: bool, force: bool) -> OperationResult<SeqNumberType> {
24        let lock_order: Vec<_> = self.non_appendable_then_appendable_segments_ids().collect();
25
26        // Grab and keep to segment RwLock's until the end of this function
27        let segments = self.read_segment_locks(lock_order.iter().cloned())?;
28
29        // We can never have zero segments
30        // Having zero segments could permanently corrupt the WAL by acknowledging u64::MAX
31        assert!(
32            !segments.is_empty(),
33            "must always have at least one segment",
34        );
35
36        // Read-lock all segments before flushing any, must prevent any writes to any segment
37        // That is to prevent any copy-on-write operation on two segments from occurring in between
38        // flushing the two segments. If that would happen, segments could end up in an
39        // inconsistent state on disk that is not recoverable after a crash.
40        //
41        // E.g.: we have a point on an immutable segment. If we use a set-payload operation, we do
42        // copy-on-write. The point from immutable segment A is deleted, the updated point is
43        // stored on appendable segment B.
44        // Because of flush ordering segment B (appendable) is flushed before segment A
45        // (not-appendable). If the copy-on-write operation happens in between, the point is
46        // deleted from A, but the new point in B is not persisted. We cannot recover this by
47        // replaying the WAL in case of a crash because the point in A does not exist anymore,
48        // making copy-on-write impossible.
49        // Locking all segments prevents copy-on-write operations from occurring in between
50        // flushes.
51        //
52        // WARNING: Ordering is very important here. Specifically:
53        // - We MUST lock non-appendable first, then appendable.
54        // - We MUST flush according to the copy-on-write dependency graph.
55        // Because of this, two rev(erse) calls are used below here.
56        //
57        // Locking must happen in this order because `apply_points_to_appendable` can take two
58        // write locks, also in this order. If we'd use different ordering we will eventually end
59        // up with a deadlock.
60        let mut segment_reads: Vec<_> = segments
61            .iter()
62            .map(|segment| Self::aloha_lock_segment_read(segment))
63            .collect();
64
65        let max_applied_version = segment_reads.iter().map(|s| s.version()).max().unwrap_or(0);
66
67        if !sync && self.is_background_flushing() {
68            // There is already a background flush ongoing, return current max persisted version
69            return Ok(self.get_max_persisted_version(segment_reads, lock_order));
70        }
71
72        // This lock also prevents multiple parallel sync flushes
73        // as it is exclusive
74        let mut background_flush_lock = self.lock_flushing()?;
75
76        sort_permutation(&mut segment_reads, &lock_order, |segment_ids| {
77            self.sort_segment_ids_by_flush_dependency(segment_ids)
78        });
79
80        // Capture all flushers first to improve data consistency
81        let flushers: Vec<_> = segment_reads
82            .iter()
83            .filter_map(|read_segment| read_segment.flusher(force))
84            .collect();
85
86        if sync {
87            for flusher in flushers {
88                flusher()?;
89            }
90            self.flush_dependency
91                .lock()
92                .retain(|_, _, version| *version > max_applied_version);
93        } else {
94            let flush_dependency = self.flush_dependency.clone();
95            *background_flush_lock = Some(
96                std::thread::Builder::new()
97                    .name("background_flush".to_string())
98                    .spawn(move || {
99                        for flusher in flushers {
100                            flusher()?;
101                        }
102                        flush_dependency
103                            .lock()
104                            .retain(|_, _, version| *version > max_applied_version);
105                        Ok(())
106                    })
107                    .unwrap(),
108            );
109        }
110
111        Ok(self.get_max_persisted_version(segment_reads, lock_order))
112    }

114    fn non_appendable_then_appendable_segments_ids(&self) -> impl Iterator<Item = SegmentId> {
115        let non_appendable_segments = self.non_appendable_segments_ids();
116        let appendable_segments = self.appendable_segments_ids();
117
118        non_appendable_segments
119            .into_iter()
120            .chain(appendable_segments)
121    }

123    fn sort_segment_ids_by_flush_dependency(&self, segment_ids: &[SegmentId]) -> Vec<SegmentId> {
124        let flush_topology = self.flush_dependency.lock().clone();
125        trace!("{flush_topology:?}");
126
127        let mut iter = flush_topology.sort_elements(segment_ids);
128        let sorted_keys: Vec<_> = iter.by_ref().collect();
129
130        // Collect any remaining elements (circular dependencies)
131        let remaining = iter.into_unordered_vec();
132
133        debug_assert!(
134            remaining.is_empty(),
135            "circular dependencies detected in flush topology: {remaining:?}",
136        );
137        #[cfg(feature = "staging")]
138        if !remaining.is_empty() {
139            log::warn!(
140                "circular dependencies detected in flush topology for segments: {remaining:?}"
141            );
142        }
143
144        // Build combined order: sorted first, then unordered remainder
145        let mut final_order: Vec<_> = sorted_keys;
146        final_order.extend(remaining);
147        final_order
148    }

150    // Joins flush thread if exists
151    // Returns lock to guarantee that there will be no other flush in a different thread
152    pub(super) fn lock_flushing(
153        &self,
154    ) -> OperationResult<parking_lot::MutexGuard<'_, Option<JoinHandle<OperationResult<()>>>>> {
155        let mut lock = self.flush_thread.lock();
156        let mut join_handle: Option<JoinHandle<OperationResult<()>>> = None;
157        std::mem::swap(&mut join_handle, &mut lock);
158        if let Some(join_handle) = join_handle {
159            // Flush result was reported to segment, so we don't need this value anymore
160            let flush_result = join_handle
161                .join()
162                .map_err(|_err| OperationError::service_error("failed to join flush thread"))?;
163            flush_result.map_err(|err| {
164                OperationError::service_error(format!("last background flush failed: {err}"))
165            })?;
166        }
167        Ok(lock)
168    }
169
170    pub(super) fn is_background_flushing(&self) -> bool {
171        let lock = self.flush_thread.lock();
172        if let Some(join_handle) = lock.as_ref() {
173            !join_handle.is_finished()
174        } else {
175            false
176        }
177    }

179    /// Calculates the version of the segments that is safe to acknowledge in WAL
180    ///
181    /// If there are unsaved changes after flush - detects lowest unsaved change version.
182    /// If all changes are saved - returns max version.
183    fn get_max_persisted_version(
184        &self,
185        segment_reads: Vec<RwLockReadGuard<'_, dyn NonAppendableSegmentEntry>>,
186        lock_order: Vec<SegmentId>,
187    ) -> SeqNumberType {
188        // Start with the max_persisted_vesrion at the set overwrite value, which may just be 0
189        // Any of the segments we flush may increase this if they have a higher persisted version
190        // The overwrite is required to ensure we acknowledge no-op operations in WAL that didn't hit any segment
191        //
192        // Only affects returned version if all changes are saved
193        let mut max_persisted_version: SeqNumberType = self
194            .max_persisted_segment_version_overwrite
195            .load(Ordering::Relaxed);
196
197        let mut min_unsaved_version: SeqNumberType = SeqNumberType::MAX;
198        let mut has_unsaved = false;
199
200        for (read_segment, segment_id) in segment_reads.into_iter().zip(lock_order.into_iter()) {
201            let segment_version = read_segment.version();
202            let segment_persisted_version = read_segment.persistent_version();
203
204            log::trace!(
205                "Flushed segment {segment_id}:{:?} version: {segment_version} to persisted: {segment_persisted_version}",
206                &read_segment.data_path(),
207            );
208
209            if segment_version > segment_persisted_version {
210                has_unsaved = true;
211                min_unsaved_version = min(min_unsaved_version, segment_persisted_version);
212            }
213
214            max_persisted_version = max(max_persisted_version, segment_persisted_version);
215
216            drop(read_segment);
217        }
218
219        if has_unsaved {
220            log::trace!(
221                "Some segments have unsaved changes, lowest unsaved version: {min_unsaved_version}"
222            );
223            min_unsaved_version
224        } else {
225            log::trace!(
226                "All segments flushed successfully, max persisted version: {max_persisted_version}"
227            );
228            max_persisted_version
229        }
230    }

232    /// Grab the RwLock's for all the given segment IDs.
233    fn read_segment_locks(
234        &self,
235        segment_ids: impl IntoIterator<Item = SegmentId>,
236    ) -> OperationResult<Vec<&RwLock<dyn NonAppendableSegmentEntry>>> {
237        segment_ids
238            .into_iter()
239            .map(|segment_id| {
240                self.get(segment_id)
241                    .ok_or_else(|| {
242                        OperationError::service_error(format!("No segment with ID {segment_id}"))
243                    })
244                    .map(LockedSegment::get_non_appendable)
245            })
246            .collect()
247    }

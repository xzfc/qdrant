169    pub fn remove(&mut self, remove_ids: &[SegmentId]) -> Vec<LockedSegment> {
170        let mut removed_segments = vec![];
171        for remove_id in remove_ids {
172            let removed_segment = self.appendable_segments.remove(remove_id);
173            if let Some(segment) = removed_segment {
174                removed_segments.push(segment);
175            }
176            let removed_segment = self.non_appendable_segments.remove(remove_id);
177            if let Some(segment) = removed_segment {
178                removed_segments.push(segment);
179            }
180        }
181        removed_segments
182    }

213    /// * `segment` - segment to replace with

301    /// Get the smallest appendable segment
302    ///
303    /// The returned segment likely has the most capacity for new points, which will help balance
304    /// new incoming data over all segments we have.
305    ///
306    /// This attempts a non-blocking read-lock on all segments to find the smallest one. Segments
307    /// that cannot be read-locked at this time are skipped. If no segment can be read-locked at
308    /// all, a random one is returned.
309    ///
310    /// If capacity is not important use `random_appendable_segment` instead because it is cheaper.
311    pub fn smallest_appendable_segment(&self) -> Option<LockedSegment> {
312        let segment_ids: Vec<_> = self.appendable_segments_ids();
313
314        // Try a non-blocking read lock on all segments and return the smallest one
315        let smallest_segment = segment_ids
316            .iter()
317            .filter_map(|segment_id| self.get(*segment_id))
318            .filter_map(|locked_segment| {
319                match locked_segment
320                    .get()
321                    .try_read()
322                    .map(|segment| segment.max_available_vectors_size_in_bytes())?
323                {
324                    Ok(size) => Some((locked_segment, size)),
325                    Err(err) => {
326                        log::error!("Failed to get segment size, ignoring: {err}");
327                        None
328                    }
329                }
330            })
331            .min_by_key(|(_, segment_size)| *segment_size);
332        if let Some((smallest_segment, _)) = smallest_segment {
333            return Some(LockedSegment::clone(smallest_segment));
334        }
335
336        // Fall back to picking a random segment
337        segment_ids
338            .choose(&mut rand::rng())
339            .and_then(|idx| self.appendable_segments.get(idx).cloned())
340    }

353    /// Select what point IDs to update and delete in each segment
354    ///
355    /// Each external point ID might have multiple point versions across all segments.
356    ///
357    /// This finds all point versions and groups them per segment. The newest point versions are
358    /// selected to be updated, all older versions are marked to be deleted.
359    ///
360    /// Deferred points are never deleted here — the optimizer handles their lifecycle.
361    ///
362    /// Points that are already soft deleted are not included.
363    fn find_points_to_update_and_delete(
364        &self,
365        ids: &[PointIdType],
366    ) -> (
367        AHashMap<SegmentId, Vec<PointIdType>>,
368        AHashMap<SegmentId, Vec<PointIdType>>,
369    ) {
370        // Two-pass approach for order-independent correctness.
371        //
372        // Rules:
373        // - Always update the latest version regardless of deferred status
374        // - Never delete deferred points — optimizer handles them
375        // - Delete older non-deferred copies only if the latest version has a non-deferred copy too
376        // - Keep older non-deferred copies when the latest is deferred
377
378        let segment_count = self.len().max(1);
379        let default_capacity = ids.len() / segment_count;
380
381        // Pass 1: collect all occurrences of each point across segments
382        let mut all_occurrences: AHashMap<PointIdType, PointOccurrences> =
383            AHashMap::with_capacity(ids.len());
384
385        for (segment_id, segment) in self.iter() {
386            let segment_arc = segment.get();
387            let segment_lock = segment_arc.read();
388            let segment_points = Self::segment_points(ids, segment_lock.deref());
389            for segment_point in segment_points {
390                let Some(point_version) = segment_lock.point_version(segment_point) else {
391                    continue;
392                };
393                let is_deferred = segment_lock.point_is_deferred(segment_point);
394                all_occurrences.entry(segment_point).or_default().push((
395                    segment_id,
396                    point_version,
397                    is_deferred,
398                ));
399            }
400        }
401
402        // Pass 2: for each point, determine what to update and what to delete
403        let mut to_update: AHashMap<SegmentId, Vec<PointIdType>> =
404            AHashMap::with_capacity(segment_count);
405        let mut to_delete: AHashMap<SegmentId, Vec<PointIdType>> = AHashMap::new();
406
407        for (point_id, occurrences) in all_occurrences {
408            let latest_version = occurrences
409                .iter()
410                .map(|(_, version, _)| *version)
411                .max()
412                .unwrap();
413
414            let latest_has_non_deferred = occurrences
415                .iter()
416                .any(|(_, version, is_deferred)| *version == latest_version && !*is_deferred);
417
418            for (segment_id, version, is_deferred) in occurrences {
419                if version == latest_version {
420                    // Latest version: always update
421                    to_update
422                        .entry(segment_id)
423                        .or_insert_with(|| Vec::with_capacity(default_capacity))
424                        .push(point_id);
425                } else if !is_deferred && latest_has_non_deferred {
426                    // Older non-deferred copy: safe to delete only if the latest
427                    // version also has a non-deferred copy
428                    to_delete
429                        .entry(segment_id)
430                        .or_insert_with(|| Vec::with_capacity(default_capacity))
431                        .push(point_id);
432                }
433                // Otherwise: deferred copies are never deleted (optimizer handles them),
434                // and older non-deferred copies are kept when the latest is deferred-only
435            }
436        }
437
438        // Assert each segment does not have overlapping updates and deletes
439        debug_assert!(
440            to_update
441                .iter()
442                .filter_map(|(segment_id, updates)| {
443                    to_delete.get(segment_id).map(|deletes| (updates, deletes))
444                })
445                .all(|(updates, deletes)| {
446                    let updates: HashSet<&ExtendedPointId> = HashSet::from_iter(updates);
447                    let deletes = HashSet::from_iter(deletes);
448                    updates.is_disjoint(&deletes)
449                }),
450            "segments should not have overlapping updates and deletes",
451        );
452
453        (to_update, to_delete)
454    }

484    pub fn apply_segments_batched<F>(&self, mut f: F) -> OperationResult<()>
485    where
486        F: FnMut(
487            &mut RwLockWriteGuard<dyn SegmentEntry + 'static>,
488            SegmentId,
489        ) -> OperationResult<bool>,
490    {
491        loop {
492            let mut did_apply = false;
493
494            // It is important to iterate over all segments for each batch
495            // to avoid blocking of a single segment with sequential updates
496            for (segment_id, segment) in self.iter() {
497                did_apply |= f(&mut segment.get().write(), segment_id)?;
498            }
499
500            // No segment update => we're done
501            if !did_apply {
502                break;
503            }
504        }
505
506        Ok(())
507    }

509    /// Apply an operation `point_operation` to a set of points `ids`.
510    ///
511    /// This operation additionally checks if there are older versions of the points we are
512    /// about to update, and deletes those older versions first.
513    /// Older points are obsolete and updating them would bump their version,
514    /// which could make the inconsistent with actual latest version of the point.
515    ///
516    /// In case of delete operations, we must apply them to all versions of a point. Otherwise
517    /// future operations may revive deletions through older point versions.
518    pub fn apply_points<F>(
519        &self,
520        ids: &[PointIdType],
521        hw_counter: &HardwareCounterCell,
522        mut point_operation: F,
523    ) -> OperationResult<usize>
524    where
525        F: FnMut(
526            PointIdType,
527            SegmentId,
528            &mut RwLockWriteGuard<dyn SegmentEntry>,
529        ) -> OperationResult<bool>,
530    {
531        let (to_update, to_delete) = self.find_points_to_update_and_delete(ids);
532
533        // Delete old points first, because we want to handle copy-on-write in multiple proxy segments properly
534        self.delete_points_from_segments(to_delete, hw_counter)?;
535
536        // Apply point operations to selected segments
537        let mut applied_points = 0;
538        for (segment_id, points) in to_update {
539            let segment = self.get(segment_id).unwrap();
540            let segment_arc = segment.get();
541            let mut write_segment = segment_arc.write();
542
543            for point_id in points {
544                let is_applied = point_operation(point_id, segment_id, &mut write_segment)?;
545                applied_points += usize::from(is_applied);
546            }
547        }
548
549        Ok(applied_points)
550    }

552    pub fn delete_points_from_segments(
553        &self,
554        to_delete: AHashMap<SegmentId, Vec<PointIdType>>,
555        hw_counter: &HardwareCounterCell,
556    ) -> OperationResult<()> {
557        for (segment_id, points) in to_delete {
558            let segment = self.get(segment_id).unwrap();
559            let segment_arc = segment.get();
560            let mut write_segment = segment_arc.write();
561
562            for point_id in points {
563                if let Some(version) = write_segment.point_version(point_id) {
564                    write_segment.delete_point(version, point_id, hw_counter)?;
565                }
566            }
567        }
568        Ok(())
569    }

583    /// Try to acquire read lock over the given segment with increasing wait time.
584    /// Should prevent deadlock in case if multiple threads tries to lock segments sequentially.
585    fn aloha_lock_segment_read(
586        segment: &'_ RwLock<dyn NonAppendableSegmentEntry>,
587    ) -> RwLockReadGuard<'_, dyn NonAppendableSegmentEntry> {
588        let mut interval = Duration::from_nanos(100);
589        loop {
590            if let Some(guard) = segment.try_read_for(interval) {
591                return guard;
592            }
593
594            interval = interval.saturating_mul(2);
595            if interval.as_secs() >= 10 {
596                log::warn!(
597                    "Trying to read-lock a segment is taking a long time. This could be a deadlock and may block new updates.",
598                );
599            }
600        }
601    }

603    /// Try to acquire write lock over random segment with increasing wait time.
604    /// Should prevent deadlock in case if multiple threads tries to lock segments sequentially.
605    pub fn aloha_random_write<F>(
606        &self,
607        segment_ids: &[SegmentId],
608        mut apply: F,
609    ) -> OperationResult<bool>
610    where
611        F: FnMut(SegmentId, &mut RwLockWriteGuard<dyn SegmentEntry>) -> OperationResult<bool>,
612    {
613        if segment_ids.is_empty() {
614            return Err(OperationError::service_error(
615                "No appendable segments exists, expected at least one",
616            ));
617        }
618
619        let mut entries: Vec<_> = Vec::with_capacity(segment_ids.len());
620
621        // Try to access each segment first without any timeout (fast)
622        for segment_id in segment_ids {
623            let segment_opt = self.get(*segment_id).map(|x| x.get());
624            match segment_opt {
625                None => {}
626                Some(segment_lock) => {
627                    match segment_lock.try_write() {
628                        None => {}
629                        Some(mut lock) => return apply(*segment_id, &mut lock),
630                    }
631                    // save segments for further lock attempts
632                    entries.push((*segment_id, segment_lock))
633                }
634            };
635        }
636
637        let mut rng = rand::rng();
638        let (segment_id, segment_lock) = entries.choose(&mut rng).unwrap();
639        let mut segment_write = segment_lock.write();
640        apply(*segment_id, &mut segment_write)
641    }

643    /// Apply an operation `point_operation` to a set of points `ids`, and, if necessary, move the
644    /// points into appendable segments.
645    ///
646    /// If the segment containing the point is appendable, then point is updated in-place without moving.
647    /// If the segment containing the point is immutable, then point is moved to a random appendable segment.
648    ///
649    /// Returns set of point ids which were successfully (already) applied to segments.
650    ///
651    /// # Warning
652    ///
653    /// This function must not be used to apply point deletions, and [`apply_points`] must be used
654    /// instead. There are two reasons for this:
655    ///
656    /// 1. moving a point first and deleting it after is unnecessary overhead.
657    /// 2. this leaves older point versions in place, which may accidentally be revived by some
658    ///    other operation later.
659    pub fn apply_points_with_conditional_move<F, G>(
660        &self,
661        op_num: SeqNumberType,
662        ids: &[PointIdType],
663        mut point_operation: F,
664        mut point_cow_operation: G,
665        hw_counter: &HardwareCounterCell,
666    ) -> OperationResult<AHashSet<PointIdType>>
667    where
668        F: FnMut(PointIdType, &mut RwLockWriteGuard<dyn SegmentEntry>) -> OperationResult<bool>,
669        for<'n, 'o, 'p> G: FnMut(PointIdType, &'n mut NamedVectors<'o>, &'p mut Payload),
670    {
671        // Choose random appendable segment from this
672        let appendable_segments = self.appendable_segments_ids();
673
674        let mut applied_points: AHashSet<PointIdType> = Default::default();
675
676        let _ = self.apply_points(ids, hw_counter, |point_id, idx, write_segment| {
677            if let Some(point_version) = write_segment.point_version(point_id)
678                && point_version >= op_num
679            {
680                applied_points.insert(point_id);
681                return Ok(false);
682            }
683
684            let can_apply_operation = !write_segment.is_proxy() && write_segment.is_appendable();
685
686            let is_applied = if can_apply_operation {
687                point_operation(point_id, write_segment)?
688            } else {
689                self.aloha_random_write(
690                    &appendable_segments,
691                    |appendable_idx, appendable_write_segment| {
692                        // If we are moving point from one segment to another,
693                        // we must guarantee, that data in new segment will be persisted before
694                        // deleting point from old segment.
695                        // Do ensure that, we add a flush dependency
696                        self.flush_dependency
697                            .lock()
698                            .add_dependency(idx, appendable_idx, op_num);
699
700                        let mut all_vectors = write_segment.all_vectors(point_id, hw_counter)?;
701                        let mut payload = write_segment.payload(point_id, hw_counter)?;
702
703                        point_cow_operation(point_id, &mut all_vectors, &mut payload);
704
705                        appendable_write_segment.upsert_point(
706                            op_num,
707                            point_id,
708                            all_vectors,
709                            hw_counter,
710                        )?;
711                        appendable_write_segment
712                            .set_full_payload(op_num, point_id, &payload, hw_counter)?;
713
714                        // Keep the source of the CoW operation as the deferred point is invisible until indexing.
715                        if !appendable_write_segment.point_is_deferred(point_id) {
716                            write_segment.delete_point(op_num, point_id, hw_counter)?;
717                        }
718
719                        Ok(true)
720                    },
721                )?
722            };
723            applied_points.insert(point_id);
724            Ok(is_applied)
725        })?;
726        Ok(applied_points)
727    }

729    /// Out of a list of point IDs, select only those that exist in at least one segment.
730    ///
731    /// Optimized for many segments and long lists of IDs:
732    /// - Removes found IDs from consideration for subsequent segments
733    /// - Early terminates when all IDs are found
734    /// - Pre-allocates result set
735    pub fn select_existing_points(&self, ids: Vec<PointIdType>) -> AHashSet<PointIdType> {
736        let mut remaining_ids = ids;
737        if remaining_ids.is_empty() {
738            return AHashSet::new();
739        }
740
741        let mut existing_points = AHashSet::with_capacity(remaining_ids.len());
742
743        // Iterate through segments in proper order (non-appendable first for consistency)
744        for segment in self.non_appendable_then_appendable_segments() {
745            let segment_guard = segment.get().read();
746
747            // Partition remaining IDs: found ones go to existing_points, rest stay in remaining
748            remaining_ids.retain(|&id| {
749                if segment_guard.has_point(id) {
750                    existing_points.insert(id);
751                    false // Remove from remaining
752                } else {
753                    true // Keep in remaining
754                }
755            });
756
757            // Early termination if all points found
758            if remaining_ids.is_empty() {
759                break;
760            }
761        }
762
763        existing_points
764    }

766    /// Create a new appendable segment and add it to the segment holder.
767    ///
768    /// The segment configuration is sourced from the given collection parameters.
769    pub fn create_appendable_segment(
770        &mut self,
771        segments_path: &Path,
772        segment_config: SegmentConfig,
773        payload_index_schema: Arc<SaveOnDisk<PayloadIndexSchema>>,
774        deferred_internal_id: Option<PointOffsetType>,

778            Some(segment_config),

787    /// Build a temporary appendable segment, usually for proxying writes into.
788    ///
789    /// The segment configuration is sourced from the given collection parameters. If none is
790    /// specified this will fall back and clone the configuration of any existing appendable
791    /// segment in the segment holder.
792    ///
793    /// # Errors
794    ///
795    /// Errors if:
796    /// - building the segment fails
797    /// - no segment configuration is provided, and no appendable segment is in the segment holder
798    ///
799    /// # Warning
800    ///
801    /// This builds a segment on disk, but does NOT add it to the current segment holder. That must
802    /// be done explicitly. `save_version` must be true for the segment to be loaded when Qdrant
803    /// restarts.
804    pub fn build_tmp_segment(
805        &self,
806        segments_path: &Path,
807        segment_config: Option<SegmentConfig>,
808        payload_index_schema: Arc<SaveOnDisk<PayloadIndexSchema>>,
809        deferred_internal_id: Option<PointOffsetType>,
810        save_version: bool,
811    ) -> OperationResult<LockedSegment> {
812        let config = match segment_config {
813            // Base config on collection params
814            Some(config) => config,
815
816            // Fall back: base config on existing appendable segment
817            None => self
818                .random_appendable_segment()
819                .ok_or_else(|| {
820                    OperationError::service_error(
821                        "No existing segment to source temporary segment configuration from",
822                    )
823                })?
824                .get()
825                .read()
826                .config()
827                .clone(),
828        };
829
830        let mut segment =
831            build_segment(segments_path, &config, deferred_internal_id, save_version)?;
832
833        // Internal operation.
834        let hw_counter = HardwareCounterCell::disposable();
835
836        let payload_schema_lock = payload_index_schema.read();
837        for (key, schema) in payload_schema_lock.schema.iter() {
838            segment.create_field_index(0, key, Some(schema), &hw_counter)?;
839        }
840
841        Ok(LockedSegment::new(segment))
842    }

844    /// Method tries to remove the segment with the given ID under the following conditions:
845    ///
846    /// - The segment exists in the holder, if not - it is ignored.
847    /// - The segment is a raw segment and not some special proxy segment.
848    /// - The segment is empty.
849    /// - We are not removing the last appendable segment.
850    ///
851    /// Returns `true` if the segment was removed, `false` otherwise.
852    pub fn remove_segment_if_not_needed(&mut self, segment_id: SegmentId) -> OperationResult<bool> {
853        let tmp_segment = {
854            let mut segments = self.remove(&[segment_id]);
855            if segments.is_empty() {
856                // Seems like segment is already removed, ignore
857                return Ok(false);
858            }
859            assert_eq!(segments.len(), 1, "expected exactly one segment");
860            segments.pop().unwrap()
861        };
862
863        // Append a temp segment to collection if it is not empty or there is no other appendable segment
864        if !self.has_appendable_segment()
865            || !tmp_segment.get().read().is_empty()
866            || !tmp_segment.is_original()
867        {
868            log::trace!(
869                "Keeping temporary segment with {} points",
870                tmp_segment.get().read().available_point_count(),
871            );
872            self.add_existing_locked(segment_id, tmp_segment);
873            Ok(false)
874        } else {
875            log::trace!("Dropping temporary segment with no changes");
876            tmp_segment.drop_data()?;
877            Ok(true)
878        }
879    }

889    /// Duplicated points can appear in case of interrupted optimization.
890    /// LocalShard can still work with duplicated points, but it is better to remove them.
891    /// Duplicated points should not affect the search results.
892    ///
893    /// Checks all segments and removes duplicated and outdated points.
894    /// If two points have the same id, the point with the highest version is kept.
895    /// If two points have the same id and version, one of them is kept.
896    pub fn deduplicate_points_tasks(&self) -> Vec<impl Fn() -> OperationResult<usize> + 'static> {
897        let points_to_remove = self.find_duplicated_points();
898
899        points_to_remove
900            .into_iter()
901            .map(|(segment_id, points)| {
902                let locked_segment = self.get(segment_id).unwrap().clone();
903
904                move || {
905                    let mut removed_points = 0;
906                    let segment_arc = locked_segment.get();
907                    let mut write_segment = segment_arc.write();
908
909                    let disposable_hw_counter = HardwareCounterCell::disposable();
910
911                    for &point_id in &points {
912                        if let Some(point_version) = write_segment.point_version(point_id) {
913                            removed_points += 1;
914                            write_segment.delete_point(point_version, point_id, &disposable_hw_counter)?; // Internal operation
915                        }
916                    }
917
918                    log::trace!("Deleted {removed_points} points from segment {segment_id} to deduplicate: {points:?}");
919
920                    OperationResult::Ok(removed_points)
921                }
922            })
923            .collect::<Vec<_>>()
924    }

926    pub fn find_duplicated_points(&self) -> AHashMap<SegmentId, Vec<PointIdType>> {
927        struct DedupPoint {
928            segment_id: SegmentId,
929            point_id: PointIdType,
930            version: Option<SeqNumberType>,
931            is_deferred: bool,
932        }
933
934        let segments = self
935            .iter()
936            .map(|(segment_id, locked_segment)| (segment_id, locked_segment.get()))
937            .collect::<Vec<_>>();
938        let locked_segments = segments
939            .iter()
940            .map(|(segment_id, locked_segment)| (*segment_id, locked_segment.read()))
941            .collect::<BTreeMap<_, _>>();
942
943        // Iterator produces groups of points by point ID
944        let point_group_iter = locked_segments
945            .iter()
946            .map(|(&segment_id, locked_segment)| {
947                locked_segment
948                    .iter_points()
949                    .map(move |point_id| DedupPoint {
950                        segment_id,
951                        point_id,
952                        version: None,
953                        is_deferred: false,
954                    })
955            })
956            .kmerge_by(|a, b| a.point_id < b.point_id)
957            .chunk_by(|entry| entry.point_id);
958
959        let mut points = Vec::new();
960        let mut points_to_remove: AHashMap<SegmentId, Vec<PointIdType>> = AHashMap::new();
961
962        for (point_id, group_iter) in &point_group_iter {
963            // Fill buffer with points of current chunk, need at least 2 points to deduplicate
964            points.clear();
965            points.extend(group_iter);
966            if points.len() < 2 {
967                continue;
968            }
969
970            // Enrich with point version and deferred status
971            for dedup_point in &mut points {
972                let segment = &locked_segments[&dedup_point.segment_id];
973                dedup_point.version = segment.point_version(point_id);
974                dedup_point.is_deferred = segment.point_is_deferred(point_id);
975            }
976
977            // Sort points from highest to lowest version
978            // If versions are equal, sort by segment ID to make the order deterministic.
979            points.sort_unstable_by_key(|p| (Reverse(p.version), p.segment_id));
980
981            let latest_version = points[0].version;
982
983            // Check if the latest version has a non-deferred copy
984            let latest_has_non_deferred = points
985                .iter()
986                .any(|p| p.version == latest_version && !p.is_deferred);
987
988            // Decide which copies to remove:
989            // - Deferred: keep the first, remove duplicates
990            // - Non-deferred with latest version: keep the first (the winner)
991            // - Older non-deferred: remove only if the latest has a non-deferred copy
992            //   (otherwise keep visible until the deferred point is indexed)
993            let mut kept_deferred = false;
994            let mut kept_non_deferred = false;
995
996            for dedup_point in &points {
997                let should_remove = if dedup_point.is_deferred {
998                    let duplicate = kept_deferred;
999                    kept_deferred = true;
1000                    duplicate
1001                } else if dedup_point.version == latest_version && !kept_non_deferred {
1002                    kept_non_deferred = true;
1003                    false
1004                } else {
1005                    latest_has_non_deferred
1006                };
1007
1008                if should_remove {
1009                    points_to_remove
1010                        .entry(dedup_point.segment_id)
1011                        .or_default()
1012                        .push(dedup_point.point_id);
1013                }
1014            }
1015        }
1016
1017        points_to_remove
1018    }

22    pub fn snapshot_manifest(&self) -> OperationResult<SnapshotManifest> {
23        let mut manifest = SnapshotManifest::default();
24
25        for (_, segment) in self.iter() {
26            let segment_manifest = segment.get().read().get_segment_manifest()?;
27            manifest.add(segment_manifest);
28        }
29
30        Ok(manifest)
31    }

33    /// Proxy all shard segments for [`proxy_all_segments_and_apply`].
34    #[allow(clippy::type_complexity)]
35    pub fn proxy_all_segments<'a>(
36        segments_lock: RwLockUpgradableReadGuard<'a, SegmentHolder>,
37        segments_path: &Path,
38        segment_config: Option<SegmentConfig>,
39        payload_index_schema: Arc<SaveOnDisk<PayloadIndexSchema>>,
40        deferred_internal_id: Option<PointOffsetType>,
41    ) -> OperationResult<(
42        Vec<(SegmentId, LockedSegment)>,
43        SegmentId,
44        RwLockUpgradableReadGuard<'a, SegmentHolder>,
45    )> {
46        // This counter will be used to measure operations on temp segment,
47        // which is part of internal process and can be ignored
48        let hw_counter = HardwareCounterCell::disposable();
49
50        // Create temporary appendable segment to direct all proxy writes into
51        let tmp_segment = segments_lock.build_tmp_segment(
52            segments_path,
53            segment_config,
54            payload_index_schema,
55            deferred_internal_id,
56            false,
57        )?;
58
59        // List all segments we want to snapshot
60        let segment_ids = segments_lock.segment_ids();
61
62        // Create proxy for all segments
63        let mut new_proxies = Vec::with_capacity(segment_ids.len());
64        for segment_id in segment_ids {
65            let segment = segments_lock.get(segment_id).unwrap();
66            let proxy = ProxySegment::new(segment.clone());
67
68            // Write segment is fresh, so it has no operations
69            // Operation with number 0 will be applied
70            proxy.replicate_field_indexes(0, &hw_counter, &tmp_segment)?;
71            new_proxies.push((segment_id, proxy));
72        }
73
74        // Save segment version once all payload indices have been converted
75        // If this ends up not being saved due to a crash, the segment will not be used
76        match &tmp_segment {
77            LockedSegment::Original(segment) => {
78                let segment_path = &segment.read().segment_path;
79                SegmentVersion::save(segment_path)?;
80            }
81            LockedSegment::Proxy(_) => unreachable!(),
82        }
83
84        // Replace all segments with proxies
85        // We cannot fail past this point to prevent only having some segments proxified
86        let mut proxies = Vec::with_capacity(new_proxies.len());
87        let mut write_segments = RwLockUpgradableReadGuard::upgrade(segments_lock);
88        for (segment_id, proxy) in new_proxies {
89            // Replicate field indexes the second time, because optimized segments could have
90            // been changed. The probability is small, though, so we can afford this operation
91            // under the full collection write lock
92            let op_num = proxy.version();
93            if let Err(err) = proxy.replicate_field_indexes(op_num, &hw_counter, &tmp_segment) {
94                log::error!("Failed to replicate proxy segment field indexes, ignoring: {err}");
95            }
96
97            // We must keep existing segment IDs because ongoing optimizations might depend on the mapping being the same
98            write_segments.replace(segment_id, proxy)?;
99            let locked_proxy_segment = write_segments
100                .get(segment_id)
101                .cloned()
102                .expect("failed to get segment from segment holder we just swapped in");
103            proxies.push((segment_id, locked_proxy_segment));
104        }
105
106        // Make sure at least one appendable segment exists
107        let temp_segment_id = write_segments.add_new_locked(tmp_segment);
108
109        let segments_lock = RwLockWriteGuard::downgrade_to_upgradable(write_segments);
110
111        Ok((proxies, temp_segment_id, segments_lock))
112    }

114    /// Try to unproxy a single shard segment for [`proxy_all_segments_and_apply`].
115    ///
116    /// # Warning
117    ///
118    /// If unproxying fails an error is returned with the lock and the proxy is left behind in the
119    /// shard holder.
120    pub fn try_unproxy_segment<'a>(
121        segments_lock: RwLockUpgradableReadGuard<'a, SegmentHolder>,
122        segment_id: SegmentId,
123        proxy_segment: LockedSegment,
124        updates_guard: UpdatesGuard<'a>,
125    ) -> Result<
126        RwLockUpgradableReadGuard<'a, SegmentHolder>,
127        RwLockUpgradableReadGuard<'a, SegmentHolder>,
128    > {
129        // We must propagate all changes in the proxy into their wrapped segments, as we'll put the
130        // wrapped segment back into the segment holder. This can be an expensive step,
131        // so it is important, that we don't block reads while doing this.
132
133        let proxy_segment = match proxy_segment {
134            LockedSegment::Proxy(proxy_segment) => proxy_segment,
135            LockedSegment::Original(_) => {
136                log::warn!(
137                    "Unproxying segment {segment_id} that is not proxified, that is unexpected, skipping",
138                );
139                return Err(segments_lock);
140            }
141        };
142
143        // propagate changes to wrapped segment with segment holder read lock
144        {
145            if let Err(err) = proxy_segment.write().propagate_to_wrapped() {
146                log::error!(
147                    "Propagating proxy segment {segment_id} changes to wrapped segment failed, ignoring: {err}",
148                );
149            }
150        }
151
152        let mut write_segments = RwLockUpgradableReadGuard::upgrade(segments_lock);
153
154        let wrapped_segment = proxy_segment.read().wrapped_segment.clone();
155        write_segments.replace(segment_id, wrapped_segment).unwrap();
156
157        drop(updates_guard); // Release updates lock as soon as possible
158
159        // Downgrade write lock to read and give it back
160        Ok(RwLockWriteGuard::downgrade_to_upgradable(write_segments))
161    }

163    /// Unproxy all shard segments for [`proxy_all_segments_and_apply`].
164    pub fn unproxy_all_segments(
165        segments_lock: RwLockUpgradableReadGuard<SegmentHolder>,
166        proxies: Vec<(SegmentId, LockedSegment)>,
167        tmp_segment_id: SegmentId,
168        updates_guard: UpdatesGuard<'_>,
169    ) -> OperationResult<()> {
170        // We must propagate all changes in the proxy into their wrapped segments, as we'll put the
171        // wrapped segment back into the segment holder. This can be an expensive step,
172        // so it is important, that we don't block reads while doing this.
173
174        // propagate changes to wrapped segment with segment holder read lock
175        proxies
176            .iter()
177            .filter_map(|(segment_id, proxy_segment)| match proxy_segment {
178                LockedSegment::Proxy(proxy_segment) => Some((segment_id, proxy_segment)),
179                LockedSegment::Original(_) => None,
180            }).for_each(|(proxy_id, proxy_segment)| {
181            if let Err(err) = proxy_segment.write().propagate_to_wrapped() {
182                log::error!("Propagating proxy segment {proxy_id} changes to wrapped segment failed, ignoring: {err}");
183            }
184        });
185
186        // Swap out each proxy with wrapped segment once changes are propagated
187        let mut write_segments = RwLockUpgradableReadGuard::upgrade(segments_lock);
188        for (segment_id, proxy_segment) in proxies {
189            match proxy_segment {
190                LockedSegment::Proxy(proxy_segment) => {
191                    let wrapped_segment = proxy_segment.read().wrapped_segment.clone();
192                    write_segments.replace(segment_id, wrapped_segment)?;
193                }
194                // If already unproxied, do nothing
195                LockedSegment::Original(_) => {}
196            }
197        }
198
199        debug_assert!(
200            write_segments.get(tmp_segment_id).is_some(),
201            "temp segment must exist",
202        );
203
204        // Remove temporary appendable segment, if we don't need it anymore
205        write_segments.remove_segment_if_not_needed(tmp_segment_id)?;
206
207        drop(updates_guard); // Release updates lock as soon as possible
208
209        Ok(())
210    }

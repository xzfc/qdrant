27pub fn process_point_operation(
28    segments: &SegmentHolder,
29    op_num: SeqNumberType,
30    point_operation: PointOperations,
31    hw_counter: &HardwareCounterCell,
32) -> OperationResult<usize> {
33    match point_operation {
34        PointOperations::UpsertPoints(operation) => {
35            let points = operation.into_point_vec();
36            let res = upsert_points(segments, op_num, points.iter(), hw_counter)?;
37            Ok(res)
38        }
39        PointOperations::UpsertPointsConditional(operation) => {
40            conditional_upsert(segments, op_num, operation, hw_counter)
41        }
42        PointOperations::DeletePoints { ids } => delete_points(segments, op_num, &ids, hw_counter),
43        PointOperations::DeletePointsByFilter(filter) => {
44            delete_points_by_filter(segments, op_num, &filter, hw_counter)
45        }
46        PointOperations::SyncPoints(operation) => {
47            let (deleted, new, updated) = sync_points(
48                segments,
49                op_num,
50                operation.from_id,
51                operation.to_id,
52                &operation.points,
53                hw_counter,
54            )?;
55            Ok(deleted + new + updated)
56        }
57    }
58}

63    op_num: SeqNumberType,

66    match operation {

97pub fn process_payload_operation(
98    segments: &SegmentHolder,
99    op_num: SeqNumberType,
100    payload_operation: PayloadOps,
101    hw_counter: &HardwareCounterCell,
102) -> OperationResult<usize> {
103    match payload_operation {
104        PayloadOps::SetPayload(sp) => {
105            let payload: Payload = sp.payload;
106            if let Some(points) = sp.points {
107                set_payload(segments, op_num, &payload, &points, &sp.key, hw_counter)
108            } else if let Some(filter) = sp.filter {
109                set_payload_by_filter(segments, op_num, &payload, &filter, &sp.key, hw_counter)
110            } else {
111                // TODO: BadRequest (prev) vs BadInput (current)!?
112                Err(OperationError::ValidationError {
113                    description: "No points or filter specified".to_string(),
114                })
115            }
116        }
117        PayloadOps::DeletePayload(dp) => {
118            if let Some(points) = dp.points {
119                delete_payload(segments, op_num, &points, &dp.keys, hw_counter)
120            } else if let Some(filter) = dp.filter {
121                delete_payload_by_filter(segments, op_num, &filter, &dp.keys, hw_counter)
122            } else {
123                // TODO: BadRequest (prev) vs BadInput (current)!?
124                Err(OperationError::ValidationError {
125                    description: "No points or filter specified".to_string(),
126                })
127            }
128        }
129        PayloadOps::ClearPayload { ref points, .. } => {
130            clear_payload(segments, op_num, points, hw_counter)
131        }
132        PayloadOps::ClearPayloadByFilter(ref filter) => {
133            clear_payload_by_filter(segments, op_num, filter, hw_counter)
134        }
135        PayloadOps::OverwritePayload(sp) => {
136            let payload: Payload = sp.payload;
137            if let Some(points) = sp.points {
138                overwrite_payload(segments, op_num, &payload, &points, hw_counter)
139            } else if let Some(filter) = sp.filter {
140                overwrite_payload_by_filter(segments, op_num, &payload, &filter, hw_counter)
141            } else {
142                // TODO: BadRequest (prev) vs BadInput (current)!?
143                Err(OperationError::ValidationError {
144                    description: "No points or filter specified".to_string(),
145                })
146            }
147        }
148    }
149}

171/// Do not insert more than this number of points in a single update operation chunk
172/// This is needed to avoid locking segments for too long, so that
173/// parallel read operations are not starved.
174const UPDATE_OP_CHUNK_SIZE: usize = 32;
175
176/// Checks point id in each segment, update point if found.
177/// All not found points are inserted into random segment.
178/// Returns: number of updated points.
179pub fn upsert_points<'a, T>(
180    segments: &SegmentHolder,
181    op_num: SeqNumberType,
182    points: T,
183    hw_counter: &HardwareCounterCell,
184) -> OperationResult<usize>
185where
186    T: IntoIterator<Item = &'a PointStructPersisted>,
187{
188    let points_map: AHashMap<PointIdType, _> = points.into_iter().map(|p| (p.id, p)).collect();
189    let ids: Vec<PointIdType> = points_map.keys().copied().collect();
190
191    let mut res = 0;
192
193    for ids_chunk in ids.chunks(UPDATE_OP_CHUNK_SIZE) {
194        // Update points in writable segments
195        let updated_points = segments.apply_points_with_conditional_move(
196            op_num,
197            ids_chunk,
198            |id, write_segment| {
199                let point = points_map[&id];
200                upsert_with_payload(
201                    write_segment,
202                    op_num,
203                    id,
204                    point.get_vectors(),
205                    point.payload.as_ref(),
206                    hw_counter,
207                )
208            },
209            |id, vectors, old_payload| {
210                let point = points_map[&id];
211                for (name, vec) in point.get_vectors() {
212                    vectors.insert(name.into(), vec.to_owned());
213                }
214                if let Some(payload) = &point.payload {
215                    *old_payload = payload.clone();
216                }
217            },
218            hw_counter,
219        )?;
220
221        res += updated_points.len();
222        // Insert new points, which was not updated or existed
223        let new_point_ids = ids_chunk
224            .iter()
225            .copied()
226            .filter(|x| !updated_points.contains(x));
227
228        {
229            let default_write_segment =
230                segments.smallest_appendable_segment().ok_or_else(|| {
231                    OperationError::service_error(
232                        "No appendable segments exist, expected at least one",
233                    )
234                })?;
235
236            let segment_arc = default_write_segment.get();
237            let mut write_segment = segment_arc.write();
238            for point_id in new_point_ids {
239                let point = points_map[&point_id];
240                res += usize::from(upsert_with_payload(
241                    &mut write_segment,
242                    op_num,
243                    point_id,
244                    point.get_vectors(),
245                    point.payload.as_ref(),
246                    hw_counter,
247                )?);
248            }
249            RwLockWriteGuard::unlock_fair(write_segment);
250        };
251    }
252
253    Ok(res)
254}

256pub fn conditional_upsert(
257    segments: &SegmentHolder,
258    op_num: SeqNumberType,
259    operation: ConditionalInsertOperationInternal,
260    hw_counter: &HardwareCounterCell,
261) -> OperationResult<usize> {
262    // Find points, which do exist, but don't match the condition.
263    // Exclude those points from the upsert operation.
264
265    let ConditionalInsertOperationInternal {
266        mut points_op,
267        condition,
268        update_mode,
269    } = operation;
270
271    let point_ids = points_op.point_ids();
272    let update_mode = update_mode.unwrap_or_default();
273
274    match update_mode {
275        UpdateMode::Upsert => {
276            // Default behavior: insert new points, update existing points that match the condition
277            let points_to_exclude =
278                select_excluded_by_filter_ids(segments, point_ids, condition, hw_counter)?;
279            points_op.retain_point_ids(|idx| !points_to_exclude.contains(idx));
280        }
281        UpdateMode::InsertOnly => {
282            // Only insert new points, skip all existing points entirely
283            let existing_points = segments.select_existing_points(point_ids);
284            points_op.retain_point_ids(|idx| !existing_points.contains(idx));
285        }
286        UpdateMode::UpdateOnly => {
287            // Only update existing points that match the condition, don't insert new points
288            let points_to_exclude =
289                select_excluded_by_filter_ids(segments, point_ids.clone(), condition, hw_counter)?;
290            let existing_points = segments.select_existing_points(point_ids);
291            // Keep only points that exist AND are not excluded by the condition
292            points_op.retain_point_ids(|idx| {
293                existing_points.contains(idx) && !points_to_exclude.contains(idx)
294            });
295        }
296    }
297
298    let points = points_op.into_point_vec();
299    let upserted_points = upsert_points(segments, op_num, points.iter(), hw_counter)?;
300
301    if upserted_points == 0 {
302        // In case we didn't hit any points, we suggest this op_num to the segment-holder to make WAL acknowledge this operation.
303        // If we don't do this, startup might take up a lot of time in some scenarios because of recovering these no-op operations.
304        segments.bump_max_segment_version_overwrite(op_num);
305    }
306
307    Ok(upserted_points)
308}

310/// Upsert to a point ID with the specified vectors and payload in the given segment.
311///
312/// If the payload is None, the existing payload will be cleared.
313///
314/// Returns
315/// - Ok(true) if the operation was successful and point replaced existing value
316/// - Ok(false) if the operation was successful and point was inserted
317/// - Err if the operation failed
318fn upsert_with_payload(
319    segment: &mut RwLockWriteGuard<dyn SegmentEntry>,
320    op_num: SeqNumberType,
321    point_id: PointIdType,
322    vectors: NamedVectors,
323    payload: Option<&Payload>,
324    hw_counter: &HardwareCounterCell,
325) -> OperationResult<bool> {
326    let mut res = segment.upsert_point(op_num, point_id, vectors, hw_counter)?;
327    if let Some(full_payload) = payload {
328        res &= segment.set_full_payload(op_num, point_id, full_payload, hw_counter)?;
329    } else {
330        res &= segment.clear_payload(op_num, point_id, hw_counter)?;
331    }
332    debug_assert!(
333        segment.has_point(point_id),
334        "the point {point_id} should be present immediately after the upsert"
335    );
336    Ok(res)
337}

339/// Max amount of points to delete in a batched deletion iteration
340const DELETION_BATCH_SIZE: usize = 512;
341
342/// Tries to delete points from all segments, returns number of actually deleted points.
343///
344/// Iterates all segments directly (rather than going through `apply_points`) to ensure
345/// that ALL copies of a point are deleted, including old non-deferred copies in optimized
346/// segments when the latest version is deferred in an appendable segment.
347pub fn delete_points(
348    segments: &SegmentHolder,
349    op_num: SeqNumberType,
350    ids: &[PointIdType],
351    hw_counter: &HardwareCounterCell,
352) -> OperationResult<usize> {
353    let mut total_deleted_points = 0;
354
355    for batch in ids.chunks(DELETION_BATCH_SIZE) {
356        for (_segment_id, segment) in segments.iter() {
357            let segment_arc = segment.get();
358            let mut write_segment = segment_arc.write();
359            for &id in batch {
360                if write_segment.delete_point(op_num, id, hw_counter)? {
361                    total_deleted_points += 1;
362                }
363            }
364        }
365    }
366
367    if total_deleted_points == 0 {
368        segments.bump_max_segment_version_overwrite(op_num);
369    }
370
371    Ok(total_deleted_points)
372}

374/// Deletes points from all segments matching the given filter
375pub fn delete_points_by_filter(
376    segments: &SegmentHolder,
377    op_num: SeqNumberType,
378    filter: &Filter,
379    hw_counter: &HardwareCounterCell,
380) -> OperationResult<usize> {
381    let mut total_deleted = 0;
382    // we don’t want to cancel this filtered read
383    let is_stopped = AtomicBool::new(false);
384    let mut points_to_delete: AHashMap<_, _> = segments
385        .iter()
386        .map(|(segment_id, segment)| {
387            (
388                segment_id,
389                segment.get().read().read_filtered(
390                    None,
391                    None,
392                    Some(filter),
393                    &is_stopped,
394                    hw_counter,
395                    // Delete also deferred points.
396                    DeferredBehavior::IncludeAll,
397                ),
398            )
399        })
400        .collect();
401
402    segments.apply_segments_batched(|s, segment_id| {
403        let Some(curr_points) = points_to_delete.get_mut(&segment_id) else {
404            return Ok(false);
405        };
406        if curr_points.is_empty() {
407            return Ok(false);
408        }
409
410        let mut deleted_in_batch = 0;
411        while let Some(point_id) = curr_points.pop() {
412            if s.delete_point(op_num, point_id, hw_counter)? {
413                total_deleted += 1;
414                deleted_in_batch += 1;
415            }
416
417            if deleted_in_batch >= DELETION_BATCH_SIZE {
418                break;
419            }
420        }
421
422        Ok(true)
423    })?;
424
425    if total_deleted == 0 {
426        // In case we didn't hit any points, we suggest this op_num to the segment-holder to make WAL acknowledge this operation.
427        // If we don't do this, startup might take up a lot of time in some scenarios because of recovering these no-op operations.
428        segments.bump_max_segment_version_overwrite(op_num);
429    }
430
431    Ok(total_deleted)
432}

434/// Sync points within a given [from_id; to_id) range.
435///
436/// 1. Retrieve existing points for a range
437/// 2. Remove points, which are not present in the sync operation
438/// 3. Retrieve overlapping points, detect which one of them are changed
439/// 4. Select new points
440/// 5. Upsert points which differ from the stored ones
441///
442/// Returns:
443///     (number of deleted points, number of new points, number of updated points)
444pub fn sync_points(
445    segments: &SegmentHolder,
446    op_num: SeqNumberType,
447    from_id: Option<PointIdType>,
448    to_id: Option<PointIdType>,
449    points: &[PointStructPersisted],
450    hw_counter: &HardwareCounterCell,
451) -> OperationResult<(usize, usize, usize)> {
452    let id_to_point: AHashMap<PointIdType, _> = points.iter().map(|p| (p.id, p)).collect();
453    let sync_points: AHashSet<_> = points.iter().map(|p| p.id).collect();
454    // 1. Retrieve existing points for a range
455    let stored_point_ids: AHashSet<_> = segments
456        .iter()
457        .flat_map(|(_, segment)| segment.get().read().read_range(from_id, to_id))
458        .collect();
459    // 2. Remove points, which are not present in the sync operation
460    let points_to_remove: Vec<_> = stored_point_ids.difference(&sync_points).copied().collect();
461    let deleted = delete_points(segments, op_num, points_to_remove.as_slice(), hw_counter)?;
462    // 3. Retrieve overlapping points, detect which one of them are changed
463    let existing_point_ids: Vec<_> = stored_point_ids
464        .intersection(&sync_points)
465        .copied()
466        .collect();
467
468    let mut points_to_update: Vec<_> = Vec::new();
469    // we don’t want to cancel this filtered read
470    let is_stopped = AtomicBool::new(false);
471    let _num_updated = segments.read_points(
472        existing_point_ids.as_slice(),
473        &is_stopped,
474        |ids, segment| {
475            let with_vector = WithVector::Bool(true);
476            let with_payload = WithPayload::from(true);
477            // Since we retrieve points, which we already know exist, we expect all of them to be found
478            let stored_records = segment.retrieve(
479                ids,
480                &with_payload,
481                &with_vector,
482                hw_counter,
483                &is_stopped,
484                DeferredBehavior::IncludeAll,
485            )?;
486            let mut updated = 0;
487
488            for (id, stored_record) in stored_records {
489                let point = id_to_point.get(&id).unwrap();
490                if !point.is_equal_to(&stored_record) {
491                    points_to_update.push(*point);
492                    updated += 1;
493                }
494            }
495
496            Ok(updated)
497        },
498    )?;
499
500    // 4. Select new points
501    let num_updated = points_to_update.len();
502    let mut num_new = 0;
503    sync_points.difference(&stored_point_ids).for_each(|id| {
504        num_new += 1;
505        points_to_update.push(*id_to_point.get(id).unwrap());
506    });
507
508    // 5. Upsert points which differ from the stored ones
509    let num_replaced = upsert_points(segments, op_num, points_to_update, hw_counter)?;
510    debug_assert!(
511        num_replaced <= num_updated,
512        "number of replaced points cannot be greater than points to update ({num_replaced} <= {num_updated})",
513    );
514
515    Ok((deleted, num_new, num_updated))
516}

521pub fn update_vectors_conditional(
522    segments: &SegmentHolder,
523    op_num: SeqNumberType,
524    points: UpdateVectorsOp,
525    hw_counter: &HardwareCounterCell,
526) -> OperationResult<usize> {
527    let UpdateVectorsOp {
528        mut points,
529        update_filter,
530    } = points;
531
532    let Some(filter_condition) = update_filter else {
533        return update_vectors(segments, op_num, points, hw_counter);
534    };
535
536    let point_ids: Vec<_> = points.iter().map(|point| point.id).collect();
537
538    let points_to_exclude =
539        select_excluded_by_filter_ids(segments, point_ids, filter_condition, hw_counter)?;
540
541    points.retain(|p| !points_to_exclude.contains(&p.id));
542    update_vectors(segments, op_num, points, hw_counter)
543}

545/// Update the specified named vectors of a point, keeping unspecified vectors intact.
546fn update_vectors(
547    segments: &SegmentHolder,
548    op_num: SeqNumberType,
549    points: Vec<PointVectorsPersisted>,
550    hw_counter: &HardwareCounterCell,
551) -> OperationResult<usize> {
552    // Build a map of vectors to update per point, merge updates on same point ID
553    let mut points_map: AHashMap<PointIdType, NamedVectors> = AHashMap::new();
554    for point in points {
555        let PointVectorsPersisted { id, vector } = point;
556        let named_vector = NamedVectors::from(vector);
557
558        let entry = points_map.entry(id).or_default();
559        entry.merge(named_vector);
560    }
561
562    let ids: Vec<PointIdType> = points_map.keys().copied().collect();
563
564    let mut total_updated_points = 0;
565    for batch in ids.chunks(VECTOR_OP_BATCH_SIZE) {
566        let updated_points = segments.apply_points_with_conditional_move(
567            op_num,
568            batch,
569            |id, write_segment| {
570                let vectors = points_map[&id].clone();
571                write_segment.update_vectors(op_num, id, vectors, hw_counter)
572            },
573            |id, owned_vectors, _| {
574                for (vector_name, vector_ref) in points_map[&id].iter() {
575                    owned_vectors.insert(vector_name.to_owned(), vector_ref.to_owned());
576                }
577            },
578            hw_counter,
579        )?;
580        check_unprocessed_points(batch, &updated_points)?;
581        total_updated_points += updated_points.len();
582    }
583
584    Ok(total_updated_points)
585}

587/// Delete the given named vectors for the given points, keeping other vectors intact.
588pub fn delete_vectors(
589    segments: &SegmentHolder,
590    op_num: SeqNumberType,
591    points: &[PointIdType],
592    vector_names: &[VectorNameBuf],
593    hw_counter: &HardwareCounterCell,
594) -> OperationResult<usize> {
595    let mut total_deleted_points = 0;
596
597    for batch in points.chunks(VECTOR_OP_BATCH_SIZE) {
598        let modified_points = segments.apply_points_with_conditional_move(
599            op_num,
600            batch,
601            |id, write_segment| {
602                let mut res = true;
603                for name in vector_names {
604                    res &= write_segment.delete_vector(op_num, id, name)?;
605                }
606                Ok(res)
607            },
608            |_, owned_vectors, _| {
609                for name in vector_names {
610                    owned_vectors.remove_ref(name);
611                }
612            },
613            hw_counter,
614        )?;
615        check_unprocessed_points(batch, &modified_points)?;
616        total_deleted_points += modified_points.len();
617    }
618
619    Ok(total_deleted_points)
620}

640    Ok(vectors_deleted)
641}

646pub fn set_payload(
647    segments: &SegmentHolder,
648    op_num: SeqNumberType,
649    payload: &Payload,
650    points: &[PointIdType],
651    key: &Option<JsonPath>,
652    hw_counter: &HardwareCounterCell,

654    let mut total_updated_points = 0;
655
656    for chunk in points.chunks(PAYLOAD_OP_BATCH_SIZE) {
657        let updated_points = segments.apply_points_with_conditional_move(
658            op_num,
659            chunk,
660            |id, write_segment| write_segment.set_payload(op_num, id, payload, key, hw_counter),
661            |_, _, old_payload| match key {
662                Some(key) => old_payload.merge_by_key(payload, key),
663                None => old_payload.merge(payload),
664            },
665            hw_counter,
666        )?;
667
668        check_unprocessed_points(chunk, &updated_points)?;
669        total_updated_points += updated_points.len();
670    }
671
672    Ok(total_updated_points)
673}

695pub fn delete_payload(
696    segments: &SegmentHolder,
697    op_num: SeqNumberType,
698    points: &[PointIdType],
699    keys: &[PayloadKeyType],
700    hw_counter: &HardwareCounterCell,
701) -> OperationResult<usize> {
702    let mut total_deleted_points = 0;
703
704    for batch in points.chunks(PAYLOAD_OP_BATCH_SIZE) {
705        let updated_points = segments.apply_points_with_conditional_move(
706            op_num,
707            batch,
708            |id, write_segment| {
709                let mut res = true;
710                for key in keys {
711                    res &= write_segment.delete_payload(op_num, id, key, hw_counter)?;
712                }
713                Ok(res)
714            },
715            |_, _, payload| {
716                for key in keys {
717                    payload.remove(key);
718                }
719            },
720            hw_counter,
721        )?;
722
723        check_unprocessed_points(batch, &updated_points)?;
724        total_deleted_points += updated_points.len();
725    }
726
727    Ok(total_deleted_points)
728}

838pub fn create_field_index(
839    segments: &SegmentHolder,
840    op_num: SeqNumberType,
841    field_name: PayloadKeyTypeRef,
842    field_schema: Option<&PayloadFieldSchema>,
843    hw_counter: &HardwareCounterCell,
844) -> OperationResult<usize> {
845    let Some(field_schema) = field_schema else {
846        return Err(OperationError::TypeInferenceError {
847            field_name: field_name.to_owned(),
848        });
849    };
850
851    segments.apply_segments(|write_segment| {
852        write_segment.with_upgraded(|segment| {
853            segment.delete_field_index_if_incompatible(op_num, field_name, field_schema)
854        })?;
855
856        let (schema, indexes) =
857            match write_segment.build_field_index(op_num, field_name, field_schema, hw_counter)? {
858                BuildFieldIndexResult::SkippedByVersion => {
859                    return Ok(false);
860                }
861                BuildFieldIndexResult::AlreadyExists => {
862                    return Ok(false);
863                }
864                BuildFieldIndexResult::IncompatibleSchema => {
865                    // This is a service error, as we should have just removed the old index
866                    // So it should not be possible to get this error
867                    return Err(OperationError::service_error(format!(
868                        "Incompatible schema for field index on field {field_name}",
869                    )));
870                }
871                BuildFieldIndexResult::Built { schema, indexes } => (schema, indexes),
872            };
873
874        write_segment.with_upgraded(|segment| {
875            segment.apply_field_index(op_num, field_name.to_owned(), schema, indexes)
876        })
877    })
878}

880pub fn delete_field_index(
881    segments: &SegmentHolder,
882    op_num: SeqNumberType,
883    field_name: PayloadKeyTypeRef,

885    segments.apply_segments(|write_segment| {
886        write_segment.with_upgraded(|segment| segment.delete_field_index(op_num, field_name))
887    })

890fn select_excluded_by_filter_ids(
891    segments: &SegmentHolder,
892    point_ids: impl IntoIterator<Item = PointIdType>,
893    filter: Filter,
894    hw_counter: &HardwareCounterCell,
895) -> OperationResult<AHashSet<PointIdType>> {
896    // Filter for points that doesn't match the condition, and have matching
897    let non_match_filter =
898        Filter::new_must_not(Condition::Filter(filter)).with_point_ids(point_ids);
899
900    Ok(points_by_filter(segments, &non_match_filter, hw_counter)?
901        .into_iter()
902        .collect())
903}
904
905fn points_by_filter(
906    segments: &SegmentHolder,
907    filter: &Filter,
908    hw_counter: &HardwareCounterCell,
909) -> OperationResult<Vec<PointIdType>> {
910    let mut affected_points: Vec<PointIdType> = Vec::new();
911    // we don’t want to cancel this filtered read
912    let is_stopped = AtomicBool::new(false);
913    segments.for_each_segment(|s| {
914        let points = s.read_filtered(
915            None,
916            None,
917            Some(filter),
918            &is_stopped,
919            hw_counter,
920            // Read operation used for updates, so we must handle all points
921            DeferredBehavior::IncludeAll,
922        );
923        affected_points.extend_from_slice(points.as_slice());
924        Ok(true)
925    })?;
926    Ok(affected_points)
927}
928
929fn check_unprocessed_points(
930    points: &[PointIdType],
931    processed: &AHashSet<PointIdType>,
932) -> OperationResult<usize> {
933    let first_missed_point = points.iter().copied().find(|p| !processed.contains(p));
934
935    match first_missed_point {
936        None => Ok(processed.len()),
937        Some(missed_point_id) => Err(OperationError::PointIdError { missed_point_id }),
938    }
939}

941#[cfg(test)]
942mod test {
943    use std::sync::Arc;
944
945    use crate::common::counter::hardware_counter::HardwareCounterCell;
946    use parking_lot::RwLock;
947    use crate::segment::types::{Condition, FieldCondition, Filter, Match, MatchValue, ValueVariants};
948    use tempfile::Builder;
949
950    use crate::shard::fixtures::{build_segment_1, build_segment_2};
951    use crate::shard::segment_holder::SegmentHolder;
952    use crate::shard::update::delete_points_by_filter;
953
954    #[test]
955    fn test_delete_by_filter_version_bump() {
956        let dir = Builder::new().prefix("segment_dir").tempdir().unwrap();
957
958        let segment1 = build_segment_1(dir.path());
959        let segment2 = build_segment_2(dir.path());
960
961        let hw_counter = HardwareCounterCell::new();
962
963        let mut holder = SegmentHolder::default();
964
965        let _sid1 = holder.add_new(segment1);
966        let _sid2 = holder.add_new(segment2);
967
968        const DELETE_OP_NUM: u64 = 16;
969
970        assert!(
971            holder
972                .iter()
973                .all(|i| i.1.get().read().version() < DELETE_OP_NUM)
974        );
975
976        let old_version = holder
977            .flush_all(true, false)
978            .expect("Failed to flush test segment holder");
979
980        let segments = Arc::new(RwLock::new(holder));
981
982        // A filter that matches no points.
983        let filter = Filter::new_must(Condition::Field(FieldCondition::new_match(
984            "color".parse().unwrap(),
985            Match::Value(MatchValue {
986                value: ValueVariants::String("white".to_string()),
987            }),
988        )));
989
990        let deleted_count =
991            delete_points_by_filter(&segments.read(), DELETE_OP_NUM, &filter, &hw_counter).unwrap();
992        assert_eq!(deleted_count, 0);
993
994        let new_version = segments
995            .read()
996            .flush_all(true, false)
997            .expect("Failed to flush test segment holder");
998
999        // Flushing again inrceases by 1 and is now equal to `DELETE_OP_NUM` as we want to acknowledge the empty
1000        // delete operation in WAL.
1001        assert_eq!(old_version + 1, new_version);
1002        assert_eq!(new_version, DELETE_OP_NUM);
1003    }

3use super::planned_query::{MergePlan, RescoreStages, Source};

6impl MergePlan {
7    /// Creates a new MergePlan and validates it.
8    ///
9    /// Validation includes:
10    /// - If fusion with weights is used, the number of weights must match the number of sources.
11    pub fn new(
12        sources: Vec<Source>,
13        rescore_stages: Option<RescoreStages>,
14    ) -> OperationResult<Self> {
15        let merge_plan = Self {
16            sources,
17            rescore_stages,
18        };
19        merge_plan.validate()?;
20        Ok(merge_plan)
21    }
22
23    /// Validates the MergePlan.
24    fn validate(&self) -> OperationResult<()> {
25        // Validate fusion weights at all levels
26        self.validate_fusion_weights()?;
27
28        // Recursively validate nested prefetches
29        for source in &self.sources {
30            match source {
31                Source::SearchesIdx(_) => {}
32                Source::ScrollsIdx(_) => {}
33                Source::Prefetch(nested_plan) => nested_plan.validate()?,
34            }
35        }
36
37        Ok(())
38    }
39
40    fn validate_fusion_weights(&self) -> OperationResult<()> {
41        let Self {
42            sources,
43            rescore_stages,
44        } = self;
45
46        if let Some(stages) = rescore_stages {
47            let RescoreStages {
48                shard_level,
49                collection_level,
50            } = stages;
51
52            if let Some(shard_level) = shard_level {
53                validate_query(&shard_level.rescore, sources)?;
54            }
55            if let Some(collection_level) = collection_level {
56                validate_query(&collection_level.rescore, sources)?;
57            }
58        }
59        Ok(())
60    }
61}
62
63fn validate_query(query: &ScoringQuery, sources: &[Source]) -> OperationResult<()> {
64    match query {
65        ScoringQuery::Vector(_) => Ok(()),
66        ScoringQuery::Fusion(fusion) => validate_fusion(fusion, sources.len()),
67        ScoringQuery::OrderBy(_) => Ok(()),
68        ScoringQuery::Formula(_) => Ok(()),
69        ScoringQuery::Sample(_) => Ok(()),
70        ScoringQuery::Mmr(_) => Ok(()),
71    }
72}
73
74fn validate_fusion(fusion: &FusionInternal, num_sources: usize) -> OperationResult<()> {
75    match fusion {
76        FusionInternal::Rrf {
77            k: _,
78            weights: Some(weights),
79        } => {
80            if weights.len() != num_sources {
81                return Err(OperationError::validation_error(format!(
82                    "RRF weights length ({}) does not match number of prefetches ({})",
83                    weights.len(),
84                    num_sources
85                )));
86            }
87            Ok(())
88        }
89        FusionInternal::Rrf {
90            k: _,
91            weights: None,
92        } => Ok(()),
93        FusionInternal::Dbsf => Ok(()),
94    }
95}

23/// Calculate the MMR (Maximal Marginal Relevance) score for a set of points with vectors.
24///
25/// Assumes the points have vectors attached. If not, the entire point will be discarded.
26///
27/// # Arguments
28///
29/// * `collection_params` - The parameters of the collection. Used to determine the right distance metric, or multivec config.
30/// * `points_with_vector` - The points with vectors.
31/// * `mmr` - The MMR parameters.
32/// * `limit` - The maximum number of points to return.
33/// * `search_runtime_handle` - The runtime handle for searching.
34/// * `timeout` - The timeout for the operation.
35/// * `hw_measurement_acc` - The hardware measurement accumulator.
36///
37/// # Returns
38///
39/// A vector of scored points.
40pub fn mmr_from_points_with_vector(
41    points_with_vector: impl IntoIterator<Item = ScoredPoint>,
42    mmr: MmrInternal,
43    distance: Distance,
44    multivector_config: Option<MultiVectorConfig>,
45    limit: usize,
46    hw_measurement_acc: HwMeasurementAcc,
47) -> OperationResult<Vec<ScoredPoint>> {
48    let (vectors, candidates): (Vec<_>, Vec<_>) = points_with_vector
49        .into_iter()
50        .unique_by(|p| p.id)
51        .filter_map(|p| {
52            let vector = p
53                .vector
54                .as_ref()
55                // silently ignore points without this named vector
56                .and_then(|v| v.get(&mmr.using))
57                .map(|v| v.to_owned())?;
58            Some((vector, p))
59        })
60        .unzip();
61
62    debug_assert_eq!(vectors.len(), candidates.len());
63
64    if candidates.is_empty() {
65        return Ok(candidates);
66    }
67
68    let volatile_storage = create_volatile_storage(
69        &vectors,
70        distance,
71        multivector_config,
72        hw_measurement_acc.get_counter_cell(),
73    )?;
74
75    if candidates.len() < 2 {
76        // can't compute MMR for less than 2 points, return with original score
77        return Ok(candidates);
78    }
79
80    // get similarities against query
81    let query_similarities = relevance_similarities(
82        &volatile_storage,
83        mmr.vector,
84        hw_measurement_acc.get_counter_cell(),
85    )?;
86
87    // get similarity matrix between candidates
88    let similarity_matrix = similarity_matrix(&volatile_storage, vectors, hw_measurement_acc)?;
89
90    // compute MMR
91    Ok(maximal_marginal_relevance(
92        candidates,
93        query_similarities,
94        similarity_matrix,
95        mmr.lambda.0,
96        limit,
97    ))
98}

100/// Creates a volatile (in-memory and not persistent) vector storage and inserts the vectors in the provided order.
101fn create_volatile_storage(
102    vectors: &[VectorInternal],
103    distance: Distance,
104    multivector_config: Option<MultiVectorConfig>,
105    hw_counter: HardwareCounterCell,
106) -> OperationResult<VectorStorageEnum> {
107    // Create temporary vector storage
108    let mut volatile_storage = {
109        match &vectors[0] {
110            VectorInternal::Dense(vector) => {
111                new_volatile_dense_vector_storage(vector.len(), distance)
112            }
113
114            VectorInternal::MultiDense(typed_multi_dense_vector) => {
115                let multivector_config = multivector_config.ok_or_else(|| {
116                    OperationError::service_error(
117                        "multivectors are present, but no multivector config provided",
118                    )
119                })?;
120
121                new_volatile_multi_dense_vector_storage(
122                    typed_multi_dense_vector.dim,
123                    distance,
124                    multivector_config,
125                )
126            }
127
128            VectorInternal::Sparse(_) => new_volatile_sparse_vector_storage(),
129        }
130    };
131
132    // Populate storage with vectors
133    for (key, vector) in (0..).zip(vectors) {
134        volatile_storage.insert_vector(key, VectorRef::from(vector), &hw_counter)?;
135    }
136
137    Ok(volatile_storage)
138}

140/// Compute the "relevance" similarity between a query vector and all vectors in the storage.
141fn relevance_similarities(
142    volatile_storage: &VectorStorageEnum,
143    query_vector: VectorInternal,
144    hw_counter: HardwareCounterCell,
145) -> OperationResult<Vec<ScoreType>> {
146    let query = QueryVector::Nearest(query_vector);
147    let query_scorer = new_raw_scorer(query, volatile_storage, hw_counter)?;
148
149    // get similarity between candidates and query
150    let ids: Vec<_> = (0..volatile_storage.total_vector_count() as u32).collect();
151    let mut similarities = vec![0.0; ids.len()];
152    query_scorer.score_points(&ids, &mut similarities);
153
154    Ok(similarities)
155}

157/// Returns a symmetric matrix where entry (i,j) represents the similarity
158/// between vector i and vector j. Diagonal entries are 0 (self-similarity is not calculated).
159/// Only computes each pair once for efficiency since similarity is symmetric.
160///
161/// Errors if there are less than 2 vectors.
162fn similarity_matrix(
163    volatile_storage: &VectorStorageEnum,
164    vectors: Vec<VectorInternal>,
165    hw_measurement_acc: HwMeasurementAcc,
166) -> OperationResult<LazyMatrix<'_>> {
167    let num_vectors = vectors.len();
168
169    // if we have less than 2 points, we can't build a matrix
170    debug_assert!(
171        num_vectors >= 2,
172        "There should be at least two vectors to calculate similarity matrix"
173    );
174
175    if num_vectors < 2 {
176        return Err(OperationError::service_error(
177            "There should be at least two vectors to calculate similarity matrix",
178        ));
179    }
180
181    LazyMatrix::new(vectors, volatile_storage, hw_measurement_acc)
182}

184/// Maximal Marginal Relevance (MMR) algorithm
185///
186/// Iteratively selects points by considering their similarity to
187/// already selected points, combining diversity and relevance.
188///
189/// # Arguments
190///
191/// * `candidates` - the list of points to select from
192/// * `query_similarities` - similarities to the query for each candidate. Offsets refer to the index of the candidate in the `candidates` vector.
193/// * `similarity_matrix` - full pairwise similarity matrix between candidates
194/// * `lambda` - the lambda parameter for the MMR algorithm (0.0 = max diversity, 1.0 = max relevance)
195/// * `limit` - the maximum number of points to select
196fn maximal_marginal_relevance(
197    candidates: Vec<ScoredPoint>,
198    query_similarities: Vec<ScoreType>,
199    mut similarity_matrix: LazyMatrix,
200    lambda: f32,
201    limit: usize,
202) -> Vec<ScoredPoint> {
203    let num_candidates = candidates.len();
204    if num_candidates == 0 || limit == 0 {
205        return Vec::new();
206    }
207
208    let mut selected_indices = Vec::with_capacity(limit);
209    let mut remaining_indices: IndexSet<usize, ahash::RandomState> = (0..num_candidates).collect();
210
211    // Select first point with highest relevance score
212    if let Some(best_idx) = remaining_indices
213        .iter()
214        .max_by_key(|&candidate_idx| OrderedFloat(query_similarities[*candidate_idx]))
215        .copied()
216    {
217        selected_indices.push(best_idx);
218        remaining_indices.swap_remove(&best_idx);
219    }
220
221    // Iteratively select remaining points using MMR
222    while selected_indices.len() < limit && !remaining_indices.is_empty() {
223        let best_candidate = remaining_indices
224            .iter()
225            .map(|&candidate_idx| {
226                let relevance_score = query_similarities[candidate_idx];
227
228                debug_assert!(
229                    selected_indices
230                        .iter()
231                        .all(|&selected_idx| selected_idx != candidate_idx)
232                );
233
234                // Find maximum similarity to any already selected point
235                let max_similarity_to_selected = selected_indices
236                    .iter()
237                    .map(|selected_idx| {
238                        similarity_matrix.get_similarity(candidate_idx, *selected_idx)
239                    })
240                    .max_by_key(|&sim| OrderedFloat(sim))
241                    .unwrap_or(0.0);
242
243                // Calculate MMR score: λ * relevance - (1 - λ) * max_similarity_to_selected
244                let mmr_score =
245                    lambda * relevance_score - (1.0 - lambda) * max_similarity_to_selected;
246
247                (candidate_idx, mmr_score)
248            })
249            .max_by_key(|(_candidate_idx, mmr_score)| OrderedFloat(*mmr_score));
250
251        if let Some((selected_idx, _mmr_score)) = best_candidate {
252            // Select the best candidate and remove from remaining
253            remaining_indices.swap_remove(&selected_idx);
254            selected_indices.push(selected_idx);
255        } else {
256            break;
257        }
258    }
259
260    // Convert selected indices to ScoredPoint results
261    selected_indices
262        .into_iter()
263        .map(|idx| {
264            // Use original score, already postprocessed.
265            //
266            // We prefer this over MMR score because:
267            // - We already selected the top candidates based on MMR score.
268            // - If this is performed at collection level, we will pass this score to the user, which is arguably more meaningful.
269            // - If this is performed at local shard, it might be combined with other shards' results.
270            //    - MMR does not make sense to compare by score with a different set of MMR results
271            //    - It makes more sense to compare by query score.
272            //    - If this isn't the last rescore before sending to collection,
273            //        we are only interested in the selection of points, not the score itself.
274            candidates[idx].clone()
275        })
276        .collect()
277}

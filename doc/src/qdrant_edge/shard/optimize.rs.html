68/// Restores original segments from proxies
69///
70/// # Arguments
71///
72/// * `segments` - segment holder
73/// * `proxy_ids` - ids of poxy-wrapped segment to restore
74///
75/// # Result
76///
77/// Original segments are pushed into `segments`, proxies removed.
78pub fn unwrap_proxy(
79    segments: &LockedSegmentHolder,
80    proxy_ids: &[SegmentId],
81) -> OperationResult<()> {
82    let mut segments_lock = segments.write();
83    for &proxy_id in proxy_ids {
84        if let Some(proxy_segment_ref) = segments_lock.get(proxy_id) {
85            let locked_proxy_segment = proxy_segment_ref.clone();
86            match locked_proxy_segment {
87                LockedSegment::Original(_) => {
88                    // Already unwrapped. It should not actually be here
89                    log::warn!("Attempt to unwrap raw segment! Should not happen.");
90                }
91                LockedSegment::Proxy(proxy_segment) => {
92                    let wrapped_segment = proxy_segment.read().wrapped_segment.clone();
93                    segments_lock.replace(proxy_id, wrapped_segment)?;
94                }
95            }
96        }
97    }
98    Ok(())
99}

101/// Accumulates approximate set of points deleted in a given set of proxies
102///
103/// This list is not synchronized (if not externally enforced),
104/// but guarantees that it contains at least all points deleted in the proxies
105/// before the call to this function.
106pub fn proxy_deleted_points(proxies: &[LockedSegment]) -> DeletedPoints {
107    let mut deleted_points = DeletedPoints::new();
108    for proxy_segment in proxies {
109        match proxy_segment {
110            LockedSegment::Original(_) => {
111                log::error!("Reading raw segment, while proxy expected");
112                debug_assert!(false, "Reading raw segment, while proxy expected");
113            }
114            LockedSegment::Proxy(proxy) => {
115                let proxy_read = proxy.read();
116                for (point_id, versions) in proxy_read.get_deleted_points() {
117                    let entry = deleted_points.entry(*point_id).or_insert(*versions);
118                    entry.operation_version =
119                        entry.operation_version.max(versions.operation_version);
120                    entry.local_version = entry.local_version.max(versions.local_version);
121                }
122            }
123        }
124    }
125    deleted_points
126}

128/// Accumulates index changes made in a given set of proxies
129///
130/// This list is not synchronized (if not externally enforced),
131/// but guarantees that it contains at least all index changes made in the proxies
132/// before the call to this function.
133pub fn proxy_index_changes(proxies: &[LockedSegment]) -> ProxyIndexChanges {
134    let mut index_changes = ProxyIndexChanges::default();
135    for proxy_segment in proxies {
136        match proxy_segment {
137            LockedSegment::Original(_) => {
138                log::error!("Reading raw segment, while proxy expected");
139                debug_assert!(false, "Reading raw segment, while proxy expected");
140            }
141            LockedSegment::Proxy(proxy) => {
142                let proxy_read = proxy.read();
143                index_changes.merge(proxy_read.get_index_changes())
144            }
145        }
146    }
147    index_changes
148}

150/// Function to wrap slow part of optimization. Performs proxy rollback in case of cancellation.
151/// Warn: this function might be _VERY_ CPU intensive,
152/// so it is necessary to avoid any locks inside this part of the code
153///
154/// Returns the newly constructed optimized segment.
155#[allow(clippy::too_many_arguments)]
156fn build_new_segment<F: ?Sized + OptimizationStrategy>(
157    factory: &F,
158    input_segments: &[LockedSegment], // Segments to optimize/merge into one
159    output_segment_uuid: Uuid,        // The UUID of the resulting optimized segment
160    deferred_internal_id: Option<PointOffsetType>,
161    proxies: &[LockedSegment],
162    permit: ResourcePermit, // IO resources for copying data
163    resource_budget: ResourceBudget,
164    stopped: &AtomicBool,
165    hw_counter: &HardwareCounterCell,
166    progress: ProgressTracker,
167    segments_path: &Path,
168) -> OperationResult<Segment> {
169    let mut segment_builder = factory.create_segment_builder(input_segments)?;
170
171    check_process_stopped(stopped)?;
172
173    let progress_copy_data = progress.subtask("copy_data");
174    let progress_populate_storages = progress.subtask("populate_vector_storages");
175    let progress_wait_permit = progress.subtask("wait_cpu_permit");
176
177    let segments: Vec<_> = input_segments
178        .iter()
179        .map(|i| match i {
180            LockedSegment::Original(o) => o.clone(),
181            LockedSegment::Proxy(_) => {
182                panic!("Trying to optimize a segment that is already being optimized!")
183            }
184        })
185        .collect();
186
187    let mut defragmentation_keys = HashSet::new();
188    for segment in &segments {
189        let payload_index = &segment.read().payload_index;
190        let payload_index = payload_index.borrow();
191        let keys = payload_index
192            .config()
193            .indices
194            .iter()
195            .filter(|(_, schema)| schema.schema.is_tenant())
196            .map(|(key, _)| key.clone());
197        defragmentation_keys.extend(keys);
198    }
199
200    if !defragmentation_keys.is_empty() {
201        segment_builder.set_defragment_keys(defragmentation_keys.into_iter().collect());
202    }
203
204    {
205        progress_copy_data.start();
206        let segment_guards = segments.iter().map(|segment| segment.read()).collect_vec();
207        segment_builder.update(
208            &segment_guards.iter().map(Deref::deref).collect_vec(),
209            stopped,
210        )?;
211        drop(progress_copy_data);
212    }
213
214    let index_changes = proxy_index_changes(proxies);
215
216    // Apply index changes to segment builder
217    // Indexes are only used for defragmentation in segment builder, so versions are ignored
218    for (field_name, change) in index_changes.iter_unordered() {
219        match change {
220            ProxyIndexChange::Create(schema, _) => {
221                segment_builder.add_indexed_field(field_name.to_owned(), schema.to_owned());
222            }
223            ProxyIndexChange::Delete(_) => {
224                segment_builder.remove_indexed_field(field_name);
225            }
226            ProxyIndexChange::DeleteIfIncompatible(_, schema) => {
227                segment_builder.remove_index_field_if_incompatible(field_name, schema);
228            }
229        }
230    }
231
232    // Before switching from IO to CPU, make sure that vectors cache is heated up,
233    // so indexing process won't need to wait for IO.
234    progress_populate_storages.start();
235    segment_builder.populate_vector_storages()?;
236    drop(progress_populate_storages);
237
238    // 000 - acquired
239    // +++ - blocked on waiting
240    //
241    // Case: 1 indexation job at a time, long indexing
242    //
243    //  IO limit = 1
244    // CPU limit = 2                         Next optimization
245    //                                       │            loop
246    //                                       │
247    //                                       ▼
248    //  IO 0  00000000000000                  000000000
249    // CPU 1              00000000000000000
250    //     2              00000000000000000
251    //
252    //
253    //  IO 0  ++++++++++++++00000000000000000
254    // CPU 1                       ++++++++0000000000
255    //     2                       ++++++++0000000000
256    //
257    //
258    //  Case: 1 indexing job at a time, short indexation
259    //
260    //
261    //   IO limit = 1
262    //  CPU limit = 2
263    //
264    //
265    //   IO 0  000000000000   ++++++++0000000000
266    //  CPU 1            00000
267    //      2            00000
268    //
269    //   IO 0  ++++++++++++00000000000   +++++++
270    //  CPU 1                       00000
271    //      2                       00000
272    // At this stage workload shifts from IO to CPU, so we can release IO permit
273
274    // Use same number of threads for indexing as for IO.
275    // This ensures that IO is equally distributed between optimization jobs.
276    progress_wait_permit.start();
277    let desired_cpus = permit.num_io as usize;
278    let indexing_permit = resource_budget
279        .replace_with(permit, desired_cpus, 0, stopped)
280        .map_err(|_| {
281            crate::segment::common::operation_error::OperationError::cancelled(
282                "optimization cancelled while waiting for budget",
283            )
284        })?;
285    drop(progress_wait_permit);
286
287    let mut rng = rand::rng();
288    let mut optimized_segment = segment_builder.build(
289        segments_path,
290        output_segment_uuid,
291        deferred_internal_id,
292        indexing_permit,
293        stopped,
294        &mut rng,
295        hw_counter,
296        progress,
297    )?;
298
299    // Delete points
300    let deleted_points_snapshot = proxy_deleted_points(proxies);
301    let index_changes = proxy_index_changes(proxies);
302
303    // Apply index changes before point deletions
304    // Point deletions bump the segment version, can cause index changes to be ignored
305    let old_optimized_segment_version = optimized_segment.version();
306    for (field_name, change) in index_changes.iter_ordered() {
307        debug_assert!(
308            change.version() >= old_optimized_segment_version,
309            "proxied index change should have newer version than segment",
310        );
311        match change {
312            ProxyIndexChange::Create(schema, version) => {
313                optimized_segment.create_field_index(
314                    *version,
315                    field_name,
316                    Some(schema),
317                    hw_counter,
318                )?;
319            }
320            ProxyIndexChange::Delete(version) => {
321                optimized_segment.delete_field_index(*version, field_name)?;
322            }
323            ProxyIndexChange::DeleteIfIncompatible(version, schema) => {
324                optimized_segment
325                    .delete_field_index_if_incompatible(*version, field_name, schema)?;
326            }
327        }
328        check_process_stopped(stopped)?;
329    }
330
331    for (point_id, versions) in deleted_points_snapshot {
332        optimized_segment
333            .delete_point(versions.operation_version, point_id, hw_counter)
334            .unwrap();
335    }
336
337    Ok(optimized_segment)
338}

340/// Create a single optimized segment from the given segments.
341///
342/// All point deletes or payload index changes made during optimization are propagated to the
343/// optimized segment at the very end.
344///
345/// This internally takes a write lock on the segments holder to block new updates when
346/// finalizing optimization. It is returned so that the optimized segment can be inserted and
347/// proxy segments can be dissolved before releasing the lock.
348///
349/// # Warning
350///
351/// This function is slow and must only be used on an optimization worker.
352#[allow(clippy::too_many_arguments)]
353fn optimize_segment_propagate_changes<F: ?Sized + OptimizationStrategy>(
354    factory: &F,
355    optimizing_segments: Vec<LockedSegment>,
356    output_segment_uuid: Uuid,
357    deferred_internal_id: Option<PointOffsetType>,
358    proxies: &[LockedSegment],
359    permit: ResourcePermit, // IO resources for copying data
360    resource_budget: ResourceBudget,
361    stopped: &AtomicBool,
362    hw_counter: &HardwareCounterCell,
363    progress: ProgressTracker,
364    segments_path: &Path,
365) -> OperationResult<(Segment, DeletedPoints)> {
366    check_process_stopped(stopped)?;
367
368    // ---- SLOW PART -----
369
370    let optimized_segment = build_new_segment(
371        factory,
372        &optimizing_segments,
373        output_segment_uuid,
374        deferred_internal_id,
375        proxies,
376        permit,
377        resource_budget,
378        stopped,
379        hw_counter,
380        progress,
381        segments_path,
382    )?;
383
384    // Avoid unnecessary point removing in the critical section:
385    // - save already removed points while avoiding long read locks
386    // - exclude already removed points from post-optimization removing
387    let already_remove_points = {
388        let mut all_removed_points = proxy_deleted_points(proxies);
389        for existing_point in optimized_segment.iter_points() {
390            all_removed_points.remove(&existing_point);
391        }
392        all_removed_points
393    };
394
395    // ---- SLOW PART ENDS HERE -----
396
397    check_process_stopped(stopped)?;
398
399    Ok((optimized_segment, already_remove_points))
400}

402/// Finish optimization: propagate remaining changes and swap segments
403#[allow(clippy::too_many_arguments)]
404fn finish_optimization(
405    segment_holder: &LockedSegmentHolder,
406    locked_proxies: Vec<LockedSegment>,
407    mut optimized_segment: Segment,
408    already_remove_points: &DeletedPoints,
409    proxy_ids: &[SegmentId],
410    cow_segment_id_opt: Option<SegmentId>,
411    stopped: &AtomicBool,
412    hw_counter: &HardwareCounterCell,
413) -> OperationResult<usize> {
414    // This block locks all write operations with collection. It should be fast.
415
416    let upgradable_segment_holder = segment_holder.upgradable_read();
417
418    // This mutex prevents update operations, which could create inconsistency during transition.
419    let update_guard = segment_holder.acquire_updates_lock();
420
421    let index_changes = proxy_index_changes(&locked_proxies);
422
423    // Apply index changes before point deletions
424    // Point deletions bump the segment version, can cause index changes to be ignored
425    for (field_name, change) in index_changes.iter_ordered() {
426        match change {
427            // Warn: change version might be lower than the segment version,
428            // because we might already applied the change earlier in optimization.
429            // Applied optimizations are not removed from `proxy_index_changes`.
430            ProxyIndexChange::Create(schema, version) => {
431                optimized_segment.create_field_index(
432                    *version,
433                    field_name,
434                    Some(schema),
435                    hw_counter,
436                )?;
437            }
438            ProxyIndexChange::Delete(version) => {
439                optimized_segment.delete_field_index(*version, field_name)?;
440            }
441            ProxyIndexChange::DeleteIfIncompatible(version, schema) => {
442                optimized_segment
443                    .delete_field_index_if_incompatible(*version, field_name, schema)?;
444            }
445        }
446        check_process_stopped(stopped)?;
447    }
448
449    let deleted_points = proxy_deleted_points(&locked_proxies);
450    let points_diff = deleted_points
451        .iter()
452        .filter(|&(point_id, _)| !already_remove_points.contains_key(point_id));
453
454    for (&point_id, &versions) in points_diff {
455        // In this specific case we're sure logical point data in the wrapped segment is not
456        // changed at all. We ensure this with an assertion at time of proxying, which makes
457        // sure we only wrap original segments. Because we're sure logical data doesn't change,
458        // we also know pending deletes are always newer. Here we assert that's actually the
459        // case.
460        debug_assert!(
461            versions.operation_version >= optimized_segment.point_version(point_id).unwrap_or(0),
462            "proxied point deletes should have newer version than point in segment",
463        );
464        optimized_segment
465            .delete_point(versions.operation_version, point_id, hw_counter)
466            .unwrap();
467    }
468
469    // Replace proxy segments with new optimized segment
470    let point_count = optimized_segment.available_point_count();
471    let mut writable_segment_holder = RwLockUpgradableReadGuard::upgrade(upgradable_segment_holder);
472
473    let (_, proxies) = writable_segment_holder.swap_new(optimized_segment, proxy_ids);
474    debug_assert_eq!(
475        proxies.len(),
476        proxy_ids.len(),
477        "swapped different number of proxies"
478    );
479
480    if let Some(cow_segment_id) = cow_segment_id_opt {
481        // Temp segment might be taken into another parallel optimization
482        // so it is not necessary exist by this time
483        writable_segment_holder.remove_segment_if_not_needed(cow_segment_id)?;
484    }
485
486    let read_segment_holder = RwLockWriteGuard::downgrade(writable_segment_holder);
487    // Can read, but can't yet write updates.
488
489    let mut deferred_points_set = AHashSet::new();
490    for proxy in &proxies {
491        deferred_points_set.extend(proxy.get().read().deferred_point_ids());
492    }
493    let deferred_points: Vec<PointIdType> = deferred_points_set.into_iter().collect();
494
495    if !deferred_points.is_empty() {
496        // Deferred points in proxy segment may become visible for optimized segment (in most cases).
497        // It's time to deduplicate them and remove older versions from optimized segment,
498        // where they were visible while deferred status.
499        // There are a situations, when deferred point is still deferred after optimization,
500        // so `deduplicate_points` also cover this case and delete only older versions of the point,
501        // which are still visible for optimized segment.
502        read_segment_holder.deduplicate_points(&deferred_points, hw_counter)?;
503    }
504
505    drop(read_segment_holder);
506    // Allow updates again
507    drop(update_guard);
508
509    // Drop all pointers to proxies, so we can de-arc them
510    drop(locked_proxies);
511
512    // Only remove data after we ensure the consistency of the collection.
513    // If remove fails - we will still have operational collection with reported error.
514    for proxy in proxies {
515        proxy.drop_data()?;
516    }
517
518    Ok(point_count)
519}

521/// Returns error if segment size is larger than available disk space
522fn check_segments_size(
523    optimizer_name: &str,
524    optimizing_segments: &[LockedSegment],
525    temp_path: &Path,
526) -> OperationResult<()> {
527    // Counting up how much space do the segments being optimized actually take on the fs.
528    // If there was at least one error while reading the size, this will be `None`.
529    let mut space_occupied = Some(0u64);
530
531    for segment in optimizing_segments {
532        match segment {
533            LockedSegment::Original(segment) => {
534                let locked_segment = segment.read();
535                space_occupied = space_occupied.and_then(|acc| {
536                    match dir_disk_size(locked_segment.data_path()) {
537                        Ok(size) => Some(size + acc),
538                        Err(err) => {
539                            log::debug!(
540                                "Could not estimate size of segment `{}`: {}",
541                                locked_segment.data_path().display(),
542                                err
543                            );
544                            None
545                        }
546                    }
547                });
548            }
549            LockedSegment::Proxy(_) => {
550                return Err(
551                    crate::segment::common::operation_error::OperationError::service_error(
552                        "Proxy segment is not expected here",
553                    ),
554                );
555            }
556        }
557    }
558
559    let space_needed = space_occupied.map(|x| 2 * x);
560
561    // Ensure temp_path exists
562    if !temp_path.exists() {
563        fs::create_dir_all(temp_path).map_err(|err| {
564            crate::segment::common::operation_error::OperationError::service_error(format!(
565                "Could not create temp directory `{}`: {}",
566                temp_path.display(),
567                err
568            ))
569        })?;
570    }
571
572    let space_available = match fs4::available_space(temp_path) {
573        Ok(available) => Some(available),
574        Err(err) => {
575            log::debug!(
576                "Could not estimate available storage space in `{}`: {}",
577                temp_path.display(),
578                err
579            );
580            None
581        }
582    };
583
584    match (space_available, space_needed) {
585        (Some(space_available), Some(space_needed)) => {
586            if space_needed > 0 {
587                log::debug!(
588                    "Available space: {}, needed for optimization: {}",
589                    bytes_to_human(space_available as usize),
590                    bytes_to_human(space_needed as usize),
591                );
592            }
593            if space_available < space_needed {
594                return Err(
595                    crate::segment::common::operation_error::OperationError::service_error(format!(
596                        "Not enough space available for optimization, needed: {}, available: {}",
597                        bytes_to_human(space_needed as usize),
598                        bytes_to_human(space_available as usize),
599                    )),
600                );
601            }
602        }
603        _ => {
604            log::warn!(
605                "Could not estimate available storage space in `{optimizer_name}`; will try optimizing anyway",
606            );
607        }
608    }
609
610    Ok(())
611}

613/// Performs optimization of segments (merge / reindex / vacuum, etc.)
614#[allow(clippy::too_many_arguments)]
615pub fn execute_optimization<F: ?Sized + OptimizationStrategy>(
616    optimizer_name: &'static str,
617    segment_holder: LockedSegmentHolder,
618    input_segment_ids: Vec<SegmentId>,
619    output_segment_uuid: Uuid,
620    deferred_internal_id: Option<PointOffsetType>,
621    paths: &OptimizationPaths,
622    permit: ResourcePermit,
623    resource_budget: ResourceBudget,
624    stopped: &AtomicBool,
625    progress: ProgressTracker,
626    telemetry_counter: &Mutex<OperationDurationsAggregator>,
627    factory: &F,
628    on_successful_start: Box<dyn FnOnce()>,
629) -> OperationResult<OptimizationResult> {
630    check_process_stopped(stopped)?;
631
632    let mut timer = ScopeDurationMeasurer::new(telemetry_counter);
633    timer.set_success(false);
634
635    let segment_holder_read = segment_holder.upgradable_read();
636
637    // Determine if we need a separate COW segment for writes
638    let appendable_segments_ids = segment_holder_read.appendable_segments_ids();
639    let has_appendable_segments_except_optimized = appendable_segments_ids
640        .iter()
641        .any(|id| !input_segment_ids.contains(id));
642    let need_extra_cow_segment = !has_appendable_segments_except_optimized;
643
644    let input_segments: Vec<_> = input_segment_ids
645        .iter()
646        .cloned()
647        .map(|id| segment_holder_read.get(id))
648        .filter_map(|x| x.cloned())
649        .collect();
650
651    // Check that all segments exist and are not already under optimization
652    let all_segments_ok = input_segments.len() == input_segment_ids.len()
653        && input_segments
654            .iter()
655            .all(|s| matches!(s, LockedSegment::Original(_)));
656    if !all_segments_ok {
657        return Ok(OptimizationResult { points_count: 0 });
658    }
659
660    check_segments_size(optimizer_name, &input_segments, &paths.temp_path)?;
661
662    check_process_stopped(stopped)?;
663
664    on_successful_start();
665
666    let hw_counter = HardwareCounterCell::disposable();
667
668    let extra_cow_segment_opt = need_extra_cow_segment
669        .then(|| factory.create_temp_segment())
670        .transpose()?;
671
672    let mut proxies = Vec::new();
673    for sg in input_segments.iter() {
674        let proxy = ProxySegment::new(sg.clone());
675        // Wrapped segment is fresh, so it has no operations
676        // Operation with number 0 will be applied
677        if let Some(extra_cow_segment) = &extra_cow_segment_opt {
678            proxy.replicate_field_indexes(0, &hw_counter, extra_cow_segment)?;
679        }
680        proxies.push(proxy);
681    }
682
683    // Save segment version once all payload indices have been converted
684    // If this ends up not being saved due to a crash, the segment will not be used
685    match &extra_cow_segment_opt {
686        Some(LockedSegment::Original(segment)) => {
687            let segment_path = &segment.read().segment_path;
688            SegmentVersion::save(segment_path)?;
689        }
690        Some(LockedSegment::Proxy(_)) => unreachable!(),
691        None => {}
692    }
693
694    let mut locked_proxies: Vec<LockedSegment> = Vec::with_capacity(proxies.len());
695
696    let (proxy_ids, cow_segment_id_opt, counter_handler): (Vec<_>, _, _) = {
697        // Exclusive lock for the segments operations.
698        let mut segment_holder_write = RwLockUpgradableReadGuard::upgrade(segment_holder_read);
699        let mut proxy_ids = Vec::new();
700        for (proxy, idx) in proxies.into_iter().zip(input_segment_ids.iter().cloned()) {
701            // During optimization, we expect that logical point data in the wrapped segment is
702            // not changed at all. But this would be possible if we wrap another proxy segment,
703            // because it can share state through it's write segment. To prevent this we assert
704            // here that we only wrap non-proxy segments.
705            // Also helps to ensure the delete propagation behavior in
706            // `optimize_segment_propagate_changes` remains  sound.
707            // See: <https://github.com/qdrant/qdrant/pull/7208>
708            debug_assert!(
709                matches!(proxy.wrapped_segment, LockedSegment::Original(_)),
710                "during optimization, wrapped segment must not be another proxy segment"
711            );
712
713            // replicate_field_indexes for the second time,
714            // because optimized segments could have been changed.
715            // The probability is small, though,
716            // so we can afford this operation under the full collection write lock
717            if let Some(extra_cow_segment) = &extra_cow_segment_opt {
718                proxy.replicate_field_indexes(0, &hw_counter, extra_cow_segment)?;
719            }
720
721            let locked_proxy = LockedSegment::from(proxy);
722            segment_holder_write.replace(idx, locked_proxy.clone())?; // Slow only in case the index is change in the gap between two calls
723
724            proxy_ids.push(idx);
725            locked_proxies.push(locked_proxy);
726        }
727
728        let cow_segment_id_opt = extra_cow_segment_opt
729            .map(|extra_cow_segment| segment_holder_write.add_new_locked(extra_cow_segment));
730
731        // Increase optimization counter right after replacing segments with proxies.
732        // We'll decrease it after inserting the optimized segment.
733        let counter_handler = segment_holder_write.running_optimizations.inc();
734
735        (proxy_ids, cow_segment_id_opt, counter_handler)
736    };
737
738    // SLOW PART: create single optimized segment and propagate all new changes to it
739    let build_result = optimize_segment_propagate_changes(
740        factory,
741        input_segments,
742        output_segment_uuid,
743        deferred_internal_id,
744        &locked_proxies,
745        permit,
746        resource_budget,
747        stopped,
748        &hw_counter,
749        progress,
750        &paths.segments_path,
751    );
752
753    let (optimized_segment, already_remove_points) = match build_result {
754        Ok(result) => result,
755        Err(err) => {
756            // Properly cancel optimization on all error kinds
757            // Unwrap proxies and add temp segment to holder
758            unwrap_proxy(&segment_holder, &proxy_ids)?;
759            return Err(err);
760        }
761    };
762
763    // Fast part: blocks updates, propagates rest of the changes, swaps optimized segment
764    let points_count = match finish_optimization(
765        &segment_holder,
766        locked_proxies,
767        optimized_segment,
768        &already_remove_points,
769        &proxy_ids,
770        cow_segment_id_opt,
771        stopped,
772        &hw_counter,
773    ) {
774        Ok(points_count) => points_count,
775        Err(err) => {
776            // Properly cancel optimization on all error kinds
777            // Unwrap proxies and add temp segment to holder
778            unwrap_proxy(&segment_holder, &proxy_ids)?;
779            return Err(err);
780        }
781    };
782
783    drop(counter_handler);
784
785    timer.set_success(true);
786
787    Ok(OptimizationResult { points_count })
788}

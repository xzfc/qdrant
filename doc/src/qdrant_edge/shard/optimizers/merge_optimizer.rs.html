16/// Optimizer that tries to reduce number of segments until it fits configured
17/// value.
18///
19/// ```text
20/// Suppose we have a set of mergeable segments, sorted by size.
21/// `A` is smallest, `M` is largest.
22///
23///     A B C D E F G H I J K L M
24///
25/// MergeOptimizer greedily arranges them in batches up to the size threshold.
26///
27///     [A B C D] [E F G] [H I J] K L M
28///     └───X───┘ └──Y──┘ └──Z──┘
29///
30/// After merging these batches, our segments would look like this:
31///
32///     ∅ X Y Z K L M
33///
34/// `∅` is the newly created appendable segment that Qdrant could potentially
35/// create because MergeOptimizer merged the last appendable segment.
36///
37/// To guarantee that the number of segments will be reduced after the merge,
38/// either merge a batch of at least 3 segments, or merge at least two batches.
39///
40/// - bad:   [A B]        →  ∅ X    (segment count is the same)
41/// - good:  [A B C]      →  ∅ X    (one segment less)
42/// - good:  [A B] [C D]  →  ∅ X Y  (one segment less)
43/// ```
44pub struct MergeOptimizer {
45    default_segments_number: usize,
46    thresholds_config: OptimizerThresholds,
47    segments_path: PathBuf,
48    temp_path: PathBuf,
49    segment_optimizer_config: SegmentOptimizerConfig,
50    hnsw_global_config: HnswGlobalConfig,
51    telemetry_durations_aggregator: Arc<Mutex<OperationDurationsAggregator>>,
52}

111    fn plan_optimizations(&self, planner: &mut OptimizationPlanner) {
112        let mut candidates = planner
113            .remaining()
114            .iter()
115            .map(|(&segment_id, segment)| {
116                let size = segment
117                    .read()
118                    .max_available_vectors_size_in_bytes()
119                    .unwrap_or_default();
120                (segment_id, size)
121            })
122            .collect_vec();
123
124        candidates.sort_by_key(|(_segment_id, size)| *size);
125        let threshold = self
126            .thresholds_config
127            .max_segment_size_kb
128            .saturating_mul(BYTES_IN_KB);
129
130        let mut first_batch = None;
131        let mut taken_candidates = 0;
132        let mut last_candidate =
133            (planner.expected_segments_number() + 2).saturating_sub(self.default_segments_number);
134        while taken_candidates < last_candidate.min(candidates.len()) {
135            let batch = candidates[taken_candidates..last_candidate.min(candidates.len())]
136                .iter()
137                .scan(0, |size_sum, &(segment_id, size)| {
138                    *size_sum += size;
139                    (*size_sum < threshold).then_some(segment_id)
140                })
141                .collect_vec();
142
143            if batch.len() < 2 {
144                return;
145            }
146            let is_first_batch = taken_candidates == 0;
147            taken_candidates += batch.len();
148            last_candidate += 1;
149            if is_first_batch && batch.len() < 3 {
150                // First batch has length 2. To guarantee that the number of
151                // segments will be reduced, we need another batch.
152                // So, hold the first batch until we find the second one.
153                first_batch = Some(batch);
154                continue;
155            }
156            if let Some(first_batch) = first_batch.take() {
157                planner.plan(first_batch);
158            }
159            planner.plan(batch);
160        }
161    }

163    fn get_telemetry_counter(&self) -> &Mutex<OperationDurationsAggregator> {
164        &self.telemetry_durations_aggregator
165    }

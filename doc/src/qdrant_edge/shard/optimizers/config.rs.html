33/// This configuration contains all necessary information to build an optimized segment.
34#[derive(Debug, Clone)]
35pub struct SegmentOptimizerConfig {
36    pub payload_storage_type: PayloadStorageType,
37    /// Configuration of dense vectors, as it should be for a plain segment (without any optimization).
38    pub plain_dense_vector_config: HashMap<VectorNameBuf, VectorDataConfig>,
39    /// Configuration of sparse vectors, as it should be for a plain segment (without any optimization).
40    pub plain_sparse_vector_config: HashMap<VectorNameBuf, SparseVectorDataConfig>,
41    /// Extra configuration for dense vectors, which _might_ be applied during optimization,
42    /// depending on the segment state.
43    pub dense_vector: HashMap<VectorNameBuf, DenseVectorOptimizerConfig>,
44    /// Extra configuration for sparse vectors, which _might_ be applied during optimization,
45    /// depending on the segment state.
46    pub sparse_vector: HashMap<VectorNameBuf, SparseVectorOptimizerConfig>,
47}

49impl SegmentOptimizerConfig {
50    pub fn plain_segment_config(&self) -> SegmentConfig {
51        SegmentConfig {
52            vector_data: self.plain_dense_vector_config.clone(),
53            sparse_vector_data: self.plain_sparse_vector_config.clone(),
54            payload_storage_type: self.payload_storage_type,
55        }
56    }
57
58    pub fn new(
59        payload_storage_type: PayloadStorageType,
60        dense_vectors: HashMap<VectorNameBuf, DenseVectorOptimizerInput>,
61        sparse_vectors: HashMap<VectorNameBuf, SparseVectorOptimizerInput>,
62    ) -> SegmentOptimizerConfig {
63        let (mut plain_dense_vector_config, mut dense_vector) = (HashMap::new(), HashMap::new());
64        for (name, input) in dense_vectors {
65            let DenseVectorOptimizerInput {
66                size,
67                distance,
68                on_disk,
69                hnsw_config,
70                quantization_config,
71                multivector_config,
72                datatype,
73            } = input;
74            plain_dense_vector_config.insert(
75                name.clone(),
76                VectorDataConfig {
77                    size,
78                    distance,
79                    index: Indexes::Plain {},
80                    storage_type: VectorStorageType::from_on_disk(on_disk.unwrap_or_default()),
81                    quantization_config: QuantizationConfig::for_appendable_segment(
82                        quantization_config.as_ref(),
83                    ),
84                    multivector_config,
85                    datatype,
86                },
87            );
88            dense_vector.insert(
89                name,
90                DenseVectorOptimizerConfig {
91                    on_disk,
92                    hnsw_config,
93                    quantization_config,
94                },
95            );
96        }
97
98        let (mut plain_sparse_vector_config, mut sparse_vector) = (HashMap::new(), HashMap::new());
99        for (name, input) in sparse_vectors {
100            let SparseVectorOptimizerInput {
101                on_disk,
102                full_scan_threshold,
103                index_datatype,
104                storage_type,
105                modifier,
106            } = input;
107            plain_sparse_vector_config.insert(
108                name.clone(),
109                SparseVectorDataConfig {
110                    index: SparseIndexConfig {
111                        full_scan_threshold,
112                        index_type: SparseIndexType::MutableRam,
113                        datatype: index_datatype,
114                    },
115                    storage_type,
116                    modifier,
117                },
118            );
119            sparse_vector.insert(name, SparseVectorOptimizerConfig { on_disk });
120        }
121
122        SegmentOptimizerConfig {
123            payload_storage_type,
124            plain_dense_vector_config,
125            plain_sparse_vector_config,
126            dense_vector,
127            sparse_vector,
128        }
129    }
130}

154/// Target segment count for the merge optimizer.
155pub fn default_segment_number() -> usize {
156    // Configure 1 segment per 2 CPUs, as a middle ground between
157    // latency and RPS.
158    let expected_segments = crate::common::cpu::get_num_cpus() / 2;
159    // Do not configure less than 2 and more than 8 segments
160    // until it is not explicitly requested
161    expected_segments.clamp(2, 8)
162}

166/// Resolve number of segments: if `default_segment_number` is 0, use CPU-based default.
167pub fn get_number_segments(requested_segment_number: usize) -> usize {
168    if requested_segment_number == 0 {
169        default_segment_number()
170    } else {
171        requested_segment_number
172    }
173}

175/// Resolve indexing threshold in KB: `None` => default, `Some(0)` => disable (usize::MAX).
176pub fn get_indexing_threshold_kb(indexing_threshold: Option<usize>) -> usize {
177    match indexing_threshold {
178        None => DEFAULT_INDEXING_THRESHOLD_KB,
179        Some(0) => usize::MAX,
180        Some(custom) => custom,
181    }
182}
183
184/// Resolve max segment size in KB: custom value or per-thread default.
185pub fn get_max_segment_size_kb(
186    max_segment_size: Option<usize>,
187    num_indexing_threads: usize,
188) -> usize {
189    if let Some(max) = max_segment_size {
190        max
191    } else {
192        num_indexing_threads.saturating_mul(DEFAULT_MAX_SEGMENT_PER_CPU_KB)
193    }
194}

196/// Build deferred points threshold in bytes when `prevent_unoptimized` is true.
197pub fn get_deferred_points_threshold_bytes(
198    prevent_unoptimized: Option<bool>,
199    indexing_threshold_kb: usize,
200) -> Option<NonZeroUsize> {
201    (prevent_unoptimized == Some(true))
202        .then(|| indexing_threshold_kb.saturating_mul(BYTES_IN_KB))
203        .and_then(NonZeroUsize::new)

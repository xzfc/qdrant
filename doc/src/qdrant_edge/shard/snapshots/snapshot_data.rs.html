11impl SnapshotData {

13        SnapshotData::Packed(MaybeTempPath::Persistent(path.as_ref().to_path_buf()))
14    }
15
16    /// Get path to the downloaded data

20            SnapshotData::Unpacked(temp_dir) => temp_dir.path(),

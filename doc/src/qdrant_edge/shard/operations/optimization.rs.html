27/// Optimizations progress for the collection

29pub struct OptimizationsResponse {

37    /// Completed optimizations.

39    #[serde(skip_serializing_if = "Option::is_none")]

41    /// Segments that don't require optimization.
42    /// Requires `?with=idle_segments`.
43    #[serde(skip_serializing_if = "Option::is_none")]
44    pub idle_segments: Option<Vec<OptimizationSegmentInfo>>,
45}

47impl OptimizationsResponse {
48    /// Merge another `OptimizationsResponse` into this one.
49    pub fn merge(&mut self, other: OptimizationsResponse) {
50        let OptimizationsResponse {
51            summary:
52                OptimizationsSummary {
53                    queued_optimizations,
54                    queued_segments,
55                    queued_points,
56                    idle_segments: idle_segments_count,
57                },
58            running,
59            queued,
60            completed,
61            idle_segments,
62        } = other;
63
64        self.running.extend(running);
65        self.summary.queued_optimizations += queued_optimizations;
66        self.summary.queued_segments += queued_segments;
67        self.summary.queued_points += queued_points;
68        self.summary.idle_segments += idle_segments_count;
69        merge_optional_vec(&mut self.completed, completed);
70        merge_optional_vec(&mut self.queued, queued);
71        merge_optional_vec(&mut self.idle_segments, idle_segments);
72    }
73}

75/// Merge two `Option<Vec<T>>` values: if either side has data, the result has data.
76fn merge_optional_vec<T>(target: &mut Option<Vec<T>>, source: Option<Vec<T>>) {
77    match (target.as_mut(), source) {
78        (Some(target), Some(source)) => target.extend(source),
79        (None, source @ Some(_)) => *target = source,
80        (Some(_) | None, None) => {}
81    }
82}

8use crate::posting_list::value_handler::{PostingValue, ValueHandler};
9use crate::posting_list::{BitPackerImpl, CHUNK_LEN};

11pub struct PostingBuilder<V> {
12    elements: Vec<PostingElement<V>>,
13}
14
15impl<V> Default for PostingBuilder<V> {
16    fn default() -> Self {
17        Self {
18            elements: Vec::new(),
19        }
20    }
21}
22
23impl<V> PostingBuilder<V> {
24    pub fn new() -> Self {
25        Self::default()
26    }
27
28    pub fn add(&mut self, id: PointOffsetType, value: V) {
29        self.elements.push(PostingElement { id, value });
30    }
31
32    /// Unified implementation that works for both fixed-size and variable-size values
33    ///
34    /// This method uses the `ValueHandler::process_values` trait function to abstract the
35    /// differences between the two implementations, allowing us to share the common logic.
36    pub fn build(mut self) -> PostingList<V>
37    where
38        V: PostingValue,
39    {
40        self.elements.sort_unstable_by_key(|e| e.id);
41
42        let num_elements = self.elements.len();
43
44        // extract ids and values into separate lists
45        let (ids, values): (Vec<_>, Vec<_>) =
46            self.elements.into_iter().map(|e| (e.id, e.value)).unzip();
47
48        // process values
49        let (sized_values, var_size_data) = V::Handler::process_values(values);
50
51        let bitpacker = BitPackerImpl::new();
52        let mut chunks = Vec::with_capacity(ids.len() / CHUNK_LEN);
53        let mut id_data_size = 0;
54
55        // process full chunks
56        let ids_chunks_iter = ids.chunks_exact(CHUNK_LEN);
57        let values_chunks_iter = sized_values.chunks_exact(CHUNK_LEN);
58        let remainder_ids = ids_chunks_iter.remainder();
59        let remainder_values = values_chunks_iter.remainder();
60
61        for (chunk_ids, chunk_values) in ids_chunks_iter.zip(values_chunks_iter) {
62            let initial = chunk_ids[0];
63            let chunk_bits = bitpacker.num_bits_sorted(initial, chunk_ids);
64            let chunk_size = BitPackerImpl::compressed_block_size(chunk_bits);
65
66            chunks.push(PostingChunk {
67                initial_id: U32::from(initial),
68                offset: u32::try_from(id_data_size)
69                    .expect("id_data_size should fit in u32, (smaller than 4GB)")
70                    .into(),
71                sized_values: chunk_values
72                    .try_into()
73                    .expect("should be a valid chunk size"),
74            });
75            id_data_size += chunk_size;
76        }
77
78        // now process remainders
79        let mut remainders = Vec::with_capacity(num_elements % CHUNK_LEN);
80        for (&id, &value) in remainder_ids.iter().zip(remainder_values) {
81            remainders.push(RemainderPosting {
82                id: U32::from(id),
83                value,
84            });
85        }
86
87        // compress id_data
88        let mut id_data = vec![0u8; id_data_size];
89        for (chunk_index, chunk_ids) in ids.chunks_exact(CHUNK_LEN).enumerate() {
90            let chunk = &chunks[chunk_index];
91            let compressed_size = PostingChunk::get_compressed_size(&chunks, &id_data, chunk_index);
92            let chunk_bits = compressed_size * u8::BITS as usize / CHUNK_LEN;
93            bitpacker.compress_sorted(
94                chunk.initial_id.get(),
95                chunk_ids,
96                &mut id_data
97                    [chunk.offset.get() as usize..chunk.offset.get() as usize + compressed_size],
98                chunk_bits as u8,
99            );
100        }
101
102        let last_id = ids.last().copied();
103
104        PostingList {
105            id_data,
106            var_size_data,
107            chunks,
108            remainders,
109            last_id,
110            _phantom: PhantomData,
111        }
112    }
113}

115impl PostingBuilder<()> {
116    /// Add an id without a value.
117    pub fn add_id(&mut self, id: PointOffsetType) {
118        self.add(id, ());
119    }
120}

122impl<V: PostingValue> From<PostingBuilder<V>> for PostingList<V> {
123    fn from(value: PostingBuilder<V>) -> Self {
124        value.build()
125    }

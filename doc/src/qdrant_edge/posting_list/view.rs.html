95    pub(crate) fn decompress_chunk(
96        &self,
97        chunk_index: usize,
98        decompressed_chunk: &mut [PointOffsetType; CHUNK_LEN],
99    ) {
100        let chunk = &self.chunks[chunk_index];
101        let compressed_size =
102            PostingChunk::get_compressed_size(self.chunks, self.id_data, chunk_index);
103        let chunk_bits = compressed_size * u8::BITS as usize / CHUNK_LEN;
104
105        let start_offset = chunk.offset.get() as usize;
106        let end_offset = start_offset + compressed_size;
107
108        BitPackerImpl::new().decompress_sorted(
109            chunk.initial_id.get(),
110            &self.id_data[start_offset..end_offset],
111            decompressed_chunk,
112            chunk_bits as u8,
113        );
114    }

136    pub(crate) fn ids_range(&self, start_chunk: usize) -> Option<RangeInclusive<u32>> {
137        // if there is no last id, it means the posting list is empty
138        let last_id = self.last_id?;
139
140        let initial_id = self
141            .chunks
142            .get(start_chunk)
143            .map(|chunk| chunk.initial_id.get())
144            .or_else(|| self.get_remainder(0).map(|elem| elem.id.get()))?;
145
146        Some(initial_id..=last_id)
147    }
148
149    /// Find the chunk that may contain the id.
150    /// It doesn't guarantee that the chunk contains the id, but if it is in the posting list, then it must be in the chunk.
151    ///
152    /// Assumes the id is in the posting list range.
153    pub fn find_chunk(&self, id: PointOffsetType, start_chunk: Option<usize>) -> Option<usize> {
154        let remainders = self.remainders;
155        let chunks = self.chunks;
156
157        // check if id might be in the remainders list
158        if remainders.first().is_some_and(|elem| id >= elem.id.get()) {
159            return None;
160        }
161
162        let start_chunk = start_chunk.unwrap_or(0);
163        let chunks_slice = chunks.get(start_chunk..)?;
164        if chunks_slice.is_empty() {
165            return None;
166        }
167
168        // No need to check if id is under range of posting list,
169        // this function assumes it is within the range
170        debug_assert!(id >= chunks_slice[0].initial_id.get());
171        debug_assert!(self.last_id.is_some_and(|last_id| id <= last_id));
172
173        // Fast-path: check if `id` falls into the first chunk's range
174        let first = &chunks_slice[0];
175        if let Some(second) = chunks_slice.get(1) {
176            let id0 = first.initial_id.get();
177            let id1 = second.initial_id.get();
178
179            if id0 <= id && id < id1 {
180                return Some(start_chunk);
181            }
182        }
183
184        match chunks_slice.binary_search_by(|chunk| chunk.initial_id.get().cmp(&id)) {
185            // id is the initial value of the chunk with index idx
186            Ok(idx) => Some(start_chunk + idx),
187
188            // id is not the initial_id of any chunk
189            Err(insert_idx) if insert_idx > 0 => {
190                // this is the index of the chunk that could contain id
191                let idx = insert_idx - 1;
192
193                // id could be within this chunk
194                Some(start_chunk + idx)
195            }
196            Err(_) => None,
197        }
198    }

200    pub(crate) fn search_in_remainders(&self, id: PointOffsetType) -> Result<usize, usize> {
201        self.remainders
202            .binary_search_by(|elem| elem.id.get().cmp(&id))
203    }

205    /// The total number of elements in the posting list.
206    pub fn len(&self) -> usize {
207        self.chunks.len() * CHUNK_LEN + self.remainders.len()
208    }

210    /// Checks if there are no elements in the posting list.

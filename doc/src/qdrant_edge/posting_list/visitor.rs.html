8/// A visitor for a posting list which caches the latest decompressed chunk of ids.
9pub struct PostingVisitor<'a, V: PostingValue> {
10    pub(crate) list: PostingListView<'a, V>,
11
12    /// Index of the decompressed chunk.
13    /// It is used to shorten the search range of chunk index for the next value.
14    decompressed_chunk_idx: Option<usize>,
15
16    /// Lazy decompressed chunk of ids. Never access this directly, prefer [`Self::decompressed_chunk`] function
17    decompressed_chunk: [PointOffsetType; CHUNK_LEN],
18}
19
20impl<'a, V: PostingValue> PostingVisitor<'a, V> {
21    pub(crate) fn new(view: PostingListView<'a, V>) -> Self {
22        Self {
23            list: view,
24            decompressed_chunk_idx: None,
25            decompressed_chunk: [0; CHUNK_LEN],
26        }
27    }
28
29    pub fn len(&self) -> usize {
30        self.list.len()
31    }
32
33    pub fn is_empty(&self) -> bool {
34        self.list.is_empty()
35    }
36
37    /// Returns the decompressed slice of ids for a chunk.
38    ///
39    /// Assumes the chunk_idx is valid.
40    fn decompressed_chunk(&mut self, chunk_idx: usize) -> &[PointOffsetType; CHUNK_LEN] {
41        if self.decompressed_chunk_idx != Some(chunk_idx) {
42            self.list
43                .decompress_chunk(chunk_idx, &mut self.decompressed_chunk);
44            self.decompressed_chunk_idx = Some(chunk_idx);
45        }
46
47        &self.decompressed_chunk
48    }
49
50    /// Returns the first offset whose element id is greater or equal to the given id.
51    ///
52    /// Returns `None` if there is no such element in the posting list
53    pub(crate) fn search_greater_or_equal(
54        &mut self,
55        id: PointOffsetType,
56        start_from_offset: Option<usize>,
57    ) -> Option<usize> {
58        let start_chunk = start_from_offset
59            .map(|offset| offset / CHUNK_LEN)
60            .unwrap_or(0);
61
62        let ids_range = self.list.ids_range(start_chunk)?;
63        // check if the first in the chunk is already greater or equal to the target id
64        if ids_range.start() >= &id {
65            return Some(start_chunk * CHUNK_LEN);
66        }
67        // check if the target id is already greater than the last id
68        if ids_range.end() < &id {
69            return None;
70        }
71
72        // Find the chunk that may contain the id and check if the id is in the chunk
73        let chunk_index = self.list.find_chunk(id, Some(start_chunk));
74
75        if let Some(chunk_index) = chunk_index {
76            let local_offset = match self.decompressed_chunk(chunk_index).binary_search(&id) {
77                Ok(found_local_offset) => found_local_offset,
78                Err(closest_local_offset) => {
79                    // If the target id is bigger than all the values here, and smaller than the first id
80                    // in the next chunk or remainders, then that next id is the closest greater id
81                    if closest_local_offset >= CHUNK_LEN {
82                        let next_offset = (chunk_index + 1) * CHUNK_LEN;
83
84                        let next_offset_exists = next_offset < self.len();
85
86                        return next_offset_exists.then_some(next_offset);
87                    }
88
89                    closest_local_offset
90                }
91            };
92
93            return Some(local_offset + (chunk_index * CHUNK_LEN));
94        }
95
96        // Check in remainders
97        let remainder_offset = match self.list.search_in_remainders(id) {
98            Ok(found_remainder_offset) => found_remainder_offset,
99            Err(closest_remainder_offset) => {
100                if closest_remainder_offset >= self.list.remainders_len() {
101                    // There is no greater or equal id in the posting list
102                    return None;
103                }
104
105                closest_remainder_offset
106            }
107        };
108
109        Some(remainder_offset + self.list.chunks_len() * CHUNK_LEN)
110    }
111
112    pub fn contains(&mut self, id: PointOffsetType) -> bool {
113        if self
114            .list
115            .ids_range(0)
116            .is_none_or(|range| !range.contains(&id))
117        {
118            return false;
119        }
120
121        // Find the chunk that may contain the id and check if the id is in the chunk
122        let chunk_index = self.list.find_chunk(id, None);
123        if let Some(chunk_idx) = chunk_index {
124            if self.list.get_chunk_unchecked(chunk_idx).initial_id == id {
125                return true;
126            }
127
128            self.decompressed_chunk(chunk_idx)
129                .binary_search(&id)
130                .is_ok()
131        } else {
132            self.list.search_in_remainders(id).is_ok()
133        }
134    }
135
136    pub(crate) fn get_by_offset(&mut self, offset: usize) -> Option<PostingElement<V>> {
137        let chunk_idx = offset / CHUNK_LEN;
138        let local_offset = offset % CHUNK_LEN;
139
140        // bound check
141        if offset >= self.list.len() {
142            return None;
143        }
144
145        // get from chunk
146        if chunk_idx < self.list.chunks_len() {
147            let id = self.decompressed_chunk(chunk_idx)[local_offset];
148            let chunk_sized_values = self.list.get_chunk_unchecked(chunk_idx).sized_values;
149            let sized_value = chunk_sized_values[local_offset];
150            let next_sized_value = || {
151                chunk_sized_values
152                    .get(local_offset + 1)
153                    .copied()
154                    // or check first of the next chunk
155                    .or_else(|| {
156                        self.list
157                            .get_chunk(chunk_idx + 1)
158                            .map(|chunk| chunk.sized_values[0])
159                    })
160                    // or, if it is the last one, check first from remainders
161                    .or_else(|| self.list.get_remainder(0).map(|e| e.value))
162            };
163
164            let value =
165                V::Handler::get_value(sized_value, next_sized_value, self.list.var_size_data);
166
167            return Some(PostingElement { id, value });
168        }
169
170        // else, get from remainder
171        self.list.get_remainder(local_offset).map(|e| {
172            let id = e.id;
173            let next_sized_value = || self.list.get_remainder(local_offset + 1).map(|r| r.value);
174            let value = V::Handler::get_value(e.value, next_sized_value, self.list.var_size_data);
175
176            PostingElement {
177                id: id.get(),
178                value,
179            }
180        })
181    }
182}

60/// Segment - an object which manages an independent group of points.
61///
62/// - Provides storage, indexing and managing operations for points (vectors + payload)
63/// - Keeps track of point versions
64/// - Persists data
65/// - Keeps track of occurred errors
66#[derive(Debug)]
67pub struct Segment {
68    pub uuid: Uuid,
69    /// Initial version this segment was created at
70    pub initial_version: Option<SeqNumberType>,
71    /// Latest update operation number, applied to this segment
72    /// If None, there were no updates and segment is empty
73    pub version: Option<SeqNumberType>,
74    /// Latest persisted version
75    /// Locked structure on which we hold the lock during flush to prevent concurrent flushes
76    pub persisted_version: Arc<Mutex<Option<SeqNumberType>>>,
77    /// Lock to prevent concurrent flushes and used for waiting for ongoing flushes to finish.
78    pub is_alive_flush_lock: IsAliveLock,
79    /// Path to the segment directory
80    pub segment_path: PathBuf,
81    pub version_tracker: VersionTracker,
82    /// Component for mapping external ids to internal and also keeping track of point versions
83    pub id_tracker: Arc<AtomicRefCell<IdTrackerEnum>>,
84    pub vector_data: HashMap<VectorNameBuf, VectorData>,
85    pub payload_index: Arc<AtomicRefCell<StructPayloadIndex>>,
86    pub payload_storage: Arc<AtomicRefCell<PayloadStorageEnum>>,
87    /// Shows if it is possible to insert more points into this segment
88    pub appendable_flag: bool,
89    /// Shows what kind of indexes and storages are used in this segment
90    pub segment_type: SegmentType,
91    pub segment_config: SegmentConfig,
92    /// Last unhandled error
93    /// If not None, all update operations will be aborted until original operation is performed properly
94    pub error_status: Option<SegmentFailedState>,
95    #[cfg(feature = "rocksdb")]
96    pub database: Option<Arc<parking_lot::RwLock<DB>>>,
97    /// Points with internal id >= this value are hidden from reads.
98    /// Available for appendable segments only.
99    pub(crate) deferred_internal_id: Option<PointOffsetType>,
100}

114impl Drop for Segment {
115    fn drop(&mut self) {
116        // Wait for all background flush operations to finish
117        self.is_alive_flush_lock.blocking_mark_dead();
118
119        // Try to remove everything from the disk cache, as it might pollute the cache
120        if let Err(e) = self.payload_storage.borrow().clear_cache() {
121            log::error!("Failed to clear cache of payload_storage: {e}");
122        }
123
124        if let Err(e) = self.payload_index.borrow().clear_cache() {
125            log::error!("Failed to clear cache of payload_index: {e}");
126        }
127
128        for (name, vector_data) in &self.vector_data {
129            let VectorData {
130                vector_index,
131                vector_storage,
132                quantized_vectors,
133            } = vector_data;
134
135            if let Err(e) = vector_index.borrow().clear_cache() {
136                log::error!("Failed to clear cache of vector index {name}: {e}");
137            }
138
139            if let Err(e) = vector_storage.borrow().clear_cache() {
140                log::error!("Failed to clear cache of vector storage {name}: {e}");
141            }
142
143            if let Some(quantized_vectors) = quantized_vectors.borrow().as_ref()
144                && let Err(e) = quantized_vectors.clear_cache()
145            {
146                log::error!("Failed to clear cache of quantized vectors {name}: {e}");
147            }
148        }
149    }
150}

153pub fn destroy_rocksdb(
154    path: &std::path::Path,
155) -> crate::segment::common::operation_error::OperationResult<()> {
156    rocksdb::DB::destroy(&Default::default(), path).map_err(|err| {
157        crate::segment::common::operation_error::OperationError::service_error(format!(
158            "failed to destroy RocksDB at {}: {err}",
159            path.display()
160        ))
161    })?;
162
163    Ok(())

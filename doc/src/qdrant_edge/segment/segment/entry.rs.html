63    fn search_batch(
64        &self,
65        vector_name: &VectorName,
66        query_vectors: &[&QueryVector],
67        with_payload: &WithPayload,
68        with_vector: &WithVector,
69        filter: Option<&Filter>,
70        top: usize,
71        params: Option<&SearchParams>,
72        query_context: &SegmentQueryContext,
73    ) -> OperationResult<Vec<Vec<ScoredPoint>>> {
74        check_query_vectors(vector_name, query_vectors, &self.segment_config)?;
75        let vector_data = &self
76            .vector_data
77            .get(vector_name)
78            .ok_or_else(|| OperationError::vector_name_not_exists(vector_name))?;
79        let vector_query_context =
80            query_context.get_vector_context(vector_name, self.deferred_internal_id);
81        let internal_results = vector_data.vector_index.borrow().search(
82            query_vectors,
83            filter,
84            top,
85            params,
86            &vector_query_context,
87        )?;
88
89        check_stopped(&vector_query_context.is_stopped())?;
90
91        let hw_counter = vector_query_context.hardware_counter();
92
93        internal_results
94            .into_iter()
95            .map(|internal_result| {
96                self.process_search_result(
97                    internal_result,
98                    with_payload,
99                    with_vector,
100                    &hw_counter,
101                    &vector_query_context.is_stopped(),
102                )
103            })
104            .collect()
105    }

172    fn retrieve(
173        &self,
174        point_ids: &[PointIdType],
175        with_payload: &WithPayload,
176        with_vector: &WithVector,
177        hw_counter: &HardwareCounterCell,
178        is_stopped: &AtomicBool,
179        deferred_behavior: DeferredBehavior,
180    ) -> OperationResult<AHashMap<ExtendedPointId, SegmentRecord>> {
181        let mut records = AHashMap::with_capacity(point_ids.len());
182
183        // Filter out deferred points. This is done in two stages to prevent cloning `point_ids` and iterating more that needed
184        // but still satisfy rusts ownership constraints.
185        let behavior_allows_filtering = !deferred_behavior.include_all_points();
186        let filter_deferred = self.has_deferred_points() && behavior_allows_filtering;
187        let filtered_point_ids = filter_deferred.then(|| {
188            point_ids
189                .iter()
190                .filter(|&&point_id| !self.point_is_deferred(point_id))
191                .copied()
192                .collect::<Vec<_>>()
193        });
194
195        // Stage two: Select the correct slice and shadow `point_ids`.
196        let point_ids = filtered_point_ids.as_deref().unwrap_or(point_ids);
197
198        let mut update_record_vector =
199            |vector_name: &VectorNameBuf,
200             point_id: PointIdType,
201             vector_internal: VectorInternal| {
202                let point_record = records
203                    .entry(point_id)
204                    .or_insert_with(|| SegmentRecord::empty(point_id));
205
206                point_record
207                    .vectors
208                    .get_or_insert_with(NamedVectorsOwned::default)
209                    .push((vector_name.clone(), vector_internal));
210            };
211
212        match with_vector {
213            WithVector::Bool(true) => {
214                for vector_name in self.vector_data.keys() {
215                    self.read_vectors(
216                        vector_name,
217                        point_ids,
218                        hw_counter,
219                        is_stopped,
220                        |point_id, vec| {
221                            update_record_vector(vector_name, point_id, vec);
222                        },
223                    )?;
224                }
225            }
226            WithVector::Bool(false) => {
227                // Do not display empty `vectors: {}` if disabled
228                for &point_id in point_ids {
229                    let point_record = records
230                        .entry(point_id)
231                        .or_insert_with(|| SegmentRecord::empty(point_id));
232                    point_record.vectors = None;
233                }
234            }
235            WithVector::Selector(selector) => {
236                for vector_name in selector {
237                    self.read_vectors(
238                        vector_name,
239                        point_ids,
240                        hw_counter,
241                        is_stopped,
242                        |point_id, vec| {
243                            update_record_vector(vector_name, point_id, vec);
244                        },
245                    )?;
246                }
247            }
248        }
249
250        for &point_id in point_ids {
251            let payload = if with_payload.enable {
252                if let Some(selector) = &with_payload.payload_selector {
253                    Some(selector.process(self.payload(point_id, hw_counter)?))
254                } else {
255                    Some(self.payload(point_id, hw_counter)?)
256                }
257            } else {
258                None
259            };
260            let point_record = records
261                .entry(point_id)
262                .or_insert_with(|| SegmentRecord::empty(point_id));
263            point_record.payload = payload;
264        }
265
266        Ok(records)
267    }

277    fn read_filtered<'a>(
278        &'a self,
279        offset: Option<PointIdType>,
280        limit: Option<usize>,
281        filter: Option<&'a Filter>,
282        is_stopped: &AtomicBool,
283        hw_counter: &HardwareCounterCell,
284        deferred_behavior: DeferredBehavior,
285    ) -> Vec<PointIdType> {
286        match filter {
287            None => self.read_by_id_stream(offset, limit, deferred_behavior),
288            Some(condition) => {
289                if self.should_pre_filter(condition, limit, hw_counter) {
290                    self.filtered_read_by_index(
291                        offset,
292                        limit,
293                        condition,
294                        is_stopped,
295                        hw_counter,
296                        deferred_behavior,
297                    )
298                } else {
299                    self.filtered_read_by_id_stream(
300                        offset,
301                        limit,
302                        condition,
303                        is_stopped,
304                        hw_counter,
305                        deferred_behavior,
306                    )
307                }
308            }
309        }
310    }

464    fn size_info(&self) -> SegmentInfo {
465        let num_vectors = self
466            .vector_data
467            .values()
468            .map(|data| data.vector_storage.borrow().available_vector_count())
469            .sum();
470
471        let mut total_average_vectors_size_bytes: usize = 0;
472
473        let vector_data_info: HashMap<_, _> = self
474            .vector_data
475            .iter()
476            .map(|(key, vector_data)| {
477                let vector_storage = vector_data.vector_storage.borrow();
478                let num_vectors = vector_storage.available_vector_count();
479                let vector_index = vector_data.vector_index.borrow();
480                let is_indexed = vector_index.is_index();
481
482                let average_vector_size_bytes = vector_index
483                    .size_of_searchable_vectors_in_bytes()
484                    .checked_div(num_vectors)
485                    .unwrap_or(0);
486                total_average_vectors_size_bytes += average_vector_size_bytes;
487
488                let vector_data_info = VectorDataInfo {
489                    num_vectors,
490                    num_indexed_vectors: if is_indexed {
491                        vector_index.indexed_vector_count()
492                    } else {
493                        0
494                    },
495                    num_deleted_vectors: vector_storage.deleted_vector_count(),
496                };
497                (key.clone(), vector_data_info)
498            })
499            .collect();
500
501        let num_indexed_vectors = if self.segment_type == SegmentType::Indexed {
502            self.vector_data
503                .values()
504                .map(|data| data.vector_index.borrow().indexed_vector_count())
505                .sum()
506        } else {
507            0
508        };
509
510        let num_points = self.available_point_count();
511
512        let vectors_size_bytes = total_average_vectors_size_bytes * num_points;
513
514        // Unwrap and default to 0 here because the RocksDB storage is the only faillible one, and we will remove it eventually.
515        let payloads_size_bytes = self
516            .payload_storage
517            .borrow()
518            .get_storage_size_bytes()
519            .unwrap_or(0);
520
521        SegmentInfo {
522            uuid: self.segment_uuid(),
523            segment_type: self.segment_type,
524            num_vectors,
525            num_indexed_vectors,
526            num_points: self.available_point_count(),
527            num_deleted_vectors: self.deleted_point_count(),
528            vectors_size_bytes,  // Considers vector storage, but not indices
529            payloads_size_bytes, // Considers payload storage, but not indices
530            ram_usage_bytes: 0,  // ToDo: Implement
531            disk_usage_bytes: 0, // ToDo: Implement
532            is_appendable: self.appendable_flag,
533            index_schema: HashMap::new(),
534            vector_data: vector_data_info,
535        }
536    }

564    fn flusher(&self, force: bool) -> Option<Flusher> {
565        let current_persisted_version: Option<SeqNumberType> = *self.persisted_version.lock();
566
567        match (self.version, current_persisted_version) {
568            (None, _) => {
569                // Segment is empty, nothing to flush
570                return None;
571            }
572            (Some(version), Some(persisted_version)) => {
573                if !force && version == persisted_version {
574                    log::trace!("not flushing because version == persisted_version");
575                    // Segment is already flushed
576                    return None;
577                }
578            }
579            (_, _) => {}
580        }
581
582        // Capture all flushers first to improve data consistency
583        let vector_storage_flushers: Vec<_> = self
584            .vector_data
585            .values()
586            .map(|v| v.vector_storage.borrow().flusher())
587            .collect();
588        let quantization_flushers: Vec<_> = self
589            .vector_data
590            .values()
591            .filter_map(|v| v.quantized_vectors.borrow().as_ref().map(|q| q.flusher()))
592            .collect();
593        let state = self.get_state();
594        let segment_path = self.segment_path.clone();
595        let id_tracker_mapping_flusher = self.id_tracker.borrow().mapping_flusher();
596        let payload_index_flusher = self.payload_index.borrow().flusher();
597        let id_tracker_versions_flusher = self.id_tracker.borrow().versions_flusher();
598        let persisted_version = self.persisted_version.clone();
599
600        // Flush order is important:
601        //
602        // 1. Flush id mapping. So during recovery the point will be recovered in proper segment.
603        // 2. Flush vectors and payloads.
604        // 3. Flush id versions last. So presence of version indicates that all other data is up-to-date.
605        //
606        // Example of recovery from WAL in case of partial flush:
607        //
608        // In-memory state:
609        //
610        //     Segment 1                  Segment 2
611        //
612        //    ID-mapping     vst.1       ID-mapping     vst.2
613        //   ext     int
614        //  ┌───┐   ┌───┐   ┌───┐       ┌───┐   ┌───┐   ┌───┐
615        //  │100├───┤1  │   │1  │       │300├───┤1  │   │1  │
616        //  └───┘   └───┘   │2  │       └───┘   └───┘   │2  │
617        //                  │   │                       │   │
618        //  ┌───┐   ┌───┐   │   │       ┌───┐   ┌───┐   │   │
619        //  │200├───┤2  │   │   │       │400├───┤2  │   │   │
620        //  └───┘   └───┘   └───┘       └───┘   └───┘   └───┘
621        //
622        //
623        //  ext - external id
624        //  int - internal id
625        //  vst - vector storage
626        //
627        //  ─────────────────────────────────────────────────
628        //   After flush, segments could be partially preserved:
629        //
630        //  ┌───┐   ┌───┐   ┌───┐       ┌───┐   ┌───┐   ┌───┐
631        //  │100├───┤1  │   │ 1 │       │300├───┤1  │   │ * │
632        //  └───┘   └───┘   │   │       └───┘   └───┘   │ * │
633        //                  │   │                       │ 3 │
634        //                  │   │       ┌───┐   ┌───┐   │   │
635        //                  │   │       │400├───┤2  │   │   │
636        //                  └───┘       └───┘   └───┘   └───┘
637        //  WAL:      ▲
638        //            │                 ┌───┐   ┌───┐
639        //  100───────┘      ┌────────► │200├───┤3  │
640        //                   |          └───┘   └───┘
641        //  200──────────────┘
642        //
643        //  300
644        //
645        //  400
646
647        let is_alive_flush_lock = self.is_alive_flush_lock.handle();
648
649        let flush_op = move || {
650            let Some(is_alive_flush_guard) = is_alive_flush_lock.lock_if_alive() else {
651                // Segment is removed, skip flush
652                log::debug!("Segment was dropped, skip flush");
653                return Ok(());
654            };
655
656            let flush_components = || {
657                // Flush mapping first to prevent having orphan internal ids.
658                id_tracker_mapping_flusher().map_err(|err| match err {
659                    OperationError::Cancelled { .. } => err,
660                    _ => OperationError::service_error(format!(
661                        "Failed to flush id_tracker mapping: {err}"
662                    )),
663                })?;
664                for vector_storage_flusher in vector_storage_flushers {
665                    vector_storage_flusher().map_err(|err| match err {
666                        OperationError::Cancelled { .. } => err,
667                        _ => OperationError::service_error(format!(
668                            "Failed to flush vector_storage: {err}"
669                        )),
670                    })?;
671                }
672                for quantization_flusher in quantization_flushers {
673                    quantization_flusher().map_err(|err| match err {
674                        OperationError::Cancelled { .. } => err,
675                        _ => OperationError::service_error(format!(
676                            "Failed to flush quantized vectors: {err}"
677                        )),
678                    })?;
679                }
680                payload_index_flusher().map_err(|err| match err {
681                    OperationError::Cancelled { .. } => err,
682                    _ => OperationError::service_error(format!(
683                        "Failed to flush payload_index: {err}"
684                    )),
685                })?;
686                // Id Tracker contains versions of points. We need to flush it after vector_storage and payload_index flush.
687                // This is because vector_storage and payload_index flush are not atomic.
688                // If payload or vector flush fails, we will be able to recover data from WAL.
689                // If Id Tracker flush fails, we are also able to recover data from WAL
690                //  by simply overriding data in vector and payload storages.
691                // Once versions are saved - points are considered persisted.
692                id_tracker_versions_flusher().map_err(|err| match err {
693                    OperationError::Cancelled { .. } => err,
694                    _ => OperationError::service_error(format!(
695                        "Failed to flush id_tracker versions: {err}"
696                    )),
697                })?;
698
699                Ok(())
700            };
701
702            match flush_components() {
703                // Only continue if all components flushed Ok
704                Ok(()) => {}
705
706                // Return early to avoid updating persisted version
707                // Flush was cancelled, bypass
708                Err(OperationError::Cancelled { description }) => {
709                    log::debug!("Segment flush cancelled: {description}");
710                    return Ok(());
711                }
712
713                // Propagate other errors
714                Err(err) => return Err(err),
715            }
716
717            let mut current_persisted_version_guard = persisted_version.lock();
718            let persisted_version_value_opt = *current_persisted_version_guard;
719
720            if persisted_version_value_opt > state.version {
721                debug_assert!(
722                    persisted_version_value_opt.is_some(),
723                    "Persisted version should never be None if it's greater than state.version"
724                );
725                // Another flush beat us to it
726                return Ok(());
727            }
728
729            Self::save_state(&state, &segment_path).map_err(|err| {
730                OperationError::service_error(format!("Failed to flush segment state: {err}"))
731            })?;
732
733            *current_persisted_version_guard = state.version;
734            debug_assert!(state.version.is_some());
735
736            // Keep the guard till the end of the flush to prevent concurrent drop/flushes
737            drop(is_alive_flush_guard);
738
739            Ok(())
740        };
741
742        Some(Box::new(flush_op))
743    }

785    fn build_field_index(
786        &self,
787        op_num: SeqNumberType,
788        key: PayloadKeyTypeRef,
789        field_type: &PayloadFieldSchema,
790        hw_counter: &HardwareCounterCell,
791    ) -> OperationResult<BuildFieldIndexResult> {
792        // Check version without updating it
793        if self.version.unwrap_or(0) > op_num {
794            return Ok(BuildFieldIndexResult::SkippedByVersion);
795        }
796
797        let field_index = match self
798            .payload_index
799            .borrow()
800            .build_index(key, field_type, hw_counter)?
801        {
802            BuildIndexResult::Built(indexes) => indexes,
803            BuildIndexResult::AlreadyBuilt => {
804                return Ok(BuildFieldIndexResult::AlreadyExists);
805            }
806            BuildIndexResult::IncompatibleSchema => {
807                // This function expects that incompatible schema is already removed
808                return Ok(BuildFieldIndexResult::IncompatibleSchema);
809            }
810        };
811
812        Ok(BuildFieldIndexResult::Built {
813            indexes: field_index,
814            schema: field_type.clone(),
815        })
816    }

892    fn fill_query_context(&self, query_context: &mut QueryContext) {
893        query_context.add_available_point_count(self.available_point_count());
894        let hw_acc = query_context.hardware_usage_accumulator();
895        let hw_counter = hw_acc.get_counter_cell();
896
897        let QueryIdfStats {
898            idf,
899            indexed_vectors,
900        } = query_context.mut_idf_stats();
901
902        for (vector_name, idf) in idf.iter_mut() {
903            if let Some(vector_data) = self.vector_data.get(vector_name) {
904                let vector_index = vector_data.vector_index.borrow();
905
906                let indexed_vector_count = vector_index.indexed_vectors();
907
908                if let Some(count) = indexed_vectors.get_mut(vector_name) {
909                    *count += indexed_vector_count;
910                } else {
911                    indexed_vectors.insert(vector_name.clone(), indexed_vector_count);
912                }
913
914                vector_index.fill_idf_statistics(idf, &hw_counter);
915            }
916        }
917    }

928    fn deferred_point_ids(&self) -> Vec<PointIdType> {
929        let Some(deferred_from) = self.deferred_internal_id else {
930            return vec![];
931        };
932        let id_tracker = self.id_tracker.borrow();
933        id_tracker
934            .iter_internal()
935            .skip_while(|&internal_id| internal_id < deferred_from)
936            .filter_map(|internal_id| id_tracker.external_id(internal_id))
937            .collect()
938    }

942    fn upsert_point(
943        &mut self,
944        op_num: SeqNumberType,
945        point_id: PointIdType,
946        mut vectors: NamedVectors,
947        hw_counter: &HardwareCounterCell,
948    ) -> OperationResult<bool> {
949        debug_assert!(self.is_appendable());
950        check_named_vectors(&vectors, &self.segment_config)?;
951        vectors.preprocess(|name| self.config().vector_data.get(name).unwrap());
952        let stored_internal_point = self.id_tracker.borrow().internal_id(point_id);
953        self.handle_point_version_and_failure(op_num, stored_internal_point, |segment| {
954            if let Some(existing_internal_id) = stored_internal_point {
955                segment.replace_all_vectors(existing_internal_id, op_num, &vectors, hw_counter)?;
956                Ok((true, Some(existing_internal_id)))
957            } else {
958                let new_index =
959                    segment.insert_new_vectors(point_id, op_num, &vectors, hw_counter)?;
960                Ok((false, Some(new_index)))
961            }
962        })
963    }

984            }

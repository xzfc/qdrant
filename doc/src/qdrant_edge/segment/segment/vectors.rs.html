1use std::sync::atomic::AtomicBool;
2
3use crate::common::counter::hardware_counter::HardwareCounterCell;
4use crate::common::iterator_ext::IteratorExt;
5
6use crate::segment::common::operation_error::OperationResult;
7use crate::segment::data_types::vectors::VectorInternal;
8use crate::segment::id_tracker::IdTracker;
9use crate::segment::segment::Segment;
10use crate::segment::types::{PointIdType, VectorName};
11
12impl Segment {
13    pub(crate) fn read_vectors(
14        &self,
15        vector_names: &VectorName,
16        point_ids: &[PointIdType],
17        hw_counter: &HardwareCounterCell,
18        is_stopped: &AtomicBool,
19        mut callback: impl FnMut(PointIdType, VectorInternal),
20    ) -> OperationResult<()> {
21        let mut error = None;
22        let internal_ids = point_ids
23            .iter()
24            .copied()
25            .stop_if(is_stopped)
26            .filter_map(|point_id| match self.lookup_internal_id(point_id) {
27                Ok(point_offset) => Some(point_offset),
28                Err(err) => {
29                    error = Some(err);
30                    None
31                }
32            });
33        self.vectors_by_offsets(
34            vector_names,
35            internal_ids,
36            hw_counter,
37            |point_offset, vector_internal| {
38                if let Some(point_id) = self.id_tracker.borrow().external_id(point_offset) {
39                    callback(point_id, vector_internal);
40                }
41            },
42        )?;
43        if let Some(err) = error {
44            return Err(err);
45        }
46        Ok(())
47    }
48}

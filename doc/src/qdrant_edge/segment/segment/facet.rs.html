20    pub(super) fn approximate_facet(
21        &self,
22        request: &FacetParams,
23        is_stopped: &AtomicBool,
24        hw_counter: &HardwareCounterCell,
25    ) -> OperationResult<HashMap<FacetValue, usize>> {
26        let payload_index = self.payload_index.borrow();
27
28        // Shortcut if this segment has no points, prevent division by zero later
29        let available_points = self.available_point_count();
30        if available_points == 0 {
31            return Ok(HashMap::new());
32        }
33
34        let facet_index = payload_index.get_facet_index(&request.key)?;
35        let context;
36
37        let hits_iter = if let Some(filter) = &request.filter {
38            let id_tracker = self.id_tracker.borrow();
39            let filter_cardinality = payload_index.estimate_cardinality(filter, hw_counter);
40
41            let percentage_filtered = filter_cardinality.exp as f64 / available_points as f64;
42
43            // TODO(facets): define a better estimate for this decision, the question is:
44            // What is more expensive, to hash the same value excessively or to check with filter too many times?
45            //
46            // For now this is defined from some rudimentary benchmarking two scenarios:
47            // - a collection with few keys
48            // - a collection with almost a unique key per point
49            let use_iterative_approach = percentage_filtered < 0.3;
50
51            let iter = if use_iterative_approach {
52                // go over the filtered points and aggregate the values
53                // aka. read from other indexes
54                let iter = payload_index
55                    .iter_filtered_points(
56                        filter,
57                        &id_tracker,
58                        &filter_cardinality,
59                        hw_counter,
60                        is_stopped,
61                        self.deferred_internal_id,
62                    )
63                    .filter(|point_id| !id_tracker.is_deleted_point(*point_id))
64                    .fold(HashMap::new(), |mut map, point_id| {
65                        facet_index
66                            .get_point_values(point_id)
67                            .unique()
68                            .for_each(|value| {
69                                *map.entry(value).or_insert(0) += 1;
70                            });
71                        map
72                    })
73                    .into_iter()
74                    .map(|(value, count)| FacetHit { value, count });
75
76                Either::Left(iter)
77            } else {
78                // go over the values and filter the points
79                // aka. read from facet index
80                //
81                // This is more similar to a full-scan, but we won't be hashing so many times.
82                context = payload_index.struct_filtered_context(filter, hw_counter);
83
84                let iter = facet_index
85                    .iter_values_map(hw_counter)
86                    .stop_if(is_stopped)
87                    .filter_map(|(value, iter)| {
88                        let count = iter
89                            .unique()
90                            .take_while(|&point_id| {
91                                point_id < self.deferred_internal_id.unwrap_or(PointOffsetType::MAX)
92                            })
93                            .filter(|&point_id| context.check(point_id))
94                            .count();
95
96                        (count > 0).then_some(FacetHit { value, count })
97                    });
98
99                Either::Right(iter)
100            };
101            Either::Left(iter)
102        } else {
103            // just count how many points each value has
104            let iter = facet_index
105                .iter_counts_per_value(self.deferred_internal_id)
106                .stop_if(is_stopped)
107                .filter(|hit| hit.count > 0);
108
109            Either::Right(iter)
110        };
111
112        // We can't just select top values, because we need to aggregate across segments,
113        // which we can't assume to select the same best top.
114        //
115        // We need all values to be able to aggregate correctly across segments
116        let hits: HashMap<_, _> = hits_iter
117            .map(|hit| (hit.value.to_owned(), hit.count))
118            .collect();
119
120        Ok(hits)
121    }

123    pub(super) fn facet_values(
124        &self,
125        key: &JsonPath,
126        filter: Option<&Filter>,
127        is_stopped: &AtomicBool,
128        hw_counter: &HardwareCounterCell,
129    ) -> OperationResult<BTreeSet<FacetValue>> {
130        let payload_index = self.payload_index.borrow();
131
132        let facet_index = payload_index.get_facet_index(key)?;
133
134        let values = if let Some(filter) = filter {
135            let id_tracker = self.id_tracker.borrow();
136            let filter_cardinality = payload_index.estimate_cardinality(filter, hw_counter);
137
138            payload_index
139                .iter_filtered_points(
140                    filter,
141                    &id_tracker,
142                    &filter_cardinality,
143                    hw_counter,
144                    is_stopped,
145                    self.deferred_internal_id,
146                )
147                .filter(|point_id| !id_tracker.is_deleted_point(*point_id))
148                .fold(BTreeSet::new(), |mut set, point_id| {
149                    set.extend(facet_index.get_point_values(point_id));
150                    set
151                })
152                .into_iter()
153                .map(|value| value.to_owned())
154                .collect()
155        } else {
156            facet_index
157                .iter_values(hw_counter, self.deferred_internal_id)
158                .stop_if(is_stopped)
159                .map(|value_ref| value_ref.to_owned())
160                .collect()
161        };
162
163        Ok(values)
164    }

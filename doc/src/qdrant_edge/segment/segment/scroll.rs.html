15    /// Estimates how many checks it would need for getting `limit` amount of points by streaming and then
16    /// filtering, versus getting all filtered points from the index and then sorting them afterwards.
17    ///
18    /// If the filter is restrictive enough to yield fewer points than the amount of points a streaming
19    /// approach would need to advance, it returns true.
20    pub(super) fn should_pre_filter(
21        &self,
22        filter: &Filter,
23        limit: Option<usize>,
24        hw_counter: &HardwareCounterCell,
25    ) -> bool {
26        let query_cardinality = {
27            let payload_index = self.payload_index.borrow();
28            payload_index.estimate_cardinality(filter, hw_counter)
29        };
30
31        // ToDo: Add telemetry for this heuristics
32
33        // Calculate expected number of condition checks required for
34        // this scroll request with is stream strategy.
35        // Example:
36        //  - cardinality = 1000
37        //  - limit = 10
38        //  - total = 10000
39        //  - point filter prob = 1000 / 10000 = 0.1
40        //  - expected_checks = 10 / 0.1  = 100
41        //  -------------------------------
42        //  - cardinality = 10
43        //  - limit = 10
44        //  - total = 10000
45        //  - point filter prob = 10 / 10000 = 0.001
46        //  - expected_checks = 10 / 0.001  = 10000
47
48        let available_points = self.available_point_count() + 1 /* + 1 for division-by-zero */;
49        // Expected number of successful checks per point
50        let check_probability =
51            (query_cardinality.exp as f64 + 1.0/* protect from zero */) / available_points as f64;
52        let exp_stream_checks =
53            (limit.unwrap_or(available_points) as f64 / check_probability) as usize;
54
55        // Assume it would require about `query cardinality` checks.
56        // We are interested in approximate number of checks, so we can
57        // use `query cardinality` as a starting point.
58        let exp_index_checks = query_cardinality.max;
59
60        exp_stream_checks > exp_index_checks
61    }

63    pub fn filtered_read_by_id_stream(
64        &self,
65        offset: Option<PointIdType>,
66        limit: Option<usize>,
67        condition: &Filter,
68        is_stopped: &AtomicBool,
69        hw_counter: &HardwareCounterCell,
70        deferred_behavior: DeferredBehavior,
71    ) -> Vec<PointIdType> {
72        let effective_deferred_id = deferred_behavior.apply(self.deferred_internal_id);
73
74        let payload_index = self.payload_index.borrow();
75        let filter_context = payload_index.filter_context(condition, hw_counter);
76        self.id_tracker
77            .borrow()
78            .iter_from_visible(offset, effective_deferred_id)
79            .stop_if(is_stopped)
80            .filter(move |(_, internal_id)| filter_context.check(*internal_id))
81            .map(|(external_id, _)| external_id)
82            .take(limit.unwrap_or(usize::MAX))
83            .collect()
84    }

86    pub(super) fn read_by_id_stream(
87        &self,
88        offset: Option<PointIdType>,
89        limit: Option<usize>,
90        deferred_behavior: DeferredBehavior,
91    ) -> Vec<PointIdType> {
92        let effective_deferred_id = deferred_behavior.apply(self.deferred_internal_id);
93
94        self.id_tracker
95            .borrow()
96            .iter_from_visible(offset, effective_deferred_id)
97            .map(|x| x.0)
98            .take(limit.unwrap_or(usize::MAX))
99            .collect()
100    }

102    pub fn filtered_read_by_index(
103        &self,
104        offset: Option<PointIdType>,
105        limit: Option<usize>,
106        condition: &Filter,
107        is_stopped: &AtomicBool,
108        hw_counter: &HardwareCounterCell,
109        deferred_behavior: DeferredBehavior,
110    ) -> Vec<PointIdType> {
111        let effective_deferred_id = deferred_behavior.apply(self.deferred_internal_id);
112
113        let payload_index = self.payload_index.borrow();
114        let id_tracker = self.id_tracker.borrow();
115        let cardinality_estimation = payload_index.estimate_cardinality(condition, hw_counter);
116
117        let ids_iterator = payload_index
118            .iter_filtered_points(
119                condition,
120                &id_tracker,
121                &cardinality_estimation,
122                hw_counter,
123                is_stopped,
124                effective_deferred_id,
125            )
126            .filter_map(|internal_id| {
127                let external_id = id_tracker.external_id(internal_id);
128                match external_id {
129                    Some(external_id) => match offset {
130                        Some(offset) if external_id < offset => None,
131                        _ => Some(external_id),
132                    },
133                    None => None,
134                }
135            });
136
137        let mut page = match limit {
138            Some(limit) => peek_top_smallest_iterable(ids_iterator, limit),
139            None => ids_iterator.collect(),
140        };
141
142        page.sort_unstable();
143
144        page
145    }

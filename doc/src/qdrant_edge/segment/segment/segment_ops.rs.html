30    /// Replace vectors in-place
31    ///
32    /// This replaces all named vectors for this point with the given set of named vectors.
33    ///
34    /// - new named vectors are inserted
35    /// - existing named vectors are replaced
36    /// - existing named vectors not specified are deleted
37    ///
38    /// This differs with [`Segment::update_vectors`], because this deletes unspecified vectors.
39    ///
40    /// # Warning
41    ///
42    /// Available for appendable segments only.
43    pub(super) fn replace_all_vectors(
44        &mut self,
45        internal_id: PointOffsetType,
46        op_num: SeqNumberType,
47        vectors: &NamedVectors,
48        hw_counter: &HardwareCounterCell,
49    ) -> OperationResult<()> {
50        debug_assert!(self.is_appendable());
51        check_named_vectors(vectors, &self.segment_config)?;
52        for (vector_name, vector_data) in self.vector_data.iter_mut() {
53            let vector = vectors.get(vector_name);
54            let mut vector_index = vector_data.vector_index.borrow_mut();
55            vector_index.update_vector(internal_id, vector, hw_counter)?;
56            self.version_tracker.set_vector(vector_name, Some(op_num));
57        }
58        Ok(())
59    }

61    /// Update vectors in-place
62    ///
63    /// This updates all specified named vectors for this point with the given set of named vectors, leaving unspecified vectors untouched.
64    ///
65    /// - new named vectors are inserted
66    /// - existing named vectors are replaced
67    /// - existing named vectors not specified are untouched and kept as-is
68    ///
69    /// This differs with [`Segment::replace_all_vectors`], because this keeps unspecified vectors as-is.
70    ///
71    /// # Warning
72    ///
73    /// Available for appendable segments only.
74    #[allow(clippy::needless_pass_by_ref_mut)] // ensure single access to AtomicRefCell vector_index
75    pub(super) fn update_vectors(
76        &mut self,
77        internal_id: PointOffsetType,
78        op_num: SeqNumberType,
79        vectors: NamedVectors,
80        hw_counter: &HardwareCounterCell,
81    ) -> OperationResult<()> {
82        debug_assert!(self.is_appendable());
83        check_named_vectors(&vectors, &self.segment_config)?;
84        for (vector_name, new_vector) in vectors {
85            let vector_data = &self.vector_data[vector_name.as_ref()];
86            let mut vector_index = vector_data.vector_index.borrow_mut();
87            vector_index.update_vector(internal_id, Some(new_vector.as_vec_ref()), hw_counter)?;
88            self.version_tracker.set_vector(&vector_name, Some(op_num));
89        }
90        Ok(())
91    }

93    /// Insert new vectors into the segment
94    ///
95    /// # Warning
96    ///
97    /// Available for appendable segments only.
98    pub(super) fn insert_new_vectors(
99        &mut self,
100        point_id: PointIdType,
101        op_num: SeqNumberType,
102        vectors: &NamedVectors,
103        hw_counter: &HardwareCounterCell,
104    ) -> OperationResult<PointOffsetType> {
105        debug_assert!(self.is_appendable());
106        check_named_vectors(vectors, &self.segment_config)?;
107        let new_index = self.id_tracker.borrow().total_point_count() as PointOffsetType;
108        for (vector_name, vector_data) in self.vector_data.iter_mut() {
109            let vector_opt = vectors.get(vector_name);
110            let mut vector_index = vector_data.vector_index.borrow_mut();
111            vector_index.update_vector(new_index, vector_opt, hw_counter)?;
112            self.version_tracker.set_vector(vector_name, Some(op_num));
113        }
114        self.id_tracker.borrow_mut().set_link(point_id, new_index)?;
115        Ok(new_index)
116    }

118    /// Operation wrapped, which handles previous and new errors in the segment, automatically
119    /// updates versions and skips operations if the segment version is too old
120    ///
121    /// # Arguments
122    ///
123    /// * `op_num` - sequential operation of the current operation
124    /// * `op` - operation to be wrapped. Should return `OperationResult` of bool (which is returned outside)
125    ///   and optionally new offset of the changed point.
126    ///
127    /// # Result
128    ///
129    /// Propagates `OperationResult` of bool (which is returned in the `op` closure)
130    pub(super) fn handle_segment_version_and_failure<F>(
131        &mut self,
132        op_num: SeqNumberType,
133        operation: F,
134    ) -> OperationResult<bool>
135    where
136        F: FnOnce(&mut Segment) -> OperationResult<bool>,
137    {
138        if let Some(SegmentFailedState {
139            version: failed_version,
140            point_id: _failed_point_id,
141            error,
142        }) = &self.error_status
143        {
144            // Failed operations should not be skipped,
145            // fail if newer operation is attempted before proper recovery
146            if *failed_version < op_num {
147                return Err(OperationError::service_error(format!(
148                    "Not recovered from previous error: {error}"
149                )));
150            } // else: Re-try operation
151        }
152
153        let res = self.handle_segment_version(op_num, operation);
154
155        if let Some(error) = get_service_error(&res) {
156            // ToDo: Recover previous segment state
157            log::error!(
158                "Segment {:?} operation error: {error}",
159                self.segment_path.as_path(),
160            );
161            self.error_status = Some(SegmentFailedState {
162                version: op_num,
163                point_id: None,
164                error,
165            });
166        }
167        res
168    }

170    /// Operation wrapped, which handles previous and new errors in the segment, automatically
171    /// updates versions and skips operations if the point version is too old
172    ///
173    /// # Arguments
174    ///
175    /// * `op_num` - sequential operation of the current operation
176    /// * `op_point_offset` - If point offset is specified, handler will use point version for comparison.
177    ///   Otherwise, it will be applied without version checks.
178    /// * `op` - operation to be wrapped. Should return `OperationResult` of bool (which is returned outside) and optionally new offset of the changed point.
179    ///
180    /// # Result
181    ///
182    /// Propagates `OperationResult` of bool (which is returned in the `op` closure)
183    pub(super) fn handle_point_version_and_failure<F>(
184        &mut self,
185        op_num: SeqNumberType,
186        op_point_offset: Option<PointOffsetType>,
187        operation: F,
188    ) -> OperationResult<bool>
189    where
190        F: FnOnce(&mut Segment) -> OperationResult<(bool, Option<PointOffsetType>)>,
191    {
192        if let Some(SegmentFailedState {
193            version: failed_version,
194            point_id: _failed_point_id,
195            error,
196        }) = &self.error_status
197        {
198            // Failed operations should not be skipped,
199            // fail if newer operation is attempted before proper recovery
200            if *failed_version < op_num {
201                return Err(OperationError::service_error(format!(
202                    "Not recovered from previous error: {error}"
203                )));
204            } // else: Re-try operation
205        }
206
207        let res = self.handle_point_version(op_num, op_point_offset, operation);
208
209        match get_service_error(&res) {
210            None => {
211                // Recover error state
212                match &self.error_status {
213                    None => {} // all good
214                    Some(error) => {
215                        let point_id = op_point_offset.and_then(|point_offset| {
216                            self.id_tracker.borrow().external_id(point_offset)
217                        });
218                        if error.point_id == point_id {
219                            // Fixed
220                            log::info!("Recovered from error: {}", error.error);
221                            self.error_status = None;
222                        }
223                    }
224                }
225            }
226            Some(error) => {
227                // ToDo: Recover previous segment state
228                log::error!(
229                    "Segment {:?} operation error: {error}",
230                    self.segment_path.as_path(),
231                );
232                let point_id = op_point_offset
233                    .and_then(|point_offset| self.id_tracker.borrow().external_id(point_offset));
234                self.error_status = Some(SegmentFailedState {
235                    version: op_num,
236                    point_id,
237                    error,
238                });
239            }
240        }
241        res
242    }

244    /// Manage segment version checking, for segment level operations
245    ///
246    /// If current version is higher than operation version - do not perform the operation
247    /// Update current version if operation successfully executed
248    fn handle_segment_version<F>(
249        &mut self,
250        op_num: SeqNumberType,
251        operation: F,
252    ) -> OperationResult<bool>
253    where
254        F: FnOnce(&mut Segment) -> OperationResult<bool>,
255    {
256        // Global version to check if operation has already been applied, then skip without execution
257        if self.version.unwrap_or(0) > op_num {
258            return Ok(false);
259        }
260
261        let applied = operation(self)?;
262        self.bump_segment_version(op_num);
263        Ok(applied)
264    }

266    /// Manage point version checking inside this segment, for point level operations
267    ///
268    /// If current version is higher than operation version - do not perform the operation
269    /// Update current version if operation successfully executed
270    pub(super) fn handle_point_version<F>(
271        &mut self,
272        op_num: SeqNumberType,
273        op_point_offset: Option<PointOffsetType>,
274        operation: F,
275    ) -> OperationResult<bool>
276    where
277        F: FnOnce(&mut Segment) -> OperationResult<(bool, Option<PointOffsetType>)>,
278    {
279        // If point does not exist or has lower version, ignore operation
280        if let Some(point_offset) = op_point_offset
281            && self
282                .id_tracker
283                .borrow()
284                .internal_version(point_offset)
285                .is_some_and(|current_version| current_version > op_num)
286        {
287            return Ok(false);
288        }
289
290        let (applied, internal_id) = operation(self)?;
291
292        self.bump_segment_version(op_num);
293        if let Some(internal_id) = internal_id {
294            self.id_tracker
295                .borrow_mut()
296                .set_internal_version(internal_id, op_num)?;
297        }
298
299        Ok(applied)
300    }

302    pub fn delete_point_internal(
303        &mut self,
304        internal_id: PointOffsetType,
305        hw_counter: &HardwareCounterCell,
306    ) -> OperationResult<()> {
307        // Mark point as deleted, drop mapping
308        self.payload_index
309            .borrow_mut()
310            .clear_payload(internal_id, hw_counter)?;
311
312        self.id_tracker.borrow_mut().drop_internal(internal_id)?;
313
314        // Before, we propagated point deletions to also delete its vectors. This turns
315        // out to be problematic because this sometimes makes us lose vector data
316        // because we cannot control the order of segment flushes.
317        // Disabled until we properly fix it or find a better way to clean up old
318        // vectors.
319        //
320        // // Propagate point deletion to all its vectors
321        // for vector_data in segment.vector_data.values() {
322        //     let mut vector_storage = vector_data.vector_storage.borrow_mut();
323        //     vector_storage.delete_vector(internal_id)?;
324        // }
325
326        Ok(())
327    }

400    /// Retrieve multiple vectors by internal ID
401    pub(super) fn vectors_by_offsets(
402        &self,
403        vector_name: &VectorName,
404        point_offsets: impl IntoIterator<Item = PointOffsetType>,
405        hw_counter: &HardwareCounterCell,
406        mut callback: impl FnMut(PointOffsetType, VectorInternal),
407    ) -> OperationResult<()> {
408        check_vector_name(vector_name, &self.segment_config)?;
409        let vector_data = &self
410            .vector_data
411            .get(vector_name)
412            .ok_or_else(|| OperationError::vector_name_not_exists(vector_name))?;
413        let vector_storage = vector_data.vector_storage.borrow();
414        let total_vectors = vector_storage.total_vector_count();
415
416        let id_tracker = self.id_tracker.borrow();
417        let non_deleted_offsets = point_offsets.into_iter().filter(|&point_offset| {
418            if total_vectors <= point_offset as usize {
419                debug_assert!(
420                    false,
421                    "Vector storage is inconsistent, total_vector_count: {total_vectors}, point_offset: {point_offset}, external_id: {:?}",
422                    id_tracker.external_id(point_offset),
423                );
424                return false;
425            }
426
427            let is_vector_deleted = vector_storage.is_deleted_vector(point_offset);
428            let is_point_deleted = id_tracker.is_deleted_point(point_offset);
429            !is_vector_deleted && !is_point_deleted
430        });
431
432        vector_storage.read_vectors::<Random>(non_deleted_offsets, |point_offset, cow_vector| {
433            if vector_storage.is_on_disk() {
434                hw_counter
435                    .vector_io_read()
436                    .incr_delta(cow_vector.estimate_size_in_bytes());
437            }
438            callback(point_offset, cow_vector.to_owned());
439        });
440
441        Ok(())
442    }

476    /// Check consistency of the segment's data and repair it if possible.
477    /// Removes partially persisted points.
478    pub fn check_consistency_and_repair(&mut self) -> OperationResult<()> {
479        // Get rid of mappingless points.
480        let ids_to_clean = self.fix_id_tracker_inconsistencies()?;
481
482        // There are some leftovers to clean from segment.
483        // After that we need to set internal version to 0, so that
484        // we won't need to clean them again.
485
486        // This is internal operation, no hw measurement needed
487        let disposable_hw_counter = HardwareCounterCell::disposable();
488        if !ids_to_clean.is_empty() {
489            log::debug!(
490                "Cleaning up {} points with version but no mapping in segment {:?}",
491                ids_to_clean.len(),
492                self.data_path()
493            );
494
495            for internal_id in ids_to_clean {
496                self.delete_point_internal(internal_id, &disposable_hw_counter)?;
497            }
498
499            self.flush(true)?;
500
501            // We do not drop version here, because it is already not loaded into memory.
502            // There are no explicit mapping between internal ID and version, so all dangling
503            // versions will be ignored automatically.
504            // Those versions could be overwritten by new points, but it is not a problem.
505            // They will also be deleted by the next optimization.
506        }
507
508        Ok(())
509    }

511    /// Update all payload/field indices to match `desired_schemas`
512    ///
513    /// Missing payload indices are created. Incorrectly configured payload indices are recreated.
514    /// Extra payload indices are NOT deleted.
515    ///
516    /// This does nothing if the current payload indices state matches `desired_schemas` exactly.
517    pub fn update_all_field_indices(
518        &mut self,
519        desired_schemas: &HashMap<PayloadKeyType, PayloadFieldSchema>,
520    ) -> OperationResult<()> {
521        let schema_applied = self.payload_index.borrow().indexed_fields();
522        let schema_config = desired_schemas;
523
524        // Create or update payload indices if they don't match configuration
525        for (key, schema) in schema_config {
526            match schema_applied.get(key) {
527                Some(existing_schema) if existing_schema == schema => continue,
528                Some(existing_schema) => log::warn!(
529                    "Segment has incorrect payload index for {key}, recreating it now (current: {:?}, configured: {:?})",
530                    existing_schema.name(),
531                    schema.name(),
532                ),
533                None => log::warn!(
534                    "Segment is missing a {} payload index for {key}, creating it now",
535                    schema.name(),
536                ),
537            }
538
539            let created = self.create_field_index(
540                self.version(),
541                key,
542                Some(schema),
543                &HardwareCounterCell::disposable(), // This function is only used in Segment::load which is unmeasured.
544            )?;
545            if !created {
546                log::warn!("Failed to create payload index for {key} in segment");
547            }
548        }
549
550        // Do not delete extra payload indices, because collection-level information about
551        // the payload indices might be incomplete due to migrations from older versions.
552
553        Ok(())
554    }

556    /// Check data consistency of the segment on its own
557    /// - internal id without external id
558    /// - external id without internal
559    /// - internal id without version
560    /// - internal id without vector
561    ///
562    /// A shard can still be consistent with an inconsistent segment as points are merged based on their version.
563    ///
564    /// Returns an error if any inconsistency is found
565    pub fn check_data_consistency(&self) -> OperationResult<()> {
566        let id_tracker = self.id_tracker.borrow();
567
568        // dangling internal ids
569        let mut has_dangling_internal_ids = false;
570        for internal_id in id_tracker.iter_internal() {
571            if id_tracker.external_id(internal_id).is_none() {
572                log::error!("Internal id {internal_id} without external id");
573                has_dangling_internal_ids = true
574            }
575        }
576
577        // dangling external ids
578        let mut has_dangling_external_ids = false;
579        for external_id in id_tracker.iter_external() {
580            if id_tracker.internal_id(external_id).is_none() {
581                log::error!("External id {external_id} without internal id");
582                has_dangling_external_ids = true;
583            }
584        }
585
586        // checking internal id without version
587        let mut has_internal_ids_without_version = false;
588        for internal_id in id_tracker.iter_internal() {
589            if id_tracker.internal_version(internal_id).is_none() {
590                log::error!("Internal id {internal_id} without version");
591                has_internal_ids_without_version = true;
592            }
593        }
594
595        // check that non deleted points exist in vector storage
596        let mut has_internal_ids_without_vector = false;
597        for internal_id in id_tracker.iter_internal() {
598            for (vector_name, vector_data) in &self.vector_data {
599                let vector_storage = vector_data.vector_storage.borrow();
600                let is_vector_deleted_storage = vector_storage.is_deleted_vector(internal_id);
601                let is_vector_deleted_tracker = id_tracker.is_deleted_point(internal_id);
602                let vector_stored = vector_storage.get_vector_opt::<Random>(internal_id);
603                if !is_vector_deleted_storage
604                    && !is_vector_deleted_tracker
605                    && vector_stored.is_none()
606                {
607                    let point_id = id_tracker.external_id(internal_id);
608                    let point_version = id_tracker.internal_version(internal_id);
609                    // ignoring initial version because the WAL replay can resurrect un-flushed points by assigning them a new initial version
610                    // those points will be deleted by the next deduplication process
611                    if point_version != Some(0) {
612                        log::error!(
613                            "Vector storage '{vector_name}' is missing point {point_id:?} point_offset: {internal_id} version: {point_version:?}",
614                        );
615                        has_internal_ids_without_vector = true;
616                    }
617                }
618            }
619        }
620
621        let is_inconsistent = has_dangling_internal_ids
622            || has_dangling_external_ids
623            || has_internal_ids_without_version
624            || has_internal_ids_without_vector;
625
626        if is_inconsistent {
627            Err(OperationError::service_error(
628                "Inconsistent segment data detected",
629            ))
630        } else {
631            Ok(())
632        }
633    }

656    /// Returns the (estimated) amount of deferred points.
657    ///
658    /// This value is an estimation because it does not account for deferred points
659    /// that have been deleted before becoming visible.
660    pub fn deferred_point_count_estimated(&self) -> usize {
661        match self.deferred_internal_id {
662            Some(internal_id) => {
663                let id_tracker = self.id_tracker.borrow();
664                let max_id = id_tracker.total_point_count();
665                max_id.saturating_sub(internal_id as usize)
666            }
667            None => 0,
668        }
669    }
670
671    /// Returns the amount of points that are not deferred.
672    pub fn non_deferred_point_count_estimated(&self) -> usize {
673        self.id_tracker
674            .borrow()
675            .available_point_count()
676            .saturating_sub(self.deferred_point_count_estimated())
677    }
678}

680fn restore_snapshot_in_place(snapshot_path: &Path) -> OperationResult<()> {
681    let segments_dir = snapshot_path
682        .parent()
683        .ok_or_else(|| OperationError::service_error("Cannot extract parent path"))?;
684
685    let file_name = snapshot_path
686        .file_name()
687        .and_then(|name| name.to_str())
688        .ok_or_else(|| {
689            OperationError::service_error("Cannot extract segment ID from snapshot path")
690        })?;
691
692    let meta = fs::metadata(snapshot_path)?;
693    let (segment_id, is_tar) = match file_name.split_once('.') {
694        Some((segment_id, "tar")) if meta.is_file() => (segment_id, true),
695        None if meta.is_dir() => (file_name, false),
696        _ => {
697            return Err(OperationError::service_error(
698                "Invalid snapshot path, expected either a directory or a .tar file",
699            ));
700        }
701    };
702
703    if !is_tar {
704        log::debug!(
705            "Extracting segment {} from {:?} snapshot",
706            segment_id,
707            SnapshotFormat::Streamable
708        );
709        unpack_snapshot(snapshot_path)?;
710    } else {
711        let segment_path = segments_dir.join(segment_id);
712        tar_unpack_file(snapshot_path, &segment_path)?;
713
714        let inner_path = segment_path.join(SNAPSHOT_PATH);
715        if inner_path.is_dir() {
716            log::debug!(
717                "Extracting segment {} from {:?} snapshot",
718                segment_id,
719                SnapshotFormat::Regular
720            );
721            unpack_snapshot(&inner_path)?;
722            utils::fs::move_all(&inner_path, &segment_path)?;
723            fs::remove_dir(&inner_path)?;
724        } else {
725            log::debug!(
726                "Extracting segment {} from {:?} snapshot",
727                segment_id,
728                SnapshotFormat::Ancient
729            );
730            // Do nothing, this format is just a plain archive.
731        }
732
733        fs::remove_file(snapshot_path)?;
734    }
735
736    Ok(())
737}
738
739fn unpack_snapshot(segment_path: &Path) -> OperationResult<()> {
740    #[cfg(feature = "rocksdb")]
741    {
742        use super::{DB_BACKUP_PATH, PAYLOAD_DB_BACKUP_PATH};
743        use crate::segment::index::struct_payload_index::StructPayloadIndex;
744
745        let db_backup_path = segment_path.join(DB_BACKUP_PATH);
746        if db_backup_path.is_dir() {
747            crate::segment::rocksdb_backup::restore(&db_backup_path, segment_path)?;
748            fs::remove_dir_all(&db_backup_path)?;
749        }
750
751        let payload_index_db_backup = segment_path.join(PAYLOAD_DB_BACKUP_PATH);
752        if payload_index_db_backup.is_dir() {
753            StructPayloadIndex::restore_database_snapshot(&payload_index_db_backup, segment_path)?;
754            fs::remove_dir_all(&payload_index_db_backup)?;
755        }
756    }
757
758    let files_path = segment_path.join(SNAPSHOT_FILES_PATH);
759    utils::fs::move_all(&files_path, segment_path)?;
760    fs::remove_dir(&files_path)?;
761
762    Ok(())
763}

61impl<T: Encodable + Numericable> Iterator for NumericIndexPairsIterator<'_, T> {
62    type Item = Point<T>;
63
64    fn next(&mut self) -> Option<Self::Item> {
65        while self.start_index < self.end_index {
66            let key = self.pairs[self.start_index].clone();
67            let deleted = self.deleted.get(key.idx as usize).unwrap_or(true);
68            self.start_index += 1;
69            if deleted {
70                continue;
71            }
72            return Some(key);
73        }
74        None
75    }
76}
77
78impl<T: Encodable + Numericable> DoubleEndedIterator for NumericIndexPairsIterator<'_, T> {
79    fn next_back(&mut self) -> Option<Self::Item> {
80        while self.start_index < self.end_index {
81            let key = self.pairs[self.end_index - 1].clone();
82            let deleted = self.deleted.get(key.idx as usize).unwrap_or(true);
83            self.end_index -= 1;
84            if deleted {
85                continue;
86            }
87            return Some(key);
88        }
89        None
90    }
91}

93impl<T: Encodable + Numericable + Default + StoredValue> MmapNumericIndex<T> {
94    pub fn build(
95        in_memory_index: InMemoryNumericIndex<T>,
96        path: &Path,
97        is_on_disk: bool,
98    ) -> OperationResult<Self> {
99        fs::create_dir_all(path)?;
100
101        let pairs_path = path.join(PAIRS_PATH);
102        let deleted_path = path.join(DELETED_PATH);
103        let config_path = path.join(CONFIG_PATH);
104
105        atomic_save_json(
106            &config_path,
107            &MmapNumericIndexConfig {
108                max_values_per_point: in_memory_index.max_values_per_point,
109            },
110        )?;
111
112        in_memory_index.histogram.save(path)?;
113
114        StoredPointToValues::<T, MmapUniversal<u8>>::from_iter(
115            path,
116            in_memory_index
117                .point_to_values
118                .iter()
119                .enumerate()
120                .map(|(idx, values)| (idx as PointOffsetType, values.iter().map(|v| v.borrow()))),
121        )?;
122
123        {
124            let pairs_file = create_and_ensure_length(
125                &pairs_path,
126                in_memory_index.map.len() * std::mem::size_of::<Point<T>>(),
127            )?;
128            let pairs_mmap = unsafe { MmapMut::map_mut(&pairs_file)? };
129            let mut pairs = unsafe { MmapSlice::<Point<T>>::try_from(pairs_mmap)? };
130            for (src, dst) in in_memory_index.map.iter().zip(pairs.iter_mut()) {
131                *dst = src.clone();
132            }
133        }
134
135        {
136            const BITS_IN_BYTE: usize = 8;
137            let deleted_flags_count = in_memory_index.point_to_values.len();
138            let deleted_file = create_and_ensure_length(
139                &deleted_path,
140                BITS_IN_BYTE
141                    * BITS_IN_BYTE
142                    * deleted_flags_count.div_ceil(BITS_IN_BYTE * BITS_IN_BYTE),
143            )?;
144            let mut deleted_mmap = unsafe { MmapMut::map_mut(&deleted_file)? };
145            deleted_mmap.fill(0);
146            let mut deleted_bitflags = MmapBitSlice::from(deleted_mmap, 0);
147            for (idx, values) in in_memory_index.point_to_values.iter().enumerate() {
148                if values.is_empty() {
149                    deleted_bitflags.set(idx, true);
150                }
151            }
152        }
153
154        Self::open(path, is_on_disk)?.ok_or_else(|| {
155            OperationError::service_error("Failed to open MmapNumericIndex after building it")
156        })
157    }

159    /// Open and load mmap numeric index from the given path
160    pub fn open(path: &Path, is_on_disk: bool) -> OperationResult<Option<Self>> {
161        let pairs_path = path.join(PAIRS_PATH);
162        let deleted_path = path.join(DELETED_PATH);
163        let config_path = path.join(CONFIG_PATH);
164
165        // If config doesn't exist, assume the index doesn't exist on disk
166        if !config_path.is_file() {
167            return Ok(None);
168        }
169
170        let histogram = Histogram::<T>::load(path)?;
171        let config: MmapNumericIndexConfig = read_json(&config_path)?;
172        let deleted = mmap::open_write_mmap(&deleted_path, AdviceSetting::Global, false)?;
173        let deleted = MmapBitSlice::from(deleted, 0);
174        let deleted_count = deleted.count_ones();
175        let do_populate = !is_on_disk;
176        let map = unsafe {
177            MmapSlice::try_from(mmap::open_write_mmap(
178                &pairs_path,
179                AdviceSetting::Global,
180                do_populate,
181            )?)?
182        };
183        let point_to_values = StoredPointToValues::open(path, do_populate)?;
184
185        Ok(Some(Self {
186            path: path.to_path_buf(),
187            storage: Storage {
188                pairs: map,
189                deleted: MmapBitSliceBufferedUpdateWrapper::new(deleted),
190                point_to_values,
191            },
192            histogram,
193            deleted_count,
194            max_values_per_point: config.max_values_per_point,
195            is_on_disk,
196        }))
197    }

199    pub fn wipe(self) -> OperationResult<()> {
200        let files = self.files();
201        let path = self.path.clone();
202        // drop mmap handles before deleting files
203        drop(self);
204        for file in files {
205            fs::remove_file(file)?;
206        }
207        let _ = fs::remove_dir(path);
208        Ok(())
209    }

279    pub(super) fn values_range<'a>(
280        &'a self,
281        start_bound: Bound<Point<T>>,
282        end_bound: Bound<Point<T>>,
283        hw_counter: &'a HardwareCounterCell,
284    ) -> impl Iterator<Item = PointOffsetType> + 'a {

332    // get iterator
333    fn values_range_iterator(
334        &self,
335        start_bound: Bound<Point<T>>,
336        end_bound: Bound<Point<T>>,
337    ) -> NumericIndexPairsIterator<'_, T> {
338        let start_index = match start_bound {
339            Bound::Included(bound) => self
340                .storage
341                .pairs
342                .binary_search(&bound)
343                .unwrap_or_else(|idx| idx),
344            Bound::Excluded(bound) => match self.storage.pairs.binary_search(&bound) {
345                Ok(idx) => idx + 1,
346                Err(idx) => idx,
347            },
348            Bound::Unbounded => 0,
349        };
350
351        if start_index >= self.storage.pairs.len() {
352            return NumericIndexPairsIterator {
353                pairs: &self.storage.pairs,
354                deleted: &self.storage.deleted,
355                start_index: self.storage.pairs.len(),
356                end_index: self.storage.pairs.len(),
357            };
358        }
359
360        let end_index = match end_bound {
361            Bound::Included(bound) => match self.storage.pairs[start_index..].binary_search(&bound)
362            {
363                Ok(idx) => idx + 1 + start_index,
364                Err(idx) => idx + start_index,
365            },
366            Bound::Excluded(bound) => {
367                let end_bound = self.storage.pairs[start_index..].binary_search(&bound);
368                end_bound.unwrap_or_else(|idx| idx) + start_index
369            }
370            Bound::Unbounded => self.storage.pairs.len(),
371        };
372
373        NumericIndexPairsIterator {
374            pairs: &self.storage.pairs,
375            deleted: &self.storage.deleted,
376            start_index,
377            end_index,
378        }
379    }

30/// Default options for Gridstore storage
31const fn default_gridstore_options<T: Sized>() -> StorageOptions {
32    let block_size = size_of::<T>();
33    StorageOptions {
34        // Size of numeric values in index
35        block_size_bytes: Some(block_size),
36        // Compressing numeric values is unreasonable
37        compression: Some(crate::gridstore::config::Compression::None),
38        // Scale page size down with block size, prevents overhead of first page when there's (almost) no values
39        page_size_bytes: Some(block_size * 8192 * 32), // 4 to 8 MiB = block_size * region_blocks * regions,
40        region_size_blocks: None,
41    }
42}

71impl<T: Encodable + Numericable> Default for InMemoryNumericIndex<T> {
72    fn default() -> Self {
73        Self {
74            map: BTreeSet::new(),
75            histogram: Histogram::new(HISTOGRAM_MAX_BUCKET_SIZE, HISTOGRAM_PRECISION),
76            points_count: 0,
77            max_values_per_point: 0,
78            point_to_values: Default::default(),

83impl<T: Encodable + Numericable + Default> FromIterator<(PointOffsetType, T)>
84    for InMemoryNumericIndex<T>
85{
86    fn from_iter<I: IntoIterator<Item = (PointOffsetType, T)>>(iter: I) -> Self {
87        let mut index = InMemoryNumericIndex::default();
88        for pair in iter {
89            let (idx, value) = pair;
90
91            if index.point_to_values.len() <= idx as usize {
92                index
93                    .point_to_values
94                    .resize_with(idx as usize + 1, Vec::new)
95            }
96
97            index.point_to_values[idx as usize].push(value);
98
99            let key = Point::new(value, idx);
100            InMemoryNumericIndex::add_to_map(&mut index.map, &mut index.histogram, key);
101        }
102        for values in &index.point_to_values {
103            if !values.is_empty() {
104                index.points_count += 1;
105                index.max_values_per_point = index.max_values_per_point.max(values.len());
106            }
107        }
108        index
109    }
110}

112impl<T: Encodable + Numericable + Default + StoredValue> InMemoryNumericIndex<T> {
113    /// Construct in-memroy index from given mmap index
114    ///
115    /// # Warning
116    ///
117    /// Expensive because this reads the full mmap index.
118    pub(super) fn from_mmap(mmap_index: &MmapNumericIndex<T>) -> Self {
119        let point_count = mmap_index.storage.point_to_values.len();
120
121        (0..point_count as PointOffsetType)
122            .filter_map(|idx| mmap_index.get_values(idx).map(|values| (idx, values)))
123            .flat_map(|(idx, values)| values.into_iter().map(move |value| (idx, value)))
124            .collect::<InMemoryNumericIndex<T>>()
125    }
126}

172    pub fn add_many_to_list(&mut self, idx: PointOffsetType, values: Vec<T>) {
173        if self.point_to_values.len() <= idx as usize {
174            self.point_to_values.resize_with(idx as usize + 1, Vec::new)
175        }
176        for value in &values {
177            let key = Point::new(*value, idx);
178            Self::add_to_map(&mut self.map, &mut self.histogram, key);
179        }
180        if !values.is_empty() {
181            self.points_count += 1;
182            self.max_values_per_point = self.max_values_per_point.max(values.len());
183        }
184        self.point_to_values[idx as usize] = values;
185    }
186
187    pub fn remove_point(&mut self, idx: PointOffsetType) {
188        if let Some(values) = self.point_to_values.get_mut(idx as usize) {
189            if !values.is_empty() {
190                self.points_count = self.points_count.saturating_sub(1);
191            }
192            for value in values.iter() {
193                let key = Point::new(*value, idx);
194                Self::remove_from_map(&mut self.map, &mut self.histogram, key);
195            }
196            *values = Default::default();
197        }
198    }
199
200    fn add_to_map(map: &mut BTreeSet<Point<T>>, histogram: &mut Histogram<T>, key: Point<T>) {
201        let was_added = map.insert(key.clone());
202        // Histogram works with unique values (idx + value) only, so we need to
203        // make sure that we don't add the same value twice.
204        // key is a combination of value + idx, so we can use it to ensure than the pair is unique
205        if was_added {
206            histogram.insert(
207                key,
208                |x| Self::get_histogram_left_neighbor(map, x.clone()),
209                |x| Self::get_histogram_right_neighbor(map, x.clone()),
210            );
211        }
212    }
213
214    fn remove_from_map(map: &mut BTreeSet<Point<T>>, histogram: &mut Histogram<T>, key: Point<T>) {
215        let was_removed = map.remove(&key);
216        if was_removed {
217            histogram.remove(
218                &key,
219                |x| Self::get_histogram_left_neighbor(map, x.clone()),
220                |x| Self::get_histogram_right_neighbor(map, x.clone()),
221            );
222        }
223    }
224
225    fn get_histogram_left_neighbor(map: &BTreeSet<Point<T>>, key: Point<T>) -> Option<Point<T>> {
226        map.range((Unbounded, Excluded(key))).next_back().cloned()
227    }
228
229    fn get_histogram_right_neighbor(map: &BTreeSet<Point<T>>, key: Point<T>) -> Option<Point<T>> {
230        map.range((Excluded(key), Unbounded)).next().cloned()
231    }
232
233    pub fn get_histogram(&self) -> &Histogram<T> {
234        &self.histogram
235    }
236
237    pub fn get_points_count(&self) -> usize {
238        self.points_count
239    }
240
241    pub fn get_max_values_per_point(&self) -> usize {
242        self.max_values_per_point
243    }
244}

250    /// Open and load mutable numeric index from RocksDB storage
251    #[cfg(feature = "rocksdb")]
252    pub fn open_rocksdb(
253        db: Arc<RwLock<DB>>,
254        field: &str,
255        create_if_missing: bool,
256    ) -> OperationResult<Option<Self>> {
257        let store_cf_name = super::numeric_index_storage_cf_name(field);
258        let db_wrapper = DatabaseColumnScheduledDeleteWrapper::new(DatabaseColumnWrapper::new(
259            db,
260            &store_cf_name,
261        ));
262        Self::open_rocksdb_db_wrapper(db_wrapper, create_if_missing)
263    }
264
265    #[cfg(feature = "rocksdb")]
266    pub fn open_rocksdb_db_wrapper(
267        db_wrapper: DatabaseColumnScheduledDeleteWrapper,
268        create_if_missing: bool,
269    ) -> OperationResult<Option<Self>> {
270        if !db_wrapper.has_column_family()? {
271            if create_if_missing {
272                db_wrapper.recreate_column_family()?;
273            } else {
274                // Column family doesn't exist, cannot load
275                return Ok(None);
276            }
277        };
278
279        // Load in-memory index from RocksDB
280        let in_memory_index = db_wrapper
281            .lock_db()
282            .iter()?
283            .map(|(key, value)| {
284                let value_idx =
285                    u32::from_be_bytes(value.as_ref().try_into().map_err(|_| {
286                        OperationError::service_error("incorrect numeric index value")
287                    })?);
288                let (idx, value) = T::decode_key(&key);
289                if idx != value_idx {
290                    return Err(OperationError::service_error(
291                        "incorrect numeric index key-value pair",
292                    ));
293                }
294                Ok((idx, value))
295            })
296            .collect::<Result<InMemoryNumericIndex<_>, OperationError>>()?;
297
298        Ok(Some(Self {
299            storage: Storage::RocksDb(db_wrapper),
300            in_memory_index,
301        }))
302    }

304    /// Open and load mutable numeric index from Gridstore storage
305    ///
306    /// The `create_if_missing` parameter indicates whether to create a new Gridstore if it does
307    /// not exist. If false and files don't exist, this will return `None` to indicate nothing
308    /// could be loaded.
309    pub fn open_gridstore(path: PathBuf, create_if_missing: bool) -> OperationResult<Option<Self>> {
310        let store = if create_if_missing {
311            let options = default_gridstore_options::<T>();
312            Gridstore::open_or_create(path, options).map_err(|err| {
313                OperationError::service_error(format!(
314                    "failed to open mutable numeric index on gridstore: {err}"
315                ))
316            })?
317        } else if path.exists() {
318            Gridstore::open(path).map_err(|err| {
319                OperationError::service_error(format!(
320                    "failed to open mutable numeric index on gridstore: {err}"
321                ))
322            })?
323        } else {
324            // Files don't exist, cannot load
325            return Ok(None);
326        };
327
328        // Load in-memory index from Gridstore
329        let mut in_memory_index = InMemoryNumericIndex::default();
330        let hw_counter = HardwareCounterCell::disposable();
331        let hw_counter_ref = hw_counter.ref_payload_index_io_write_counter();
332        store
333            .iter::<_, GridstoreError>(
334                |idx, values: Vec<T>| {
335                    in_memory_index.add_many_to_list(idx, values);
336                    Ok(true)
337                },
338                hw_counter_ref,
339            )
340            // unwrap safety: never returns an error
341            .unwrap();
342
343        Ok(Some(Self {
344            storage: Storage::Gridstore(store),
345            in_memory_index,
346        }))
347    }

431    pub fn add_many_to_list(
432        &mut self,
433        idx: PointOffsetType,
434        values: Vec<T>,
435        hw_counter: &HardwareCounterCell,
436    ) -> OperationResult<()> {
437        // Update persisted storage
438        match &mut self.storage {
439            #[cfg(feature = "rocksdb")]
440            Storage::RocksDb(db_wrapper) => {
441                let mut hw_cell_wb = hw_counter
442                    .payload_index_io_write_counter()
443                    .write_back_counter();
444                for value in &values {
445                    let key = value.encode_key(idx);
446                    db_wrapper.put(&key, idx.to_be_bytes())?;
447                    hw_cell_wb.incr_delta(size_of_val(&key) + size_of_val(&idx));
448                }
449            }
450            // We cannot store empty value, then delete instead
451            Storage::Gridstore(store) if values.is_empty() => {
452                store.delete_value(idx)?;
453            }
454            Storage::Gridstore(store) => {
455                let hw_counter_ref = hw_counter.ref_payload_index_io_write_counter();
456                store
457                    .put_value(idx, &values, hw_counter_ref)
458                    .map_err(|err| {
459                        OperationError::service_error(format!(
460                            "failed to put value in mutable numeric index gridstore: {err}"
461                        ))
462                    })?;
463            }
464        }
465
466        self.in_memory_index.add_many_to_list(idx, values);
467        Ok(())
468    }

470    pub fn remove_point(&mut self, idx: PointOffsetType) -> OperationResult<()> {
471        // Update persisted storage
472        match &mut self.storage {
473            #[cfg(feature = "rocksdb")]
474            Storage::RocksDb(db_wrapper) => {
475                self.in_memory_index
476                    .get_values(idx)
477                    .map(|mut values| {
478                        values.try_for_each(|value| {
479                            let key = value.encode_key(idx);
480                            db_wrapper.remove(key)
481                        })
482                    })
483                    .transpose()?;
484            }
485            Storage::Gridstore(store) => {
486                store.delete_value(idx)?;
487            }
488        }
489
490        self.in_memory_index.remove_point(idx);
491        Ok(())
492    }

58#[derive(Debug, Clone, PartialEq)]
59pub struct CardinalityEstimation {
60    /// Conditions that could be used to make a primary point selection.
61    pub primary_clauses: Vec<PrimaryCondition>,
62    /// Minimal possible matched points in best case for a query
63    pub min: usize,
64    /// Expected number of matched points for a query, assuming even random distribution if stored data
65    pub exp: usize,
66    /// The largest possible number of matched points in a worst case for a query
67    pub max: usize,
68}
69
70impl CardinalityEstimation {
71    pub const fn exact(count: usize) -> Self {
72        CardinalityEstimation {
73            primary_clauses: vec![],
74            min: count,
75            exp: count,
76            max: count,
77        }
78    }
79
80    /// Generate estimation for unknown filter
81    pub const fn unknown(total: usize) -> Self {
82        CardinalityEstimation {
83            primary_clauses: vec![],
84            min: 0,
85            exp: total / 2,
86            max: total,
87        }
88    }
89
90    /// Push a primary clause to the estimation
91    pub fn with_primary_clause(mut self, clause: PrimaryCondition) -> Self {
92        self.primary_clauses.push(clause);
93        self
94    }
95
96    #[cfg(test)]
97    pub const fn equals_min_exp_max(&self, other: &Self) -> bool {
98        self.min == other.min && self.exp == other.exp && self.max == other.max
99    }
100
101    /// Checks that the given condition is a primary condition of the estimation.
102    pub fn is_primary(&self, condition: &Condition) -> bool {
103        self.primary_clauses
104            .iter()
105            .any(|primary_condition| match primary_condition {
106                PrimaryCondition::Condition(primary_field_condition) => match condition {
107                    Condition::Field(field_condition) => {
108                        primary_field_condition.as_ref() == field_condition
109                    }
110                    _ => false,
111                },
112                PrimaryCondition::Ids(ids) => match condition {
113                    Condition::HasId(has_id) => ids.point_ids.deref() == has_id.has_id.deref(),
114                    _ => false,
115                },
116                PrimaryCondition::HasVector(has_vector) => match condition {
117                    Condition::HasVector(vector_condition) => {
118                        has_vector == &vector_condition.has_vector
119                    }
120                    _ => false,
121                },
122            })
123    }
124}

126pub trait EstimationMerge: Iterator<Item = CardinalityEstimation> {
127    fn merge_independent(self) -> CardinalityEstimation
128    where
129        Self: Sized,
130    {
131        self.fold(CardinalityEstimation::exact(0), |acc, x| {
132            CardinalityEstimation {
133                primary_clauses: vec![],
134                min: acc.min + x.min,
135                exp: acc.exp + x.exp,
136                max: acc.max + x.max,
137            }
138        })
139    }
140}

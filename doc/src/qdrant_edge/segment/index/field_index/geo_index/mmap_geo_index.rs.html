91    pub fn build(
92        dynamic_index: InMemoryGeoMapIndex,
93        path: &Path,
94        is_on_disk: bool,
95    ) -> OperationResult<Self> {
96        fs::create_dir_all(path)?;
97
98        let deleted_path = path.join(DELETED_PATH);
99        let stats_path = path.join(STATS_PATH);
100        let counts_per_hash_path = path.join(COUNTS_PER_HASH);
101        let points_map_path = path.join(POINTS_MAP);
102        let points_map_ids_path = path.join(POINTS_MAP_IDS);
103
104        // Create the point-to-value mapping and persist in the mmap file
105        StoredPointToValues::<GeoPoint, MmapUniversal<u8>>::from_iter(
106            path,
107            dynamic_index
108                .point_to_values
109                .iter()
110                .enumerate()
111                .map(|(idx, values)| (idx as PointOffsetType, values.iter())),
112        )?;
113
114        {
115            let points_map_file = create_and_ensure_length(
116                &points_map_path,
117                dynamic_index.points_map.len() * std::mem::size_of::<PointKeyValue>(),
118            )?;
119            let points_map_file = unsafe { MmapMut::map_mut(&points_map_file)? };
120            let mut points_map = unsafe { MmapSlice::<PointKeyValue>::try_from(points_map_file)? };
121
122            let points_map_ids_file = create_and_ensure_length(
123                &points_map_ids_path,
124                dynamic_index
125                    .points_map
126                    .values()
127                    .map(|v| v.len())
128                    .sum::<usize>()
129                    * std::mem::size_of::<PointOffsetType>(),
130            )?;
131            let points_map_ids_file = unsafe { MmapMut::map_mut(&points_map_ids_file)? };
132            let mut points_map_ids =
133                unsafe { MmapSlice::<PointOffsetType>::try_from(points_map_ids_file)? };
134
135            let mut ids_offset = 0;
136            for (i, (hash, ids)) in dynamic_index.points_map.iter().enumerate() {
137                points_map[i].hash = *hash;
138                points_map[i].ids_start = ids_offset as u32;
139                points_map[i].ids_end = (ids_offset + ids.len()) as u32;
140                points_map_ids[ids_offset..ids_offset + ids.len()].copy_from_slice(
141                    &ids.iter()
142                        .map(|v| *v as PointOffsetType)
143                        .collect::<Vec<_>>(),
144                );
145                ids_offset += ids.len();
146            }
147        }
148
149        {
150            let counts_per_hash_file = create_and_ensure_length(
151                &counts_per_hash_path,
152                std::cmp::min(
153                    dynamic_index.points_per_hash.len(),
154                    dynamic_index.values_per_hash.len(),
155                ) * std::mem::size_of::<Counts>(),
156            )?;
157            let counts_per_hash_file = unsafe { MmapMut::map_mut(&counts_per_hash_file)? };
158            let mut counts_per_hash =
159                unsafe { MmapSlice::<Counts>::try_from(counts_per_hash_file)? };
160            for ((hash, points), dst) in dynamic_index
161                .points_per_hash
162                .iter()
163                .zip(counts_per_hash.iter_mut())
164            {
165                if let Some(values) = dynamic_index.values_per_hash.get(hash) {
166                    dst.hash = *hash;
167                    dst.points = *points as u32;
168                    dst.values = *values as u32;
169                }
170            }
171        }
172
173        {
174            let deleted_flags_count = dynamic_index.point_to_values.len();
175            let deleted_file = create_and_ensure_length(
176                &deleted_path,
177                deleted_flags_count
178                    .div_ceil(u8::BITS as usize)
179                    .next_multiple_of(std::mem::size_of::<usize>()),
180            )?;
181            let mut deleted_mmap = unsafe { MmapMut::map_mut(&deleted_file)? };
182            deleted_mmap.fill(0);
183            let mut deleted_bitflags = MmapBitSlice::from(deleted_mmap, 0);
184            for (idx, values) in dynamic_index.point_to_values.iter().enumerate() {
185                if values.is_empty() {
186                    deleted_bitflags.set(idx, true);
187                }
188            }
189        }
190
191        atomic_save_json(
192            &stats_path,
193            &MmapGeoMapIndexStat {
194                points_values_count: dynamic_index.points_values_count,
195                max_values_per_point: dynamic_index.max_values_per_point,
196            },
197        )?;
198
199        Self::open(path, is_on_disk)?.ok_or_else(|| {
200            OperationError::service_error("Failed to open MmapGeoMapIndex after building it")
201        })
202    }

204    pub fn open(path: &Path, is_on_disk: bool) -> OperationResult<Option<Self>> {
205        let deleted_path = path.join(DELETED_PATH);
206        let stats_path = path.join(STATS_PATH);
207        let counts_per_hash_path = path.join(COUNTS_PER_HASH);
208        let points_map_path = path.join(POINTS_MAP);
209        let points_map_ids_path = path.join(POINTS_MAP_IDS);
210
211        // If stats file doesn't exist, assume the index doesn't exist on disk
212        if !stats_path.is_file() {
213            return Ok(None);
214        }
215
216        let populate = !is_on_disk;
217        let stats: MmapGeoMapIndexStat = read_json(&stats_path)?;
218        let counts_per_hash = unsafe {
219            MmapSlice::try_from(open_write_mmap(
220                &counts_per_hash_path,
221                AdviceSetting::Global,
222                populate,
223            )?)?
224        };
225        let points_map = unsafe {
226            MmapSlice::try_from(open_write_mmap(
227                &points_map_path,
228                AdviceSetting::Global,
229                populate,
230            )?)?
231        };
232        let points_map_ids = unsafe {
233            MmapSlice::try_from(open_write_mmap(
234                &points_map_ids_path,
235                AdviceSetting::Global,
236                populate,
237            )?)?
238        };
239        let point_to_values = StoredPointToValues::open(path, true)?;
240
241        let deleted = open_write_mmap(&deleted_path, AdviceSetting::Global, populate)?;
242        let deleted = MmapBitSlice::from(deleted, 0);
243        let deleted_count = deleted.count_ones();
244
245        Ok(Some(Self {
246            path: path.to_owned(),
247            storage: Storage {
248                counts_per_hash,
249                points_map,
250                points_map_ids,
251                point_to_values,
252                deleted: MmapBitSliceBufferedUpdateWrapper::new(deleted),
253            },
254            deleted_count,
255            points_values_count: stats.points_values_count,
256            max_values_per_point: stats.max_values_per_point,
257            is_on_disk,
258        }))
259    }

261    pub fn check_values_any(
262        &self,
263        idx: PointOffsetType,
264        hw_counter: &HardwareCounterCell,
265        check_fn: impl Fn(&GeoPoint) -> bool,
266    ) -> bool {
267        let hw_counter = self.make_conditioned_counter(hw_counter);
268        self.storage
269            .deleted
270            .get(idx as usize)
271            .filter(|b| !b)
272            .map(|_| {
273                self.storage
274                    .point_to_values
275                    .check_values_any(idx, |v| check_fn(v), &hw_counter)
276            })
277            .map(|r| r.unwrap_or(false))
278            .unwrap_or(false)
279    }

306    pub fn points_of_hash(&self, hash: &GeoHash, hw_counter: &HardwareCounterCell) -> usize {
307        let hw_counter = self.make_conditioned_counter(hw_counter);
308
309        hw_counter
310            .payload_index_io_read_counter()
311            // Simulate binary search complexity as IO read estimation
312            .incr_delta(
313                (self.storage.counts_per_hash.len() as f32).log2().ceil() as usize
314                    * size_of::<Counts>(),
315            );
316
317        if let Ok(index) = self
318            .storage
319            .counts_per_hash
320            .binary_search_by(|x| x.hash.cmp(hash))
321        {
322            self.storage.counts_per_hash[index].points as usize
323        } else {
324            0
325        }
326    }

350    pub fn wipe(self) -> OperationResult<()> {
351        let files = self.files();
352        let path = self.path.clone();
353        // drop mmap handles before deleting files
354        drop(self);
355        for file in files {
356            fs::remove_file(file)?;
357        }
358        let _ = fs::remove_dir(path);
359        Ok(())
360    }

389    pub fn remove_point(&mut self, idx: PointOffsetType) {
390        let idx = idx as usize;
391        if let Some(deleted) = self.storage.deleted.get(idx)
392            && !deleted
393        {
394            self.storage.deleted.set(idx, true);
395            self.deleted_count += 1;
396        }
397    }

399    /// Returns an iterator over all point IDs which have the `geohash` prefix.
400    /// Note. Point ID may be repeated multiple times in the iterator.
401    pub fn stored_sub_regions<'a>(
402        &'a self,
403        geohash: GeoHash,
404    ) -> impl Iterator<Item = PointOffsetType> + 'a {
405        let start_index = self
406            .storage
407            .points_map
408            .binary_search_by(|point_key_value| point_key_value.hash.cmp(&geohash))
409            .unwrap_or_else(|index| index);
410        self.storage.points_map[start_index..]
411            .iter()
412            .take_while(move |point_key_value| point_key_value.hash.starts_with(geohash))
413            .filter_map(|point_key_value| {
414                Some(
415                    self.storage
416                        .points_map_ids
417                        .get(point_key_value.ids_start as usize..point_key_value.ids_end as usize)?
418                        .iter()
419                        .copied()
420                        .filter(|idx| !self.storage.deleted.get(*idx as usize).unwrap_or(true)),
421                )
422            })
423            .flatten()
424    }

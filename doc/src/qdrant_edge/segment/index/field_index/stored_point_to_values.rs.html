16const POINT_TO_VALUES_PATH: &str = "point_to_values.bin";
17const NOT_ENOUGH_BYTES_ERROR_MESSAGE: &str = "Not enough bytes to operate with slice in file `point_to_values.bin`. Is the storage corrupted?";
18const PADDING_SIZE: usize = 4096;

20/// Trait for values that can be stored in a file. It's used in `StoredPointToValues` to store values.
21pub trait StoredValue: ToOwned {
22    /// Get the size in bytes that the value will take when stored in a slice.
23    fn stored_size(value: &Self) -> usize;
24
25    /// Read one value from the beginning of the bytes slice.
26    fn read_from_prefix(bytes: &[u8]) -> Option<&Self>;
27
28    /// Write the value into the beginning of the bytes slice. The slice must have enough capacity for the value.
29    fn write_to_prefix(&self, bytes: &mut [u8]) -> Option<()>;
30}
31
32impl<T: bytemuck::Pod> StoredValue for T {
33    fn stored_size(_value: &Self) -> usize {
34        std::mem::size_of::<Self>()
35    }
36
37    fn read_from_prefix(bytes: &[u8]) -> Option<&Self> {
38        bytemuck::try_cast_slice(bytes).ok()?.first()
39    }
40
41    fn write_to_prefix(&self, bytes: &mut [u8]) -> Option<()> {
42        let value_bytes = bytemuck::bytes_of(self);
43        <[u8] as zerocopy::IntoBytes>::write_to_prefix(value_bytes, bytes).ok()
44    }
45}
46
47impl StoredValue for str {
48    fn stored_size(value: &str) -> usize {
49        value.len() + std::mem::size_of::<u32>()
50    }
51
52    fn read_from_prefix(bytes: &[u8]) -> Option<&Self> {
53        let (len, rest) = <u32 as zerocopy::FromBytes>::read_from_prefix(bytes).ok()?;
54        std::str::from_utf8(rest.get(..len as usize)?).ok()
55    }
56
57    fn write_to_prefix(&self, bytes: &mut [u8]) -> Option<()> {
58        <u32 as zerocopy::IntoBytes>::write_to_prefix(&(self.len() as u32), bytes).ok()?;
59        bytes
60            .get_mut(std::mem::size_of::<u32>()..std::mem::size_of::<u32>() + self.len())?
61            .copy_from_slice(self.as_bytes());
62        Some(())
63    }
64}

66/// Flattened memmapped points-to-values map
67/// It's an analogue of `Vec<Vec<N>>` but in memmapped file.
68/// This structure is immutable.
69/// It's used in mmap field indices like `MmapMapIndex`, `MmapNumericIndex`, etc to store points-to-values map.
70/// This structure is not generic to avoid boxing lifetimes for `&str` values.
71pub struct StoredPointToValues<T: StoredValue + ?Sized, S: UniversalRead<u8>> {
72    file_name: PathBuf,
73    store: S,
74    header: Header,
75    phantom: std::marker::PhantomData<T>,
76}
77
78/// Memory and IO overhead of accessing mmap index.
79pub const MMAP_PTV_ACCESS_OVERHEAD: usize = size_of::<MmapRange>();
80
81#[repr(C)]
82#[derive(Copy, Clone, Debug, Default, FromBytes, Immutable, IntoBytes, KnownLayout)]
83struct MmapRange {
84    start: u64,
85    count: u64,
86}
87
88#[repr(C)]
89#[derive(Copy, Clone, Debug, FromBytes, Immutable, IntoBytes, KnownLayout)]
90struct Header {
91    ranges_start: u64,
92    points_count: u64,
93}
94
95impl<T, S> StoredPointToValues<T, S>
96where
97    T: StoredValue + ?Sized,
98    S: UniversalRead<u8>,
99{
100    pub fn from_iter<'a>(
101        path: &Path,
102        iter: impl Iterator<Item = (PointOffsetType, impl Iterator<Item = &'a T>)> + Clone,
103    ) -> OperationResult<Self>
104    where
105        T: 'a,
106    {
107        // calculate file size
108        let mut points_count: usize = 0;
109        let mut values_size = 0;
110        for (point_id, values) in iter.clone() {
111            points_count = max(points_count, (point_id + 1) as usize);
112            values_size += values.map(|v| T::stored_size(v)).sum::<usize>();
113        }
114        let ranges_size = points_count * std::mem::size_of::<MmapRange>();
115        let file_size = PADDING_SIZE + ranges_size + values_size;
116
117        // create new file and mmap
118        let file_name = path.join(POINT_TO_VALUES_PATH);
119        create_and_ensure_length(&file_name, file_size)?;
120        let mut mmap = open_write_mmap(&file_name, AdviceSetting::Global, false)?;
121
122        // fill mmap file data
123        let header = Header {
124            ranges_start: PADDING_SIZE as u64,
125            points_count: points_count as u64,
126        };
127        header
128            .write_to_prefix(mmap.as_mut())
129            .map_err(|_| OperationError::service_error(NOT_ENOUGH_BYTES_ERROR_MESSAGE))?;
130
131        // counter for values offset
132        let mut point_values_offset = header.ranges_start as usize + ranges_size;
133        for (point_id, values) in iter {
134            let start = point_values_offset;
135            let mut values_count = 0;
136            for value in values {
137                values_count += 1;
138                let bytes = mmap
139                    .get_mut(point_values_offset..)
140                    .ok_or_else(|| OperationError::service_error(NOT_ENOUGH_BYTES_ERROR_MESSAGE))?;
141                value
142                    .write_to_prefix(bytes)
143                    .ok_or_else(|| OperationError::service_error(NOT_ENOUGH_BYTES_ERROR_MESSAGE))?;
144                point_values_offset += T::stored_size(value);
145            }
146
147            let range = MmapRange {
148                start: start as u64,
149                count: values_count as u64,
150            };
151            mmap.get_mut(
152                header.ranges_start as usize
153                    + point_id as usize * std::mem::size_of::<MmapRange>()..,
154            )
155            .and_then(|bytes| range.write_to_prefix(bytes).ok())
156            .ok_or_else(|| OperationError::service_error(NOT_ENOUGH_BYTES_ERROR_MESSAGE))?;
157        }
158
159        mmap.flush()?;
160        drop(mmap);
161
162        Self::open(path, true)
163    }

165    pub fn open(path: &Path, populate: bool) -> OperationResult<Self> {
166        let file_name = path.join(POINT_TO_VALUES_PATH);
167
168        let open_options = crate::common::universal_io::OpenOptions {
169            need_sequential: false,
170            disk_parallel: None,
171            populate: Some(populate),
172            advice: None,
173        };
174
175        let store = S::open(&file_name, open_options)?;
176
177        let header_bytes = store.read::<false>(ElementsRange {
178            start: 0,
179            length: std::mem::size_of::<Header>() as u64,
180        })?;
181
182        let (header, _) = Header::read_from_prefix(&header_bytes).map_err(|_| {
183            OperationError::InconsistentStorage {
184                description: NOT_ENOUGH_BYTES_ERROR_MESSAGE.to_owned(),
185            }
186        })?;
187
188        Ok(Self {
189            file_name,
190            store,
191            header,
192            phantom: std::marker::PhantomData,
193        })
194    }

205    pub fn check_values_any(
206        &self,
207        point_id: PointOffsetType,
208        check_fn: impl Fn(&T) -> bool,
209        hw_counter: &ConditionedCounter,
210    ) -> OperationResult<bool> {
211        let Some(mut values_iter) = self.values_iter(point_id, *hw_counter)? else {
212            return Ok(false);
213        };
214
215        while let Some(satisfy) = values_iter.map_next(&check_fn) {
216            if satisfy {
217                return Ok(true);
218            }
219        }
220
221        Ok(false)
222    }

224    pub fn values_iter(
225        &self,
226        point_id: PointOffsetType,
227        hw_counter: ConditionedCounter, // TODO: make it by reference
228    ) -> OperationResult<Option<ValuesIter<'_, T>>> {
229        let hw_cell = hw_counter.payload_index_io_read_counter();
230
231        // first, get range of values for point
232        let Some(bytes_range) = self.get_bytes_range(point_id)?.map(|range| {
233            let range = universal_io::ElementsRange {
234                start: range.start,
235                length: range.end - range.start,
236            };
237            // Measure IO overhead of `self.get_bytes_range()` and the length of the values
238            hw_cell.incr_delta(MMAP_PTV_ACCESS_OVERHEAD + range.length as usize);
239
240            range
241        }) else {
242            return Ok(None);
243        };
244
245        let bytes = self.store.read::<false>(bytes_range)?;
246        let count = self.get_values_count(point_id)?.unwrap_or(0);
247
248        let iter = ValuesIter::new(bytes, count);
249
250        Ok(Some(iter))
251    }
252
253    pub fn get_values_count(&self, point_id: PointOffsetType) -> OperationResult<Option<usize>> {
254        self.get_range(point_id)
255            .map_some(|range| range.count as usize)
256    }

258    pub fn len(&self) -> usize {
259        self.header.points_count as usize
260    }

265    }

267    fn get_range(&self, point_id: PointOffsetType) -> OperationResult<Option<MmapRange>> {
268        if point_id < self.header.points_count as PointOffsetType {
269            let range_offset = (self.header.ranges_start as usize)
270                + (point_id as usize) * std::mem::size_of::<MmapRange>();
271
272            let bytes = self.store.read::<false>(ElementsRange {
273                start: range_offset as u64,
274                length: std::mem::size_of::<MmapRange>() as u64,
275            })?;
276            Ok(MmapRange::read_from_prefix(&bytes)
277                .ok()
278                .map(|(range, _)| range))
279        } else {
280            Ok(None)
281        }
282    }
283
284    fn get_bytes_range(&self, point_id: PointOffsetType) -> OperationResult<Option<Range<u64>>> {
285        let Some(start) = self.get_range(point_id).map_some(|range| range.start)? else {
286            return Ok(None);
287        };
288
289        // Use next point's start as end offset for this one.
290        let end = match self.get_range(point_id + 1).map_some(|range| range.start)? {
291            Some(end) => end,
292            None => {
293                // if there is no next point, then use end of file
294                self.store.len()?
295            }
296        };
297
298        Ok(Some(start..end))
299    }

324pub struct ValuesIter<'a, T: ?Sized> {
325    bytes: Cow<'a, [u8]>,
326    start: usize,
327    count: usize,
328    _type: std::marker::PhantomData<T>,
329}
330
331impl<'a, T: StoredValue + ?Sized + 'a> ValuesIter<'a, T> {
332    fn new(bytes: Cow<'a, [u8]>, count: usize) -> Self {
333        Self {
334            bytes,
335            start: 0,
336            count,
337            _type: std::marker::PhantomData,
338        }
339    }
340
341    /// Similar to [`Iterator::next()`], but immediately process the reference, so that we don't
342    /// need to `Cow` the value to be able to return it.
343    fn map_next<U>(&mut self, map: impl Fn(&T) -> U) -> Option<U> {
344        if self.count == 0 {
345            return None;
346        }
347
348        let value = T::read_from_prefix(self.bytes.get(self.start..)?)?;
349
350        let out = map(value);
351
352        self.start += T::stored_size(value);
353        self.count = self.count.saturating_sub(1);
354
355        Some(out)
356    }
357}
358
359impl<'a, T: StoredValue + ?Sized + 'a> Iterator for ValuesIter<'a, T> {
360    type Item = Cow<'a, T>;
361
362    fn next(&mut self) -> Option<Self::Item> {
363        if self.count == 0 {
364            return None;
365        }
366
367        let cow = match &self.bytes {
368            Cow::Borrowed(slice) => Cow::Borrowed(T::read_from_prefix(slice.get(self.start..)?)?),
369            Cow::Owned(vec) => Cow::Owned(T::read_from_prefix(vec.get(self.start..)?)?.to_owned()),
370        };
371
372        self.start += T::stored_size(cow.as_ref());
373        self.count = self.count.saturating_sub(1);
374
375        Some(cow)
376    }
377}

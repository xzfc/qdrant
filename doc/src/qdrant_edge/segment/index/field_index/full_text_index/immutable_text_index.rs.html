23pub struct ImmutableFullTextIndex {
24    pub(super) inverted_index: ImmutableInvertedIndex,
25    pub(super) tokenizer: Tokenizer,
26    // Backing storage, source of state, persists deletions
27    pub(super) storage: Storage,
28}
29
30pub(super) enum Storage {
31    #[cfg(feature = "rocksdb")]
32    RocksDb(DatabaseColumnScheduledDeleteWrapper),
33    Mmap(Box<MmapFullTextIndex>),
34}
35
36impl ImmutableFullTextIndex {
37    /// Open and load immutable full text index from RocksDB storage
38    #[cfg(feature = "rocksdb")]
39    pub fn open_rocksdb(
40        db_wrapper: DatabaseColumnScheduledDeleteWrapper,
41        config: TextIndexParams,
42    ) -> OperationResult<Option<Self>> {
43        let tokenizer = Tokenizer::new_from_text_index_params(&config);
44
45        if !db_wrapper.has_column_family()? {
46            return Ok(None);
47        };
48
49        let db = db_wrapper.clone();
50        let db = db.lock_db();
51        let phrase_matching = config.phrase_matching.unwrap_or_default();
52        let iter = db.iter()?.map(|(key, value)| {
53            let idx = FullTextIndex::restore_key(&key);
54            let tokens = FullTextIndex::deserialize_document(&value)?;
55            Ok((idx, tokens))
56        });
57
58        let mutable = MutableInvertedIndex::build_index(iter, phrase_matching)?;
59
60        Ok(Some(Self {
61            inverted_index: ImmutableInvertedIndex::from(mutable),
62            tokenizer,
63            storage: Storage::RocksDb(db_wrapper),
64        }))
65    }
66
67    /// Open and load immutable full text index from mmap storage
68    pub fn open_mmap(index: MmapFullTextIndex) -> Self {
69        let inverted_index = ImmutableInvertedIndex::from(&index.inverted_index);
70
71        // ToDo(rocksdb): this is a duplication of tokenizer,
72        // ToDo(rocksdb): But once the RocksDB is removed, we can always use the tokenizer from the index.
73        let tokenizer = index.tokenizer.clone();
74
75        // Index is now loaded into memory, clear cache of backing mmap storage
76        if let Err(err) = index.clear_cache() {
77            log::warn!("Failed to clear mmap cache of ram mmap full text index: {err}");
78        }
79
80        Self {
81            inverted_index,
82            storage: Storage::Mmap(Box::new(index)),
83            tokenizer,
84        }
85    }
86
87    #[cfg_attr(not(feature = "rocksdb"), expect(clippy::unnecessary_wraps))]
88    pub fn remove_point(&mut self, id: PointOffsetType) -> OperationResult<()> {
89        if self.inverted_index.remove(id) {
90            match self.storage {
91                #[cfg(feature = "rocksdb")]
92                Storage::RocksDb(ref db_wrapper) => {
93                    let db_doc_id = FullTextIndex::store_key(id);
94                    db_wrapper.remove(db_doc_id)?;
95                }
96                Storage::Mmap(ref mut index) => {
97                    index.remove_point(id);
98                }
99            }
100        }
101        Ok(())
102    }
103
104    pub fn wipe(self) -> OperationResult<()> {
105        match self.storage {
106            #[cfg(feature = "rocksdb")]
107            Storage::RocksDb(db_wrapper) => db_wrapper.remove_column_family(),
108            Storage::Mmap(index) => index.wipe(),
109        }
110    }
111
112    /// Clear cache
113    ///
114    /// Only clears cache of mmap storage if used. Does not clear in-memory representation of
115    /// index.
116    pub fn clear_cache(&self) -> OperationResult<()> {
117        match &self.storage {
118            #[cfg(feature = "rocksdb")]
119            Storage::RocksDb(_) => Ok(()),
120            Storage::Mmap(index) => index.clear_cache().map_err(|err| {
121                OperationError::service_error(format!(
122                    "Failed to clear immutable full text index gridstore cache: {err}"
123                ))
124            }),
125        }
126    }
127
128    pub fn files(&self) -> Vec<PathBuf> {
129        match self.storage {
130            #[cfg(feature = "rocksdb")]
131            Storage::RocksDb(_) => vec![],
132            Storage::Mmap(ref index) => index.files(),
133        }
134    }
135
136    pub fn immutable_files(&self) -> Vec<PathBuf> {
137        match self.storage {
138            #[cfg(feature = "rocksdb")]
139            Storage::RocksDb(_) => vec![],
140            Storage::Mmap(ref index) => index.immutable_files(),
141        }
142    }
143
144    pub fn flusher(&self) -> Flusher {
145        match self.storage {
146            #[cfg(feature = "rocksdb")]
147            Storage::RocksDb(ref db_wrapper) => db_wrapper.flusher(),
148            Storage::Mmap(ref index) => index.flusher(),
149        }
150    }
151
152    #[cfg(feature = "rocksdb")]
153    pub fn from_rocksdb_mutable(mutable: MutableFullTextIndex) -> Self {
154        let MutableFullTextIndex {
155            inverted_index,
156            config: _,
157            tokenizer,
158            storage,
159        } = mutable;
160
161        let mutable_text_index::Storage::RocksDb(db) = storage else {
162            unreachable!(
163                "There is no Gridstore-backed immutable text index, it should be Mmap-backed instead",
164            );
165        };
166
167        Self {
168            inverted_index: ImmutableInvertedIndex::from(inverted_index),
169            tokenizer,
170            storage: Storage::RocksDb(db),
171        }
172    }
173
174    pub fn storage_type(&self) -> StorageType {
175        match &self.storage {
176            #[cfg(feature = "rocksdb")]
177            Storage::RocksDb(_) => StorageType::RocksDb,
178            Storage::Mmap(index) => StorageType::Mmap {
179                is_on_disk: index.is_on_disk(),
180            },
181        }
182    }
183
184    #[cfg(feature = "rocksdb")]
185    pub fn is_rocksdb(&self) -> bool {
186        match self.storage {
187            Storage::RocksDb(_) => true,
188            Storage::Mmap(_) => false,
189        }
190    }

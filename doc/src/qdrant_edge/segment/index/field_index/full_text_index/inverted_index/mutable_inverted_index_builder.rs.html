9pub struct MutableInvertedIndexBuilder {
10    index: MutableInvertedIndex,
11}
12
13impl MutableInvertedIndexBuilder {
14    pub fn new(phrase_matching: bool) -> Self {
15        let index = MutableInvertedIndex::new(phrase_matching);
16        Self { index }
17    }
18
19    /// Add a vector to the inverted index builder
20    pub fn add(&mut self, idx: PointOffsetType, str_tokens: impl IntoIterator<Item = String>) {
21        self.index.points_count += 1;
22
23        // resize point_to_* structures if needed
24        if self.index.point_to_tokens.len() <= idx as usize {
25            self.index
26                .point_to_tokens
27                .resize_with(idx as usize + 1, Default::default);
28
29            if let Some(point_to_doc) = self.index.point_to_doc.as_mut() {
30                point_to_doc.resize_with(idx as usize + 1, Default::default);
31            }
32        }
33
34        let tokens = self.index.register_tokens(str_tokens);
35
36        // insert as whole document
37        if let Some(point_to_doc) = self.index.point_to_doc.as_mut() {
38            point_to_doc[idx as usize] = Some(Document::new(tokens.clone()));
39        }
40
41        // insert as tokenset
42        let tokens_set = TokenSet::from_iter(tokens);
43        self.index.point_to_tokens[idx as usize] = Some(tokens_set);
44    }
45
46    #[cfg(feature = "rocksdb")]
47    pub fn add_iter(
48        &mut self,
49        iter: impl Iterator<Item = OperationResult<(PointOffsetType, Vec<String>)>>,
50        // TODO(phrase-index): add param for including phrase field
51    ) -> OperationResult<()> {
52        for item in iter {
53            let (idx, str_tokens) = item?;
54            self.add(idx, str_tokens);
55        }
56        Ok(())
57    }
58
59    /// Consumes the builder and returns a MutableInvertedIndex
60    pub fn build(mut self) -> MutableInvertedIndex {
61        // build postings from point_to_tokens
62        // build in order to increase point id
63        for (idx, tokenset) in self.index.point_to_tokens.iter().enumerate() {
64            if let Some(tokenset) = tokenset {
65                for token_idx in tokenset.tokens() {
66                    if self.index.postings.len() <= *token_idx as usize {
67                        self.index
68                            .postings
69                            .resize_with(*token_idx as usize + 1, Default::default);
70                    }
71                    self.index
72                        .postings
73                        .get_mut(*token_idx as usize)
74                        .expect("posting must exist")
75                        .insert(idx as PointOffsetType);
76                }
77            }
78        }
79
80        self.index
81    }
82}

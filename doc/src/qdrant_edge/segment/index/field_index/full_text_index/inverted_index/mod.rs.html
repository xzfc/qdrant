25/// Contains the set of tokens that are in a document.
26///
27/// Internally, it keeps them unique and sorted, so that we can binary-search over them
28#[derive(Default, Debug, Clone)]
29pub struct TokenSet(Vec<TokenId>);
30
31impl TokenSet {
32    pub fn len(&self) -> usize {
33        self.0.len()
34    }
35
36    pub fn is_empty(&self) -> bool {
37        self.0.is_empty()
38    }
39
40    pub fn tokens(&self) -> &[TokenId] {
41        &self.0
42    }
43
44    pub fn inner(self) -> Vec<TokenId> {
45        self.0
46    }
47
48    pub fn contains(&self, token: &TokenId) -> bool {
49        self.0.binary_search(token).is_ok()
50    }
51
52    /// Checks if the current set contains all given tokens.
53    ///
54    /// Returns false if the subset is empty
55    pub fn has_subset(&self, subset: &TokenSet) -> bool {
56        if subset.is_empty() {
57            return false;
58        }
59        subset.0.iter().all(|token| self.contains(token))
60    }
61
62    /// Checks if the current set contains any of the given tokens.
63    /// Returns false if the subset is empty
64    pub fn has_any(&self, subset: &TokenSet) -> bool {
65        if subset.is_empty() {
66            return false;
67        }
68        subset.0.iter().any(|token| self.contains(token))
69    }
70}

72impl From<AHashSet<TokenId>> for TokenSet {
73    fn from(tokens: AHashSet<TokenId>) -> Self {
74        let sorted_unique = tokens.into_iter().sorted_unstable().collect();
75
76        Self(sorted_unique)
77    }

119    /// Checks if the current document contains the given phrase.
120    ///
121    /// Returns false if the phrase is empty
122    pub fn has_phrase(&self, phrase: &Document) -> bool {
123        let doc = self.0.as_slice();
124        let phrase = phrase.0.as_slice();
125
126        if doc.is_empty() || phrase.is_empty() {
127            return false;
128        }
129
130        // simple check for tokens in the same order as phrase
131        doc.windows(phrase.len()).any(|window| window == phrase)
132    }
133}

153pub enum ParsedQuery {
154    /// All these tokens must be present in the document, regardless of order.
155    ///
156    /// In other words this should be a subset of the document's token set.
157    AllTokens(TokenSet),
158
159    /// At least one of these tokens must be present in the document.
160    AnyTokens(TokenSet),
161
162    /// All these tokens must be present in the document, in the same order as this query.
163    Phrase(Document),
164}

169    /// Translate the string tokens into token ids.
170    /// If it is an unseen token, it is added to the vocabulary and a new token id is generated.
171    ///
172    /// The order of the tokens is preserved.
173    fn register_tokens<'a>(
174        &mut self,
175        str_tokens: impl IntoIterator<Item = impl AsRef<str>> + 'a,
176    ) -> Vec<TokenId> {
177        str_tokens
178            .into_iter()
179            .map(|token| self.register_token(token))
180            .collect()
181    }
182
183    /// Translate the string token into token id.
184    /// If it is an unseen token, it is added to the vocabulary and a new token id is generated.
185    fn register_token<S: AsRef<str>>(&mut self, token_str: S) -> TokenId {
186        let vocab = self.get_vocab_mut();
187        match vocab.get(token_str.as_ref()) {
188            Some(&idx) => idx,
189            None => {
190                let next_token_id = vocab.len() as TokenId;
191                vocab.insert(token_str.as_ref().to_string(), next_token_id);
192                next_token_id
193            }
194        }
195    }

222    fn estimate_cardinality(
223        &self,
224        query: &ParsedQuery,
225        condition: &FieldCondition,
226        hw_counter: &HardwareCounterCell,
227    ) -> CardinalityEstimation {
228        match query {
229            ParsedQuery::AllTokens(tokens) => {
230                self.estimate_has_subset_cardinality(tokens, condition, hw_counter)
231            }
232            ParsedQuery::Phrase(phrase) => {
233                self.estimate_has_phrase_cardinality(phrase, condition, hw_counter)
234            }
235            ParsedQuery::AnyTokens(tokens) => {
236                self.estimate_has_any_cardinality(tokens, condition, hw_counter)
237            }
238        }
239    }

241    fn estimate_has_subset_cardinality(
242        &self,
243        tokens: &TokenSet,
244        condition: &FieldCondition,
245        hw_counter: &HardwareCounterCell,
246    ) -> CardinalityEstimation {
247        let points_count = self.points_count();
248
249        let posting_lengths: Option<Vec<usize>> = tokens
250            .tokens()
251            .iter()
252            .map(|&vocab_idx| self.get_posting_len(vocab_idx, hw_counter))
253            .collect();
254        if posting_lengths.is_none() || points_count == 0 {
255            // There are unseen tokens -> no matches
256            return CardinalityEstimation::exact(0)
257                .with_primary_clause(PrimaryCondition::Condition(Box::new(condition.clone())));
258        }
259        let postings = posting_lengths.unwrap();
260        if postings.is_empty() {
261            // Empty request -> no matches
262            return CardinalityEstimation::exact(0)
263                .with_primary_clause(PrimaryCondition::Condition(Box::new(condition.clone())));
264        }
265        // Smallest posting is the largest possible cardinality
266        let smallest_posting = postings.iter().min().copied().unwrap();
267
268        if postings.len() == 1 {
269            return CardinalityEstimation::exact(smallest_posting)
270                .with_primary_clause(PrimaryCondition::Condition(Box::new(condition.clone())));
271        }
272
273        let expected_frac: f64 = postings
274            .iter()
275            .map(|posting| *posting as f64 / points_count as f64)
276            .product();
277        let exp = (expected_frac * points_count as f64) as usize;
278        CardinalityEstimation {
279            primary_clauses: vec![PrimaryCondition::Condition(Box::new(condition.clone()))],
280            min: 0, // ToDo: make better estimation
281            exp,
282            max: smallest_posting,
283        }
284    }
285
286    fn estimate_has_any_cardinality(
287        &self,
288        tokens: &TokenSet,
289        condition: &FieldCondition,
290        hw_counter: &HardwareCounterCell,
291    ) -> CardinalityEstimation {
292        let points_count = self.points_count();
293
294        let posting_lengths: Vec<_> = tokens
295            .tokens()
296            .iter()
297            .filter_map(|&vocab_idx| self.get_posting_len(vocab_idx, hw_counter))
298            .collect();
299
300        if posting_lengths.is_empty() {
301            // Empty request -> no matches
302            return CardinalityEstimation::exact(0)
303                .with_primary_clause(PrimaryCondition::Condition(Box::new(condition.clone())));
304        }
305
306        // At least one posting is the largest possible cardinality
307        let largest_posting = posting_lengths.iter().max().copied().unwrap();
308
309        if posting_lengths.len() == 1 {
310            return CardinalityEstimation::exact(largest_posting)
311                .with_primary_clause(PrimaryCondition::Condition(Box::new(condition.clone())));
312        }
313
314        let sum: usize = posting_lengths.iter().sum();
315
316        let exp = expected_should_estimation(posting_lengths.into_iter(), points_count);
317
318        CardinalityEstimation {
319            primary_clauses: vec![PrimaryCondition::Condition(Box::new(condition.clone()))],
320            min: largest_posting,
321            exp,
322            max: min(sum, points_count),
323        }
324    }
325
326    fn estimate_has_phrase_cardinality(
327        &self,
328        phrase: &Document,
329        condition: &FieldCondition,
330        hw_counter: &HardwareCounterCell,
331    ) -> CardinalityEstimation {
332        if phrase.is_empty() {
333            return CardinalityEstimation::exact(0)
334                .with_primary_clause(PrimaryCondition::Condition(Box::new(condition.clone())));
335        }
336
337        // Start with same cardinality estimation as has_subset
338        let tokenset = phrase.to_token_set();
339        let subset_estimation =
340            self.estimate_has_subset_cardinality(&tokenset, condition, hw_counter);
341
342        // But we can restrict it by considering the phrase length
343        let phrase_sq = phrase.len() * phrase.len();
344
345        CardinalityEstimation {
346            primary_clauses: vec![PrimaryCondition::Condition(Box::new(condition.clone()))],
347            min: subset_estimation.min / phrase_sq,
348            exp: subset_estimation.exp / phrase_sq,
349            max: subset_estimation.max / phrase_sq,
350        }
351    }

354
355    fn payload_blocks(
356        &self,
357        threshold: usize,
358        key: PayloadKeyType,
359    ) -> impl Iterator<Item = PayloadBlockCondition> + '_ {
360        let map_filter_condition = move |(token, postings_len): (&str, usize)| {
361            if postings_len >= threshold {
362                Some(PayloadBlockCondition {
363                    condition: FieldCondition::new_match(key.clone(), Match::new_text(token)),
364                    cardinality: postings_len,
365                })
366            } else {
367                None
368            }
369        };
370
371        // It might be very hard to predict possible combinations of conditions,
372        // so we only build it for individual tokens
373        self.vocab_with_postings_len_iter()
374            .filter_map(map_filter_condition)
375    }

441    fn mutable_inverted_index(
442        indexed_count: u32,
443        deleted_count: u32,
444        with_positions: bool,
445    ) -> MutableInvertedIndex {
446        let mut index = MutableInvertedIndex::new(with_positions);
447
448        let hw_counter = HardwareCounterCell::new();
449
450        for idx in 0..indexed_count {
451            // Generate 10 to 30-word documents
452            let doc_len = rand::rng().random_range(10..=30);
453            let tokens: Vec<String> = (0..doc_len).map(|_| generate_word()).collect();
454            let token_ids = index.register_tokens(&tokens);
455            if with_positions {
456                index
457                    .index_document(idx, Document(token_ids.clone()), &hw_counter)
458                    .unwrap();
459            }
460            let token_set = TokenSet::from_iter(token_ids);
461            index.index_tokens(idx, token_set, &hw_counter).unwrap();
462        }
463
464        // Remove some points
465        let mut points_to_delete = (0..indexed_count).collect::<Vec<_>>();
466        points_to_delete.shuffle(&mut rand::rng());
467        for idx in &points_to_delete[..deleted_count as usize] {
468            index.remove(*idx);
469        }
470
471        index
472    }

677        imm_mmap_parsed_queries: &[Option<ParsedQuery>],
678        mut_index: &MutableInvertedIndex,
679        mmap_index: &MmapInvertedIndex,
680        imm_mmap_index: &ImmutableInvertedIndex,
681        hw_counter: &HardwareCounterCell,
682    ) {
683        for queries in mut_parsed_queries.iter().cloned().zip(
684            mmap_parsed_queries
685                .iter()
686                .cloned()
687                .zip(imm_mmap_parsed_queries.iter().cloned()),
688        ) {
689            let (Some(mut_query), (Some(imm_query), Some(imm_mmap_query))) = queries else {
690                // Immutable index can have a smaller vocabulary, since it only contains tokens that have
691                // non-empty posting lists.
692                // Since we removed some documents from the mutable index, it can happen that the immutable
693                // index returns None when parsing the query, even if the mutable index returns Some.
694                //
695                // In this case both queries would filter to an empty set of documents.
696                continue;
697            };
698            let mut_filtered = mut_index.filter(mut_query, hw_counter).collect::<Vec<_>>();
699            let imm_filtered = mmap_index.filter(imm_query, hw_counter).collect::<Vec<_>>();
700            let imm_mmap_filtered = imm_mmap_index
701                .filter(imm_mmap_query, hw_counter)
702                .collect::<Vec<_>>();
703
704            assert_eq!(mut_filtered, imm_filtered);
705            assert_eq!(imm_filtered, imm_mmap_filtered);
706        }
707    }
708}

6/// Represents a list of positions of a token in a document.
7#[derive(Default, Clone, Debug)]
8pub struct Positions(Vec<u32>);
9
10impl Positions {
11    pub fn is_empty(&self) -> bool {
12        self.0.is_empty()
13    }
14
15    pub fn push(&mut self, position: u32) {
16        self.0.push(position);
17    }
18
19    pub fn to_token_positions(&self, token_id: TokenId) -> Vec<TokenPosition> {
20        self.0
21            .iter()
22            .map(|pos| TokenPosition {
23                token_id,
24                position: *pos,
25            })
26            .collect()
27    }
28}
29
30impl PostingValue for Positions {
31    type Handler = UnsizedHandler<Self>;
32}
33
34impl UnsizedValue for Positions {
35    fn write_len(&self) -> usize {
36        self.0.as_bytes().len()
37    }
38
39    fn write_to(&self, dst: &mut [u8]) {
40        self.0
41            .as_slice()
42            .write_to(dst)
43            .expect("write_len should provide correct length");
44    }
45
46    fn from_bytes(data: &[u8]) -> Self {
47        let positions =
48            <[u32]>::ref_from_bytes(data).expect("write_len should provide correct length");
49        Positions(positions.to_vec())
50    }
51}

53#[derive(Debug, Eq, PartialEq)]
54pub struct TokenPosition {
55    token_id: TokenId,
56    position: u32,
57}
58
59/// A reconstructed partial document which stores [`TokenPosition`]s, ordered by positions
60#[derive(Debug)]
61pub struct PartialDocument(Vec<TokenPosition>);
62
63impl PartialDocument {
64    pub fn new(mut tokens_positions: Vec<TokenPosition>) -> Self {
65        tokens_positions.sort_by_key(|tok_pos| tok_pos.position);
66
67        // There should be no duplicate token with same position
68        debug_assert!(tokens_positions.array_windows().all(|[a, b]| a != b));
69
70        Self(tokens_positions)
71    }
72
73    /// Returns true if any sequential window of tokens match the given phrase.
74    pub fn has_phrase(&self, phrase: &Document) -> bool {
75        match phrase.tokens() {
76            // no tokens in query -> no match
77            [] => false,
78
79            // single token -> match if any token matches
80            [token] => self.0.iter().any(|tok_pos| tok_pos.token_id == *token),
81
82            // multiple tokens -> match if any sequential window matches
83            phrase => self.sequential_windows(phrase.len()).any(|seq_window| {
84                seq_window
85                    .zip(phrase)
86                    .all(|(doc_token, query_token)| &doc_token == query_token)
87            }),
88        }
89    }
90
91    /// Returns an iterator over windows which have sequential sequence of tokens.
92    ///
93    /// Will only return a window if:
94    /// - the window is as large as the window size
95    /// - all positions in the window are sequential
96    fn sequential_windows(
97        &self,
98        window_size: usize,
99    ) -> impl Iterator<Item = impl Iterator<Item = TokenId>> {
100        debug_assert!(window_size >= 2, "Window size must be at least 2");
101        self.0.windows(window_size).filter_map(|window| {
102            // make sure the positions are sequential
103            window
104                .array_windows()
105                .all(|[a, b]| a.position + 1 == b.position)
106                .then_some(window.iter().map(|tok_pos| tok_pos.token_id))
107        })
108    }
109}

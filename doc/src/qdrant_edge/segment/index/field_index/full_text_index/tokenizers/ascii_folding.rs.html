1// ASCII folding implementation inspired by Lucene's ASCIIFoldingFilter.
2// Converts non-ASCII Latin letters and various symbols to ASCII equivalents.
3// This aims to be compatible with Lucene's mapping list used in ASCIIFoldingFilter.
4
5use std::borrow::Cow;
6
7pub fn fold_to_ascii_cow<'a>(input: Cow<'a, str>) -> Cow<'a, str> {
8    // Cheap check if already ASCII
9    if input.is_ascii() {
10        return input;
11    }
12
13    Cow::Owned(fold_to_ascii(input.as_ref()))
14}
15
16pub fn fold_to_ascii(input: &str) -> String {
17    let mut out = String::with_capacity(input.len());
18    for ch in input.chars() {
19        if ch.is_ascii() {
20            out.push(ch);
21            continue;
22        }
23        match fold_char(ch) {
24            Some(repl) => out.push_str(repl),
25            None => out.push(ch),
26        }
27    }
28    out.shrink_to_fit();
29    out
30}

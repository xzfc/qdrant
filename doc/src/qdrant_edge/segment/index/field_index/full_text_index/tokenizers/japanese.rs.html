8/// Vaporetto prediction model. Source: https://github.com/daac-tools/vaporetto-models/releases/tag/v0.5.0
9const MODEL: &[u8] = include_bytes!(concat!(
10    env!("CARGO_MANIFEST_DIR"),
11    "/tokenizer/bccwj-suw_c1.0.model"
12));

14/// Sha512 checksum of the model to ensure integrity and make modifications or corrupt model file easier to detect.
15#[cfg(test)]
16const MODEL_CHECKSUM: [u8; 64] = [
17    34, 108, 156, 130, 7, 199, 31, 24, 147, 156, 119, 202, 98, 129, 109, 101, 114, 8, 250, 182,
18    159, 28, 112, 122, 214, 50, 51, 191, 118, 112, 143, 237, 70, 15, 96, 45, 78, 76, 90, 62, 178,

20    43, 203,

23// Global initialization of the Japanese tokenizer.
24static GLOBAL_JAPANESE_TOKENIZER: LazyLock<JapaneseTokenizer> =
25    LazyLock::new(JapaneseTokenizer::init);
26
27/// Tokenizer for Japanese text using vaporetto tokenizer.
28struct JapaneseTokenizer {
29    predictor: Predictor,
30}
31
32impl JapaneseTokenizer {
33    /// Initializes a new `JapaneseTokenizer`. Should only called once and then kept allocated somewhere for efficient reuse.
34    fn init() -> Self {
35        let model = Model::read_slice(MODEL).unwrap().0;
36        let predictor = Predictor::new(model, false).unwrap();
37        Self { predictor }
38    }
39
40    fn tokenize<'a, C: FnMut(Cow<'a, str>)>(
41        &self,
42        input: &'a str,
43        tokens_processor: &TokensProcessor,
44        mut cb: C,
45    ) {
46        let Ok(mut s) = Sentence::from_raw(Cow::Borrowed(input)) else {
47            return;
48        };
49
50        self.predictor.predict(&mut s);
51
52        // TODO(multilingual): Implement similar method to `iter_tokens()` that allows returning borrowed Cows instead of needlessly cloning here.
53        for i in s.iter_tokens() {
54            let surface = i.surface();
55
56            // Skip if all characters are not alphanumeric or if the surface is empty.
57            if tokens_processor.is_stopword(surface)
58                || surface.chars().all(|char| !char.is_alphabetic())
59            {
60                continue;
61            }
62
63            let surface = if tokens_processor.lowercase {
64                Cow::Owned(surface.to_lowercase())
65            } else {
66                Cow::Owned(surface.to_string())
67            };
68            cb(surface);
69        }
70    }
71}
72
73/// Tokenizes the given `input` of Japanese text and calls `cb` with each tokens.
74pub fn tokenize<'a, C: FnMut(Cow<'a, str>)>(input: &'a str, config: &TokensProcessor, cb: C) {
75    GLOBAL_JAPANESE_TOKENIZER.tokenize(input, config, cb);
76}

86    #[test]
87    fn test_assert_model_integrity() {
88        // Sha512 checksum of model
89        let mut sha512 = Sha512::new();
90        sha512.write_all(MODEL).unwrap();
91        let sum = sha512.finalize();
92
93        assert!(
94            sum.as_ref() == MODEL_CHECKSUM,
95            "Japanese Tokenizer Model integrity check failed! The file might be modified or corrupted."
96        );
97
98        // The init() function is completely deterministic, since the model that gets loaded is included in
99        // qdrant binary file. We test initialization here once to ensure it won't panic on runtime (eg. when a model has changed).
100        let _ = JapaneseTokenizer::init();
101    }

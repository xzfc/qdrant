8/// Default normalizer options from charabia(https://github.com/meilisearch/charabia/blob/main/charabia/src/normalizer/mod.rs#L82) used
9/// in `str::tokenize()`.
10const DEFAULT_NORMALIZER: NormalizerOption = NormalizerOption {
11    create_char_map: false,
12    lossy: true,
13    classifier: ClassifierOption {
14        stop_words: None,
15        separators: None,
16    },
17};
18
19pub struct MultilingualTokenizer;
20
21impl MultilingualTokenizer {
22    pub fn tokenize<'a, C: FnMut(Cow<'a, str>)>(
23        input: &'a str,
24        config: &'a TokensProcessor,
25        cb: C,
26    ) {
27        let script = detect_script_of_language(input);
28
29        // If the script of the input is latin and we don't need to stem early, tokenize as-is.
30        // This skips language detection, reduces overhead, and improves performance.
31        if script_is_latin(script) {
32            Self::tokenize_charabia(input, config, cb);
33            return;
34        }
35
36        // If the script of the input is Japanese, use vaporetto to segment.
37        if detect_language(input) == Some(Language::Jpn) {
38            japanese::tokenize(input, config, cb);
39            return;
40        }
41
42        Self::tokenize_charabia(input, config, cb);
43    }
44
45    // Tokenize input using charabia. Automatically applies stemming and filters stopwords if configured.
46    fn tokenize_charabia<'a, C>(input: &'a str, tokens_processor: &'a TokensProcessor, mut cb: C)
47    where
48        C: FnMut(Cow<'a, str>),
49    {
50        for token in charabia_token_iter(input) {
51            let lemma = token.lemma;
52
53            if lemma.chars().all(|char| !char.is_alphabetic()) {
54                // Skip tokens that are not alphanumeric.
55                continue;
56            }
57
58            if let Some(processed_token) = tokens_processor.process_token_cow(lemma, true) {
59                cb(processed_token);
60            }
61        }
62    }
63}
64
65// Tokenize::tokenize() function from charabia unrolled due to lifetime issues
66// when using .tokenize() on a `str` directly.
67fn charabia_token_iter(inp: &str) -> NormalizedTokenIter<'_, '_, '_, '_> {
68    inp.segment().normalize(&DEFAULT_NORMALIZER)
69}

71// Detect the script of the given input using charabia.
72fn detect_script_of_language(input: &str) -> Script {
73    StrDetection::new(input, Some(SUPPORTED_LANGUAGES)).script()
74}
75
76// Detect the script of the given input using charabia.
77fn detect_language(input: &str) -> Option<charabia::Language> {
78    StrDetection::new(input, Some(SUPPORTED_LANGUAGES)).language()
79}
80
81/// Returns `true` if the given `script` is latin.
82#[inline]
83fn script_is_latin(script: Script) -> bool {
84    matches!(script, Script::Latin)
85}

87/// Languages that are supported by rust-stemmers and thus should be used in language detection white-list.
88/// Also includes Languages that we manually need to check against, such as Japanese and Chinese.
89const SUPPORTED_LANGUAGES: &[charabia::Language] = &[
90    charabia::Language::Eng,
91    charabia::Language::Rus,

119        // Japanese
120        let input = "日本語のテキストです。Qdrantのコードで単体テストで使用されています。";
121        assert_eq!(detect_language(input), Some(Language::Jpn));
122
123        let input = "This is english text. It's being used within Qdrant's code in a unit test.";
124        assert_eq!(detect_language(input), Some(Language::Eng));
125
126        let input =
127            "Das ist ein deutscher Text. Er wird in Qdrants code in einem unit Test benutzt."; // codespell:ignore ist

129
130        // Chinese traditional
131        let input = "這是一段德文文本。它用於 Qdrant 程式碼的單元測試中。";
132        assert_eq!(detect_language(input), Some(Language::Cmn));
133
134        // Chinese simplified
135        let input = "这是一段德语文本。它用于 Qdrant 代码的单元测试中。";

152    fn assert_tokenization(inp: &str, expected: &str) {
153        let tokens_processor = TokensProcessor::default();
154
155        let mut out = vec![];
156        MultilingualTokenizer::tokenize(inp, &tokens_processor, |i| out.push(i.to_string()));
157        let expected: Vec<_> = expected.split('|').collect();
158        for i in out.iter().zip(expected.iter()) {
159            assert_eq!(i.0, i.1);
160        }
161        assert_eq!(out, expected)
162    }

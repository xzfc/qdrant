15pub fn is_quantized_search(
16    quantized_storage: Option<&QuantizedVectors>,
17    params: Option<&SearchParams>,
18) -> bool {
19    let ignore_quantization = params
20        .and_then(|p| p.quantization)
21        .map(|q| q.ignore)
22        .unwrap_or(default_quantization_ignore_value());
23    let exact = params.map(|p| p.exact).unwrap_or(false);
24    quantized_storage.is_some() && !ignore_quantization && !exact
25}

27pub fn get_oversampled_top(
28    quantized_storage: Option<&QuantizedVectors>,
29    params: Option<&SearchParams>,
30    top: usize,
31) -> usize {
32    let quantization_enabled = is_quantized_search(quantized_storage, params);
33
34    let oversampling_value = params
35        .and_then(|p| p.quantization)
36        .map(|q| q.oversampling)
37        .unwrap_or(default_quantization_oversampling_value());
38
39    match oversampling_value {
40        Some(oversampling) if quantization_enabled && oversampling > 1.0 => {
41            (oversampling * top as f64) as usize
42        }
43        _ => top,
44    }
45}

47#[allow(clippy::too_many_arguments)]
48pub fn postprocess_search_result(
49    mut search_result: Vec<ScoredPointOffset>,
50    point_deleted: &BitSlice,
51    vector_storage: &VectorStorageEnum,
52    quantized_vectors: Option<&QuantizedVectors>,
53    vector: &QueryVector,
54    params: Option<&SearchParams>,
55    top: usize,
56    hardware_counter: HardwareCounterCell,
57) -> OperationResult<Vec<ScoredPointOffset>> {
58    let quantization_enabled = is_quantized_search(quantized_vectors, params);
59
60    let default_rescoring = quantized_vectors
61        .as_ref()
62        .map(|q| q.default_rescoring())
63        .unwrap_or(false);
64    let rescore = quantization_enabled
65        && params
66            .and_then(|p| p.quantization)
67            .and_then(|q| q.rescore)
68            .unwrap_or(default_rescoring);
69    if rescore {
70        let mut scorer = FilteredScorer::new(
71            vector.to_owned(),
72            vector_storage,
73            None,
74            None,
75            point_deleted,
76            hardware_counter,
77        )?;
78
79        search_result = scorer
80            .score_points(&mut search_result.iter().map(|x| x.idx).collect_vec(), 0)
81            .collect();
82        search_result.sort_unstable();
83        search_result.reverse();
84    }
85    search_result.truncate(top);
86    Ok(search_result)
87}

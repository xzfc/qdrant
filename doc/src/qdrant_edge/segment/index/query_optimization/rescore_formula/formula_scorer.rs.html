24/// A scorer to evaluate the same formula for many points
25pub struct FormulaScorer<'a> {
26    /// The formula to evaluate
27    formula: ParsedExpression,
28    /// One hashmap for each prefetch results
29    prefetches_scores: &'a [AHashMap<PointOffsetType, ScoreType>],
30    /// Payload key -> retriever function
31    payload_retrievers: HashMap<JsonPath, VariableRetrieverFn<'a>>,
32    /// Condition id -> checker function
33    condition_checkers: Vec<OptimizedCondition<'a>>,
34    /// Default values for all variables
35    defaults: HashMap<VariableId, Value>,
36}

60impl StructPayloadIndex {
61    pub fn formula_scorer<'s, 'q>(
62        &'s self,
63        parsed_formula: &'q ParsedFormula,
64        prefetches_scores: &'q [AHashMap<PointOffsetType, ScoreType>],
65        hw_counter: &'q HardwareCounterCell,
66    ) -> FormulaScorer<'q>
67    where
68        's: 'q,
69    {
70        let ParsedFormula {
71            payload_vars,
72            conditions,
73            defaults,
74            formula,
75        } = parsed_formula;
76
77        let payload_retrievers = self.retrievers_map(payload_vars.clone(), hw_counter);
78
79        let payload_provider = PayloadProvider::new(self.payload.clone());
80        let total = self.available_point_count();
81        let condition_checkers = self
82            .convert_conditions(conditions, payload_provider, total, hw_counter)
83            .into_iter()
84            .map(|(checker, _estimation)| checker)
85            .collect();
86
87        FormulaScorer {
88            formula: formula.clone(),
89            prefetches_scores,
90            payload_retrievers,
91            condition_checkers,
92            defaults: defaults.clone(),
93        }
94    }
95}

97impl FormulaScorer<'_> {
98    /// Evaluate the formula for the given point
99    pub fn score(&self, point_id: PointOffsetType) -> OperationResult<ScoreType> {
100        self.eval_expression(&self.formula, point_id)
101            .and_then(|score| {
102                let score_f32 = score as f32;
103                if !score_f32.is_finite() {
104                    return Err(OperationError::NonFiniteNumber {
105                        expression: format!("{score} as f32 = {score_f32}"),
106                    });
107                }
108                Ok(score_f32)
109            })
110    }
111
112    /// Evaluate the expression recursively
113    fn eval_expression(
114        &self,
115        expression: &ParsedExpression,
116        point_id: PointOffsetType,
117    ) -> OperationResult<PreciseScore> {
118        match expression {
119            ParsedExpression::Constant(c) => Ok(c.0),
120            ParsedExpression::Variable(v) => match v {
121                VariableId::Score(prefetch_idx) => Ok(self
122                    .prefetches_scores
123                    .get(*prefetch_idx)
124                    .and_then(|scores| scores.get(&point_id))
125                    .map(|score| PreciseScore::from(*score))
126                    .or_else(|| {
127                        self.defaults
128                            // if there is no score, or it isn't a number, we use the default score
129                            .get(&VariableId::Score(*prefetch_idx))
130                            .and_then(|value| value.as_f64())
131                    })
132                    .unwrap_or(DEFAULT_SCORE)),
133                VariableId::Payload(path) => {
134                    self.get_parsed_payload_value(path, point_id, |value| {
135                        value.as_f64().ok_or("Value is not a number")
136                    })
137                }
138                VariableId::Condition(id) => {
139                    let value = check_condition(&self.condition_checkers[*id], point_id);
140                    let score = if value { 1.0 } else { 0.0 };
141                    Ok(score)
142                }
143            },
144            ParsedExpression::GeoDistance { origin, key } => {
145                let value = self.get_parsed_payload_value(
146                    key,
147                    point_id,
148                    serde_json::from_value::<GeoPoint>,
149                )?;
150
151                Ok(Haversine.distance((*origin).into(), value.into()))
152            }
153            ParsedExpression::Datetime(dt_expr) => {
154                let datetime = match dt_expr {
155                    DatetimeExpression::Constant(dt) => *dt,
156                    DatetimeExpression::PayloadVariable(json_path) => {
157                        self.get_parsed_payload_value(json_path, point_id, |value| {
158                            value
159                                // datetime index also returns the Serialize impl of datetime which is a string
160                                .as_str()
161                                .ok_or("Value is not a string")?
162                                .parse::<DateTimePayloadType>()
163                                .map_err(|e| e.to_string())
164                        })?
165                    }
166                };
167                // Convert from i64 to f64.
168                // f64's 53 bits of sign + mantissa for microseconds means a span of exact equivalence of
169                // about 285 years, after which precision starts dropping
170                let float_micros = datetime.timestamp() as PreciseScore;
171
172                // Convert to seconds
173                let float_seconds = float_micros / 1_000_000.0;
174
175                Ok(float_seconds)
176            }
177            ParsedExpression::Mult(expressions) => {
178                let mut product = 1.0;
179                for expr in expressions {
180                    let value = self.eval_expression(expr, point_id)?;
181                    // shortcut on multiplication by zero
182                    if value == 0.0 {
183                        return Ok(0.0);
184                    }
185                    product *= value;
186                }
187                Ok(product)
188            }
189            ParsedExpression::Sum(expressions) => expressions.iter().try_fold(0.0, |acc, expr| {
190                let value = self.eval_expression(expr, point_id)?;
191                Ok(acc + value)
192            }),
193            ParsedExpression::Div {
194                left,
195                right,
196                by_zero_default,
197            } => {
198                let left = self.eval_expression(left, point_id)?;
199                // shortcut on numerator zero
200                if left == 0.0 {
201                    return Ok(0.0);
202                }
203                let right = self.eval_expression(right, point_id)?;
204
205                if right == 0.0
206                    && let Some(default) = by_zero_default
207                {
208                    return Ok(default.0);
209                }
210
211                let div_value = left / right;
212                if div_value.is_finite() {
213                    return Ok(div_value);
214                }
215                Err(OperationError::NonFiniteNumber {
216                    expression: format!("{left}/{right} = {div_value}"),
217                })
218            }
219            ParsedExpression::Neg(expr) => {
220                let value = self.eval_expression(expr, point_id)?;
221                Ok(value.neg())
222            }
223            ParsedExpression::Sqrt(expr) => {
224                let value = self.eval_expression(expr, point_id)?;
225                let sqrt_value = value.sqrt();
226                if sqrt_value.is_finite() {
227                    return Ok(sqrt_value);
228                }
229                Err(OperationError::NonFiniteNumber {
230                    expression: format!("√{value} = {sqrt_value}"),
231                })
232            }
233            ParsedExpression::Pow { base, exponent } => {
234                let base_value = self.eval_expression(base, point_id)?;
235                let exponent_value = self.eval_expression(exponent, point_id)?;
236                let power = base_value.powf(exponent_value);
237                if power.is_finite() {
238                    return Ok(power);
239                }
240                Err(OperationError::NonFiniteNumber {
241                    expression: format!("{base_value}^{exponent_value} = {power}"),
242                })
243            }
244            ParsedExpression::Exp(parsed_expression) => {
245                let value = self.eval_expression(parsed_expression, point_id)?;
246                let exp_value = value.exp();
247                if exp_value.is_finite() {
248                    return Ok(exp_value);
249                }
250                Err(OperationError::NonFiniteNumber {
251                    expression: format!("exp({value}) = {exp_value}"),
252                })
253            }
254            ParsedExpression::Log10(expr) => {
255                let value = self.eval_expression(expr, point_id)?;
256                let log_value = value.log10();
257                if log_value.is_finite() {
258                    return Ok(log_value);
259                }
260                Err(OperationError::NonFiniteNumber {
261                    expression: format!("log10({value}) = {log_value}"),
262                })
263            }
264            ParsedExpression::Ln(expr) => {
265                let value = self.eval_expression(expr, point_id)?;
266                let ln_value = value.ln();
267                if ln_value.is_finite() {
268                    return Ok(ln_value);
269                }
270                Err(OperationError::NonFiniteNumber {
271                    expression: format!("ln({value}) = {ln_value}"),
272                })
273            }
274            ParsedExpression::Abs(expr) => {
275                let value = self.eval_expression(expr, point_id)?;
276                Ok(value.abs())
277            }
278            // Interactive formulas in https://www.desmos.com/calculator/htg0vrfmks
279            ParsedExpression::Decay {
280                kind,
281                target,
282                lambda,
283                x,
284            } => {
285                let x = self.eval_expression(x, point_id)?;
286                let target = if let Some(target) = target {
287                    self.eval_expression(target, point_id)?
288                } else {
289                    DEFAULT_DECAY_TARGET
290                };
291                let decay = match kind {
292                    DecayKind::Exp => exp_decay(x, target, lambda.0),
293                    DecayKind::Gauss => gauss_decay(x, target, lambda.0),
294                    DecayKind::Lin => linear_decay(x, target, lambda.0),
295                };
296
297                // All decay functions have a range of [0, 1], no need to check for bounds
298                debug_assert!((0.0..=1.0).contains(&decay));
299
300                Ok(decay)
301            }
302        }
303    }

305    fn get_payload_value(&self, json_path: &JsonPath, point_id: PointOffsetType) -> Option<Value> {
306        self.payload_retrievers
307            .get(json_path)
308            .and_then(|retriever| {
309                let mut multi_value = retriever(point_id);
310                match multi_value.len() {
311                    0 => None,
312                    1 => Some(multi_value.pop().unwrap()),
313                    _ => Some(Value::Array(multi_value.into_iter().collect())),
314                }
315            })
316    }
317
318    /// Tries to get a value from payload or from the defaults. Then tries to convert it to the desired type.
319    fn get_parsed_payload_value<T, F, E>(
320        &self,
321        json_path: &JsonPath,
322        point_id: PointOffsetType,
323        from_value: F,
324    ) -> OperationResult<T>
325    where
326        F: Fn(Value) -> Result<T, E>,
327        E: ToString,
328        T: FriendlyName,
329    {
330        let value = self
331            .get_payload_value(json_path, point_id)
332            .or_else(|| {
333                self.defaults
334                    .get(&VariableId::Payload(json_path.clone()))
335                    .cloned()
336            })
337            .ok_or_else(|| OperationError::VariableTypeError {
338                field_name: json_path.clone(),
339                expected_type: T::friendly_name().to_owned(),
340                description: "No value found in a payload nor defaults".to_string(),
341            })?;
342
343        from_value(value).map_err(|e| OperationError::VariableTypeError {
344            field_name: json_path.clone(),
345            expected_type: T::friendly_name().to_owned(),
346            description: e.to_string(),
347        })
348    }
349}

351fn exp_decay(x: PreciseScore, target: PreciseScore, lambda: PreciseScore) -> PreciseScore {
352    let diff = (x - target).abs();
353    (lambda * diff).exp()
354}
355
356fn gauss_decay(x: PreciseScore, target: PreciseScore, lambda: PreciseScore) -> PreciseScore {
357    let diff = x - target;
358    (lambda * diff * diff).exp()
359}
360
361fn linear_decay(x: PreciseScore, target: PreciseScore, lambda: PreciseScore) -> PreciseScore {
362    let diff = (x - target).abs();
363    (-lambda * diff + 1.0).max(0.0)
364}

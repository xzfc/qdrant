1use std::collections::BinaryHeap;
2
3use crate::common::fixed_length_priority_queue::FixedLengthPriorityQueue;
4use crate::common::types::{ScoreType, ScoredPointOffset};
5use num_traits::float::FloatCore;
6
7/// Structure that holds context of the search
8pub struct SearchContext {
9    /// Overall nearest points found so far
10    pub nearest: FixedLengthPriorityQueue<ScoredPointOffset>,
11    /// Current candidates to process
12    pub candidates: BinaryHeap<ScoredPointOffset>,
13}
14
15impl SearchContext {
16    pub fn new(ef: usize) -> Self {
17        SearchContext {
18            nearest: FixedLengthPriorityQueue::new(ef),
19            candidates: BinaryHeap::new(),
20        }
21    }
22
23    pub fn lower_bound(&self) -> ScoreType {
24        match self.nearest.top() {
25            None => ScoreType::min_value(),
26            Some(worst_of_the_best) => worst_of_the_best.score,
27        }
28    }
29
30    /// Updates search context with new scored point.
31    /// If it is closer than existing - also add it to candidates for further search
32    pub fn process_candidate(&mut self, score_point: ScoredPointOffset) {
33        let was_added = match self.nearest.push(score_point) {
34            None => true,
35            Some(removed) => removed.idx != score_point.idx,
36        };
37        if was_added {
38            self.candidates.push(score_point);
39        }
40    }
41}

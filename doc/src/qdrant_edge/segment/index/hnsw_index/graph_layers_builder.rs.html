101    /// For a given sub-graph defined by points, returns connectivity estimation.
102    /// How it works:
103    ///  - Select entry point, it would be a point with the highest level. If there are several, pick first one.
104    ///  - Start Breadth-First Search (BFS) from the entry point, on each edge flip a coin to decide if the edge is removed or not.
105    ///  - Count number of nodes reachable from the entry point.
106    ///  - Use visited points as entry points for the next layer below and repeat until layer 0 has reached.
107    ///  - Return the fraction of reachable nodes to the total number of nodes in the sub-graph.
108    ///
109    /// Coin probability `q` is a parameter of this function. By default, it is 0.5.
110    pub fn subgraph_connectivity(&self, points: &[PointOffsetType], q: f32) -> f32 {
111        if points.is_empty() {
112            return 1.0;
113        }
114
115        let max_point_id = *points.iter().max().unwrap();
116
117        let mut visited: BitVec = BitVec::repeat(false, max_point_id as usize + 1);
118        let mut point_selection: BitVec = BitVec::repeat(false, max_point_id as usize + 1);
119
120        for point_id in points {
121            point_selection.set(*point_id as usize, true);
122        }
123
124        let mut rnd = rand::rng();
125
126        // Try to get entry point from the entry points list
127        // If not found, select the point with the highest level
128        let entry_point = self
129            .entry_points
130            .lock()
131            .get_random_entry_point(&mut rnd, |point_id| {
132                point_selection.get_bit(point_id as usize).unwrap_or(false)
133            })
134            .map(|ep| ep.point_id);
135
136        // Select entry point by selecting the point with the highest level
137        let entry_point = entry_point.unwrap_or_else(|| {
138            points
139                .iter()
140                .max_by_key(|point_id| self.links_layers[**point_id as usize].len())
141                .cloned()
142                .unwrap()
143        });
144        let entry_layer = self.get_point_level(entry_point);
145
146        let mut queue: Vec<u32> = vec![];
147
148        // Amount of points reached when searching the graph.
149        let mut reached_points = 1;
150
151        // Total points visited (also across retries).
152        let mut spent_budget = 0;
153
154        // Retry loop, in case some budget is left.
155        loop {
156            visited.set(entry_point as usize, true);
157
158            // Points visited in the previous layer (Get used as entry point in the iteration over the next layer)
159            let mut previous_visited_points = vec![entry_point];
160
161            // For each layer in HNSW lower than the entry point layer
162            for current_layer in (0..=entry_layer).rev() {
163                // Set entry points to visited points of previous layer.
164                queue.extend_from_slice(&previous_visited_points);
165
166                // Do BFS through all points on the current layer.
167                while let Some(current_point) = queue.pop() {
168                    let links = self.links_layers[current_point as usize][current_layer].read();
169
170                    for link in links.iter() {
171                        spent_budget += 1;
172
173                        // Flip a coin to decide if the edge is removed or not
174                        let coin_flip = rnd.random_range(0.0..1.0);
175                        if coin_flip < q {
176                            continue;
177                        }
178
179                        let is_selected = point_selection.get_bit(link as usize).unwrap_or(false);
180                        let is_visited = visited.get_bit(link as usize).unwrap_or(false);
181
182                        if !is_visited && is_selected {
183                            visited.set(link as usize, true);
184                            reached_points += 1;
185                            queue.push(link);
186                            previous_visited_points.push(link);
187                        }
188                    }
189                }
190            }
191
192            // Budget exhausted, don't retry.
193            if spent_budget > SUBGRAPH_CONNECTIVITY_SEARCH_BUDGET {
194                break;
195            }
196
197            queue.clear();
198            reached_points = 1; // Reset reached points
199            visited.fill(false);
200        }
201
202        reached_points as f32 / points.len() as f32
203    }

205    pub fn into_graph_layers(
206        self,
207        path: &Path,
208        format_param: GraphLinksFormatParam,
209        on_disk: bool,
210    ) -> OperationResult<GraphLayers> {
211        let links_path = GraphLayers::get_links_path(path, format_param.as_format());
212
213        let edges = Self::links_layers_to_edges(self.links_layers);
214        let links;
215        if on_disk {
216            // Save memory by serializing directly to disk, then re-loading as mmap.
217            atomic_save(&links_path, |writer| {
218                serialize_graph_links(edges, format_param, self.hnsw_m, writer)
219            })?;
220            links = GraphLinks::load_from_file(&links_path, true, format_param.as_format())?;
221        } else {
222            // Since we'll keep it in the RAM anyway, we can afford to build in the RAM too.
223            links = GraphLinks::new_from_edges(edges, format_param, self.hnsw_m)?;
224            atomic_save(&links_path, |writer| writer.write_all(links.as_bytes()))?;
225        }
226
227        let entry_points = self.entry_points.into_inner();
228
229        let data = GraphLayerData {
230            m: self.hnsw_m.m,
231            m0: self.hnsw_m.m0,
232            ef_construct: self.ef_construct,
233            entry_points: Cow::Borrowed(&entry_points),
234        };
235        atomic_save_bin(&GraphLayers::get_path(path), &data)?;
236
237        Ok(GraphLayers {
238            hnsw_m: self.hnsw_m,
239            links,
240            entry_points,
241            visited_pool: self.visited_pool,
242        })
243    }

288    pub fn new_with_params(
289        num_vectors: usize, // Initial number of points in index
290        hnsw_m: HnswM,
291        ef_construct: usize,
292        entry_points_num: usize, // Depends on number of points
293        use_heuristic: bool,
294        reserve: bool,
295    ) -> Self {
296        let links_layers = std::iter::repeat_with(|| {
297            let capacity = if reserve { hnsw_m.m0 } else { 0 };
298            vec![RwLock::new(LinksContainer::with_capacity(capacity))]
299        })

305        Self {
306            max_level: AtomicUsize::new(0),
307            hnsw_m,
308            ef_construct,
309            level_factor: 1.0 / (max(hnsw_m.m, 2) as f64).ln(),
310            use_heuristic,
311            links_layers,
312            entry_points: Mutex::new(EntryPoints::new(entry_points_num)),
313            visited_pool: VisitedPool::new(),
314            ready_list,

335    pub fn merge_from_other(&mut self, other: GraphLayersBuilder) {
336        self.max_level = AtomicUsize::new(max(
337            self.max_level.load(std::sync::atomic::Ordering::Relaxed),
338            other.max_level.load(std::sync::atomic::Ordering::Relaxed),
339        ));
340        let mut visited_list = self.visited_pool.get(self.num_points());
341        if other.links_layers.len() > self.links_layers.len() {
342            self.links_layers
343                .resize_with(other.links_layers.len(), Vec::new);
344        }
345        for (point_id, layers) in other.links_layers.into_iter().enumerate() {
346            let current_layers = &mut self.links_layers[point_id];
347            for (level, other_links) in layers.into_iter().enumerate() {
348                if current_layers.len() <= level {
349                    current_layers.push(other_links);
350                } else {
351                    let other_links = other_links.into_inner();
352                    visited_list.next_iteration();
353                    let mut current_links = current_layers[level].write();
354                    current_links.iter().for_each(|x| {
355                        visited_list.check_and_update_visited(x);
356                    });
357                    for other_link in other_links
358                        .into_vec()
359                        .into_iter()
360                        .filter(|x| !visited_list.check_and_update_visited(*x))
361                    {
362                        current_links.push(other_link);
363                    }
364                }
365            }
366        }
367        self.entry_points
368            .lock()
369            .merge_from_other(other.entry_points.into_inner());
370    }

376    /// Generate random level for a new point, according to geometric distribution
377    pub fn get_random_layer<R>(&self, rng: &mut R) -> usize
378    where
379        R: Rng + ?Sized,
380    {
381        let distribution = Uniform::new(0.0, 1.0).unwrap();
382        let sample: f64 = rng.sample(distribution);
383        let picked_level = -sample.ln() * self.level_factor;
384        picked_level.round() as usize
385    }

391    pub fn set_levels(&mut self, point_id: PointOffsetType, level: usize) {
392        if self.links_layers.len() <= point_id as usize {
393            while self.links_layers.len() <= point_id as usize {
394                self.links_layers.push(vec![]);
395            }
396        }
397        let point_layers = &mut self.links_layers[point_id as usize];
398        while point_layers.len() <= level {
399            let links = LinksContainer::with_capacity(self.hnsw_m.level_m(level));
400            point_layers.push(RwLock::new(links));
401        }
402        self.max_level
403            .fetch_max(level, std::sync::atomic::Ordering::Relaxed);
404    }

406    pub fn link_new_point(&self, point_id: PointOffsetType, mut points_scorer: FilteredScorer) {
407        // Check if there is an suitable entry point
408        //   - entry point level if higher or equal
409        //   - it satisfies filters
410
411        let level = self.get_point_level(point_id);
412
413        let entry_point_opt = self
414            .entry_points
415            .lock()
416            .get_entry_point(|point_id| points_scorer.filters().check_vector(point_id));
417        if let Some(entry_point) = entry_point_opt {
418            let mut level_entry = if entry_point.level > level {
419                // The entry point is higher than a new point
420                // Let's find closest one on same level
421
422                // greedy search for a single closest point
423                self.search_entry(
424                    entry_point.point_id,
425                    entry_point.level,
426                    level,
427                    &mut points_scorer,
428                    &AtomicBool::new(false),
429                )
430                .unwrap()
431            } else {
432                ScoredPointOffset {
433                    idx: entry_point.point_id,
434                    score: points_scorer.score_internal(point_id, entry_point.point_id),
435                }
436            };
437            // minimal common level for entry points
438            let linking_level = min(level, entry_point.level);
439
440            for curr_level in (0..=linking_level).rev() {
441                level_entry = self.link_new_point_on_level(
442                    point_id,
443                    curr_level,
444                    &mut points_scorer,
445                    level_entry,
446                );
447            }
448        } else {
449            // New point is a new empty entry (for this filter, at least)
450            // We can't do much here, so just quit
451        }
452        debug_assert!(
453            !self.ready_list[point_id as usize],
454            "Point {point_id} was already marked as ready"
455        );
456        self.ready_list.set_aliased(point_id as usize, true);
457        self.entry_points
458            .lock()
459            .new_point(point_id, level, |point_id| {
460                points_scorer.filters().check_vector(point_id)
461            });
462    }

464    /// Add a new point using pre-existing links.
465    /// Mutually exclusive with [`Self::link_new_point`].
466    pub fn add_new_point(
467        &self,
468        point_id: PointOffsetType,
469        links_by_level: Vec<Vec<PointOffsetType>>,
470    ) {
471        let level = self.get_point_level(point_id);
472        debug_assert_eq!(links_by_level.len(), level + 1);
473
474        for (level, neighbours) in links_by_level.iter().enumerate() {
475            let mut links = self.links_layers[point_id as usize][level].write();
476            links.fill_from(neighbours.iter().copied());
477        }
478
479        debug_assert!(
480            !self.ready_list[point_id as usize],
481            "Point {point_id} was already marked as ready"
482        );
483        self.ready_list.set_aliased(point_id as usize, true);
484        self.entry_points
485            .lock()
486            .new_point(point_id, level, |_| true);
487    }

489    /// Link a new point on a specific level.
490    /// Returns an entry point for the level below.
491    fn link_new_point_on_level(
492        &self,
493        point_id: PointOffsetType,
494        curr_level: usize,
495        points_scorer: &mut FilteredScorer,
496        mut level_entry: ScoredPointOffset,
497    ) -> ScoredPointOffset {
498        let nearest = self
499            .search_on_level(
500                level_entry,
501                curr_level,
502                self.ef_construct,
503                points_scorer,
504                &AtomicBool::new(false),
505            )
506            .unwrap();
507
508        if let Some(the_nearest) = nearest.iter_unsorted().max() {
509            level_entry = *the_nearest;
510        }
511
512        if self.use_heuristic {
513            self.link_with_heuristic(point_id, curr_level, points_scorer, nearest);
514        } else {
515            self.link_without_heuristic(point_id, curr_level, points_scorer, nearest);
516        }
517
518        level_entry
519    }
520
521    fn link_with_heuristic(
522        &self,
523        point_id: PointOffsetType,
524        curr_level: usize,
525        points_scorer: &FilteredScorer,
526        nearest: FixedLengthPriorityQueue<ScoredPointOffset>,
527    ) {
528        let level_m = self.hnsw_m.level_m(curr_level);
529        let scorer = |a, b| points_scorer.score_internal(a, b);
530
531        let selected_nearest = {
532            let iter = nearest.into_iter_sorted();
533            let mut existing_links = self.links_layers[point_id as usize][curr_level].write();
534            existing_links.fill_from_sorted_with_heuristic(iter, level_m, scorer);
535            existing_links.links().to_vec()
536        };
537
538        // Insert backlinks.
539        let mut items = ItemsBuffer::default();
540        for &other_point in &selected_nearest {
541            self.links_layers[other_point as usize][curr_level]
542                .write()
543                .connect_with_heuristic(point_id, other_point, level_m, scorer, &mut items);
544        }
545    }
546
547    fn link_without_heuristic(
548        &self,
549        point_id: PointOffsetType,
550        curr_level: usize,
551        points_scorer: &FilteredScorer,
552        nearest: FixedLengthPriorityQueue<ScoredPointOffset>,
553    ) {
554        let level_m = self.hnsw_m.level_m(curr_level);
555        let scorer = |a, b| points_scorer.score_internal(a, b);
556        for nearest_point in nearest.iter_unsorted() {
557            {
558                let mut links = self.links_layers[point_id as usize][curr_level].write();
559                links.connect(nearest_point.idx, point_id, level_m, scorer);
560            }
561
562            {
563                let mut links = self.links_layers[nearest_point.idx as usize][curr_level].write();
564                links.connect(point_id, nearest_point.idx, level_m, scorer);
565            }
566        }
567    }

569    /// This function returns average number of links per node in HNSW graph
570    /// on specified level.
571    ///
572    /// Useful for:
573    /// - estimating memory consumption
574    /// - percolation threshold estimation
575    /// - debugging
576    pub fn get_average_connectivity_on_level(&self, level: usize) -> f32 {
577        let mut sum = 0;
578        let mut count = 0;
579        for links in self.links_layers.iter() {
580            if links.len() > level {
581                sum += links[level].read().links().len();
582                count += 1;
583            }
584        }
585        if count == 0 {
586            0.0
587        } else {
588            sum as f32 / count as f32
589        }
590    }

659    fn create_graph_layer<R>(
660        num_vectors: usize,
661        dim: usize,
662        use_heuristic: bool,
663        use_quantization: bool,
664        distance: Distance,
665        rng: &mut R,
666    ) -> (TestRawScorerProducer, GraphLayersBuilder)
667    where
668        R: Rng + ?Sized,
669    {
670        let m = M;
671        let ef_construct = 16;
672        let entry_points_num = 10;
673
674        let vector_holder =
675            TestRawScorerProducer::new(dim, distance, num_vectors, use_quantization, rng);
676
677        let mut graph_layers = GraphLayersBuilder::new(
678            num_vectors,
679            HnswM::new2(m),
680            ef_construct,
681            entry_points_num,
682            use_heuristic,
683        );
684
685        for idx in 0..(num_vectors as PointOffsetType) {
686            let level = graph_layers.get_random_layer(rng);
687            graph_layers.set_levels(idx, level);
688        }
689
690        for idx in 0..(num_vectors as PointOffsetType) {
691            let scorer = vector_holder.internal_scorer(idx);
692            graph_layers.link_new_point(idx, scorer);
693        }
694
695        (vector_holder, graph_layers)
696    }

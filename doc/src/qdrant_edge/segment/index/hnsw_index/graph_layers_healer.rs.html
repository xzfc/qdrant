17pub struct GraphLayersHealer<'a> {
18    links_layers: Vec<LockedLayersContainer>,
19    to_heal: Vec<(PointOffsetType, usize)>,
20    old_to_new: &'a [Option<PointOffsetType>],
21    hnsw_m: HnswM,
22    ef_construct: usize,
23    visited_pool: VisitedPool,

26impl<'a> GraphLayersHealer<'a> {
27    pub fn new(
28        graph_layers: &GraphLayers,
29        old_to_new: &'a [Option<PointOffsetType>],
30        ef_construct: usize,
31    ) -> Self {
32        let mut to_heal = Vec::new();
33        let links_layers = {
34            graph_layers.links.to_edges_impl(|point_id, level| {
35                let level_m = graph_layers.hnsw_m.level_m(level);
36                let mut container = LinksContainer::with_capacity(level_m);
37                container.fill_from(graph_layers.links.links(point_id, level).take(level_m));
38                if container
39                    .iter()
40                    .any(|neighbor| old_to_new[neighbor as usize].is_none())
41                {
42                    to_heal.push((point_id, level));
43                }
44                RwLock::new(container)
45            })
46        };
47
48        Self {
49            links_layers,
50            to_heal,
51            old_to_new,
52            hnsw_m: graph_layers.hnsw_m,
53            ef_construct,
54            visited_pool: VisitedPool::new(),
55        }
56    }

58    fn point_deleted(&self, point: PointOffsetType) -> bool {
59        self.old_to_new[point as usize].is_none()
60    }

62    /// Greedy search for non-deleted points accessible through deleted points.
63    ///
64    /// Unlike the regular search ([`search_on_level`]) which:
65    /// - BFS (queue-based).
66    /// - Deleted points are ignored.
67    /// - Non-deleted points are added into the result AND the search queue.
68    ///
69    /// This method:
70    /// - DFS (stack-based).
71    /// - Deleted points are added into the search queue, but not into the
72    ///   result.
73    /// - Non-deleted points are added into the result, but not into the search
74    ///   queue.
75    ///
76    /// In other words, we search in the scope of deleted points, but
77    /// we want to use points on the border between deleted and non-deleted as candidates
78    /// for the shortcut.
79    ///
80    /// [`search_on_level`]: crate::index::hnsw_index::graph_layers::GraphLayersBase::search_on_level
81    fn search_shortcuts_on_level(
82        &self,
83        offset: PointOffsetType,
84        level: usize,
85        scorer: &dyn RawScorer,
86    ) -> FixedLengthPriorityQueue<ScoredPointOffset> {
87        let mut visited_list = self.visited_pool.get(self.links_layers.len());
88
89        // Result of the search is stored here.
90        let mut nearest = FixedLengthPriorityQueue::<ScoredPointOffset>::new(self.ef_construct);
91
92        let limit = self.hnsw_m.level_m(level);
93
94        let mut neighbours: Vec<PointOffsetType> = Vec::with_capacity(2 * limit);
95        let mut scores_buffer = Vec::with_capacity(limit);
96
97        // Candidates for the search stack.
98        // ToDo: Try later, instead of using stack, we can use proper priority queue
99        // ToDo: So that in the deleted sub-graph we can navigate towards the point with better scores
100        let mut pending = Vec::new();
101
102        // Find entry into "deleted" sub-graph, do not consider non-deleted neighbors
103        // as they already connected to the "healing" point.
104        visited_list.check_and_update_visited(offset);
105        {
106            let links = self.links_layers[offset as usize][level].read();
107            for &point in links.links() {
108                if !self.point_deleted(point) {
109                    visited_list.check_and_update_visited(point);
110                } else {
111                    pending.push(ScoredPointOffset {
112                        idx: point,
113                        score: scorer.score_point(point),
114                    });
115                }
116            }
117        }
118
119        // At this moment `pending` is initialized with at least one deleted point,
120        // now we need to find borders of all "deleted" points sub-graphs
121        while let Some(candidate) = pending.pop() {
122            if nearest.is_full() && candidate.score < nearest.top().unwrap().score {
123                // Stop the search branch early, if it is not promising
124                continue;
125            }
126            if visited_list.check_and_update_visited(candidate.idx) {
127                continue;
128            }
129
130            neighbours.clear();
131            neighbours.extend(
132                self.links_layers[candidate.idx as usize][level]
133                    .read()
134                    .links()
135                    .iter()
136                    .filter(|&&link| !visited_list.check(link)),
137            );
138
139            if scores_buffer.len() < neighbours.len() {
140                scores_buffer.resize(neighbours.len(), 0.0);
141            }
142
143            scorer.score_points(&neighbours, &mut scores_buffer[..neighbours.len()]);
144            for (&idx, &score) in neighbours.iter().zip(&scores_buffer) {
145                if !self.point_deleted(idx) {
146                    // This point is on the "border", as it is reachable from the deleted
147                    // And is not deleted itself
148                    nearest.push(ScoredPointOffset { idx, score });
149                } else {
150                    // This is just another deleted point
151                    pending.push(ScoredPointOffset { idx, score });
152                }
153            }
154        }
155
156        nearest
157    }

159    fn heal_point_on_level(&self, offset: PointOffsetType, level: usize, scorer: &dyn RawScorer) {
160        let level_m = self.hnsw_m.level_m(level);
161
162        // Get current links and filter out deleted ones
163        let mut valid_links = Vec::with_capacity(level_m);
164        valid_links.extend(
165            self.links_layers[offset as usize][level]
166                .read()
167                .links()
168                .iter()
169                .filter(|&&idx| !self.point_deleted(idx)),
170        );
171
172        // First: generate list of candidates using shortcuts search
173        let shortcuts = self.search_shortcuts_on_level(offset, level, scorer);
174
175        // Second: process list of candidates with heuristic
176        let mut container = LinksContainer::with_capacity(level_m);
177        let scorer_fn = |a, b| scorer.score_internal(a, b);
178        container.fill_from_sorted_with_heuristic(
179            shortcuts.into_iter_sorted(),
180            level_m - valid_links.len(),
181            scorer_fn,
182        );
183        for &link in &valid_links {
184            container.push(link);
185        }
186        let container = container.into_vec();
187
188        self.links_layers[offset as usize][level]
189            .write()
190            .fill_from(container.iter().copied());
191
192        // Insert backlinks.
193        let mut items = ItemsBuffer::default();
194        for other_point in container {
195            let mut other_container = self.links_layers[other_point as usize][level].write();
196            if !other_container.iter().any(|link| link == offset) {
197                other_container.connect_with_heuristic(
198                    offset,
199                    other_point,
200                    level_m,
201                    scorer_fn,
202                    &mut items,
203                );
204            }
205        }
206    }

208    pub fn heal(
209        &mut self,
210        pool: &ThreadPool,
211        vector_storage: &VectorStorageEnum,
212        quantized_vectors: Option<&QuantizedVectors>,
213    ) -> OperationResult<()> {
214        pool.install(|| {
215            std::mem::take(&mut self.to_heal)
216                .into_par_iter()
217                .try_for_each(|(offset, level)| {
218                    // Internal operation. No measurements needed.
219                    let internal_hardware_counter = HardwareCounterCell::disposable();
220                    let query = vector_storage
221                        .get_vector::<Random>(offset)
222                        .as_vec_ref()
223                        .into();
224                    let scorer = if let Some(quantized_vectors) = quantized_vectors {
225                        quantized_vectors.raw_scorer(query, internal_hardware_counter)?
226                    } else {
227                        new_raw_scorer(query, vector_storage, internal_hardware_counter)?
228                    };
229                    self.heal_point_on_level(offset, level, scorer.as_ref());
230                    Ok(())
231                })
232        })
233    }

235    pub fn save_into_builder(self, builder: &GraphLayersBuilder) {
236        for (old_offset, layers) in self.links_layers.into_iter().enumerate() {
237            let Some(new_offset) = self.old_to_new[old_offset] else {
238                continue;
239            };
240
241            let links_by_level = layers
242                .into_iter()
243                .map(|layer| {
244                    layer
245                        .into_inner()
246                        .into_vec()
247                        .into_iter()
248                        .filter_map(|link| self.old_to_new[link as usize])
249                        .collect()
250                })
251                .collect();
252
253            builder.add_new_point(new_offset, links_by_level);
254        }
255    }

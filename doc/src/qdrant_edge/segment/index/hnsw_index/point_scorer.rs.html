21/// Scorers composition:
22///
23/// ```plaintext
24///                                                               Metric
25///                                                              ┌─────────────┐
26///                                                              │ - Cosine    │
27///  FilteredScorer      RawScorer          QueryScorer          │ - Dot       │
28/// ┌─────────────────┐ ┌───────────────┐   ┌────────────────┐ ┌─┤ - Euclidean │
29/// │ RawScorer ◄─────┼─┤ QueryScorer ◄─┼───│ Metric ◄───────┼─┘ └─────────────┘
30/// │                 │ └───────────────┘   │                │    - Vector Distance
31/// │ FilterContext   │  - Access patterns  │ Query  ◄───────┼─┐
32/// │                 │                     │                │ │  Query
33/// │ deleted_points  │                     │ TVectorStorage │ │ ┌──────────────────┐
34/// │ deleted_vectors │                     └────────────────┘ └─┤ - RecoQuery      │
35/// └─────────────────┘                                          │ - DiscoverQuery  │
36///                                                              │ - ContextQuery   │
37///                                                              └──────────────────┘
38///                                                              - Scoring logic
39///                                                              - Complex queries
40/// ```
41///
42/// The `BatchFilteredSearcher` contains an array of `RawScorer`s, a common filter and certain parameters.
43///
44/// ```plaintext
45/// BatchFilteredSearcher  RawScorer
46///  ┌─────────────────┐  ┌───────────────┐
47///  │ [RawScorer] ◄───┼──┤ QueryScorer ◄─┼── (ditto)
48///  │                 │  └───────────────┘
49///  │ FilterContext   │
50///  └─────────────────┘
51/// ```
52pub struct FilteredScorer<'a> {
53    raw_scorer: Box<dyn RawScorer + 'a>,
54    filters: ScorerFilters<'a>,
55    /// Temporary buffer for scores.
56    scores_buffer: Vec<ScoreType>,
57}
58
59pub struct ScorerFilters<'a> {
60    filter_context: Option<BoxCow<'a, dyn FilterContext + 'a>>,
61    /// Point deleted flags should be explicitly present as `false`
62    /// for each existing point in the segment.
63    /// If there are no flags for some points, they are considered deleted.
64    /// [`BitSlice`] defining flags for deleted points (and thus these vectors).
65    point_deleted: &'a BitSlice,
66    /// [`BitSlice`] defining flags for deleted vectors in this segment.
67    vec_deleted: &'a BitSlice,
68}
69
70impl<'a> ScorerFilters<'a> {
71    /// Return true if vector satisfies current search context for given point:
72    /// exists, not deleted, and satisfies filter context.
73    pub fn check_vector(&self, point_id: PointOffsetType) -> bool {
74        check_deleted_condition(point_id, self.vec_deleted, self.point_deleted)
75            && self
76                .filter_context
77                .as_ref()
78                .is_none_or(|f| f.check(point_id))
79    }
80
81    fn as_borrowed(&'a self) -> Self {
82        ScorerFilters {
83            filter_context: self.filter_context.as_ref().map(BoxCow::as_borrowed),
84            point_deleted: self.point_deleted,
85            vec_deleted: self.vec_deleted,
86        }
87    }
88}

90pub struct FilteredBytesScorer<'a> {
91    scorer_bytes: &'a dyn QueryScorerBytes,
92    filters: ScorerFilters<'a>,
93}
94
95impl<'a> FilteredBytesScorer<'a> {
96    pub fn score_points(
97        &self,
98        points: &mut Vec<(PointOffsetType, &[u8])>,
99        limit: usize,
100    ) -> impl Iterator<Item = ScoredPointOffset> {
101        points.retain(|(point_id, _)| self.filters.check_vector(*point_id));
102        if limit != 0 {
103            points.truncate(limit);
104        }
105
106        points.iter().map(|&(idx, bytes)| ScoredPointOffset {
107            idx,
108            score: self.scorer_bytes.score_bytes(bytes),
109        })
110    }
111}

113impl<'a> FilteredScorer<'a> {
114    /// Create a new filtered scorer.
115    ///
116    /// If present, `quantized_vectors` will be used for scoring, otherwise `vectors` will be used.
117    pub fn new(
118        query: QueryVector,
119        vectors: &'a VectorStorageEnum,
120        quantized_vectors: Option<&'a QuantizedVectors>,
121        filter_context: Option<BoxCow<'a, dyn FilterContext + 'a>>,
122        point_deleted: &'a BitSlice,
123        hardware_counter: HardwareCounterCell,
124    ) -> OperationResult<Self> {
125        let raw_scorer = match quantized_vectors {
126            Some(quantized_vectors) => quantized_vectors.raw_scorer(query, hardware_counter)?,
127            None => new_raw_scorer(query, vectors, hardware_counter)?,
128        };
129        Ok(FilteredScorer {
130            raw_scorer,
131            filters: ScorerFilters {
132                filter_context,
133                point_deleted,
134                vec_deleted: vectors.deleted_vector_bitslice(),
135            },
136            scores_buffer: Vec::new(),
137        })
138    }
139
140    pub fn new_internal(
141        point_id: PointOffsetType,
142        vectors: &'a VectorStorageEnum,
143        quantized_vectors: Option<&'a QuantizedVectors>,
144        filter_context: Option<BoxCow<'a, dyn FilterContext + 'a>>,
145        point_deleted: &'a BitSlice,
146        hardware_counter: HardwareCounterCell,
147    ) -> OperationResult<Self> {
148        // This is a fallback function, which is used if quantized vector storage
149        // is not capable of reconstructing the query vector.
150        let original_query_fn = || {
151            let query = vectors.get_vector::<Random>(point_id);
152            let query: QueryVector = query.as_vec_ref().into();
153            query
154        };
155        let raw_scorer = match quantized_vectors {
156            Some(quantized_vectors) => quantized_vectors
157                .raw_internal_scorer(point_id, hardware_counter)
158                .or_else(|InternalScorerUnsupported(hardware_counter)| {
159                    quantized_vectors.raw_scorer(original_query_fn(), hardware_counter)
160                })?,
161            None => {
162                let query = original_query_fn();
163                new_raw_scorer(query, vectors, hardware_counter)?
164            }
165        };
166        Ok(FilteredScorer {
167            raw_scorer,
168            filters: ScorerFilters {
169                filter_context,
170                point_deleted,
171                vec_deleted: vectors.deleted_vector_bitslice(),
172            },
173            scores_buffer: Vec::new(),
174        })
175    }

215    /// Filters and calculates scores for the given slice of points IDs.
216    ///
217    /// For performance reasons this method mutates `point_ids`.
218    ///
219    /// # Arguments
220    ///
221    /// * `point_ids` - list of points to score.
222    ///   **Warning**: This input will be wrecked during the execution.
223    /// * `limit` - limits the number of points to process after filtering.
224    ///   `0` means no limit.
225    pub fn score_points(
226        &mut self,
227        point_ids: &mut Vec<PointOffsetType>,
228        limit: usize,
229    ) -> impl Iterator<Item = ScoredPointOffset> {
230        point_ids.retain(|point_id| self.filters.check_vector(*point_id));
231        if limit != 0 {
232            point_ids.truncate(limit);
233        }
234
235        self.score_points_unfiltered(point_ids)
236    }
237
238    pub fn score_points_unfiltered(
239        &mut self,
240        point_ids: &[PointOffsetType],
241    ) -> impl Iterator<Item = ScoredPointOffset> {
242        if self.scores_buffer.len() < point_ids.len() {
243            self.scores_buffer.resize(point_ids.len(), 0.0);
244        }
245
246        self.raw_scorer
247            .score_points(point_ids, &mut self.scores_buffer[..point_ids.len()]);
248
249        std::iter::zip(point_ids, &self.scores_buffer)
250            .map(|(&idx, &score)| ScoredPointOffset { idx, score })
251    }
252
253    pub fn score_point(&self, point_id: PointOffsetType) -> ScoreType {
254        self.raw_scorer.score_point(point_id)
255    }
256
257    pub fn score_internal(&self, point_a: PointOffsetType, point_b: PointOffsetType) -> ScoreType {
258        self.raw_scorer.score_internal(point_a, point_b)
259    }
260}

262// We keep each scorer with its queue to reduce allocations and improve data locality.
263struct BatchSearch<'a> {
264    raw_scorer: Box<dyn RawScorer + 'a>,
265    pq: FixedLengthPriorityQueue<ScoredPointOffset>,
266}
267
268pub struct BatchFilteredSearcher<'a> {
269    scorer_batch: SmallVec<[BatchSearch<'a>; 1]>,
270    filters: ScorerFilters<'a>,
271}
272
273impl<'a> BatchFilteredSearcher<'a> {
274    /// Create a new batch filtered searcher.
275    ///
276    /// If present, `quantized_vectors` will be used for scoring, otherwise `vectors` will be used.
277    pub fn new(
278        queries: &[&QueryVector],
279        vectors: &'a VectorStorageEnum,
280        quantized_vectors: Option<&'a QuantizedVectors>,
281        filter_context: Option<BoxCow<'a, dyn FilterContext + 'a>>,
282        top: usize,
283        point_deleted: &'a BitSlice,
284        hardware_counter: HardwareCounterCell,
285    ) -> OperationResult<Self> {
286        let scorer_batch = queries
287            .iter()
288            .map(|&query| {
289                let query = query.to_owned();
290                let hardware_counter = hardware_counter.fork();
291                let raw_scorer = match quantized_vectors {
292                    Some(quantized_vectors) => {
293                        quantized_vectors.raw_scorer(query, hardware_counter)
294                    }
295                    None => new_raw_scorer(query, vectors, hardware_counter),
296                };
297                let pq = FixedLengthPriorityQueue::new(top);
298                raw_scorer.map(|raw_scorer| BatchSearch { raw_scorer, pq })
299            })
300            .collect::<Result<_, _>>()?;
301        let filters = ScorerFilters {
302            filter_context,
303            point_deleted,
304            vec_deleted: vectors.deleted_vector_bitslice(),
305        };
306        Ok(Self {
307            scorer_batch,
308            filters,
309        })
310    }

312    /// Create a new batched filtered searcher for testing purposes.

314    /// # Panics

349    pub fn peek_top_all(
350        self,
351        is_stopped: &AtomicBool,
352        deferred_internal_id: Option<PointOffsetType>,
353    ) -> CancellableResult<Vec<Vec<ScoredPointOffset>>> {
354        let iter = self
355            .filters
356            .point_deleted
357            .iter_zeros()
358            .map(|p| p as PointOffsetType)
359            .take_while(|&point_id| {
360                // Early exit if we hit the max point ID (e.g. a deferred point).
361                point_id < deferred_internal_id.unwrap_or(PointOffsetType::MAX)
362            });
363
364        self.peek_top_iter(iter, is_stopped)
365    }
366
367    /// This function expects deferred points to be already filtered from the iterator.
368    pub fn peek_top_iter(
369        mut self,
370        mut points: impl Iterator<Item = PointOffsetType>,
371        is_stopped: &AtomicBool,
372    ) -> CancellableResult<Vec<Vec<ScoredPointOffset>>> {
373        // Reuse the same buffer for all chunks, to avoid reallocation
374        let mut chunk = [0; VECTOR_READ_BATCH_SIZE];
375        let mut scores_buffer = [0.0; VECTOR_READ_BATCH_SIZE];
376
377        loop {
378            check_process_stopped(is_stopped)?;
379
380            let mut chunk_size = 0;
381            for point_id in &mut points {
382                check_process_stopped(is_stopped)?;
383
384                if !self.filters.check_vector(point_id) {
385                    continue;
386                }
387                chunk[chunk_size] = point_id;
388                chunk_size += 1;
389                if chunk_size == VECTOR_READ_BATCH_SIZE {
390                    break;
391                }
392            }
393
394            if chunk_size == 0 {
395                break;
396            }
397
398            // Switching the loops improves batching performance, but slightly degrades single-query performance.
399            for BatchSearch { raw_scorer, pq } in &mut self.scorer_batch {
400                raw_scorer.score_points(&chunk[..chunk_size], &mut scores_buffer[..chunk_size]);
401
402                for i in 0..chunk_size {
403                    pq.push(ScoredPointOffset {
404                        idx: chunk[i],
405                        score: scores_buffer[i],
406                    });
407                }
408            }
409        }
410
411        let results = self
412            .scorer_batch
413            .into_iter()
414            .map(|BatchSearch { pq, .. }| pq.into_sorted_vec())
415            .collect();
416        Ok(results)
417    }

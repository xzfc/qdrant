205    pub fn build<R: Rng + ?Sized>(
206        open_args: HnswIndexOpenArgs<'_>,
207        build_args: VectorIndexBuildArgs<'_, R>,
208    ) -> OperationResult<Self> {
209        if HnswGraphConfig::get_config_path(open_args.path).exists()
210            || GraphLayers::get_path(open_args.path).exists()
211        {
212            log::warn!(
213                "HNSW index already exists at {:?}, skipping building",
214                open_args.path
215            );
216            debug_assert!(false);
217            return Self::open(open_args);
218        }
219
220        let HnswIndexOpenArgs {
221            path,
222            id_tracker,
223            vector_storage,
224            quantized_vectors,
225            payload_index,
226            hnsw_config,
227        } = open_args;
228        let VectorIndexBuildArgs {
229            permit,
230            old_indices,
231            gpu_device,
232            rng,
233            stopped,
234            hnsw_global_config,
235            feature_flags,
236            progress,
237        } = build_args;
238
239        fs::create_dir_all(path)?;
240
241        let id_tracker_ref = id_tracker.borrow();
242        let vector_storage_ref = vector_storage.borrow();
243        let quantized_vectors_ref = quantized_vectors.borrow();
244        let payload_index_ref = payload_index.borrow();
245
246        let total_vector_count = vector_storage_ref.total_vector_count();
247
248        let full_scan_threshold = vector_storage_ref
249            .size_of_available_vectors_in_bytes()
250            .checked_div(total_vector_count)
251            .and_then(|avg_vector_size| {
252                hnsw_config
253                    .full_scan_threshold
254                    .saturating_mul(BYTES_IN_KB)
255                    .checked_div(avg_vector_size)
256            })
257            .unwrap_or(1);
258
259        let mut config = HnswGraphConfig::new(
260            hnsw_config.m,
261            hnsw_config.ef_construct,
262            full_scan_threshold,
263            hnsw_config.max_indexing_threads,
264            hnsw_config.payload_m,
265            total_vector_count,
266        );
267
268        #[allow(unused_mut)]
269        let mut build_main_graph = config.m > 0;
270        if !build_main_graph {
271            debug!("skip building main HNSW graph");
272        }
273
274        let payload_m = HnswM::new(
275            config.payload_m.unwrap_or(config.m),
276            config.payload_m0.unwrap_or(config.m0),
277        );
278
279        // Progress subtasks
280        let progress_migrate = build_main_graph.then(|| progress.subtask("migrate"));
281        let progress_main_graph = build_main_graph.then(|| progress.subtask("main_graph"));
282        let additional_links_params: Option<(ProgressTracker, Vec<(ProgressTracker, JsonPath)>)> =
283            (payload_m.m > 0)
284                .then(|| payload_index_ref.indexed_fields())
285                .filter(|fields| !fields.is_empty())
286                .map(|fields| {
287                    let progress_additional_links = progress.subtask("additional_links");
288                    let fields = fields
289                        .into_iter()
290                        .filter_map(|(field, payload_schema)| {
291                            let subtask_name = format!("{}:{field}", payload_schema.name());
292                            if payload_schema.enable_hnsw() {
293                                Some((progress_additional_links.subtask(subtask_name), field))
294                            } else {
295                                debug!("enable_hnsw=false. Skip building additional index for field {}", &field);
296                                None
297                            }
298                        })
299                        .collect::<Vec<_>>();
300                    (progress_additional_links, fields)
301                });
302
303        let old_index = old_indices
304            .iter()
305            .filter_map(|old_index| {
306                OldIndexCandidate::evaluate(
307                    &feature_flags,
308                    old_index,
309                    &config,
310                    hnsw_global_config,
311                    &vector_storage_ref,
312                    &quantized_vectors_ref,
313                    id_tracker_ref.deref(),
314                )
315            })
316            .max_by_key(|old_index| old_index.valid_points);
317
318        // Build main index graph
319        let deleted_bitslice = vector_storage_ref.deleted_vector_bitslice();
320
321        #[cfg(feature = "gpu")]
322        let gpu_name_postfix = if let Some(gpu_device) = gpu_device {
323            format!(" and GPU {}", gpu_device.device().name())
324        } else {
325            Default::default()
326        };
327        #[cfg(not(feature = "gpu"))]
328        let gpu_name_postfix = "";
329        #[cfg(not(feature = "gpu"))]
330        let _ = gpu_device;
331        debug!(
332            "building HNSW for {total_vector_count} vectors with {} CPUs{gpu_name_postfix}",
333            permit.num_cpus,
334        );
335
336        let num_entries = std::cmp::max(
337            1,
338            total_vector_count
339                .checked_div(full_scan_threshold)
340                .unwrap_or(0)
341                * 10,
342        );
343        let mut graph_layers_builder = GraphLayersBuilder::new(
344            total_vector_count,
345            HnswM::new(config.m, config.m0),
346            config.ef_construct,
347            num_entries,
348            HNSW_USE_HEURISTIC,
349        );
350
351        let pool = rayon::ThreadPoolBuilder::new()
352            .thread_name(|idx| format!("hnsw-build-{idx}"))
353            .num_threads(permit.num_cpus as usize)
354            .spawn_handler(|thread| {
355                let mut b = thread::Builder::new();
356                if let Some(name) = thread.name() {
357                    b = b.name(name.to_owned());
358                }
359                if let Some(stack_size) = thread.stack_size() {
360                    b = b.stack_size(stack_size);
361                }
362                b.spawn(|| {
363                    // On Linux, use lower thread priority so we interfere less with serving traffic
364                    #[cfg(target_os = "linux")]
365                    if let Err(err) = linux_low_thread_priority() {
366                        log::debug!(
367                            "Failed to set low thread priority for HNSW building, ignoring: {err}"
368                        );
369                    }
370
371                    thread.run()
372                })?;
373                Ok(())
374            })
375            .build()?;
376
377        let old_index = old_index.map(|old_index| old_index.reuse(total_vector_count));
378
379        let mut indexed_vectors = 0;
380        for vector_id in id_tracker_ref.iter_internal_excluding(deleted_bitslice) {
381            check_process_stopped(stopped)?;
382            indexed_vectors += 1;
383
384            let level = old_index
385                .as_ref()
386                .and_then(|old_index| old_index.point_level(vector_id))
387                .unwrap_or_else(|| graph_layers_builder.get_random_layer(rng));
388            graph_layers_builder.set_levels(vector_id, level);
389        }
390
391        // Try to build the main graph on GPU if possible.
392        // Store created gpu vectors to reuse them for payload links.
393        #[cfg(feature = "gpu")]
394        let gpu_vectors = if build_main_graph {
395            let timer = std::time::Instant::now();
396            let gpu_vectors = Self::create_gpu_vectors(
397                gpu_device,
398                &vector_storage_ref,
399                &quantized_vectors_ref,
400                stopped,
401            )?;
402            if let Some(gpu_constructed_graph) = Self::build_main_graph_on_gpu(
403                id_tracker_ref.deref(),
404                &vector_storage_ref,
405                &quantized_vectors_ref,
406                gpu_vectors.as_ref(),
407                &graph_layers_builder,
408                deleted_bitslice,
409                num_entries,
410                stopped,
411            )? {
412                graph_layers_builder = gpu_constructed_graph;
413                build_main_graph = false;
414                debug!("{FINISH_MAIN_GRAPH_LOG_MESSAGE} {:?}", timer.elapsed());
415            }
416            gpu_vectors
417        } else {
418            None
419        };
420
421        if build_main_graph {
422            let progress_main_graph = progress_main_graph.unwrap();
423            let progress_migrate = progress_migrate.unwrap();
424
425            let mut ids = Vec::with_capacity(total_vector_count);
426            let mut first_few_ids = Vec::with_capacity(SINGLE_THREADED_HNSW_BUILD_THRESHOLD);
427
428            let mut ids_iter = id_tracker_ref.iter_internal_excluding(deleted_bitslice);
429            if let Some(old_index) = old_index {
430                progress_migrate.start();
431
432                let timer = std::time::Instant::now();
433
434                let mut healer = GraphLayersHealer::new(
435                    old_index.graph(),
436                    &old_index.old_to_new,
437                    config.ef_construct,
438                );
439                let old_vector_storage = old_index.index.vector_storage.borrow();
440                let old_quantized_vectors = old_index.index.quantized_vectors.borrow();
441                healer.heal(&pool, &old_vector_storage, old_quantized_vectors.as_ref())?;
442                healer.save_into_builder(&graph_layers_builder);
443
444                for vector_id in ids_iter {
445                    if old_index.new_to_old[vector_id as usize].is_none() {
446                        if first_few_ids.len() < SINGLE_THREADED_HNSW_BUILD_THRESHOLD {
447                            first_few_ids.push(vector_id);
448                        } else {
449                            ids.push(vector_id);
450                        }
451                    }
452                }
453
454                debug!("Migrated in {:?}", timer.elapsed());
455            } else {
456                first_few_ids.extend(ids_iter.by_ref().take(SINGLE_THREADED_HNSW_BUILD_THRESHOLD));
457                ids.extend(ids_iter);
458            }
459            drop(progress_migrate);
460
461            let timer = std::time::Instant::now();
462
463            progress_main_graph.start();
464            let counter = progress_main_graph
465                .track_progress(Some(first_few_ids.len() as u64 + ids.len() as u64));
466            let counter = counter.deref();
467
468            let insert_point = |vector_id| {
469                check_process_stopped(stopped)?;
470                // No need to accumulate hardware, since this is an internal operation
471                let internal_hardware_counter = HardwareCounterCell::disposable();
472
473                let points_scorer = FilteredScorer::new_internal(
474                    vector_id,
475                    vector_storage_ref.deref(),
476                    quantized_vectors_ref.as_ref(),
477                    None,
478                    id_tracker_ref.deleted_point_bitslice(),
479                    internal_hardware_counter,
480                )?;
481
482                graph_layers_builder.link_new_point(vector_id, points_scorer);
483
484                counter.fetch_add(1, Ordering::Relaxed);
485
486                Ok::<_, OperationError>(())
487            };
488
489            for vector_id in first_few_ids {
490                insert_point(vector_id)?;
491            }
492
493            if !ids.is_empty() {
494                pool.install(|| ids.into_par_iter().try_for_each(insert_point))?;
495            }
496
497            drop(progress_main_graph);
498            debug!("{FINISH_MAIN_GRAPH_LOG_MESSAGE} {:?}", timer.elapsed());
499        } else {
500            drop(old_index);
501        }
502
503        if let Some((progress_additional_links, indexed_fields)) = additional_links_params {
504            progress_additional_links.start();
505
506            // Calculate true average number of links per vertex in the HNSW graph
507            // to better estimate percolation threshold
508            let average_links_per_0_level =
509                graph_layers_builder.get_average_connectivity_on_level(0);
510            let average_links_per_0_level_int = (average_links_per_0_level as usize).max(1);
511
512            // Estimate connectivity of the main graph
513            let all_points = id_tracker_ref
514                .iter_internal_excluding(deleted_bitslice)
515                .collect::<Vec<_>>();
516
517            // According to percolation theory, random graph becomes disconnected
518            // if 1/K points are left, where K is average number of links per point
519            // So we need to sample connectivity relative to this bifurcation point, but
520            // not exactly at 1/K, as at this point graph is very sensitive to noise.
521            //
522            // Instead, we choose sampling point at 2/K, which expects graph to still be
523            // mostly connected, but still have some measurable disconnected components.
524
525            let percolation = 1. - 2. / (average_links_per_0_level_int as f32);
526
527            let required_connectivity = if average_links_per_0_level_int >= 4 {
528                let global_graph_connectivity = [
529                    graph_layers_builder.subgraph_connectivity(&all_points, percolation),
530                    graph_layers_builder.subgraph_connectivity(&all_points, percolation),
531                    graph_layers_builder.subgraph_connectivity(&all_points, percolation),
532                ];
533
534                debug!("graph connectivity: {global_graph_connectivity:?} @ {percolation}");
535
536                global_graph_connectivity
537                    .iter()
538                    .copied()
539                    .max_by(|a, b| a.partial_cmp(b).unwrap())
540            } else {
541                // Main graph is too small to estimate connectivity,
542                // we can't shortcut sub-graph building
543                None
544            };
545
546            let mut indexed_vectors_set = if config.m != 0 {
547                // Every vector is already indexed in the main graph, so skip counting.
548                BitVec::new()
549            } else {
550                BitVec::repeat(false, total_vector_count)
551            };
552
553            let visited_pool = VisitedPool::new();
554            let mut block_filter_list = visited_pool.get(total_vector_count);
555
556            #[cfg(feature = "gpu")]
557            let mut gpu_insert_context = if let Some(gpu_vectors) = gpu_vectors.as_ref() {
558                Some(GpuInsertContext::new(
559                    gpu_vectors,
560                    get_gpu_groups_count(),
561                    payload_m,
562                    config.ef_construct,
563                    false,
564                    1..=GPU_MAX_VISITED_FLAGS_FACTOR,
565                )?)
566            } else {
567                None
568            };
569            #[cfg(not(feature = "gpu"))]
570            let mut gpu_insert_context = None;
571
572            for (index_pos, (field_progress, field)) in indexed_fields.into_iter().enumerate() {
573                field_progress.start();
574
575                debug!("building additional index for field {}", &field);
576
577                let is_tenant = payload_index_ref.is_tenant(&field);
578
579                // It is expected, that graph will become disconnected less than
580                // $1/m$ points left.
581                // So blocks larger than $1/m$ are not needed.
582                // We add multiplier for the extra safety.
583                const PERCOLATION_MULTIPLIER: usize = 4;
584                let max_block_size = if config.m > 0 {
585                    total_vector_count / average_links_per_0_level_int * PERCOLATION_MULTIPLIER
586                } else {
587                    usize::MAX
588                };
589
590                let counter = field_progress.track_progress(None);
591
592                for payload_block in payload_index_ref.payload_blocks(&field, full_scan_threshold) {
593                    check_process_stopped(stopped)?;
594
595                    if payload_block.cardinality > max_block_size {
596                        continue;
597                    }
598
599                    let points_to_index = Self::condition_points(
600                        payload_block.condition,
601                        id_tracker_ref.deref(),
602                        &payload_index_ref,
603                        &vector_storage_ref,
604                        stopped,
605                    );
606
607                    // This is a heuristic to skip building graph for mostly deleted blocks.
608                    // It might be, that majority of points do not actually have vectors
609                    // (vectors marked as deleted), so we can avoid building graph for such blocks.
610                    //
611                    // FYI: query heuristic does account
612                    // for deleted vectors via [`adjust_to_available_vectors`]
613                    const DELETED_POINTS_FACTOR: usize = 4; // allow block to have up to 75% of deleted points and still be indexed
614
615                    if points_to_index.len() <= full_scan_threshold / DELETED_POINTS_FACTOR {
616                        continue;
617                    }
618
619                    if !is_tenant
620                        && index_pos > 0
621                        && let Some(required_connectivity) = required_connectivity
622                    {
623                        // Always build for tenants
624                        let graph_connectivity = graph_layers_builder
625                            .subgraph_connectivity(&points_to_index, percolation);
626
627                        if graph_connectivity >= required_connectivity {
628                            trace!(
629                                "skip building additional HNSW links for {field}, connectivity {graph_connectivity:.4} >= {required_connectivity:.4}"
630                            );
631                            continue;
632                        }
633                        trace!("graph connectivity: {graph_connectivity} for {field}");
634                    }
635
636                    // ToDo: reuse graph layer for same payload
637                    let mut additional_graph = GraphLayersBuilder::new_with_params(
638                        total_vector_count,
639                        payload_m,
640                        config.ef_construct,
641                        1,
642                        HNSW_USE_HEURISTIC,
643                        false,
644                    );
645
646                    Self::build_filtered_graph(
647                        id_tracker_ref.deref(),
648                        &vector_storage_ref,
649                        &quantized_vectors_ref,
650                        &mut gpu_insert_context,
651                        &payload_index_ref,
652                        &pool,
653                        stopped,
654                        &mut additional_graph,
655                        points_to_index,
656                        &mut block_filter_list,
657                        &mut indexed_vectors_set,
658                        &counter,
659                    )?;
660                    graph_layers_builder.merge_from_other(additional_graph);
661                }
662            }
663
664            let indexed_payload_vectors = indexed_vectors_set.count_ones();
665
666            debug_assert!(indexed_vectors >= indexed_payload_vectors || config.m == 0);
667            indexed_vectors = indexed_vectors.max(indexed_payload_vectors);
668            debug_assert!(indexed_payload_vectors <= total_vector_count);
669        } else {
670            debug!("skip building additional HNSW links");
671        }
672
673        config.indexed_vector_count.replace(indexed_vectors);
674
675        // Always skip loading graph to RAM on build
676        // as it will be discarded anyway
677        let is_on_disk = true;
678
679        let graph_links_vectors = hnsw_config
680            .inline_storage
681            .unwrap_or_default()
682            .then(|| {
683                // NOTE: the configuration is silently ignored if try_new fails.
684                StorageGraphLinksVectors::try_new(
685                    &vector_storage_ref,
686                    quantized_vectors_ref.as_ref(),
687                )
688            })
689            .flatten();
690        let format_param = match graph_links_vectors.as_ref() {
691            Some(v) => GraphLinksFormatParam::CompressedWithVectors(v),
692            None => GraphLinksFormatParam::Compressed,
693        };
694
695        let graph: GraphLayers =
696            graph_layers_builder.into_graph_layers(path, format_param, is_on_disk)?;
697
698        #[cfg(debug_assertions)]
699        {
700            for (idx, deleted) in deleted_bitslice.iter().enumerate() {
701                if *deleted {
702                    graph.links.for_each_link(idx as PointOffsetType, 0, |_| {
703                        panic!("Deleted point in the graph");
704                    });
705                }
706            }
707        }
708
709        debug!("finish additional payload field indexing");
710
711        config.save(&HnswGraphConfig::get_config_path(path))?;
712
713        drop(id_tracker_ref);
714        drop(vector_storage_ref);
715        drop(quantized_vectors_ref);
716        drop(payload_index_ref);
717
718        Ok(HNSWIndex {
719            id_tracker,
720            vector_storage,
721            quantized_vectors,
722            payload_index,
723            config,
724            path: path.to_owned(),
725            graph,
726            searches_telemetry: HNSWSearchesTelemetry::new(),
727            is_on_disk,
728        })
729    }

731    /// Get list of points for indexing, associated with payload block filtering condition
732    fn condition_points(
733        condition: FieldCondition,
734        id_tracker: &IdTrackerEnum,
735        payload_index: &StructPayloadIndex,
736        vector_storage: &VectorStorageEnum,
737        stopped: &AtomicBool,
738    ) -> Vec<PointOffsetType> {
739        let filter = Filter::new_must(Field(condition));
740
741        let disposed_hw_counter = HardwareCounterCell::disposable(); // Internal operation. No measurements needed
742
743        let deleted_bitslice = vector_storage.deleted_vector_bitslice();
744
745        let cardinality_estimation =
746            payload_index.estimate_cardinality(&filter, &disposed_hw_counter);
747
748        payload_index
749            .iter_filtered_points(
750                &filter,
751                id_tracker,
752                &cardinality_estimation,
753                &disposed_hw_counter,
754                stopped,
755                None,
756            )
757            .filter(|&point_id| !deleted_bitslice.get_bit(point_id as usize).unwrap_or(false))
758            .collect()
759    }

761    #[allow(clippy::too_many_arguments)]
762    #[allow(unused_variables)]
763    #[allow(clippy::needless_pass_by_ref_mut)]
764    fn build_filtered_graph(
765        id_tracker: &IdTrackerEnum,
766        vector_storage: &VectorStorageEnum,
767        quantized_vectors: &Option<QuantizedVectors>,
768        #[allow(unused_variables)] gpu_insert_context: &mut Option<GpuInsertContext<'_>>,
769        payload_index: &StructPayloadIndex,
770        pool: &ThreadPool,
771        stopped: &AtomicBool,
772        graph_layers_builder: &mut GraphLayersBuilder,
773        points_to_index: Vec<PointOffsetType>,
774        block_filter_list: &mut VisitedListHandle,
775        indexed_vectors_set: &mut BitVec,
776        counter: &AtomicU64,
777    ) -> OperationResult<()> {
778        block_filter_list.next_iteration();
779
780        for block_point_id in points_to_index.iter().copied() {
781            block_filter_list.check_and_update_visited(block_point_id);
782            if !indexed_vectors_set.is_empty() {
783                indexed_vectors_set.set(block_point_id as usize, true);
784            }
785        }
786
787        #[cfg(feature = "gpu")]
788        if let Some(gpu_constructed_graph) = Self::build_filtered_graph_on_gpu(
789            id_tracker,
790            vector_storage,
791            quantized_vectors,
792            gpu_insert_context.as_mut(),
793            graph_layers_builder,
794            block_filter_list,
795            &points_to_index,
796            stopped,
797        )? {
798            *graph_layers_builder = gpu_constructed_graph;
799            return Ok(());
800        }
801
802        let insert_points = |block_point_id| {
803            check_process_stopped(stopped)?;
804
805            // This hardware counter can be discarded, since it is only used for internal operations
806            let internal_hardware_counter = HardwareCounterCell::disposable();
807
808            let block_condition_checker = BuildConditionChecker {
809                filter_list: block_filter_list,
810                current_point: block_point_id,
811            };
812            let points_scorer = FilteredScorer::new_internal(
813                block_point_id,
814                vector_storage,
815                quantized_vectors.as_ref(),
816                Some(BoxCow::Borrowed(&block_condition_checker)),
817                id_tracker.deleted_point_bitslice(),
818                internal_hardware_counter,
819            )?;
820
821            graph_layers_builder.link_new_point(block_point_id, points_scorer);
822
823            counter.fetch_add(1, Ordering::Relaxed);
824
825            Ok::<_, OperationError>(())
826        };
827
828        let first_points = points_to_index
829            .len()
830            .min(SINGLE_THREADED_HNSW_BUILD_THRESHOLD);
831
832        // First index points in single thread so ensure warm start for parallel indexing process
833        for point_id in points_to_index[..first_points].iter().copied() {
834            insert_points(point_id)?;
835        }
836        // Once initial structure is built, index remaining points in parallel
837        // So that each thread will insert points in different parts of the graph,
838        // it is less likely that they will compete for the same locks
839        if points_to_index.len() > first_points {
840            pool.install(|| {
841                points_to_index
842                    .into_par_iter()
843                    .skip(first_points)
844                    .try_for_each(insert_points)
845            })?;
846        }
847        Ok(())
848    }

935    #[allow(clippy::too_many_arguments)]
936    fn build_graph_on_gpu<'a, 'b>(
937        gpu_insert_context: Option<&mut GpuInsertContext<'b>>,
938        graph_layers_builder: &GraphLayersBuilder,
939        points_to_index: impl Iterator<Item = PointOffsetType>,
940        entry_points_num: usize,
941        points_scorer_builder: impl Fn(PointOffsetType) -> OperationResult<FilteredScorer<'a>>
942        + Send
943        + Sync,
944        stopped: &AtomicBool,
945    ) -> OperationResult<Option<GraphLayersBuilder>> {

947            let gpu_constructed_graph = build_hnsw_on_gpu(
948                gpu_insert_context,
949                graph_layers_builder,
950                get_gpu_groups_count(),
951                entry_points_num,
952                SINGLE_THREADED_HNSW_BUILD_THRESHOLD,
953                points_to_index.collect::<Vec<_>>(),
954                points_scorer_builder,
955                stopped,
956            );
957
958            // GPU construction does not return an error. If it fails, it will fall back to CPU.
959            // To cover stopping case, we need to check stopping flag here.
960            check_process_stopped(stopped)?;
961
962            match gpu_constructed_graph {
963                Ok(gpu_constructed_graph) => Ok(Some(gpu_constructed_graph)),
964                Err(gpu_error) => {
965                    log::warn!("Failed to build HNSW on GPU: {gpu_error}. Falling back to CPU.");
966                    Ok(None)
967                }
968            }
969        } else {
970            Ok(None)
971        }
972    }

1011    #[allow(clippy::too_many_arguments)]
1012    fn search_with_graph(
1013        &self,
1014        vector: &QueryVector,
1015        filter: Option<&Filter>,
1016        top: usize,
1017        params: Option<&SearchParams>,
1018        custom_entry_points: Option<&[PointOffsetType]>,
1019        vector_query_context: &VectorQueryContext,
1020    ) -> OperationResult<Vec<ScoredPointOffset>> {
1021        let ef = params
1022            .and_then(|params| params.hnsw_ef)
1023            .unwrap_or(self.config.ef);
1024        let acorn_enabled = params
1025            .and_then(|params| params.acorn)
1026            .is_some_and(|acorn| acorn.enable);
1027        let acorn_max_selectivity = params
1028            .and_then(|params| params.acorn)
1029            .and_then(|acorn| acorn.max_selectivity)
1030            .map_or(ACORN_MAX_SELECTIVITY_DEFAULT, |v| *v);
1031
1032        let is_stopped = vector_query_context.is_stopped();
1033
1034        let id_tracker = self.id_tracker.borrow();
1035        let payload_index = self.payload_index.borrow();
1036        let vector_storage = self.vector_storage.borrow();
1037        let quantized_vectors = self.quantized_vectors.borrow();
1038
1039        let deleted_points = vector_query_context
1040            .deleted_points()
1041            .unwrap_or_else(|| id_tracker.deleted_point_bitslice());
1042
1043        let hw_counter = vector_query_context.hardware_counter();
1044        let oversampled_top = get_oversampled_top(quantized_vectors.as_ref(), params, top);
1045
1046        let mut algorithm = SearchAlgorithm::Hnsw;
1047        if acorn_enabled
1048            && self.config.m0 != 0
1049            && let Some(filter) = filter
1050        {
1051            // NOTE: technically we also might want to use ACORN for unfiltered
1052            // searches for segments with a lot of deleted points. But in
1053            // practice, such segments most likely to be picked by an optimizer
1054            // soon.
1055
1056            let available_vector_count = vector_storage.available_vector_count();
1057            let selectivity = if available_vector_count == 0 {
1058                1.0
1059            } else {
1060                let query_point_cardinality =
1061                    payload_index.estimate_cardinality(filter, &hw_counter);
1062                let query_cardinality = adjust_to_available_vectors(
1063                    query_point_cardinality,
1064                    available_vector_count,
1065                    id_tracker.available_point_count(),
1066                );
1067                query_cardinality.exp as f64 / available_vector_count as f64
1068            };
1069            if selectivity <= acorn_max_selectivity {
1070                algorithm = SearchAlgorithm::Acorn;
1071            }
1072        }
1073
1074        let search_with_vectors = || -> OperationResult<Option<Vec<ScoredPointOffset>>> {
1075            match algorithm {
1076                SearchAlgorithm::Hnsw => (),
1077                // ACORN is not implemented for graph with vectors yet (but possible)
1078                SearchAlgorithm::Acorn => return Ok(None),
1079            }
1080            if !self.graph.has_inline_vectors()
1081                || !is_quantized_search(quantized_vectors.as_ref(), params)
1082            {
1083                return Ok(None);
1084            }
1085            let Some(quantized_vectors) = quantized_vectors.as_ref() else {
1086                return Ok(None);
1087            };
1088
1089            // Quantized vectors are "link vectors"
1090            let link_scorer_filtered = FilteredScorer::new(
1091                vector.to_owned(),
1092                &vector_storage,
1093                Some(quantized_vectors),
1094                filter.map(|f| BoxCow::Owned(payload_index.filter_context(f, &hw_counter))),
1095                deleted_points,
1096                vector_query_context.hardware_counter(),
1097            )?;
1098            let Some(link_scorer_filtered_bytes) = link_scorer_filtered.scorer_bytes() else {
1099                return Ok(None);
1100            };
1101
1102            // Full vectors are "base vectors"
1103            let base_scorer = new_raw_scorer(
1104                vector.to_owned(),
1105                &vector_storage,
1106                vector_query_context.hardware_counter(),
1107            )?;
1108            let Some(base_scorer_bytes) = base_scorer.scorer_bytes() else {
1109                return Ok(None);
1110            };
1111
1112            Ok(Some(self.graph.search_with_vectors(
1113                top,
1114                std::cmp::max(ef, oversampled_top),
1115                &link_scorer_filtered,
1116                &link_scorer_filtered_bytes,
1117                base_scorer_bytes,
1118                custom_entry_points,
1119                &vector_query_context.is_stopped(),
1120            )?))
1121        };
1122
1123        let regular_search = || -> OperationResult<Vec<ScoredPointOffset>> {
1124            let filter_context = filter.map(|f| payload_index.filter_context(f, &hw_counter));
1125            let points_scorer = Self::construct_search_scorer(
1126                vector,
1127                &vector_storage,
1128                quantized_vectors.as_ref(),
1129                deleted_points,
1130                params,
1131                vector_query_context.hardware_counter(),
1132                filter_context,
1133            )?;
1134
1135            let search_result = self.graph.search(
1136                oversampled_top,
1137                ef,
1138                algorithm,
1139                points_scorer,
1140                custom_entry_points,
1141                &is_stopped,
1142            )?;
1143
1144            postprocess_search_result(
1145                search_result,
1146                id_tracker.deleted_point_bitslice(),
1147                &vector_storage,
1148                quantized_vectors.as_ref(),
1149                vector,
1150                params,
1151                top,
1152                vector_query_context.hardware_counter(),
1153            )
1154        };
1155
1156        // Try to use graph with vectors first.
1157        if let Some(search_result) = search_with_vectors()? {
1158            Ok(search_result)
1159        } else {
1160            // Graph with vectors is not available, fallback to regular graph search.
1161            regular_search()
1162        }
1163    }

1277        let query_cardinality = payload_index.estimate_cardinality(filter, hw_counter);
1278        // Assume query is already estimated to be small enough so we can iterate over all matched ids
1279        let filtered_points = payload_index.iter_filtered_points(
1280            filter,
1281            &id_tracker,
1282            &query_cardinality,
1283            hw_counter,
1284            is_stopped,
1285            // No deferred filtering here since it's HNSW index.
1286            None,
1287        );
1288        self.search_plain_batched(vectors, filtered_points, top, params, vector_query_context)
1289    }

1301

1305            .search_with_graph(
1306                &query_vector,
1307                filter,

1309                params,
1310                None,
1311                vector_query_context,
1312            )
1313            .map(|search_result| search_result.iter().map(|x| x.idx).collect())?;
1314
1315        // Stage 2: Discover search with entry points
1316        let query_vector = QueryVector::Discover(discover_query);
1317
1318        self.search_with_graph(
1319            &query_vector,
1320            filter,
1321            top,
1322            params,
1323            Some(&custom_entry_points),
1324            vector_query_context,
1325        )
1326    }

1386    fn search(
1387        &self,
1388        vectors: &[&QueryVector],
1389        filter: Option<&Filter>,
1390        top: usize,
1391        params: Option<&SearchParams>,
1392        query_context: &VectorQueryContext,
1393    ) -> OperationResult<Vec<Vec<ScoredPointOffset>>> {
1394        if top == 0 {
1395            return Ok(vec![vec![]; vectors.len()]);
1396        }
1397
1398        // If neither `m` nor `payload_m` is set, HNSW doesn't have any links.
1399        // And if so, we need to fall back to plain search (optionally, with quantization).
1400
1401        let is_hnsw_disabled = self.config.m == 0 && self.config.payload_m.unwrap_or(0) == 0;
1402        let exact = params.map(|params| params.exact).unwrap_or(false);
1403
1404        let exact_params = if exact {
1405            params.map(|params| {
1406                let mut params = *params;
1407                params.quantization = Some(QuantizationSearchParams {
1408                    ignore: true,
1409                    rescore: Some(false),
1410                    oversampling: None,
1411                }); // disable quantization for exact search
1412                params
1413            })
1414        } else {
1415            None
1416        };
1417
1418        match filter {
1419            None => {
1420                let vector_storage = self.vector_storage.borrow();
1421
1422                // Determine whether to do a plain or graph search, and pick search timer aggregator
1423                // Because an HNSW graph is built, we'd normally always assume to search the graph.
1424                // But because a lot of points may be deleted in this graph, it may just be faster
1425                // to do a plain search instead.
1426                let plain_search = exact
1427                    || is_hnsw_disabled
1428                    || vector_storage.available_vector_count() < self.config.full_scan_threshold;
1429
1430                // Do plain or graph search
1431                if plain_search {
1432                    let _timer = ScopeDurationMeasurer::new(if exact {
1433                        &self.searches_telemetry.exact_unfiltered
1434                    } else {
1435                        &self.searches_telemetry.unfiltered_plain
1436                    });
1437
1438                    let params_ref = if exact { exact_params.as_ref() } else { params };
1439                    self.search_plain_unfiltered_batched(vectors, top, params_ref, query_context)
1440                } else {
1441                    let _timer =
1442                        ScopeDurationMeasurer::new(&self.searches_telemetry.unfiltered_hnsw);
1443                    self.search_vectors_with_graph(vectors, None, top, params, query_context)
1444                }
1445            }
1446            Some(query_filter) => {
1447                // depending on the amount of filtered-out points the optimal strategy could be
1448                // - to retrieve possible points and score them after
1449                // - to use HNSW index with filtering condition
1450
1451                // if exact search is requested, we should not use HNSW index
1452                if exact || is_hnsw_disabled {
1453                    let _timer = ScopeDurationMeasurer::new(if exact {
1454                        &self.searches_telemetry.exact_filtered
1455                    } else {
1456                        &self.searches_telemetry.filtered_plain
1457                    });
1458
1459                    let params_ref = if exact { exact_params.as_ref() } else { params };
1460
1461                    return self.search_vectors_plain(
1462                        vectors,
1463                        query_filter,
1464                        top,
1465                        params_ref,
1466                        query_context,
1467                    );
1468                }
1469
1470                let payload_index = self.payload_index.borrow();
1471                let vector_storage = self.vector_storage.borrow();
1472                let id_tracker = self.id_tracker.borrow();
1473                let available_vector_count = vector_storage.available_vector_count();
1474
1475                let hw_counter = query_context.hardware_counter();
1476
1477                let query_point_cardinality =
1478                    payload_index.estimate_cardinality(query_filter, &hw_counter);
1479                let query_cardinality = adjust_to_available_vectors(
1480                    query_point_cardinality,
1481                    available_vector_count,
1482                    id_tracker.available_point_count(),
1483                );
1484
1485                if query_cardinality.max < self.config.full_scan_threshold {
1486                    // if cardinality is small - use plain index
1487                    let _timer =
1488                        ScopeDurationMeasurer::new(&self.searches_telemetry.small_cardinality);
1489                    return self.search_vectors_plain(
1490                        vectors,
1491                        query_filter,
1492                        top,
1493                        params,
1494                        query_context,
1495                    );
1496                }
1497
1498                if query_cardinality.min > self.config.full_scan_threshold {
1499                    // if cardinality is high enough - use HNSW index
1500                    let _timer =
1501                        ScopeDurationMeasurer::new(&self.searches_telemetry.large_cardinality);
1502                    return self.search_vectors_with_graph(
1503                        vectors,
1504                        filter,
1505                        top,
1506                        params,
1507                        query_context,
1508                    );
1509                }
1510
1511                let filter_context = payload_index.filter_context(query_filter, &hw_counter);
1512
1513                // Fast cardinality estimation is not enough, do sample estimation of cardinality
1514                let id_tracker = self.id_tracker.borrow();
1515                if sample_check_cardinality(
1516                    id_tracker.sample_ids(Some(vector_storage.deleted_vector_bitslice())),
1517                    |idx| filter_context.check(idx),
1518                    self.config.full_scan_threshold,
1519                    available_vector_count, // Check cardinality among available vectors
1520                ) {
1521                    // if cardinality is high enough - use HNSW index
1522                    let _timer =
1523                        ScopeDurationMeasurer::new(&self.searches_telemetry.large_cardinality);
1524                    self.search_vectors_with_graph(vectors, filter, top, params, query_context)
1525                } else {
1526                    // if cardinality is small - use plain index
1527                    let _timer =
1528                        ScopeDurationMeasurer::new(&self.searches_telemetry.small_cardinality);
1529                    self.search_vectors_plain(vectors, query_filter, top, params, query_context)
1530                }
1531            }
1532        }
1533    }

1587/// Contains enough information to decide which one old index to use.
1588/// Once decided, it is converted to [`OldIndex`].
1589struct OldIndexCandidate<'a> {
1590    index: AtomicRef<'a, HNSWIndex>,
1591    /// Mapping from old index to new index.
1592    /// `old_to_new[old_idx] == Some(new_idx)`.
1593    old_to_new: Vec<Option<PointOffsetType>>,
1594    /// Count of successfully mapped points.
1595    valid_points: usize,
1596    /// Count of points that are missing in the new index.
1597    missing_points: usize,
1598}
1599
1600pub(super) struct OldIndex<'a> {
1601    index: AtomicRef<'a, HNSWIndex>,
1602    /// Mapping from old index to new index.
1603    /// `old_to_new[old_idx] == Some(new_idx)`.
1604    pub old_to_new: Vec<Option<PointOffsetType>>,
1605    /// Mapping from new index to old index.
1606    /// `new_to_old[new_idx] == Some(old_idx)`.
1607    pub new_to_old: Vec<Option<PointOffsetType>>,
1608}
1609
1610impl<'a> OldIndexCandidate<'a> {
1611    /// Evaluate whether we can use the old index.
1612    fn evaluate(
1613        feature_flags: &FeatureFlags,
1614        old_index: &'a Arc<AtomicRefCell<VectorIndexEnum>>,
1615        config: &HnswGraphConfig,
1616        hnsw_global_config: &HnswGlobalConfig,
1617        vector_storage: &VectorStorageEnum,
1618        quantized_vectors: &Option<QuantizedVectors>,
1619        id_tracker: &IdTrackerEnum,
1620    ) -> Option<Self> {
1621        if !feature_flags.incremental_hnsw_building {
1622            return None;
1623        }
1624
1625        let old_index = AtomicRef::filter_map(old_index.borrow(), |index| match index {
1626            VectorIndexEnum::Hnsw(old_index) => Some(old_index),
1627            _ => None,
1628        })?;
1629
1630        let old_quantized_vectors_ref = old_index.quantized_vectors.borrow();
1631        let old_quantization_config = old_quantized_vectors_ref
1632            .as_ref()
1633            .map(|qv| &qv.config().quantization_config);
1634        let new_quantization_config = quantized_vectors
1635            .as_ref()
1636            .map(|qv| &qv.config().quantization_config);
1637
1638        let no_main_graph = config.m == 0;
1639        let configuration_mismatch = config.m != old_index.config.m
1640            || config.m0 != old_index.config.m0
1641            || config.ef_construct != old_index.config.ef_construct
1642            || new_quantization_config != old_quantization_config;
1643        // If old graph has vectors, reusing it will cause a lot of random reads,
1644        // making it slower than building from scratch.
1645        let old_graph_is_with_vectors = old_index.graph.has_inline_vectors();
1646        if no_main_graph || configuration_mismatch || old_graph_is_with_vectors {
1647            return None;
1648        }
1649        drop(old_quantized_vectors_ref);
1650
1651        let new_deleted = vector_storage.deleted_vector_bitslice();
1652        let old_id_tracker = old_index.id_tracker.borrow();
1653
1654        let healing_enabled = hnsw_global_config.healing_threshold > 0.0;
1655
1656        if old_id_tracker.deleted_point_count() != 0 {
1657            // Old index has deleted points.
1658            if !healing_enabled {
1659                return None;
1660            }
1661        }
1662
1663        // Rough check whether the point is included in the old graph.
1664        // If it's included, it almost certainly has at least one outgoing link at level 0.
1665        let old_graph_has_point = |id: PointOffsetType| !old_index.graph.links.links_empty(id, 0);
1666
1667        // Build old_to_new mapping.
1668        let mut valid_points = 0;
1669        let mut missing_points = 0;
1670        let mut old_to_new = vec![None; old_id_tracker.total_point_count()];
1671
1672        // Loop 1: Loop through all external ids, which present in either new or old index.
1673        // So that we have 3 possible cases: `in old`, `in new`, `in both`.
1674        //
1675        // If we have `in both` case, we need to fill the `old_to_new` mapping.
1676        // Otherwise, we are interested in counts of "missing" points - which absence in the new
1677        for item in itertools::merge_join_by(
1678            id_tracker.iter_from(None),
1679            old_id_tracker.iter_from(None),
1680            |(new_external_id, _), (old_external_id, _)| new_external_id.cmp(old_external_id),
1681        ) {
1682            let (new_offset, old_offset): (Option<PointOffsetType>, Option<PointOffsetType>) =
1683                match item {
1684                    EitherOrBoth::Both((_, new_offset), (_, old_offset)) => {
1685                        (Some(new_offset), Some(old_offset))
1686                    }
1687                    EitherOrBoth::Left((_, new_offset)) => (Some(new_offset), None),
1688                    EitherOrBoth::Right((_, old_offset)) => (None, Some(old_offset)),
1689                };
1690
1691            let new_offset =
1692                new_offset.filter(|&id| !new_deleted.get_bit(id as usize).unwrap_or(false));
1693
1694            // If vector is not present in the old graph, assume it's deleted.
1695            let old_offset = old_offset.filter(|&id| old_graph_has_point(id));
1696
1697            match (new_offset, old_offset) {
1698                (_, None) => (),
1699                (None, Some(_)) => {
1700                    // Vector was in the old index, but not in the new one.
1701                    missing_points += 1;
1702                    if !healing_enabled {
1703                        return None;
1704                    }
1705                }
1706                (Some(new_offset), Some(old_offset)) => {
1707                    let new_version = id_tracker.internal_version(new_offset);
1708                    let old_version = old_id_tracker.internal_version(old_offset);
1709
1710                    if old_version == new_version {
1711                        old_to_new[old_offset as usize] = Some(new_offset);
1712                        valid_points += 1;
1713                    } else {
1714                        // Version is different.
1715                        missing_points += 1;
1716                        if !healing_enabled {
1717                            return None;
1718                        }
1719                    }
1720                }
1721            }
1722        }
1723
1724        // Loop 2: find points that once were in the old index, but deleted in
1725        // its id tracker.
1726        // Since loop 1 doesn't detect deleted points, we need to check them explicitly.
1727        for old_offset in 0..old_id_tracker.total_point_count() {
1728            let old_offset = old_offset as PointOffsetType;
1729            if old_id_tracker.is_deleted_point(old_offset) && old_graph_has_point(old_offset) {
1730                missing_points += 1;
1731                if !healing_enabled {
1732                    return None;
1733                }
1734            }
1735        }
1736
1737        if valid_points == 0 {
1738            return None;
1739        }
1740
1741        let missing_ratio = missing_points as f64 / (missing_points + valid_points) as f64;
1742        let do_heal = missing_ratio <= hnsw_global_config.healing_threshold;
1743        debug!(
1744            "valid points: {valid_points}, missing points: {missing_points}, missing ratio: {missing_ratio:.3}, do_heal: {do_heal}"
1745        );
1746        if !do_heal {
1747            return None;
1748        }
1749
1750        drop(old_id_tracker);
1751
1752        Some(OldIndexCandidate {
1753            index: old_index,
1754            old_to_new,
1755            valid_points,
1756            missing_points,
1757        })
1758    }
1759
1760    fn reuse(self, total_vector_count: usize) -> OldIndex<'a> {
1761        let mut new_to_old = vec![None; total_vector_count];
1762        for (old_offset, new_offset) in self.old_to_new.iter().copied().enumerate() {
1763            if let Some(new_offset) = new_offset {
1764                new_to_old[new_offset as usize] = Some(old_offset as PointOffsetType);
1765            }
1766        }
1767
1768        log::debug!(
1769            "Reusing {} points from the old index, healing {} points",
1770            self.valid_points,
1771            self.missing_points,
1772        );
1773
1774        OldIndex {
1775            index: self.index,
1776            old_to_new: self.old_to_new,
1777            new_to_old,
1778        }
1779    }
1780}

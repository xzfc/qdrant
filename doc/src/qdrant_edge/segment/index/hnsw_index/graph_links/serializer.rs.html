23pub fn serialize_graph_links<W: Write + Seek>(
24    mut edges: Vec<Vec<Vec<PointOffsetType>>>,
25    format_param: GraphLinksFormatParam,
26    hnsw_m: HnswM,
27    writer: &mut W,
28) -> OperationResult<()> {
29    let bits_per_unsorted =
30        packed_bits(u32::try_from(edges.len().saturating_sub(1)).unwrap()).max(MIN_BITS_PER_VALUE);
31
32    let vectors_layout = match format_param {
33        GraphLinksFormatParam::Plain => None,
34        GraphLinksFormatParam::Compressed => None,
35        GraphLinksFormatParam::CompressedWithVectors(v) => {
36            let vectors_layout = v.vectors_layout();
37            if vectors_layout.base.size() % vectors_layout.base.align() != 0 {
38                return Err(OperationError::service_error(
39                    "Base vector size must be a multiple of its alignment",
40                ));
41            }
42            if vectors_layout.link.size() % vectors_layout.link.align() != 0 {
43                return Err(OperationError::service_error(
44                    "Link vector size must be a multiple of its alignment",
45                ));
46            }
47            Some(vectors_layout)
48        }
49    };
50
51    // create map from index in `offsets` to point_id
52    let mut back_index: Vec<PointOffsetType> = (0..edges.len()).map(|i| i as _).collect();
53    // sort by max layer and use this map to build `reindex`
54    back_index.sort_unstable_by_key(|&i| Reverse(edges[i as usize].len()));
55
56    let levels_count = back_index.first().map_or(0, |&id| edges[id as usize].len());
57    let mut point_count_by_level = vec![0; levels_count];
58    for point in &edges {
59        point_count_by_level[point.len() - 1] += 1;
60    }
61
62    // 1. Write header (placeholder, will be rewritten later)
63    writer.write_zeros(match &format_param {
64        GraphLinksFormatParam::Plain => size_of::<HeaderPlain>(),
65        GraphLinksFormatParam::Compressed => size_of::<HeaderCompressed>(),
66        GraphLinksFormatParam::CompressedWithVectors(_) => size_of::<HeaderCompressedWithVectors>(),
67    })?;
68
69    // 2. Write level offsets
70    let mut total_offsets_len = 0;
71    {
72        let mut suffix_sum = point_count_by_level.iter().sum::<u64>();
73        for &value in point_count_by_level.iter() {
74            writer.write_all(total_offsets_len.as_bytes())?;
75            total_offsets_len += suffix_sum;
76            suffix_sum -= value;
77        }
78        total_offsets_len += 1;
79    }
80
81    // 3. Write reindex (aka map from point id to index in `offsets`)
82    {
83        let mut reindex = vec![0; back_index.len()];
84        for i in 0..back_index.len() {
85            reindex[back_index[i] as usize] = i as PointOffsetType;
86        }
87        writer.write_all(reindex.as_bytes())?;
88    }
89
90    // 4. Write neighbors padding (if applicable)
91    if let Some(vectors_layout) = vectors_layout.as_ref() {
92        let pos = writer.stream_position()? as usize;
93        let alignment = std::cmp::max(vectors_layout.base.align(), vectors_layout.link.align());
94        writer.write_zeros(pos.next_multiple_of(alignment) - pos)?;
95    }
96
97    // 5. Write neighbors (and calculate `offsets`)
98    let mut links_buf = Vec::new();
99    let mut offset = 0; // elements for Plain, bytes for Compressed/CompressedWithVectors
100    let mut offsets = Vec::with_capacity(total_offsets_len as usize);
101    offsets.push(0);
102
103    #[expect(clippy::needless_range_loop)]
104    // this clippy lint is positively demented, can't wait till they remove it 🙄
105    for level in 0..levels_count {
106        let count = point_count_by_level.iter().skip(level).sum::<u64>() as usize;
107        let (level_m, mut iter) = match level {
108            0 => (hnsw_m.m0, Either::Left((0..count).map(|x| x as u32))),
109            _ => (hnsw_m.m, Either::Right(back_index[..count].iter().copied())),
110        };
111
112        iter.try_for_each(|id| {
113            let mut raw_links = std::mem::take(&mut edges[id as usize][level]);
114            match format_param {
115                GraphLinksFormatParam::Plain => {
116                    writer.write_all(raw_links.as_bytes())?;
117                    offset += raw_links.len();
118                }
119                GraphLinksFormatParam::Compressed => {
120                    pack_links(&mut links_buf, &mut raw_links, bits_per_unsorted, level_m);
121                    writer.write_all(&links_buf)?;
122                    offset += links_buf.len();
123                }
124                GraphLinksFormatParam::CompressedWithVectors(vectors) => {
125                    // Unwrap safety: `vectors_layout` is `Some` for `CompressedWithVectors`.
126                    let vectors_layout = vectors_layout.as_ref().unwrap();
127
128                    // 1. Base vector (`B` in the doc, only on level 0).
129                    if level == 0 {
130                        let vector = vectors.get_base_vector(id)?;
131                        if vector.len() != vectors_layout.base.size() {
132                            return Err(OperationError::service_error("Vector size mismatch"));
133                        }
134                        writer.write_all(&vector)?;
135                        offset += vector.len();
136                    }
137
138                    // 2. The varint-encoded length (`#` in the doc).
139                    writer.write_varint(raw_links.len() as u64)?;
140                    offset += VarInt::required_space(raw_links.len() as u64);
141
142                    // 3. Compressed links (`c` in the doc)
143                    pack_links(&mut links_buf, &mut raw_links, bits_per_unsorted, level_m);
144                    writer.write_all(&links_buf)?;
145                    offset += links_buf.len();
146
147                    // 4. Padding to align link vectors (`_` in the doc).
148                    let padding = offset.next_multiple_of(vectors_layout.link.align()) - offset;
149                    writer.write_zeros(padding)?;
150                    offset += padding;
151
152                    // 5. Link vectors (`L` in the doc).
153                    // Write them in the same order as `raw_links`.
154                    for i in raw_links {
155                        let vector = vectors.get_link_vector(i)?;
156                        if vector.len() != vectors_layout.link.size() {
157                            return Err(OperationError::service_error("Vector size mismatch"));
158                        }
159                        writer.write_all(&vector)?;
160                        offset += vector.len();
161                    }
162
163                    // 6. Padding to align the next base vector (`_` in the doc).
164                    if level == 0 {
165                        let padding = offset.next_multiple_of(vectors_layout.base.align()) - offset;
166                        writer.write_zeros(padding)?;
167                        offset += padding;
168                    }
169                }
170            }
171            offsets.push(offset as u64);
172            links_buf.clear();
173            Ok(())
174        })?;
175    }
176    drop(back_index);
177
178    // 7. Write offsets (and get some info for the header)
179    let (offsets_padding, offsets_parameters) = match &format_param {
180        GraphLinksFormatParam::Plain => {
181            let len = writer.stream_position()? as usize;
182            let offsets_padding = len.next_multiple_of(size_of::<u64>()) - len;
183            writer.write_zeros(offsets_padding)?;
184            writer.write_all(offsets.as_bytes())?;
185            (Some(offsets_padding), None)
186        }
187        GraphLinksFormatParam::Compressed | GraphLinksFormatParam::CompressedWithVectors(_) => {
188            let (compressed_offsets, offsets_parameters) = bitpacking_ordered::compress(&offsets);
189            writer.write_all(&compressed_offsets)?;
190            (None, Some(offsets_parameters))
191        }
192    };
193
194    // 8. Write header (not a placeholder anymore)
195    writer.seek(std::io::SeekFrom::Start(0))?;
196    match format_param {
197        GraphLinksFormatParam::Plain => {
198            let header = HeaderPlain {
199                point_count: edges.len() as u64,
200                levels_count: levels_count as u64,
201                total_neighbors_count: offset as u64,
202                total_offset_count: offsets.len() as u64,
203                offsets_padding_bytes: offsets_padding.unwrap() as u64,
204                zero_padding: [0; 24],
205            };
206            writer.write_all(header.as_bytes())?;
207        }
208        GraphLinksFormatParam::Compressed => {
209            let header = HeaderCompressed {
210                version: LittleU64::from(HEADER_VERSION_COMPRESSED),
211                point_count: LittleU64::new(edges.len() as u64),
212                total_neighbors_bytes: LittleU64::new(offset as u64),
213                offsets_parameters: offsets_parameters.unwrap(),
214                levels_count: LittleU64::new(levels_count as u64),
215                m: LittleU64::new(hnsw_m.m as u64),
216                m0: LittleU64::new(hnsw_m.m0 as u64),
217                zero_padding: [0; 5],
218            };
219            writer.write_all(header.as_bytes())?;
220        }
221        GraphLinksFormatParam::CompressedWithVectors(_) => {
222            let vectors_layout = vectors_layout.as_ref().unwrap();
223            let header = HeaderCompressedWithVectors {
224                version: LittleU64::from(HEADER_VERSION_COMPRESSED_WITH_VECTORS),
225                point_count: LittleU64::new(edges.len() as u64),
226                total_neighbors_bytes: LittleU64::new(offset as u64),
227                offsets_parameters: offsets_parameters.unwrap(),
228                levels_count: LittleU64::new(levels_count as u64),
229                m: LittleU64::new(hnsw_m.m as u64),
230                m0: LittleU64::new(hnsw_m.m0 as u64),
231                base_vector_layout: pack_layout(&vectors_layout.base),
232                link_vector_layout: pack_layout(&vectors_layout.link),
233                zero_padding: [0; 3],
234            };
235            writer.write_all(header.as_bytes())?;
236        }
237    };
238
239    Ok(())
240}

242fn pack_layout(layout: &Layout) -> PackedVectorLayout {
243    PackedVectorLayout {
244        size: LittleU64::new(layout.size() as u64),
245        alignment: u8::try_from(layout.align()).expect("Alignment must fit in u8"),
246    }

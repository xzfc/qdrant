12/// Trait for payload data storage. Should allow filter checks
13pub trait PayloadStorage {
14    /// Overwrite payload for point_id. If payload already exists, replace it
15    fn overwrite(
16        &mut self,
17        point_id: PointOffsetType,
18        payload: &Payload,
19        hw_counter: &HardwareCounterCell,
20    ) -> OperationResult<()>;
21
22    /// Set payload for point_id. If payload already exists, merge it with existing
23    fn set(
24        &mut self,
25        point_id: PointOffsetType,
26        payload: &Payload,
27        hw_counter: &HardwareCounterCell,
28    ) -> OperationResult<()>;
29
30    /// Set payload to a point_id by key. If payload already exists, merge it with existing
31    fn set_by_key(
32        &mut self,
33        point_id: PointOffsetType,
34        payload: &Payload,
35        key: &JsonPath,
36        hw_counter: &HardwareCounterCell,
37    ) -> OperationResult<()>;
38
39    fn get(
40        &self,
41        point_id: PointOffsetType,
42        hw_counter: &HardwareCounterCell,
43    ) -> OperationResult<Payload>;
44
45    fn get_sequential(
46        &self,
47        point_id: PointOffsetType,
48        hw_counter: &HardwareCounterCell,
49    ) -> OperationResult<Payload>;
50
51    /// Delete payload by point_id and key
52    fn delete(
53        &mut self,
54        point_id: PointOffsetType,
55        key: &JsonPath,
56        hw_counter: &HardwareCounterCell,
57    ) -> OperationResult<Vec<Value>>;
58
59    /// Clear all payload of the point
60    fn clear(
61        &mut self,
62        point_id: PointOffsetType,
63        hw_counter: &HardwareCounterCell,
64    ) -> OperationResult<Option<Payload>>;
65
66    /// Completely delete payload storage, without keeping allocated memory. Pufff!
67    #[cfg(test)]
68    fn clear_all(&mut self, hw_counter: &HardwareCounterCell) -> OperationResult<()>;
69
70    /// Return function that forces persistence of current storage state.
71    fn flusher(&self) -> Flusher;
72
73    /// Iterate over all stored payload and apply the provided callback.
74    /// Stop iteration if callback returns false or error.
75    ///
76    /// Required for building payload index.
77    fn iter<F>(&self, callback: F, hw_counter: &HardwareCounterCell) -> OperationResult<()>
78    where
79        F: FnMut(PointOffsetType, &Payload) -> OperationResult<bool>;
80
81    /// Return all files that are used by storage to include in snapshots.
82    /// RocksDB storages are captured outside of this trait.
83    fn files(&self) -> Vec<PathBuf>;
84
85    /// Returns a list of files, that are immutable, to exclude from partial snapshots.
86    fn immutable_files(&self) -> Vec<PathBuf> {
87        Vec::new()
88    }
89
90    /// Return storage size in bytes
91    fn get_storage_size_bytes(&self) -> OperationResult<usize>;
92
93    /// Whether this storage is on-disk or in-memory.
94    fn is_on_disk(&self) -> bool;
95}

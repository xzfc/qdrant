40impl FacetValueRef<'_> {
41    pub fn to_owned(&self) -> FacetValue {
42        match self {
43            FacetValueRef::Keyword(str) => FacetValue::Keyword(str.to_string()),
44            &FacetValueRef::Int(int) => FacetValue::Int(int),
45            &FacetValueRef::Uuid(uuid) => FacetValue::Uuid(uuid),
46            &FacetValueRef::Bool(bool) => FacetValue::Bool(bool),
47        }
48    }
49}

101pub type FacetValueHit = FacetHit<FacetValue>;

103#[derive(PartialEq, Eq, Clone, Debug)]
104pub struct FacetHit<T: FacetValueTrait> {
105    pub value: T,
106    pub count: usize,
107}
108
109#[derive(Clone, Debug, Default)]
110pub struct FacetResponse {
111    pub hits: Vec<FacetValueHit>,
112}
113
114impl FacetResponse {
115    /// Convert a count map to top `limit` hits sorted by count descending.
116    ///
117    /// Shared utility used by Edge and Collection facet implementations.
118    pub fn top_hits(counts: HashMap<FacetValue, usize>, limit: usize) -> Self {
119        let hits = counts
120            .into_iter()
121            .map(|(value, count)| FacetValueHit { value, count })
122            .k_largest(limit)
123            .collect();
124
125        Self { hits }
126    }
127}
128
129impl<T: FacetValueTrait> Ord for FacetHit<T> {
130    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
131        self.count
132            .cmp(&other.count)
133            // Reverse so that descending order has ascending values when having the same count
134            .then_with(|| Reverse(&self.value).cmp(&Reverse(&other.value)))
135    }
136}
137
138impl<T: FacetValueTrait> PartialOrd for FacetHit<T> {
139    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
140        Some(self.cmp(other))
141    }
142}

660        let count = vector_storage.total_vector_count();
661        let distance = vector_storage.distance();
662        let datatype = vector_storage.datatype();
663        let vectors = (0..count as PointOffsetType).map(|i| {
664            match vector_storage.get_dense::<Sequential>(i) {

666                    quantization_config,

671                    // TODO: Preprocess without reallocating

674                        distance,

678                ),
679            }

682

684            Self::construct_vector_parameters(distance, dim, count, storage_type);

692                count,

744    fn create_multi_impl<
745        TElement: PrimitiveVectorElement + 'static,
746        TVectorStorage: MultiVectorStorage<TElement> + Send + Sync,
747    >(
748        vector_storage: &TVectorStorage,
749        quantization_config: &QuantizationConfig,
750        storage_type: QuantizedVectorsStorageType,
751        path: &Path,
752        max_threads: usize,
753        stopped: &AtomicBool,
754    ) -> OperationResult<Self> {
755        let dim = vector_storage.vector_dim();
756        let distance = vector_storage.distance();
757        let datatype = vector_storage.datatype();
758        let multi_vector_config = *vector_storage.multi_vector_config();
759        let vectors = vector_storage.iterate_inner_vectors().map(|v| match v {
760            Cow::Borrowed(slice) => PrimitiveVectorElement::quantization_preprocess(
761                quantization_config,
762                distance,
763                slice,
764            ),
765            // TODO: avoid reallocation by accepting Cow in quantization_preprocess
766            Cow::Owned(vec) => Cow::Owned(
767                PrimitiveVectorElement::quantization_preprocess(
768                    quantization_config,
769                    distance,
770                    &vec,
771                )
772                .into_owned(),
773            ),
774        });
775        let inner_vectors_count = vectors.clone().count();
776        let vectors_count = vector_storage.total_vector_count();
777        let on_disk_vector_storage = vector_storage.is_on_disk();
778
779        let vector_parameters =
780            Self::construct_vector_parameters(distance, dim, inner_vectors_count, storage_type);
781
782        let offsets = (0..vectors_count as PointOffsetType)
783            .map(|idx| {
784                vector_storage
785                    .get_multi::<Random>(idx)
786                    .as_ref()
787                    .vectors_count() as PointOffsetType
788            })
789            .scan(0, |offset_acc, multi_vector_len| {
790                let offset = *offset_acc;
791                *offset_acc += multi_vector_len;
792                Some(MultivectorOffset {
793                    start: offset,
794                    count: multi_vector_len,
795                })
796            });
797
798        let quantized_storage = match quantization_config {
799            QuantizationConfig::Scalar(ScalarQuantization {
800                scalar: scalar_config,
801            }) => Self::create_scalar_multi(
802                vectors,
803                offsets,
804                &vector_parameters,
805                vectors_count,
806                inner_vectors_count,
807                scalar_config,
808                storage_type,
809                multi_vector_config,
810                path,
811                on_disk_vector_storage,
812                stopped,
813            )?,
814            QuantizationConfig::Product(ProductQuantization { product: pq_config }) => {
815                Self::create_pq_multi(
816                    vectors,
817                    offsets,
818                    &vector_parameters,
819                    vectors_count,
820                    inner_vectors_count,
821                    pq_config,
822                    storage_type,
823                    multi_vector_config,
824                    path,
825                    on_disk_vector_storage,
826                    max_threads,
827                    stopped,
828                )?
829            }
830            QuantizationConfig::Binary(BinaryQuantization {
831                binary: binary_config,
832            }) => Self::create_binary_multi(
833                vectors,
834                offsets,
835                &vector_parameters,
836                vectors_count,
837                inner_vectors_count,
838                binary_config,
839                storage_type,
840                multi_vector_config,
841                path,
842                on_disk_vector_storage,
843                stopped,
844            )?,
845        };
846
847        let quantized_vectors_config = QuantizedVectorsConfig {
848            quantization_config: quantization_config.clone(),
849            vector_parameters,
850            storage_type,
851        };
852
853        let quantized_vectors = QuantizedVectors {
854            storage_impl: quantized_storage,
855            config: quantized_vectors_config,
856            path: path.to_path_buf(),
857            distance,
858            datatype,
859        };
860
861        atomic_save_json(&path.join(QUANTIZED_CONFIG_PATH), &quantized_vectors.config)?;
862        Ok(quantized_vectors)
863    }

865    pub fn load(
866        quantization_config: &QuantizationConfig,
867        vector_storage: &VectorStorageEnum,
868        path: &Path,
869        stopped: &AtomicBool,
870    ) -> OperationResult<Option<Self>> {
871        let config_path = Self::get_config_path(path);
872        if config_path.exists() {
873            let config: QuantizedVectorsConfig = read_json(&config_path)?;
874            return Ok(Some(Self::load_impl(config, vector_storage, path)?));
875        }
876
877        // If we don't have an appendable quantization feature, do not create a new one.
878        if !crate::common::flags::feature_flags().appendable_quantization {
879            return Ok(None);
880        }
881
882        let quantized_vectors = Self::create(
883            vector_storage,
884            quantization_config,
885            QuantizedVectorsStorageType::Mutable,
886            path,
887            1,
888            stopped,
889        )?;
890        Ok(Some(quantized_vectors))
891    }

899            vector_storage.try_multi_vector_config()

916                        vector_storage,

918                        &config,

922                }
923            }

936        };

941            storage_impl: quantized_store,

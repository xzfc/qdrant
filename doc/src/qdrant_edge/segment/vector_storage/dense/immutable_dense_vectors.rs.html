28const HEADER_SIZE: usize = 4;
29const VECTORS_HEADER: &[u8; HEADER_SIZE] = b"data";
30const DELETED_HEADER: &[u8; HEADER_SIZE] = b"drop";

32/// Immutable storage for dense vectors.
33#[derive(Debug)]
34pub struct ImmutableDenseVectors<
35    T: PrimitiveVectorElement,
36    S: UniversalRead<u8> = MmapUniversal<u8>,
37> {
38    pub dim: usize,
39    pub num_vectors: usize,
40    /// Byte-addressable vector data storage, providing read access via [`UniversalRead<u8>`].
41    storage: S,
42    /// Context for io_uring-based async IO
43    #[cfg_attr(not(target_os = "linux"), allow(dead_code))]
44    uring_reader: Option<Mutex<UringReader<T>>>,
45    /// Memory mapped deletion flags
46    deleted: MmapBitSlice,
47    /// Current number of deleted vectors.
48    pub deleted_count: usize,
49}
50
51impl<T: PrimitiveVectorElement, S: UniversalRead<u8>> ImmutableDenseVectors<T, S> {
52    pub fn open(
53        vectors_path: &Path,
54        deleted_path: &Path,
55        dim: usize,
56        with_async_io: bool,
57        populate: bool,
58    ) -> OperationResult<Self> {
59        // Allocate/open vectors file
60        ensure_mmap_file_size(vectors_path, VECTORS_HEADER, None)
61            .describe("Create mmap data file")?;
62
63        let file_len = fs_err::metadata(vectors_path)?.len() as usize;
64        let num_vectors = file_len.saturating_sub(HEADER_SIZE) / dim / size_of::<T>();
65
66        let options = UniversalOpenOptions {
67            need_sequential: true,
68            disk_parallel: None,
69            populate: Some(populate),
70            advice: None,
71        };
72        let storage = UniversalRead::<u8>::open(vectors_path, options).map_err(|e| {
73            crate::segment::common::operation_error::OperationError::service_error(format!(
74                "Failed to open vector mmap at {}: {e}",
75                vectors_path.display()
76            ))
77        })?;
78
79        // Allocate/open deleted mmap
80        let deleted_mmap_size = deleted_mmap_size(num_vectors);
81        ensure_mmap_file_size(deleted_path, DELETED_HEADER, Some(deleted_mmap_size as u64))
82            .describe("Create mmap deleted file")?;
83        let deleted_mmap = mmap::open_write_mmap(deleted_path, AdviceSetting::Global, false)
84            .describe("Open mmap deleted for writing")?;
85
86        // Advise kernel that we'll need this page soon so the kernel can prepare
87        #[cfg(unix)]
88        if let Err(err) = deleted_mmap.advise(memmap2::Advice::WillNeed) {
89            log::error!("Failed to advise MADV_WILLNEED for deleted flags: {err}");
90        }
91
92        // Transform into mmap BitSlice
93        let deleted = MmapBitSlice::try_from(deleted_mmap, deleted_mmap_data_start())?;
94        let deleted_count = deleted.count_ones();
95
96        let uring_reader = if with_async_io {
97            // Keep file handle open for async IO
98            let vectors_file = File::open(vectors_path)?;
99            let raw_size = dim * size_of::<T>();
100            Some(UringReader::new(vectors_file, raw_size, HEADER_SIZE)?)
101        } else {
102            None
103        };
104
105        Ok(ImmutableDenseVectors {
106            dim,
107            num_vectors,
108            storage,
109            uring_reader: uring_reader.map(Mutex::new),
110            deleted,
111            deleted_count,
112        })
113    }

123    /// Returns the byte offset within the file at which the vector for `key` begins.
124    ///
125    /// File layout:
126    /// ```text
127    /// [HEADER_SIZE] [vector_0] [vector_1] ... [vector_N-1]
128    /// ```
129    /// Each vector occupies `dim * size_of::<T>()` bytes.
130    pub fn data_offset(&self, key: PointOffsetType) -> Option<usize> {
131        let vector_data_length = self.dim * size_of::<T>();
132        let offset = (key as usize) * vector_data_length + HEADER_SIZE;
133        if key >= (self.num_vectors as PointOffsetType) {
134            return None;
135        }
136        Some(offset)
137    }
138
139    /// Size in bytes of a single vector.
140    pub fn raw_size(&self) -> usize {
141        self.dim * size_of::<T>()
142    }
143
144    /// Read one vector's worth of bytes from storage at `byte_offset` and reinterpret
145    /// the byte slice as `&[T]`.
146    fn raw_vector_offset<P: AccessPattern>(&self, byte_offset: usize) -> Cow<'_, [T]> {
147        let range = ElementsRange {
148            start: byte_offset as u64,
149            length: self.raw_size() as u64,
150        };
151
152        let cow: Cow<'_, [u8]> = if P::IS_SEQUENTIAL {
153            self.storage.read::<true>(range)
154        } else {
155            self.storage.read::<false>(range)
156        }
157        .expect("vector read from storage failed");
158
159        match cow {
160            Cow::Borrowed(byte_slice) => Cow::Borrowed(bytemuck::cast_slice(byte_slice)),
161            Cow::Owned(byte_vec) => Cow::Owned(bytemuck::cast_vec(byte_vec)),
162        }
163    }
164
165    /// Returns vector data by key
166    fn get_vector<P: AccessPattern>(&self, key: PointOffsetType) -> Cow<'_, [T]> {
167        self.get_vector_opt::<P>(key).expect("vector not found")
168    }
169
170    /// Returns an optional vector data by key
171    pub fn get_vector_opt<P: AccessPattern>(&self, key: PointOffsetType) -> Option<Cow<'_, [T]>> {
172        self.data_offset(key)
173            .map(|offset| self.raw_vector_offset::<P>(offset))
174    }

176    pub fn for_each_in_batch<F: FnMut(usize, &[T])>(&self, keys: &[PointOffsetType], mut f: F) {
177        debug_assert!(keys.len() <= VECTOR_READ_BATCH_SIZE);
178
179        // The `f` is most likely a scorer function.
180        // Fetching all vectors first then scoring them is more cache friendly
181        // than fetching and scoring in a single loop.
182
183        let mut vectors_buffer = [const { MaybeUninit::uninit() }; VECTOR_READ_BATCH_SIZE];
184        let vectors = if is_read_with_prefetch_efficient(keys) {
185            let iter = keys.iter().map(|key| self.get_vector::<Sequential>(*key));
186            maybe_uninit_fill_from(&mut vectors_buffer, iter).0
187        } else {
188            let iter = keys.iter().map(|key| self.get_vector::<Random>(*key));
189            maybe_uninit_fill_from(&mut vectors_buffer, iter).0
190        };
191
192        for (i, vec) in vectors.iter().enumerate() {
193            f(i, vec);
194        }
195    }

197    /// Marks the key as deleted.
198    ///
199    /// Returns true if the key was not deleted before, and it is now deleted.
200    pub fn delete(&mut self, key: PointOffsetType) -> bool {
201        let is_deleted = !self.deleted.replace(key as usize, true);
202        if is_deleted {
203            self.deleted_count += 1;
204        }
205        is_deleted
206    }
207
208    pub fn is_deleted_vector(&self, key: PointOffsetType) -> bool {
209        self.deleted.get_bit(key as usize).unwrap_or(false)
210    }

212    /// Get [`BitSlice`] representation for deleted vectors with deletion flags
213    ///
214    /// The size of this slice is not guaranteed. It may be smaller/larger than the number of
215    /// vectors in this segment.
216    pub fn deleted_vector_bitslice(&self) -> &BitSlice {
217        &self.deleted
218    }

220    fn process_points_simple(
221        &self,
222        points: impl Iterator<Item = PointOffsetType>,
223        mut callback: impl FnMut(usize, PointOffsetType, &[T]),
224    ) {
225        for (idx, point) in points.enumerate() {
226            let vector = self.get_vector::<Random>(point);
227            callback(idx, point, &vector);
228        }
229    }
230
231    /// Reads vectors for the given ids and calls the callback for each vector.
232    /// Tries to utilize asynchronous IO if possible.
233    /// In particular, uses io_uring on Linux and simple synchronous IO otherwise.
234    pub fn read_vectors_async(
235        &self,
236        points: impl Iterator<Item = PointOffsetType>,
237        callback: impl FnMut(usize, PointOffsetType, &[T]),
238    ) -> OperationResult<()> {
239        match &self.uring_reader {
240            None => self.process_points_simple(points, callback),
241
242            #[cfg(target_os = "linux")]
243            Some(uring_reader) => {
244                // Use `UringReader` on Linux
245                let mut uring_guard = uring_reader.lock();
246                uring_guard.read_stream(points, callback)?;
247            }
248
249            #[cfg(not(target_os = "linux"))]
250            Some(_) => {
251                // Fallback to synchronous processing on non-Linux platforms
252                self.process_points_simple(points, callback);
253            }
254        }
255        Ok(())
256    }
257
258    pub fn populate(&self) {
259        if let Err(err) = self.storage.populate() {
260            log::error!("Failed to populate vector storage: {err}");
261        }
262    }

265/// Ensure the given mmap file exists and is the given size.
266///
267/// # Arguments
268/// * `path`: path of the file.
269/// * `header`: header to set when the file is newly created.
270/// * `size`: set the file size in bytes, filled with zeroes.
271fn ensure_mmap_file_size(path: &Path, header: &[u8], size: Option<u64>) -> OperationResult<()> {
272    // If it exists, only set the length
273    if path.exists() {
274        if let Some(size) = size {
275            let file = OpenOptions::new().write(true).open(path)?;
276            file.set_len(size)?;
277        }
278        return Ok(());
279    }
280
281    // Create file, and make it the correct size
282    let mut file = File::create(path)?;
283    file.write_all(header)?;
284    if let Some(size) = size
285        && size > header.len() as u64
286    {
287        file.set_len(size)?;
288    }
289    Ok(())
290}

292/// Get start position of flags `BitSlice` in deleted mmap.
293#[inline]
294const fn deleted_mmap_data_start() -> usize {
295    let align = mem::align_of::<usize>();
296    HEADER_SIZE.div_ceil(align) * align
297}
298
299/// Calculate size for deleted mmap to hold the given number of vectors.
300///
301/// The mmap will hold a file header and an aligned `BitSlice`.
302fn deleted_mmap_size(num: usize) -> usize {
303    let unit_size = mem::size_of::<usize>();
304    let num_bytes = num.div_ceil(8);
305    let num_usizes = num_bytes.div_ceil(unit_size);
306    let data_size = num_usizes * unit_size;
307    deleted_mmap_data_start() + data_size

22/// In-memory vector storage that is volatile
23///
24/// This storage is not persisted and intended for temporary use in tests.
25#[derive(Debug)]
26pub struct VolatileDenseVectorStorage<T: PrimitiveVectorElement> {
27    dim: usize,
28    distance: Distance,
29    vectors: VolatileChunkedVectors<T>,
30    /// BitVec for deleted flags. Grows dynamically upto last set flag.
31    deleted: BitVec,
32    /// Current number of deleted vectors.
33    deleted_count: usize,
34}
35
36pub fn new_volatile_dense_vector_storage(dim: usize, distance: Distance) -> VectorStorageEnum {
37    VectorStorageEnum::DenseVolatile(VolatileDenseVectorStorage::new(dim, distance))
38}

50impl<T: PrimitiveVectorElement> VolatileDenseVectorStorage<T> {
51    pub fn new(dim: usize, distance: Distance) -> Self {
52        Self {
53            dim,
54            distance,
55            vectors: VolatileChunkedVectors::new(dim),
56            deleted: BitVec::new(),
57            deleted_count: 0,
58        }
59    }
60
61    /// Set deleted flag for given key. Returns previous deleted state.
62    #[inline]
63    fn set_deleted(&mut self, key: PointOffsetType, deleted: bool) -> bool {
64        if !deleted && key as usize >= self.vectors.len() {
65            return false;
66        }
67        let was_deleted = bitvec_set_deleted(&mut self.deleted, key, deleted);
68        if was_deleted != deleted {
69            if !was_deleted {
70                self.deleted_count += 1;
71            } else {
72                self.deleted_count = self.deleted_count.saturating_sub(1);
73            }
74        }
75        was_deleted
76    }
77}

89impl<T: PrimitiveVectorElement> VectorStorage for VolatileDenseVectorStorage<T> {
90    fn distance(&self) -> Distance {
91        self.distance
92    }
93
94    fn datatype(&self) -> VectorStorageDatatype {
95        T::datatype()
96    }
97
98    fn is_on_disk(&self) -> bool {
99        false
100    }
101
102    fn total_vector_count(&self) -> usize {
103        self.vectors.len()
104    }
105
106    fn get_vector<P: AccessPattern>(&self, key: PointOffsetType) -> CowVector<'_> {
107        self.get_vector_opt::<P>(key).expect("vector not found")
108    }
109
110    /// Get vector by key, if it exists.
111    fn get_vector_opt<P: AccessPattern>(&self, key: PointOffsetType) -> Option<CowVector<'_>> {
112        // In memory so no optimization to be done for access pattern
113        self.vectors
114            .get_opt(key as VectorOffsetType)
115            .map(|slice| CowVector::from(T::slice_to_float_cow(slice.into())))
116    }
117
118    fn insert_vector(
119        &mut self,
120        key: PointOffsetType,
121        vector: VectorRef,
122        _hw_counter: &HardwareCounterCell,
123    ) -> OperationResult<()> {
124        let vector: &[VectorElementType] = vector.try_into()?;
125        let vector = T::slice_from_float_cow(Cow::from(vector));
126        self.vectors
127            .insert(key as VectorOffsetType, vector.as_ref())?;
128        self.set_deleted(key, false);
129        Ok(())
130    }
131
132    fn update_from<'a>(
133        &mut self,
134        other_vectors: &'a mut impl Iterator<Item = (CowVector<'a>, bool)>,
135        stopped: &AtomicBool,
136    ) -> OperationResult<Range<PointOffsetType>> {
137        let start_index = self.vectors.len() as PointOffsetType;
138        for (other_vector, other_deleted) in other_vectors {
139            check_process_stopped(stopped)?;
140            // Do not perform preprocessing - vectors should be already processed
141            let other_vector = T::slice_from_float_cow(Cow::try_from(other_vector)?);
142            let new_id = self.vectors.push(other_vector.as_ref())? as PointOffsetType;
143            self.set_deleted(new_id, other_deleted);
144        }
145        let end_index = self.vectors.len() as PointOffsetType;
146        Ok(start_index..end_index)
147    }
148
149    fn flusher(&self) -> Flusher {
150        Box::new(|| Ok(()))
151    }
152
153    fn files(&self) -> Vec<std::path::PathBuf> {
154        vec![]
155    }
156
157    fn delete_vector(&mut self, key: PointOffsetType) -> OperationResult<bool> {
158        let is_deleted = !self.set_deleted(key, true);
159        Ok(is_deleted)
160    }
161
162    fn is_deleted_vector(&self, key: PointOffsetType) -> bool {
163        self.deleted.get_bit(key as usize).unwrap_or(false)
164    }
165
166    fn deleted_vector_count(&self) -> usize {
167        self.deleted_count
168    }
169
170    fn deleted_vector_bitslice(&self) -> &BitSlice {
171        self.deleted.as_bitslice()
172    }
173}

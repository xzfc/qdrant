1use bitvec::vec::BitVec;
2use crate::common::types::PointOffsetType;

4/// Set deleted state in given bitvec.
5///
6/// Grows bitvec automatically if it is not big enough.
7///
8/// Returns previous deleted state of the given point.
9#[inline]
10pub fn bitvec_set_deleted(bitvec: &mut BitVec, point_id: PointOffsetType, deleted: bool) -> bool {
11    // Set deleted flag if bitvec is large enough, no need to check bounds
12    if (point_id as usize) < bitvec.len() {
13        return unsafe { bitvec.replace_unchecked(point_id as usize, deleted) };
14    }
15
16    // Bitvec is too small; grow and set the deletion flag, no need to check bounds
17    if deleted {
18        bitvec.resize(point_id as usize + 1, false);
19        unsafe { bitvec.set_unchecked(point_id as usize, true) };
20    }
21    false
22}

78/// Trait for vector storage
79/// El - type of vector element, expected numerical type
80/// Storage operates with internal IDs (`PointOffsetType`), which always starts with zero and have no skips
81pub trait VectorStorage {
82    fn distance(&self) -> Distance;
83
84    fn datatype(&self) -> VectorStorageDatatype;
85
86    fn is_on_disk(&self) -> bool;
87
88    /// Number of vectors
89    ///
90    /// - includes soft deleted vectors, as they are still stored
91    fn total_vector_count(&self) -> usize;
92
93    /// Get the number of available vectors, considering deleted points and vectors
94    ///
95    /// This uses [`VectorStorage::total_vector_count`] and [`VectorStorage::deleted_vector_count`] internally.
96    ///
97    /// # Warning
98    ///
99    /// This number may not always be accurate. See warning in [`VectorStorage::deleted_vector_count`] documentation.
100    fn available_vector_count(&self) -> usize {
101        self.total_vector_count()
102            .saturating_sub(self.deleted_vector_count())
103    }
104
105    /// Get the vector by the given key
106    /// Get the vector by the given key with potential optimizations for sequential reads.
107    fn get_vector<P: AccessPattern>(&self, key: PointOffsetType) -> CowVector<'_>;
108
109    /// Get multiple vectors by the given keys
110    /// Potentially optimized for internal parallel reads.
111    fn read_vectors<P: AccessPattern>(
112        &self,
113        keys: impl IntoIterator<Item = PointOffsetType>,
114        mut callback: impl FnMut(PointOffsetType, CowVector<'_>),
115    ) {
116        for key in keys {
117            callback(key, self.get_vector::<P>(key));
118        }
119    }
120
121    /// Get the vector by the given key if it exists
122    fn get_vector_opt<P: AccessPattern>(&self, key: PointOffsetType) -> Option<CowVector<'_>>;
123
124    fn insert_vector(
125        &mut self,
126        key: PointOffsetType,
127        vector: VectorRef,
128        hw_counter: &HardwareCounterCell,
129    ) -> OperationResult<()>;
130
131    /// Add the given vectors to the storage.
132    ///
133    /// # Returns
134    /// The range of point offsets that were added to the storage.
135    ///
136    /// If stopped, the operation returns a cancellation error.
137    fn update_from<'a>(
138        &mut self,
139        other_vectors: &'a mut impl Iterator<Item = (CowVector<'a>, bool)>,
140        stopped: &AtomicBool,
141    ) -> OperationResult<Range<PointOffsetType>>;
142
143    fn flusher(&self) -> Flusher;
144
145    fn files(&self) -> Vec<PathBuf>;
146
147    fn immutable_files(&self) -> Vec<PathBuf> {
148        Vec::new()
149    }
150
151    /// Flag the vector by the given key as deleted
152    ///
153    /// Returns true if the vector was not deleted before and is now deleted
154    fn delete_vector(&mut self, key: PointOffsetType) -> OperationResult<bool>;
155
156    /// Check whether the vector at the given key is flagged as deleted
157    fn is_deleted_vector(&self, key: PointOffsetType) -> bool;
158
159    /// Get the number of deleted vectors, considering deleted points and vectors
160    ///
161    /// Vectors may be deleted at two levels, as point or as vector. Deleted points should
162    /// propagate to deleting the vectors. That means that the deleted vector count includes the
163    /// number of deleted points as well.
164    ///
165    /// This includes any vectors that were deleted at creation.
166    ///
167    /// # Warning
168    ///
169    /// In some very exceptional cases it is possible for this count not to include some deleted
170    /// points. That may happen when flushing a segment to disk fails. This should be recovered
171    /// when loading/recovering the segment, but that isn't guaranteed. You should therefore use
172    /// the deleted count with care.
173    fn deleted_vector_count(&self) -> usize;
174
175    /// Get [`BitSlice`] representation for deleted vectors with deletion flags
176    ///
177    /// The size of this slice is not guaranteed. It may be smaller/larger than the number of
178    /// vectors in this segment.
179    fn deleted_vector_bitslice(&self) -> &BitSlice;
180}

182pub trait DenseVectorStorage<T: PrimitiveVectorElement>: VectorStorage {
183    fn vector_dim(&self) -> usize;
184
185    fn get_dense<P: AccessPattern>(&self, key: PointOffsetType) -> Cow<'_, [T]>;
186
187    /// Get the raw bytes of the vector by the given key if it exists
188    fn get_dense_bytes_opt<P: AccessPattern>(&self, key: PointOffsetType) -> Option<Cow<'_, [u8]>> {
189        ((key as usize) < self.total_vector_count()).then(|| match self.get_dense::<P>(key) {
190            Cow::Borrowed(slice) => Cow::Borrowed(bytemuck::cast_slice(slice)),
191            Cow::Owned(vec) => Cow::Owned(bytemuck::cast_vec(vec)),
192        })
193    }
194
195    /// Get layout for a single vector
196    fn get_dense_vector_layout(&self) -> OperationResult<Layout> {
197        Layout::array::<T>(self.vector_dim())
198            .map_err(|_| OperationError::service_error("Layout is too big"))
199    }
200
201    /// Run given function for each vector in the dense batch.
202    ///
203    /// Implementation can assume that the keys are consecutive
204    fn for_each_in_dense_batch<F: FnMut(usize, &[T])>(&self, keys: &[PointOffsetType], mut f: F) {
205        for (idx, &key) in keys.iter().enumerate() {
206            f(idx, &self.get_dense::<Random>(key));
207        }
208    }
209
210    fn size_of_available_vectors_in_bytes(&self) -> usize {
211        self.available_vector_count() * self.vector_dim() * std::mem::size_of::<T>()
212    }
213}

223pub trait MultiVectorStorage<T: PrimitiveVectorElement>: VectorStorage {
224    fn vector_dim(&self) -> usize;
225    fn get_multi<P: AccessPattern>(&self, key: PointOffsetType) -> CowMultiVector<'_, T>;
226    fn get_multi_opt<P: AccessPattern>(
227        &self,
228        key: PointOffsetType,
229    ) -> Option<CowMultiVector<'_, T>>;
230    fn get_batch_multi<'a>(
231        &'a self,
232        keys: &[PointOffsetType],
233        vectors: &'a mut [MaybeUninit<CowMultiVector<'a, T>>],
234    ) -> &'a [CowMultiVector<'a, T>] {
235        debug_assert_eq!(keys.len(), vectors.len());
236        debug_assert!(keys.len() <= VECTOR_READ_BATCH_SIZE);
237        let iter = keys.iter().map(|key| self.get_multi::<Random>(*key));
238        maybe_uninit_fill_from(vectors, iter).0
239    }
240    fn iterate_inner_vectors(&self) -> impl Iterator<Item = Cow<'_, [T]>> + Clone + Send;
241    fn multi_vector_config(&self) -> &MultiVectorConfig;
242
243    fn size_of_available_vectors_in_bytes(&self) -> usize;
244}

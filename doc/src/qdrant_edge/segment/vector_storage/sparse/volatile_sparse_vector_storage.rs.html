21pub const SPARSE_VECTOR_DISTANCE: Distance = Distance::Dot;
22
23/// In-memory vector storage with on-update persistence using `store`
24#[derive(Default, Debug)]
25pub struct VolatileSparseVectorStorage {
26    vectors: Vec<Option<SparseVector>>,
27    /// BitVec for deleted flags. Grows dynamically upto last set flag.
28    deleted: BitVec,
29    /// Current number of deleted vectors.
30    deleted_count: usize,
31    total_vector_count: usize,
32    /// Total number of non-zero elements in all vectors. Used to estimate average vector size.
33    total_sparse_size: usize,
34}
35
36pub fn new_volatile_sparse_vector_storage() -> VectorStorageEnum {
37    VectorStorageEnum::SparseVolatile(VolatileSparseVectorStorage::default())
38}
39
40impl VolatileSparseVectorStorage {
41    /// Set deleted flag for given key. Returns previous deleted state.
42    #[inline]
43    fn set_deleted(&mut self, key: PointOffsetType, deleted: bool) -> bool {
44        if !deleted && key as usize >= self.total_vector_count {
45            return false;
46        }
47        let was_deleted = bitvec_set_deleted(&mut self.deleted, key, deleted);
48        if was_deleted != deleted {
49            if !was_deleted {
50                self.deleted_count += 1;
51            } else {
52                self.deleted_count = self.deleted_count.saturating_sub(1);
53            }
54        }
55        was_deleted
56    }
57
58    fn update_stored(
59        &mut self,
60        key: PointOffsetType,
61        deleted: bool,
62        vector: Option<&SparseVector>,
63    ) {
64        // Resize sparse vector container if needed
65        if key as usize >= self.vectors.len() {
66            if deleted {
67                return;
68            }
69            self.vectors.resize(key as usize + 1, None);
70        }
71
72        let entry = &mut self.vectors[key as usize];
73
74        // Update bookkeeping of total sparse size
75        let elements_removed = entry.as_ref().map_or(0, |v| v.indices.len());
76        let elements_added = vector
77            .as_ref()
78            .filter(|_| !deleted)
79            .map_or(0, |v| v.indices.len());
80        self.total_sparse_size = self
81            .total_sparse_size
82            .saturating_sub(elements_removed)
83            .saturating_add(elements_added);
84
85        if deleted {
86            entry.take();
87        } else {
88            *entry = vector.cloned();
89        }
90    }
91
92    pub fn size_of_available_vectors_in_bytes(&self) -> usize {
93        if self.total_vector_count == 0 {
94            return 0;
95        }
96        let available_fraction =
97            (self.total_vector_count - self.deleted_count) as f32 / self.total_vector_count as f32;
98        let available_size = (self.total_sparse_size as f32 * available_fraction) as usize;
99        available_size * (std::mem::size_of::<DimWeight>() + std::mem::size_of::<DimId>())
100    }
101}

103impl SparseVectorStorage for VolatileSparseVectorStorage {
104    fn get_sparse<P: AccessPattern>(&self, key: PointOffsetType) -> OperationResult<SparseVector> {
105        let vector = self
106            .get_sparse_opt::<P>(key)?
107            .ok_or_else(|| OperationError::service_error("Sparse vector not found in storage"))?;
108        Ok(vector)
109    }
110
111    fn get_sparse_opt<P: AccessPattern>(
112        &self,
113        key: PointOffsetType,
114    ) -> OperationResult<Option<SparseVector>> {
115        // Already in memory, so no sequential optimizations available.
116        let opt_vector = self.vectors.get(key as usize).cloned().flatten();
117        Ok(opt_vector)
118    }
119}
120
121impl VectorStorage for VolatileSparseVectorStorage {
122    fn distance(&self) -> Distance {
123        SPARSE_VECTOR_DISTANCE
124    }
125
126    fn datatype(&self) -> VectorStorageDatatype {
127        VectorStorageDatatype::Float32
128    }
129
130    fn is_on_disk(&self) -> bool {
131        false
132    }
133
134    fn total_vector_count(&self) -> usize {
135        self.total_vector_count
136    }
137
138    fn get_vector<P: AccessPattern>(&self, key: PointOffsetType) -> CowVector<'_> {
139        let vector = self.get_vector_opt::<P>(key);
140        vector.unwrap_or_else(CowVector::default_sparse)
141    }
142
143    /// Get vector by key, if it exists.
144    ///
145    /// ignore any error
146    fn get_vector_opt<P: AccessPattern>(&self, key: PointOffsetType) -> Option<CowVector<'_>> {
147        // In memory, so no sequential read optimization.
148        match self.get_sparse_opt::<P>(key) {
149            Ok(Some(vector)) => Some(CowVector::from(vector)),
150            _ => None,
151        }
152    }
153
154    fn insert_vector(
155        &mut self,
156        key: PointOffsetType,
157        vector: VectorRef,
158        _hw_counter: &HardwareCounterCell,
159    ) -> OperationResult<()> {
160        let vector: &SparseVector = vector.try_into()?;
161        debug_assert!(vector.is_sorted());
162        self.total_vector_count = std::cmp::max(self.total_vector_count, key as usize + 1);
163        self.set_deleted(key, false);
164        self.update_stored(key, false, Some(vector));
165        Ok(())
166    }
167
168    fn update_from<'a>(
169        &mut self,
170        other_vectors: &'a mut impl Iterator<Item = (CowVector<'a>, bool)>,
171        stopped: &AtomicBool,
172    ) -> OperationResult<Range<PointOffsetType>> {
173        let start_index = self.total_vector_count as PointOffsetType;
174        for (other_vector, other_deleted) in other_vectors {
175            check_process_stopped(stopped)?;
176            // Do not perform preprocessing - vectors should be already processed
177            let other_vector = other_vector.as_vec_ref().try_into()?;
178            let new_id = self.total_vector_count as PointOffsetType;
179            self.total_vector_count += 1;
180            self.set_deleted(new_id, other_deleted);
181            self.update_stored(new_id, other_deleted, Some(other_vector));
182        }
183        Ok(start_index..self.total_vector_count as PointOffsetType)
184    }
185
186    fn flusher(&self) -> Flusher {
187        Box::new(|| Ok(()))
188    }
189
190    fn files(&self) -> Vec<std::path::PathBuf> {
191        vec![]
192    }
193
194    fn delete_vector(&mut self, key: PointOffsetType) -> OperationResult<bool> {
195        let is_deleted = !self.set_deleted(key, true);
196        if is_deleted {
197            let old_vector = self.get_sparse_opt::<Random>(key).ok().flatten();
198            self.update_stored(key, true, old_vector.as_ref());
199        }
200        Ok(is_deleted)
201    }
202
203    fn is_deleted_vector(&self, key: PointOffsetType) -> bool {
204        self.deleted.get_bit(key as usize).unwrap_or(false)
205    }
206
207    fn deleted_vector_count(&self) -> usize {
208        self.deleted_count
209    }
210
211    fn deleted_vector_bitslice(&self) -> &BitSlice {
212        self.deleted.as_bitslice()
213    }

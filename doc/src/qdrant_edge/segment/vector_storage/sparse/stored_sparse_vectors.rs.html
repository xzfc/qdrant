7use crate::segment::common::operation_error::OperationError;

9#[derive(Debug, PartialEq, Clone, Serialize, Deserialize)]
10pub struct StoredSparseVector {
11    /// Compressed u64 indices
12    pub indices: Vec<u8>,
13    /// Values and indices must be the same length
14    pub values: Vec<DimWeight>,
15}
16
17impl StoredSparseVector {
18    /// Convert indices into a byte array
19    /// Use bitpacking and delta-encoding for additional compression
20    fn serialize_indices(indices: &[DimId64]) -> Vec<u8> {
21        delta_pack(indices)
22    }
23
24    /// Recover indices from a byte array
25    fn deserialize_indices(data: &[u8]) -> Vec<DimId64> {
26        delta_unpack(data)
27    }
28}
29
30impl From<&SparseVector> for StoredSparseVector {
31    fn from(vector: &SparseVector) -> Self {
32        let mut stored_indices: Vec<_> =
33            vector.indices.iter().copied().map(DimId64::from).collect();
34        let mut stored_values = vector.values.clone();
35
36        double_sort(&mut stored_indices, &mut stored_values);
37
38        let compressed_indices = StoredSparseVector::serialize_indices(&stored_indices);
39
40        Self {
41            indices: compressed_indices,
42            values: stored_values,
43        }
44    }
45}
46
47impl TryFrom<StoredSparseVector> for SparseVector {
48    type Error = OperationError;
49
50    fn try_from(value: StoredSparseVector) -> Result<Self, Self::Error> {
51        let decompressed_indices = StoredSparseVector::deserialize_indices(&value.indices);
52
53        Ok(SparseVector {
54            indices: decompressed_indices
55                .into_iter()
56                .map(DimId::try_from)
57                .collect::<Result<_, _>>()
58                .map_err(|err| {
59                    OperationError::service_error(format!("Failed to convert indices: {err}"))
60                })?,
61            values: value.values,
62        })
63    }
64}
65
66impl Blob for StoredSparseVector {
67    fn to_bytes(&self) -> Vec<u8> {
68        bincode::serialize(&self).expect("Sparse vector serialization should not fail")
69    }
70
71    fn from_bytes(data: &[u8]) -> Self {
72        bincode::deserialize(data).expect("Sparse vector deserialization should not fail")
73    }

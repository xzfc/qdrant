45#[derive(Debug)]
46pub struct AppendableMmapMultiDenseVectorStorage<T: PrimitiveVectorElement> {
47    vectors: ChunkedVectors<T, MmapUniversal<T>>,
48    offsets: ChunkedVectors<MultivectorMmapOffset, MmapUniversal<MultivectorMmapOffset>>,
49    /// Flags marking deleted vectors
50    ///
51    /// Structure grows dynamically, but may be smaller than actual number of vectors. Must not
52    /// depend on its length.
53    deleted: BitvecFlags,
54    distance: Distance,
55    multi_vector_config: MultiVectorConfig,
56    deleted_count: usize,
57    _phantom: std::marker::PhantomData<T>,
58}

60impl<T: PrimitiveVectorElement> AppendableMmapMultiDenseVectorStorage<T> {
61    /// Set deleted flag for given key. Returns previous deleted state.
62    #[inline]
63    fn set_deleted(&mut self, key: PointOffsetType, deleted: bool) -> bool {
64        if !deleted && self.vectors.len() <= key as usize {
65            return false;
66        }
67
68        // set value
69        let previous = self.deleted.set(key, deleted);
70
71        // update counter
72        if !previous && deleted {
73            self.deleted_count += 1;
74        } else if previous && !deleted {
75            self.deleted_count -= 1;
76        }
77
78        previous
79    }
80
81    /// Populate all pages in the mmap.
82    /// Block until all pages are populated.
83    pub fn populate(&self) -> OperationResult<()> {
84        // deleted bitvec is already loaded
85        self.vectors.populate()?;
86        self.offsets.populate()?;
87        Ok(())
88    }
89
90    /// Drop disk cache.
91    pub fn clear_cache(&self) -> OperationResult<()> {
92        self.vectors.clear_cache()?;
93        self.offsets.clear_cache()?;
94        Ok(())
95    }
96}

98impl<T: PrimitiveVectorElement> MultiVectorStorage<T> for AppendableMmapMultiDenseVectorStorage<T> {
99    fn vector_dim(&self) -> usize {
100        self.vectors.dim()
101    }
102
103    /// Panics if key is not found
104    fn get_multi<P: AccessPattern>(&self, key: PointOffsetType) -> CowMultiVector<'_, T> {
105        self.get_multi_opt::<P>(key).expect("vector not found")
106    }
107
108    /// Returns None if key is not found
109    fn get_multi_opt<P: AccessPattern>(
110        &self,
111        key: PointOffsetType,
112    ) -> Option<CowMultiVector<'_, T>> {
113        self.offsets
114            .get::<P>(key as VectorOffsetType)
115            .and_then(|mmap_offset| {
116                let mmap_offset = mmap_offset.first().expect("mmap_offset must not be empty");
117                self.vectors.get_many::<P>(
118                    mmap_offset.offset as VectorOffsetType,
119                    mmap_offset.count as usize,
120                )
121            })
122            .map(|flattened_vectors| match flattened_vectors {
123                Cow::Borrowed(slice) => CowMultiVector::Borrowed(TypedMultiDenseVectorRef {
124                    flattened_vectors: slice,
125                    dim: self.vectors.dim(),
126                }),
127                Cow::Owned(vec) => CowMultiVector::Owned(TypedMultiDenseVector {
128                    flattened_vectors: vec,
129                    dim: self.vectors.dim(),
130                }),
131            })
132    }
133
134    fn iterate_inner_vectors(&self) -> impl Iterator<Item = Cow<'_, [T]>> + Clone + Send {
135        (0..self.total_vector_count()).flat_map(move |key| {
136            let mmap_offset = self
137                .offsets
138                .get::<Sequential>(key as VectorOffsetType)
139                .unwrap()
140                .first()
141                .copied()
142                .unwrap();
143            (0..mmap_offset.count).map(move |i| {
144                self.vectors
145                    .get::<Sequential>((mmap_offset.offset + i) as VectorOffsetType)
146                    .unwrap()
147            })
148        })
149    }
150
151    fn multi_vector_config(&self) -> &MultiVectorConfig {
152        &self.multi_vector_config
153    }
154
155    fn size_of_available_vectors_in_bytes(&self) -> usize {
156        if self.total_vector_count() > 0 {
157            let total_size = self.vectors.len() * self.vector_dim() * std::mem::size_of::<T>();
158            (total_size as u128 * self.available_vector_count() as u128
159                / self.total_vector_count() as u128) as usize
160        } else {
161            0
162        }
163    }
164}

166impl<T: PrimitiveVectorElement> VectorStorage for AppendableMmapMultiDenseVectorStorage<T> {
167    fn distance(&self) -> Distance {
168        self.distance
169    }
170
171    fn datatype(&self) -> VectorStorageDatatype {
172        T::datatype()
173    }
174
175    fn is_on_disk(&self) -> bool {
176        self.vectors.is_on_disk()
177    }
178
179    fn total_vector_count(&self) -> usize {
180        self.offsets.len()
181    }
182
183    fn get_vector<P: AccessPattern>(&self, key: PointOffsetType) -> CowVector<'_> {
184        self.get_vector_opt::<P>(key).expect("vector not found")
185    }
186
187    fn get_vector_opt<P: AccessPattern>(&self, key: PointOffsetType) -> Option<CowVector<'_>> {
188        self.get_multi_opt::<P>(key).map(|multi_dense_vector| {
189            CowVector::MultiDense(T::into_float_multivector(multi_dense_vector))
190        })
191    }
192
193    fn insert_vector(
194        &mut self,
195        key: PointOffsetType,
196        vector: VectorRef,
197        hw_counter: &HardwareCounterCell,
198    ) -> OperationResult<()> {
199        let multi_vector: TypedMultiDenseVectorRef<VectorElementType> = vector.try_into()?;
200        let multi_vector = T::from_float_multivector(CowMultiVector::Borrowed(multi_vector));
201        let multi_vector = multi_vector.as_vec_ref();
202        assert_eq!(multi_vector.dim, self.vectors.dim());
203        let multivector_size_in_bytes = std::mem::size_of_val(multi_vector.flattened_vectors);
204        let max_vector_size_bytes = self.vectors.max_vector_size_bytes();
205        if multivector_size_in_bytes >= max_vector_size_bytes {
206            return Err(OperationError::service_error(format!(
207                "Cannot insert multi vector of size {multivector_size_in_bytes} to the mmap vector storage.\
208                 It's too large, maximum size is {max_vector_size_bytes}."
209            )));
210        }
211
212        let mut offset = self
213            .offsets
214            .get::<Random>(key as VectorOffsetType)
215            .map(|x| x.first().copied().unwrap_or_default())
216            .unwrap_or_default();
217
218        if multi_vector.vectors_count() > offset.capacity as usize {
219            // append vector to the end
220            let mut new_key = self.vectors.len();
221            let chunk_left_keys = self.vectors.get_remaining_chunk_keys(new_key);
222            if multi_vector.vectors_count() > chunk_left_keys {
223                new_key += chunk_left_keys;
224            }
225
226            offset = MultivectorMmapOffset {
227                offset: new_key as PointOffsetType,
228                count: multi_vector.vectors_count() as PointOffsetType,
229                capacity: multi_vector.vectors_count() as PointOffsetType,
230            };
231        } else {
232            // use existing place to insert vector
233            offset.count = multi_vector.vectors_count() as PointOffsetType;
234        }
235
236        self.vectors.insert_many(
237            offset.offset as VectorOffsetType,
238            multi_vector.flattened_vectors,
239            multi_vector.vectors_count(),
240            hw_counter,
241        )?;
242        self.offsets
243            .insert(key as VectorOffsetType, &[offset], hw_counter)?;
244        self.set_deleted(key, false);
245
246        Ok(())
247    }
248
249    fn update_from<'a>(
250        &mut self,
251        other_vectors: &'a mut impl Iterator<Item = (CowVector<'a>, bool)>,
252        stopped: &AtomicBool,
253    ) -> OperationResult<Range<PointOffsetType>> {
254        let start_index = self.offsets.len() as PointOffsetType;
255        let disposed_hw_counter = HardwareCounterCell::disposable(); // Internal operation
256        for (other_vector, other_deleted) in other_vectors {
257            check_process_stopped(stopped)?;
258            // Do not perform preprocessing - vectors should be already processed
259            let other_vector: VectorRef = other_vector.as_vec_ref();
260            let new_id = self.offsets.len() as PointOffsetType;
261            self.insert_vector(new_id, other_vector, &disposed_hw_counter)?;
262            self.set_deleted(new_id, other_deleted);
263        }
264        let end_index = self.offsets.len() as PointOffsetType;
265        Ok(start_index..end_index)
266    }
267
268    fn flusher(&self) -> Flusher {
269        Box::new({
270            let vectors_flusher = self.vectors.flusher();
271            let offsets_flusher = self.offsets.flusher();
272            let deleted_flusher = self.deleted.flusher();
273            move || {
274                vectors_flusher()?;
275                offsets_flusher()?;
276                deleted_flusher()?;
277                Ok(())
278            }
279        })
280    }
281
282    fn files(&self) -> Vec<PathBuf> {
283        let mut files = self.vectors.files();
284        files.extend(self.offsets.files());
285        files.extend(self.deleted.files());
286        files
287    }
288
289    fn immutable_files(&self) -> Vec<PathBuf> {
290        let mut files = self.vectors.immutable_files();
291        files.extend(self.offsets.immutable_files());
292        files
293    }
294
295    fn delete_vector(&mut self, key: PointOffsetType) -> OperationResult<bool> {
296        Ok(self.set_deleted(key, true))
297    }
298
299    fn is_deleted_vector(&self, key: PointOffsetType) -> bool {
300        self.deleted.get(key)
301    }
302
303    fn deleted_vector_count(&self) -> usize {
304        self.deleted_count
305    }
306
307    fn deleted_vector_bitslice(&self) -> &BitSlice {
308        self.deleted.get_bitslice()
309    }
310}

448pub fn open_appendable_memmap_multi_vector_storage_impl<T: PrimitiveVectorElement>(
449    path: &Path,
450    dim: usize,
451    distance: Distance,
452    multi_vector_config: MultiVectorConfig,
453    madvise: AdviceSetting,
454    populate: bool,
455) -> OperationResult<AppendableMmapMultiDenseVectorStorage<T>> {
456    fs::create_dir_all(path)?;
457
458    let vectors_path = path.join(VECTORS_DIR_PATH);
459    let offsets_path = path.join(OFFSETS_DIR_PATH);
460    let deleted_path = path.join(DELETED_DIR_PATH);
461
462    let vectors = ChunkedVectors::open(&vectors_path, dim, madvise, Some(populate))?;
463    let offsets = ChunkedVectors::open(&offsets_path, 1, madvise, Some(populate))?;
464
465    let deleted = BitvecFlags::new(DynamicMmapFlags::open(&deleted_path, populate)?);
466    let deleted_count = deleted.count_trues();
467
468    Ok(AppendableMmapMultiDenseVectorStorage {
469        vectors,
470        offsets,
471        deleted,
472        distance,
473        multi_vector_config,
474        deleted_count,
475        _phantom: Default::default(),
476    })
477}

479/// Find files related to this dense vector storage
480#[cfg(any(test, feature = "rocksdb"))]
481pub(crate) fn find_storage_files(vector_storage_path: &Path) -> OperationResult<Vec<PathBuf>> {
482    let vectors_path = vector_storage_path.join(VECTORS_DIR_PATH);
483    let offsets_path = vector_storage_path.join(OFFSETS_DIR_PATH);
484    let deleted_path = vector_storage_path.join(DELETED_DIR_PATH);
485
486    let mut files = vec![];
487    files.extend(crate::common::disk::list_files(&vectors_path)?);
488    files.extend(crate::common::disk::list_files(&offsets_path)?);
489    files.extend(crate::common::disk::list_files(&deleted_path)?);
490    Ok(files)
491}

493#[cfg(test)]
494mod tests {
495    use std::collections::HashSet;
496
497    use rand::rngs::StdRng;
498    use rand::{RngExt, SeedableRng};
499    use tempfile::Builder;
500
501    use super::*;
502    use crate::segment::data_types::vectors::MultiDenseVectorInternal;
503
504    const RAND_SEED: u64 = 42;
505
506    /// Test that `find_storage_files` finds all files that are reported by the storage.
507    #[test]
508    fn test_find_storage_files() {
509        // Numbers chosen so we get 3 data chunks, not just 1
510        const POINT_COUNT: PointOffsetType = 1000;
511        const DIM: usize = 128;
512
513        let mutli_vector_config = MultiVectorConfig::default();
514        let dir = Builder::new().prefix("storage_dir").tempdir().unwrap();
515        let mut storage = open_appendable_memmap_multi_vector_storage_full(
516            dir.path(),
517            DIM,
518            Distance::Dot,
519            mutli_vector_config,
520            AdviceSetting::Global,
521            false,
522        )
523        .unwrap();
524
525        let mut rng = StdRng::seed_from_u64(RAND_SEED);
526        let hw_counter = HardwareCounterCell::disposable();
527
528        // Insert points, delete 10% of it, and flush
529        for internal_id in 0..POINT_COUNT {
530            let size = rng.random_range(1..=4);
531            let vectors = std::iter::repeat_with(|| {
532                std::iter::repeat_with(|| rng.random_range(-1.0..1.0))
533                    .take(DIM)
534                    .collect()
535            })
536            .take(size)
537            .collect::<Vec<Vec<_>>>();
538            let multivec = MultiDenseVectorInternal::try_from(vectors).unwrap();
539            storage
540                .insert_vector(internal_id, VectorRef::from(&multivec), &hw_counter)
541                .unwrap();
542        }
543        for internal_id in 0..POINT_COUNT {
544            if !rng.random_bool(0.1) {
545                continue;
546            }
547            storage.delete_vector(internal_id).unwrap();
548        }
549        storage.flusher()().unwrap();
550
551        let storage_files = storage.files().into_iter().collect::<HashSet<_>>();
552        let found_files = find_storage_files(dir.path())
553            .unwrap()
554            .into_iter()
555            .collect::<HashSet<_>>();
556
557        assert_eq!(
558            storage_files, found_files,
559            "find_storage_files must find same files that storage reports",
560        );
561    }

19pub trait QueryScorer {
20    type TVector: ?Sized;
21
22    fn score_stored(&self, idx: PointOffsetType) -> ScoreType;
23
24    /// Score a batch of points
25    ///
26    /// Enable underlying storage to optimize pre-fetching of data
27    fn score_stored_batch(&self, ids: &[PointOffsetType], scores: &mut [ScoreType]) {
28        debug_assert!(ids.len() <= VECTOR_READ_BATCH_SIZE);
29        debug_assert_eq!(ids.len(), scores.len());
30
31        // no specific implementation for batch scoring
32        for (idx, id) in ids.iter().enumerate() {
33            scores[idx] = self.score_stored(*id);
34        }
35    }
36
37    fn score(&self, v2: &Self::TVector) -> ScoreType;
38
39    fn score_internal(&self, point_a: PointOffsetType, point_b: PointOffsetType) -> ScoreType;
40
41    type SupportsBytes: TBool;
42    fn score_bytes(&self, _: Self::SupportsBytes, bytes: &[u8]) -> ScoreType;
43}

67/// Colbert MaxSim metric, metric for multi-dense vectors
68/// <https://arxiv.org/pdf/2112.01488.pdf>, figure 1
69/// This metric is also implemented in `QuantizedMultivectorStorage` structure for quantized data.
70///
71/// Disclaimer: this score is not equivalent to the original Colbert metric for Euclidean space because we do not apply the post-processing step.
72/// In that case the score value will be different but the ranking will be the same.
73/// We do that because of performance reasons and complex sort ordering of the vectors when applying the post-processing step.
74pub fn score_max_similarity<T: PrimitiveVectorElement, TMetric: Metric<T>>(
75    multi_dense_a: TypedMultiDenseVectorRef<'_, T>,
76    multi_dense_b: TypedMultiDenseVectorRef<'_, T>,
77) -> ScoreType {
78    debug_assert!(!multi_dense_a.is_empty());
79    debug_assert!(!multi_dense_b.is_empty());
80    let mut sum = 0.0;
81    for dense_a in multi_dense_a.multi_vectors() {
82        let mut max_sim = ScoreType::NEG_INFINITY;
83        // manual `max_by` for performance
84        for dense_b in multi_dense_b.multi_vectors() {
85            let sim = TMetric::similarity(dense_a, dense_b);
86            if sim > max_sim {
87                max_sim = sim;
88            }
89        }
90        // sum of max similarity
91        sum += max_sim;
92    }
93    sum
94}
95
96fn score_multi<T: PrimitiveVectorElement, TMetric: Metric<T>>(
97    multi_vector_config: &MultiVectorConfig,
98    multi_dense_a: TypedMultiDenseVectorRef<'_, T>,
99    multi_dense_b: TypedMultiDenseVectorRef<'_, T>,
100) -> ScoreType {
101    match multi_vector_config.comparator {
102        MultiVectorComparator::MaxSim => {
103            score_max_similarity::<T, TMetric>(multi_dense_a, multi_dense_b)
104        }
105    }
106}

108/// Check if ids are rather contiguous to enable further optimizations
109/// TODO: this can be smarter, but requires experiments with actual mmap behaviour
110/// TODO: For example
111///
112/// - If the whole batch is less than one page - don't use prefetch
113/// - If one vector is bigger then the prefetch size - don't use prefetch
114/// - ???
115pub fn is_read_with_prefetch_efficient<O: VectorOffset>(ids: &[O]) -> bool {
116    let mut min = usize::MAX;
117    let mut max = 0;
118    let mut n = 0;
119
120    for id in ids {
121        let id = id.offset();
122        if id < min {
123            min = id;
124        }
125        if id > max {
126            max = id;
127        }
128        n += 1;
129    }
130
131    if n < 2 {
132        return false;
133    }
134
135    let diff = max.saturating_sub(min);
136
137    diff < n * 2
138}

164    #[test]
165    fn test_score_multi_euclidean() {
166        let a = MultiDenseVectorInternal::try_from(vec![
167            vec![1.0, 2.0, 3.0],
168            vec![3.0, 3.0, 3.0],
169            vec![4.0, 5.0, 6.0],
170        ])
171        .unwrap();
172        // distance to itself
173        let score = score_max_similarity::<f32, EuclidMetric>((&a).into(), (&a).into());
174        assert_eq!(score, -0.0);
175
176        let b = MultiDenseVectorInternal::try_from(vec![vec![3.0, 3.0, 3.0], vec![4.0, 2.0, 1.0]])
177            .unwrap();
178        let score = score_max_similarity::<f32, EuclidMetric>((&a).into(), (&b).into());
179        // proper value according to theory should be `5.9777255` but we do not apply post-processing step
180        assert_eq!(score, -19.);
181    }

13const BATCH_SIZE: usize = 256;
14
15/// Define location of the point source during segment construction.
16pub struct PointData {
17    pub external_id: CompactExtendedPointId,
18    /// [`CompactExtendedPointId`] is 17 bytes, we reduce
19    /// `segment_index` to 3 bytes to avoid paddings and align nicely.
20    pub segment_index: U24,
21    pub internal_id: PointOffsetType,
22    pub version: u64,
23    pub ordering: u64,
24}
25
26/// Batched iterator over points to insert.
27/// This structure should read `BATCH_SIZE` points into a buffer,
28/// and then iterate over them.
29pub struct BatchedVectorReader<'a> {
30    points_to_insert: &'a [PointData],
31    source_vector_storages: &'a [AtomicRef<'a, VectorStorageEnum>],
32    buffer: Vec<(CowVector<'a>, bool)>,
33    seg_to_points_buffer: AHashMap<U24, Vec<(&'a PointData, usize)>>,
34    /// Global position of the iterator.
35    /// From 0 to `points_to_insert.len()`.
36    position: usize,
37}
38
39impl<'a> BatchedVectorReader<'a> {
40    pub fn new(
41        points_to_insert: &'a [PointData],
42        source_vector_storages: &'a [AtomicRef<'a, VectorStorageEnum>],
43    ) -> BatchedVectorReader<'a> {
44        // We need to allocate the buffer with the size of the batch,
45        // but we don't know the size of the vectors.
46        // So we use a placeholder vector with size 0.
47        let buffer = vec![(CowVector::default(), false); BATCH_SIZE];
48
49        BatchedVectorReader {
50            points_to_insert,
51            source_vector_storages,
52            buffer,
53            seg_to_points_buffer: AHashMap::default(),
54            position: 0,
55        }
56    }
57
58    /// Fills the buffer with the next batch of points.
59    ///
60    /// Reading of a single point looks like this:
61    ///
62    /// ```text
63    ///  let source_vector_storage = &source_vector_storages[point_data.segment_index.get() as usize];
64    ///  let vec = source_vector_storage.get_vector(point_data.internal_id);
65    ///  let vector_deleted = source_vector_storage.is_deleted_vector(point_data.internal_id);
66    ///  (vec, vector_deleted)
67    /// ```
68    fn refill_buffer(&mut self) {
69        let start_pos = self.position;
70        let end_pos = min(self.position + BATCH_SIZE, self.points_to_insert.len());
71
72        // Read by segments, as we want to localize reads as much as possible.
73        for pos in start_pos..end_pos {
74            let point_data = &self.points_to_insert[pos];
75            let offset_in_batch = pos - start_pos;
76
77            self.seg_to_points_buffer
78                .entry(point_data.segment_index)
79                .or_default()
80                .push((point_data, offset_in_batch))
81        }
82
83        for (segment_index, points) in self.seg_to_points_buffer.drain() {
84            let source_vector_storage = &self.source_vector_storages[segment_index.get() as usize];
85            for (point_data, offset_in_batch) in points {
86                let vec = source_vector_storage.get_vector::<Sequential>(point_data.internal_id);
87                let vector_deleted =
88                    source_vector_storage.is_deleted_vector(point_data.internal_id);
89                self.buffer[offset_in_batch] = (vec, vector_deleted);
90            }
91        }
92    }
93
94    fn refill_buffer_if_needed(&mut self) {
95        if self.position.is_multiple_of(BATCH_SIZE) {
96            self.refill_buffer();
97        }
98    }
99}
100
101impl<'a> Iterator for BatchedVectorReader<'a> {
102    type Item = (CowVector<'a>, bool);
103
104    fn next(&mut self) -> Option<Self::Item> {
105        if self.position >= self.points_to_insert.len() {
106            return None;
107        }
108
109        self.refill_buffer_if_needed();
110
111        let item = self.buffer[self.position % BATCH_SIZE].clone();
112        self.position += 1;
113
114        Some(item)
115    }
116}

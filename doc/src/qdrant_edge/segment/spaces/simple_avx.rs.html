8#[target_feature(enable = "avx")]
9#[allow(clippy::missing_safety_doc)]
10pub unsafe fn hsum256_ps_avx(x: __m256) -> f32 {
11    let lr_sum: __m128 = _mm_add_ps(_mm256_extractf128_ps(x, 1), _mm256_castps256_ps128(x));
12    let hsum = _mm_hadd_ps(lr_sum, lr_sum);
13    let p1 = _mm_extract_ps(hsum, 0);
14    let p2 = _mm_extract_ps(hsum, 1);
15    f32::from_bits(p1 as u32) + f32::from_bits(p2 as u32)
16}

18/// Calculates the hsum (horizontal sum) of four 32 byte registers.
19#[target_feature(enable = "avx")]
20#[allow(clippy::missing_safety_doc)]
21pub unsafe fn four_way_hsum(a: __m256, b: __m256, c: __m256, d: __m256) -> f32 {
22    unsafe {
23        let sum1 = _mm256_add_ps(a, b);
24        let sum2 = _mm256_add_ps(c, d);
25        let total = _mm256_add_ps(sum1, sum2);
26        hsum256_ps_avx(total)
27    }
28}

30#[target_feature(enable = "avx")]
31#[target_feature(enable = "fma")]
32pub(crate) unsafe fn euclid_similarity_avx(
33    v1: &[VectorElementType],
34    v2: &[VectorElementType],
35) -> ScoreType {
36    unsafe {
37        let n = v1.len();
38        let m = n - (n % 32);
39        let mut ptr1: *const f32 = v1.as_ptr();
40        let mut ptr2: *const f32 = v2.as_ptr();
41        let mut sum256_1: __m256 = _mm256_setzero_ps();
42        let mut sum256_2: __m256 = _mm256_setzero_ps();
43        let mut sum256_3: __m256 = _mm256_setzero_ps();
44        let mut sum256_4: __m256 = _mm256_setzero_ps();
45        let mut i: usize = 0;
46        while i < m {
47            let sub256_1: __m256 =
48                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(0)), _mm256_loadu_ps(ptr2.add(0)));
49            sum256_1 = _mm256_fmadd_ps(sub256_1, sub256_1, sum256_1);
50
51            let sub256_2: __m256 =
52                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(8)), _mm256_loadu_ps(ptr2.add(8)));
53            sum256_2 = _mm256_fmadd_ps(sub256_2, sub256_2, sum256_2);
54
55            let sub256_3: __m256 =
56                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(16)), _mm256_loadu_ps(ptr2.add(16)));
57            sum256_3 = _mm256_fmadd_ps(sub256_3, sub256_3, sum256_3);
58
59            let sub256_4: __m256 =
60                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(24)), _mm256_loadu_ps(ptr2.add(24)));
61            sum256_4 = _mm256_fmadd_ps(sub256_4, sub256_4, sum256_4);
62
63            ptr1 = ptr1.add(32);
64            ptr2 = ptr2.add(32);
65            i += 32;
66        }
67
68        let mut result = four_way_hsum(sum256_1, sum256_2, sum256_3, sum256_4);
69
70        for i in 0..n - m {
71            result += (*ptr1.add(i) - *ptr2.add(i)).powi(2);
72        }
73        -result
74    }
75}

77#[target_feature(enable = "avx")]
78#[target_feature(enable = "fma")]
79pub(crate) unsafe fn manhattan_similarity_avx(
80    v1: &[VectorElementType],
81    v2: &[VectorElementType],
82) -> ScoreType {
83    unsafe {
84        let mask: __m256 = _mm256_set1_ps(-0.0f32); // 1 << 31 used to clear sign bit to mimic abs
85
86        let n = v1.len();
87        let m = n - (n % 32);
88        let mut ptr1: *const f32 = v1.as_ptr();
89        let mut ptr2: *const f32 = v2.as_ptr();
90        let mut sum256_1: __m256 = _mm256_setzero_ps();
91        let mut sum256_2: __m256 = _mm256_setzero_ps();
92        let mut sum256_3: __m256 = _mm256_setzero_ps();
93        let mut sum256_4: __m256 = _mm256_setzero_ps();
94        let mut i: usize = 0;
95        while i < m {
96            let sub256_1: __m256 = _mm256_sub_ps(_mm256_loadu_ps(ptr1), _mm256_loadu_ps(ptr2));
97            sum256_1 = _mm256_add_ps(_mm256_andnot_ps(mask, sub256_1), sum256_1);
98
99            let sub256_2: __m256 =
100                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(8)), _mm256_loadu_ps(ptr2.add(8)));
101            sum256_2 = _mm256_add_ps(_mm256_andnot_ps(mask, sub256_2), sum256_2);
102
103            let sub256_3: __m256 =
104                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(16)), _mm256_loadu_ps(ptr2.add(16)));
105            sum256_3 = _mm256_add_ps(_mm256_andnot_ps(mask, sub256_3), sum256_3);
106
107            let sub256_4: __m256 =
108                _mm256_sub_ps(_mm256_loadu_ps(ptr1.add(24)), _mm256_loadu_ps(ptr2.add(24)));
109            sum256_4 = _mm256_add_ps(_mm256_andnot_ps(mask, sub256_4), sum256_4);
110
111            ptr1 = ptr1.add(32);
112            ptr2 = ptr2.add(32);
113            i += 32;
114        }
115
116        let mut result = four_way_hsum(sum256_1, sum256_2, sum256_3, sum256_4);
117
118        for i in 0..n - m {
119            result += (*ptr1.add(i) - *ptr2.add(i)).abs();
120        }
121        -result
122    }
123}

125#[target_feature(enable = "avx")]
126#[target_feature(enable = "fma")]
127pub(crate) unsafe fn cosine_preprocess_avx(vector: DenseVector) -> DenseVector {
128    unsafe {
129        let n = vector.len();
130        let m = n - (n % 32);
131        let mut ptr: *const f32 = vector.as_ptr();
132        let mut sum256_1: __m256 = _mm256_setzero_ps();
133        let mut sum256_2: __m256 = _mm256_setzero_ps();
134        let mut sum256_3: __m256 = _mm256_setzero_ps();
135        let mut sum256_4: __m256 = _mm256_setzero_ps();
136        let mut i: usize = 0;
137        while i < m {
138            let m256_1 = _mm256_loadu_ps(ptr);
139            sum256_1 = _mm256_fmadd_ps(m256_1, m256_1, sum256_1);
140
141            let m256_2 = _mm256_loadu_ps(ptr.add(8));
142            sum256_2 = _mm256_fmadd_ps(m256_2, m256_2, sum256_2);
143
144            let m256_3 = _mm256_loadu_ps(ptr.add(16));
145            sum256_3 = _mm256_fmadd_ps(m256_3, m256_3, sum256_3);
146
147            let m256_4 = _mm256_loadu_ps(ptr.add(24));
148            sum256_4 = _mm256_fmadd_ps(m256_4, m256_4, sum256_4);
149
150            ptr = ptr.add(32);
151            i += 32;
152        }
153
154        let mut length = four_way_hsum(sum256_1, sum256_2, sum256_3, sum256_4);
155
156        for i in 0..n - m {
157            length += (*ptr.add(i)).powi(2);
158        }
159        if is_length_zero_or_normalized(length) {
160            return vector;
161        }
162        length = length.sqrt();
163        vector.into_iter().map(|x| x / length).collect()
164    }
165}

167#[target_feature(enable = "avx")]
168#[target_feature(enable = "fma")]
169pub(crate) unsafe fn dot_similarity_avx(
170    v1: &[VectorElementType],
171    v2: &[VectorElementType],
172) -> ScoreType {
173    unsafe {
174        let n = v1.len();
175        let m = n - (n % 32);
176        let mut ptr1: *const f32 = v1.as_ptr();
177        let mut ptr2: *const f32 = v2.as_ptr();
178        let mut sum256_1: __m256 = _mm256_setzero_ps();
179        let mut sum256_2: __m256 = _mm256_setzero_ps();
180        let mut sum256_3: __m256 = _mm256_setzero_ps();
181        let mut sum256_4: __m256 = _mm256_setzero_ps();
182        let mut i: usize = 0;
183        while i < m {
184            sum256_1 = _mm256_fmadd_ps(_mm256_loadu_ps(ptr1), _mm256_loadu_ps(ptr2), sum256_1);
185            sum256_2 = _mm256_fmadd_ps(
186                _mm256_loadu_ps(ptr1.add(8)),
187                _mm256_loadu_ps(ptr2.add(8)),
188                sum256_2,
189            );
190            sum256_3 = _mm256_fmadd_ps(
191                _mm256_loadu_ps(ptr1.add(16)),
192                _mm256_loadu_ps(ptr2.add(16)),
193                sum256_3,
194            );
195            sum256_4 = _mm256_fmadd_ps(
196                _mm256_loadu_ps(ptr1.add(24)),
197                _mm256_loadu_ps(ptr2.add(24)),
198                sum256_4,
199            );
200
201            ptr1 = ptr1.add(32);
202            ptr2 = ptr2.add(32);
203            i += 32;
204        }
205
206        let mut result = four_way_hsum(sum256_1, sum256_2, sum256_3, sum256_4);
207
208        for i in 0..n - m {
209            result += (*ptr1.add(i)) * (*ptr2.add(i));
210        }
211        result
212    }
213}

12#[allow(clippy::missing_safety_doc)]
13pub unsafe fn hsum128_ps_sse(x: __m128) -> f32 {
14    let x64: __m128 = _mm_add_ps(x, _mm_movehl_ps(x, x));
15    let x32: __m128 = _mm_add_ss(x64, _mm_shuffle_ps(x64, x64, 0x55));
16    _mm_cvtss_f32(x32)
17}

19#[target_feature(enable = "sse")]
20pub(crate) unsafe fn euclid_similarity_sse(
21    v1: &[VectorElementType],
22    v2: &[VectorElementType],
23) -> ScoreType {
24    unsafe {
25        let n = v1.len();
26        let m = n - (n % 16);
27        let mut ptr1: *const f32 = v1.as_ptr();
28        let mut ptr2: *const f32 = v2.as_ptr();
29        let mut sum128_1: __m128 = _mm_setzero_ps();
30        let mut sum128_2: __m128 = _mm_setzero_ps();
31        let mut sum128_3: __m128 = _mm_setzero_ps();
32        let mut sum128_4: __m128 = _mm_setzero_ps();
33        let mut i: usize = 0;
34        while i < m {
35            let sub128_1 = _mm_sub_ps(_mm_loadu_ps(ptr1), _mm_loadu_ps(ptr2));
36            sum128_1 = _mm_add_ps(_mm_mul_ps(sub128_1, sub128_1), sum128_1);
37
38            let sub128_2 = _mm_sub_ps(_mm_loadu_ps(ptr1.add(4)), _mm_loadu_ps(ptr2.add(4)));
39            sum128_2 = _mm_add_ps(_mm_mul_ps(sub128_2, sub128_2), sum128_2);
40
41            let sub128_3 = _mm_sub_ps(_mm_loadu_ps(ptr1.add(8)), _mm_loadu_ps(ptr2.add(8)));
42            sum128_3 = _mm_add_ps(_mm_mul_ps(sub128_3, sub128_3), sum128_3);
43
44            let sub128_4 = _mm_sub_ps(_mm_loadu_ps(ptr1.add(12)), _mm_loadu_ps(ptr2.add(12)));
45            sum128_4 = _mm_add_ps(_mm_mul_ps(sub128_4, sub128_4), sum128_4);
46
47            ptr1 = ptr1.add(16);
48            ptr2 = ptr2.add(16);
49            i += 16;
50        }
51
52        let mut result = hsum128_ps_sse(sum128_1)
53            + hsum128_ps_sse(sum128_2)
54            + hsum128_ps_sse(sum128_3)
55            + hsum128_ps_sse(sum128_4);
56        for i in 0..n - m {
57            result += (*ptr1.add(i) - *ptr2.add(i)).powi(2);
58        }
59        -result
60    }
61}

63#[target_feature(enable = "sse")]
64pub(crate) unsafe fn manhattan_similarity_sse(
65    v1: &[VectorElementType],
66    v2: &[VectorElementType],
67) -> ScoreType {
68    unsafe {
69        let mask: __m128 = _mm_set1_ps(-0.0f32); // 1 << 31 used to clear sign bit to mimic abs
70
71        let n = v1.len();
72        let m = n - (n % 16);
73        let mut ptr1: *const f32 = v1.as_ptr();
74        let mut ptr2: *const f32 = v2.as_ptr();
75        let mut sum128_1: __m128 = _mm_setzero_ps();
76        let mut sum128_2: __m128 = _mm_setzero_ps();
77        let mut sum128_3: __m128 = _mm_setzero_ps();
78        let mut sum128_4: __m128 = _mm_setzero_ps();
79        let mut i: usize = 0;
80        while i < m {
81            let sub128_1 = _mm_sub_ps(_mm_loadu_ps(ptr1), _mm_loadu_ps(ptr2));
82            sum128_1 = _mm_add_ps(_mm_andnot_ps(mask, sub128_1), sum128_1);
83
84            let sub128_2 = _mm_sub_ps(_mm_loadu_ps(ptr1.add(4)), _mm_loadu_ps(ptr2.add(4)));
85            sum128_2 = _mm_add_ps(_mm_andnot_ps(mask, sub128_2), sum128_2);
86
87            let sub128_3 = _mm_sub_ps(_mm_loadu_ps(ptr1.add(8)), _mm_loadu_ps(ptr2.add(8)));
88            sum128_3 = _mm_add_ps(_mm_andnot_ps(mask, sub128_3), sum128_3);
89
90            let sub128_4 = _mm_sub_ps(_mm_loadu_ps(ptr1.add(12)), _mm_loadu_ps(ptr2.add(12)));
91            sum128_4 = _mm_add_ps(_mm_andnot_ps(mask, sub128_4), sum128_4);
92
93            ptr1 = ptr1.add(16);
94            ptr2 = ptr2.add(16);
95            i += 16;
96        }
97
98        let mut result = hsum128_ps_sse(sum128_1)
99            + hsum128_ps_sse(sum128_2)
100            + hsum128_ps_sse(sum128_3)
101            + hsum128_ps_sse(sum128_4);
102        for i in 0..n - m {
103            result += (*ptr1.add(i) - *ptr2.add(i)).abs();
104        }
105        -result
106    }
107}

109#[target_feature(enable = "sse")]
110pub(crate) unsafe fn cosine_preprocess_sse(vector: DenseVector) -> DenseVector {
111    unsafe {
112        let n = vector.len();
113        let m = n - (n % 16);
114        let mut ptr: *const f32 = vector.as_ptr();
115        let mut sum128_1: __m128 = _mm_setzero_ps();
116        let mut sum128_2: __m128 = _mm_setzero_ps();
117        let mut sum128_3: __m128 = _mm_setzero_ps();
118        let mut sum128_4: __m128 = _mm_setzero_ps();
119
120        let mut i: usize = 0;
121        while i < m {
122            let m128_1 = _mm_loadu_ps(ptr);
123            sum128_1 = _mm_add_ps(_mm_mul_ps(m128_1, m128_1), sum128_1);
124
125            let m128_2 = _mm_loadu_ps(ptr.add(4));
126            sum128_2 = _mm_add_ps(_mm_mul_ps(m128_2, m128_2), sum128_2);
127
128            let m128_3 = _mm_loadu_ps(ptr.add(8));
129            sum128_3 = _mm_add_ps(_mm_mul_ps(m128_3, m128_3), sum128_3);
130
131            let m128_4 = _mm_loadu_ps(ptr.add(12));
132            sum128_4 = _mm_add_ps(_mm_mul_ps(m128_4, m128_4), sum128_4);
133
134            ptr = ptr.add(16);
135            i += 16;
136        }
137
138        let mut length = hsum128_ps_sse(sum128_1)
139            + hsum128_ps_sse(sum128_2)
140            + hsum128_ps_sse(sum128_3)
141            + hsum128_ps_sse(sum128_4);
142        for i in 0..n - m {
143            length += (*ptr.add(i)).powi(2);
144        }
145        if is_length_zero_or_normalized(length) {
146            return vector;
147        }
148        length = length.sqrt();
149        vector.into_iter().map(|x| x / length).collect()
150    }
151}

153#[target_feature(enable = "sse")]
154pub(crate) unsafe fn dot_similarity_sse(
155    v1: &[VectorElementType],
156    v2: &[VectorElementType],
157) -> ScoreType {
158    unsafe {
159        let n = v1.len();
160        let m = n - (n % 16);
161        let mut ptr1: *const f32 = v1.as_ptr();
162        let mut ptr2: *const f32 = v2.as_ptr();
163        let mut sum128_1: __m128 = _mm_setzero_ps();
164        let mut sum128_2: __m128 = _mm_setzero_ps();
165        let mut sum128_3: __m128 = _mm_setzero_ps();
166        let mut sum128_4: __m128 = _mm_setzero_ps();
167
168        let mut i: usize = 0;
169        while i < m {
170            sum128_1 = _mm_add_ps(_mm_mul_ps(_mm_loadu_ps(ptr1), _mm_loadu_ps(ptr2)), sum128_1);
171
172            sum128_2 = _mm_add_ps(
173                _mm_mul_ps(_mm_loadu_ps(ptr1.add(4)), _mm_loadu_ps(ptr2.add(4))),
174                sum128_2,
175            );
176
177            sum128_3 = _mm_add_ps(
178                _mm_mul_ps(_mm_loadu_ps(ptr1.add(8)), _mm_loadu_ps(ptr2.add(8))),
179                sum128_3,
180            );
181
182            sum128_4 = _mm_add_ps(
183                _mm_mul_ps(_mm_loadu_ps(ptr1.add(12)), _mm_loadu_ps(ptr2.add(12))),
184                sum128_4,
185            );
186
187            ptr1 = ptr1.add(16);
188            ptr2 = ptr2.add(16);
189            i += 16;
190        }
191
192        let mut result = hsum128_ps_sse(sum128_1)
193            + hsum128_ps_sse(sum128_2)
194            + hsum128_ps_sse(sum128_3)
195            + hsum128_ps_sse(sum128_4);
196        for i in 0..n - m {
197            result += (*ptr1.add(i)) * (*ptr2.add(i));
198        }
199        result
200    }
201}

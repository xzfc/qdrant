13pub unsafe fn avx_dot_similarity_half(
14    v1: &[VectorElementTypeHalf],
15    v2: &[VectorElementTypeHalf],
16) -> ScoreType {
17    unsafe {
18        let n = v1.len();
19        let m = n - (n % 32);
20        let mut ptr1: *const __m128i = v1.as_ptr().cast::<__m128i>();
21        let mut ptr2: *const __m128i = v2.as_ptr().cast::<__m128i>();
22        let mut sum256_1: __m256 = _mm256_setzero_ps();
23        let mut sum256_2: __m256 = _mm256_setzero_ps();
24        let mut sum256_3: __m256 = _mm256_setzero_ps();
25        let mut sum256_4: __m256 = _mm256_setzero_ps();
26
27        let mut addr1s: __m128i;
28        let mut addr2s: __m128i;
29        let mut i: usize = 0;
30        while i < m {
31            addr1s = _mm_loadu_si128(ptr1);
32            addr2s = _mm_loadu_si128(ptr2);
33
34            sum256_1 = _mm256_fmadd_ps(_mm256_cvtph_ps(addr1s), _mm256_cvtph_ps(addr2s), sum256_1);
35
36            addr1s = _mm_loadu_si128(ptr1.wrapping_add(1));
37            addr2s = _mm_loadu_si128(ptr2.wrapping_add(1));
38
39            sum256_2 = _mm256_fmadd_ps(_mm256_cvtph_ps(addr1s), _mm256_cvtph_ps(addr2s), sum256_2);
40
41            addr1s = _mm_loadu_si128(ptr1.wrapping_add(2));
42            addr2s = _mm_loadu_si128(ptr2.wrapping_add(2));
43
44            sum256_3 = _mm256_fmadd_ps(_mm256_cvtph_ps(addr1s), _mm256_cvtph_ps(addr2s), sum256_3);
45
46            addr1s = _mm_loadu_si128(ptr1.wrapping_add(3));
47            addr2s = _mm_loadu_si128(ptr2.wrapping_add(3));
48
49            sum256_4 = _mm256_fmadd_ps(_mm256_cvtph_ps(addr1s), _mm256_cvtph_ps(addr2s), sum256_4);
50
51            ptr1 = ptr1.wrapping_add(4);
52            ptr2 = ptr2.wrapping_add(4);
53            i += 32;
54        }
55
56        let ptr1_f16: *const f16 = ptr1.cast::<f16>();
57        let ptr2_f16: *const f16 = ptr2.cast::<f16>();
58
59        let mut result = hsum256_ps_avx(sum256_1)
60            + hsum256_ps_avx(sum256_2)
61            + hsum256_ps_avx(sum256_3)
62            + hsum256_ps_avx(sum256_4);
63
64        for i in 0..n - m {
65            result += f16::to_f32(*ptr1_f16.add(i)) * f16::to_f32(*ptr2_f16.add(i));
66        }
67        result
68    }
69}

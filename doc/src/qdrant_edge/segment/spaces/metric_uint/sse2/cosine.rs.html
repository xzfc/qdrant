5#[target_feature(enable = "sse")]
6#[target_feature(enable = "sse2")]
7#[allow(clippy::missing_safety_doc)]
8pub unsafe fn sse_cosine_similarity_bytes(v1: &[u8], v2: &[u8]) -> f32 {
9    debug_assert!(v1.len() == v2.len());
10
11    let mut ptr1: *const u8 = v1.as_ptr();
12    let mut ptr2: *const u8 = v2.as_ptr();
13
14    unsafe {
15        // sum accumulator for 4x32 bit integers
16        let mut dot_acc = _mm_setzero_si128();
17        let mut norm1_acc = _mm_setzero_si128();
18        let mut norm2_acc = _mm_setzero_si128();
19        // mask to take only lower 8 bits from 16 bits
20        let mask_epu16_epu8 = _mm_set1_epi16(0xFF);
21        let len = v1.len();
22        for _ in 0..len / 16 {
23            // load 16 bytes
24            let p1 = _mm_loadu_si128(ptr1.cast::<__m128i>());
25            let p2 = _mm_loadu_si128(ptr2.cast::<__m128i>());
26            ptr1 = ptr1.add(16);
27            ptr2 = ptr2.add(16);
28
29            // convert 16x8 bit integers into 8x16 bit integers using bitwise AND
30            // conversion is done by taking only lower 8 bits from 16 bits
31            // p1 = [byte0, byte1, byte2, byte3, ..]
32            // p1_low = [0, byte1, 0, byte3, ..]
33            // p1_high = [0, byte0, 0, byte2, ..]
34            let p1_low = _mm_and_si128(p1, mask_epu16_epu8);
35            let p1_high = _mm_and_si128(_mm_bsrli_si128(p1, 1), mask_epu16_epu8);
36            let p2_low = _mm_and_si128(p2, mask_epu16_epu8);
37            let p2_high = _mm_and_si128(_mm_bsrli_si128(p2, 1), mask_epu16_epu8);
38
39            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
40            let norm1_low = _mm_madd_epi16(p1_low, p1_low);
41            norm1_acc = _mm_add_epi32(norm1_acc, norm1_low);
42
43            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
44            let norm2_low = _mm_madd_epi16(p2_low, p2_low);
45            norm2_acc = _mm_add_epi32(norm2_acc, norm2_low);
46
47            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
48            let dot_low = _mm_madd_epi16(p1_low, p2_low);
49            dot_acc = _mm_add_epi32(dot_acc, dot_low);
50
51            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
52            let norm1_high = _mm_madd_epi16(p1_high, p1_high);
53            norm1_acc = _mm_add_epi32(norm1_acc, norm1_high);
54
55            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
56            let norm2_high = _mm_madd_epi16(p2_high, p2_high);
57            norm2_acc = _mm_add_epi32(norm2_acc, norm2_high);
58
59            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
60            let dot_high = _mm_madd_epi16(p1_high, p2_high);
61            dot_acc = _mm_add_epi32(dot_acc, dot_high);
62        }
63
64        // convert 4x32 bit integers into 8x32 bit floats and calculate horizontal sum
65        let dot_ps = _mm_cvtepi32_ps(dot_acc);
66        let mut dot_product = hsum128_ps_sse(dot_ps);
67
68        // convert 4x32 bit integers into 8x32 bit floats and calculate horizontal sum
69        let norm1_ps = _mm_cvtepi32_ps(norm1_acc);
70        let mut norm1 = hsum128_ps_sse(norm1_ps);
71
72        // convert 4x32 bit integers into 8x32 bit floats and calculate horizontal sum
73        let norm2_ps = _mm_cvtepi32_ps(norm2_acc);
74        let mut norm2 = hsum128_ps_sse(norm2_ps);
75
76        let remainder = len % 16;
77        if remainder != 0 {
78            let mut remainder_dot_product = 0;
79            let mut remainder_norm1 = 0;
80            let mut remainder_norm2 = 0;
81            for _ in 0..remainder {
82                let v1 = *ptr1;
83                let v2 = *ptr2;
84                ptr1 = ptr1.add(1);
85                ptr2 = ptr2.add(1);
86                remainder_dot_product += i32::from(v1) * i32::from(v2);
87                remainder_norm1 += i32::from(v1) * i32::from(v1);
88                remainder_norm2 += i32::from(v2) * i32::from(v2);
89            }
90            dot_product += remainder_dot_product as f32;
91            norm1 += remainder_norm1 as f32;
92            norm2 += remainder_norm2 as f32;
93        }
94
95        let denominator = norm1 * norm2;
96        if denominator == 0.0 {
97            return 0.0;
98        }
99
100        dot_product / denominator.sqrt()
101    }
102}

5#[target_feature(enable = "sse")]
6#[target_feature(enable = "sse2")]
7#[allow(clippy::missing_safety_doc)]
8pub unsafe fn sse_euclid_similarity_bytes(v1: &[u8], v2: &[u8]) -> f32 {
9    debug_assert!(v1.len() == v2.len());
10
11    let mut ptr1: *const u8 = v1.as_ptr();
12    let mut ptr2: *const u8 = v2.as_ptr();
13
14    unsafe {
15        // sum accumulator for 4x32 bit integers
16        let mut acc = _mm_setzero_si128();
17        // mask to take only lower 8 bits from 16 bits
18        let mask_epu16_epu8 = _mm_set1_epi16(0xFF);
19        let len = v1.len();
20        for _ in 0..len / 16 {
21            // load 16 bytes
22            let p1 = _mm_loadu_si128(ptr1.cast::<__m128i>());
23            let p2 = _mm_loadu_si128(ptr2.cast::<__m128i>());
24            ptr1 = ptr1.add(16);
25            ptr2 = ptr2.add(16);
26
27            // Compute the difference in both directions and take the maximum for abs
28            let diff1 = _mm_subs_epu8(p1, p2);
29            let diff2 = _mm_subs_epu8(p2, p1);
30
31            let abs_diff = _mm_max_epu8(diff1, diff2);
32            let abs_diff_low = _mm_and_si128(abs_diff, mask_epu16_epu8);
33            let abs_diff_high = _mm_and_si128(_mm_bsrli_si128(abs_diff, 1), mask_epu16_epu8);
34
35            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
36            let mul16 = _mm_madd_epi16(abs_diff_low, abs_diff_low);
37            acc = _mm_add_epi32(acc, mul16);
38
39            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
40            let mul16 = _mm_madd_epi16(abs_diff_high, abs_diff_high);
41            acc = _mm_add_epi32(acc, mul16);
42        }
43
44        // convert 4x32 bit integers into 8x32 bit floats and calculate horizontal sum
45        let mul_ps = _mm_cvtepi32_ps(acc);
46        let mut score = hsum128_ps_sse(mul_ps);
47
48        let remainder = len % 16;
49        if remainder != 0 {
50            let mut remainder_score = 0;
51            for _ in 0..remainder {
52                let v1 = i32::from(*ptr1);
53                let v2 = i32::from(*ptr2);
54                ptr1 = ptr1.add(1);
55                ptr2 = ptr2.add(1);
56                let diff = v1 - v2;
57                remainder_score += diff * diff;
58            }
59            score += remainder_score as f32;
60        }
61
62        -score
63    }
64}

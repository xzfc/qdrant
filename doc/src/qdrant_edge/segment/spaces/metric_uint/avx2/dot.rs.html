1use std::arch::x86_64::*;
2
3use crate::segment::spaces::simple_avx::hsum256_ps_avx;
4
5#[target_feature(enable = "avx")]
6#[target_feature(enable = "avx2")]
7#[target_feature(enable = "fma")]
8#[allow(clippy::missing_safety_doc)]
9pub unsafe fn avx_dot_similarity_bytes(v1: &[u8], v2: &[u8]) -> f32 {
10    debug_assert!(v1.len() == v2.len());
11    debug_assert!(is_x86_feature_detected!("avx"));
12    debug_assert!(is_x86_feature_detected!("avx2"));
13    debug_assert!(is_x86_feature_detected!("fma"));
14
15    let mut ptr1: *const u8 = v1.as_ptr();
16    let mut ptr2: *const u8 = v2.as_ptr();
17
18    unsafe {
19        // sum accumulator for 8x32 bit integers
20        let mut dot_acc = _mm256_setzero_si256();
21        // mask to take only lower 8 bits from 16 bits
22        let mask_epu16_epu8 = _mm256_set1_epi16(0xFF);
23        let len = v1.len();
24        for _ in 0..len / 32 {
25            // load 32 bytes
26            let p1 = _mm256_loadu_si256(ptr1.cast::<__m256i>());
27            let p2 = _mm256_loadu_si256(ptr2.cast::<__m256i>());
28            ptr1 = ptr1.add(32);
29            ptr2 = ptr2.add(32);
30
31            // convert 32x8 bit integers into 16x16 bit integers using bitwise AND
32            // conversion is done by taking only lower 8 bits from 16 bits
33            // p1 = [byte0, byte1, byte2, byte3, ..]
34            // p1_low = [0, byte1, 0, byte3, ..]
35            // p1_high = [0, byte0, 0, byte2, ..]
36            let p1_low = _mm256_and_si256(p1, mask_epu16_epu8);
37            let p1_high = _mm256_and_si256(_mm256_bsrli_epi128(p1, 1), mask_epu16_epu8);
38            let p2_low = _mm256_and_si256(p2, mask_epu16_epu8);
39            let p2_high = _mm256_and_si256(_mm256_bsrli_epi128(p2, 1), mask_epu16_epu8);
40
41            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
42            let dot_low = _mm256_madd_epi16(p1_low, p2_low);
43            dot_acc = _mm256_add_epi32(dot_acc, dot_low);
44
45            // calculate 16bit multiplication with taking lower 16 bits and adding to accumulator
46            let dot_high = _mm256_madd_epi16(p1_high, p2_high);
47            dot_acc = _mm256_add_epi32(dot_acc, dot_high);
48        }
49
50        // convert 8x32 bit integers into 8x32 bit floats and calculate horizontal sum
51        let mul_ps = _mm256_cvtepi32_ps(dot_acc);
52        let mut score = hsum256_ps_avx(mul_ps);
53
54        let remainder = len % 32;
55        if remainder != 0 {
56            let mut remainder_dot = 0;
57            for _ in 0..remainder {
58                let v1 = *ptr1;
59                let v2 = *ptr2;
60                ptr1 = ptr1.add(1);
61                ptr2 = ptr2.add(1);
62                remainder_dot += i32::from(v1) * i32::from(v2);
63            }
64            score += remainder_dot as f32;
65        }
66
67        score
68    }
69}

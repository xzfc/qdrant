1use std::arch::x86_64::*;
2
3use crate::segment::spaces::simple_avx::hsum256_ps_avx;
4
5#[target_feature(enable = "avx")]
6#[target_feature(enable = "avx2")]
7#[target_feature(enable = "fma")]
8#[allow(clippy::missing_safety_doc)]
9pub unsafe fn avx_manhattan_similarity_bytes(v1: &[u8], v2: &[u8]) -> f32 {
10    debug_assert!(v1.len() == v2.len());
11    debug_assert!(is_x86_feature_detected!("avx"));
12    debug_assert!(is_x86_feature_detected!("avx2"));
13    debug_assert!(is_x86_feature_detected!("fma"));
14
15    let mut ptr1: *const u8 = v1.as_ptr();
16    let mut ptr2: *const u8 = v2.as_ptr();
17
18    unsafe {
19        // sum accumulator for 8x32 bit integers
20        let mut acc = _mm256_setzero_si256();
21        let len = v1.len();
22        for _ in 0..len / 32 {
23            // load 32 bytes
24            let p1 = _mm256_loadu_si256(ptr1.cast::<__m256i>());
25            let p2 = _mm256_loadu_si256(ptr2.cast::<__m256i>());
26            ptr1 = ptr1.add(32);
27            ptr2 = ptr2.add(32);
28
29            // Computes the absolute differences of packed unsigned 8-bit integers in a and b
30            // with horizontal sum and adding to accumulator
31            let sad = _mm256_sad_epu8(p1, p2);
32            acc = _mm256_add_epi32(acc, sad);
33        }
34
35        // convert 8x32 bit integers into 8x32 bit floats and calculate horizontal sum
36        let mul_ps = _mm256_cvtepi32_ps(acc);
37        let mut score = hsum256_ps_avx(mul_ps);
38
39        let remainder = len % 32;
40        if remainder != 0 {
41            let mut remainder_score = 0;
42            for _ in 0..len % 32 {
43                let v1 = i32::from(*ptr1);
44                let v2 = i32::from(*ptr2);
45                ptr1 = ptr1.add(1);
46                ptr2 = ptr2.add(1);
47                remainder_score += (v1 - v2).abs();
48            }
49            score += remainder_score as f32;
50        }
51
52        -score
53    }
54}

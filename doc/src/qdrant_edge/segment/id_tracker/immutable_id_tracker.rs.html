35enum ExternalIdType {
36    Number = 0,
37    Uuid = 1,
38}

40impl ExternalIdType {
41    fn from_byte(byte: u8) -> Option<Self> {
42        match byte {
43            x if x == Self::Number as u8 => Some(Self::Number),
44            x if x == Self::Uuid as u8 => Some(Self::Uuid),
45            _ => None,
46        }
47    }
48
49    fn from_point_id(point_id: &PointIdType) -> Self {
50        match point_id {
51            PointIdType::NumId(_) => Self::Number,
52            PointIdType::Uuid(_) => Self::Uuid,
53        }
54    }
55}

81    /// Loads a `CompressedPointMappings` from the given reader. Applies an optional filter of deleted items
82    /// to prevent allocating unneeded data.
83    fn load_mapping<R: BufRead>(
84        mut reader: R,
85        deleted: Option<BitVec>,
86    ) -> OperationResult<CompressedPointMappings> {
87        // Deserialize the header
88        let len = reader.read_u64::<FileEndianess>()? as usize;
89
90        let mut deleted = deleted.unwrap_or_else(|| BitVec::repeat(false, len));
91
92        deleted.truncate(len);
93
94        let mut internal_to_external = CompressedInternalToExternal::with_capacity(len);
95        let mut external_to_internal_num: Vec<(u64, PointOffsetType)> = Vec::new();
96        let mut external_to_internal_uuid: Vec<(Uuid, PointOffsetType)> = Vec::new();
97
98        // Deserialize the list entries
99        for i in 0..len {
100            let (internal_id, external_id) = Self::read_entry(&mut reader)
101                .map_err(|err| {
102                    OperationError::inconsistent_storage(format!("Immutable ID tracker failed to read next mapping, reading {} out of {len}, assuming malformed storage: {err}", i + 1))
103                })?;
104
105            // Need to push this regardless of point deletion as the vecs index represents the internal id
106            // which would become wrong if we leave out entries.
107            if internal_to_external.len() <= internal_id as usize {
108                internal_to_external.resize(internal_id as usize + 1, PointIdType::NumId(0));
109            }
110
111            internal_to_external.set(internal_id, external_id);
112
113            let point_deleted = deleted.get_bit(i).unwrap_or(false);
114            if point_deleted {
115                continue;
116            }
117
118            match external_id {
119                ExtendedPointId::NumId(num) => {
120                    external_to_internal_num.push((num, internal_id));
121                }
122                ExtendedPointId::Uuid(uuid) => {
123                    external_to_internal_uuid.push((uuid, internal_id));
124                }
125            }
126        }
127
128        // Check that the file has been fully read.
129        #[cfg(debug_assertions)] // Only for dev builds
130        {
131            debug_assert_eq!(reader.bytes().map(Result::unwrap).count(), 0,);
132        }
133
134        let external_to_internal = CompressedExternalToInternal::from_vectors(
135            external_to_internal_num,
136            external_to_internal_uuid,
137        );
138
139        Ok(CompressedPointMappings::new(
140            deleted,
141            internal_to_external,
142            external_to_internal,
143        ))
144    }

146    /// Loads a single entry from a reader. Expects the reader to be aligned so, that the next read
147    /// byte is the first byte of a new entry.
148    /// This function reads exact one entry which means after calling this function, the reader
149    /// will be at the start of the next entry.
150    pub(crate) fn read_entry<R: Read>(
151        mut reader: R,
152    ) -> OperationResult<(PointOffsetType, ExtendedPointId)> {
153        let point_id_type = reader.read_u8().map_err(|err| {
154            OperationError::inconsistent_storage(format!(
155                "failed to read point ID type from file: {err}"
156            ))
157        })?;
158
159        let external_id = match ExternalIdType::from_byte(point_id_type) {
160            None => {
161                return Err(OperationError::inconsistent_storage(
162                    "invalid byte for point ID type",
163                ));
164            }
165            Some(ExternalIdType::Number) => {
166                let num = reader.read_u64::<FileEndianess>().map_err(|err| {
167                    OperationError::inconsistent_storage(format!(
168                        "failed to read numeric point ID from file: {err}"
169                    ))
170                })?;
171                PointIdType::NumId(num)
172            }
173            Some(ExternalIdType::Uuid) => {
174                let uuid_u128 = reader.read_u128::<FileEndianess>().map_err(|err| {
175                    OperationError::inconsistent_storage(format!(
176                        "failed to read UUID point ID from file: {err}"
177                    ))
178                })?;
179                PointIdType::Uuid(Uuid::from_u128_le(uuid_u128))
180            }
181        };
182
183        let internal_id = reader.read_u32::<FileEndianess>().map_err(|err| {
184            OperationError::inconsistent_storage(format!(
185                "failed to read internal point ID from file: {err}"
186            ))
187        })? as PointOffsetType;
188        Ok((internal_id, external_id))
189    }

191    /// Serializes the `PointMappings` into the given writer using the file format specified below.
192    ///
193    /// ## File format
194    /// In general the format looks like this:
195    /// +---------------------------+-----------------+
196    /// | Header (list length: u64) | List of entries |
197    /// +---------------------------+-----------------+
198    ///
199    /// A single list entry:
200    /// +-----------------+-----------------------+------------------+
201    /// | PointIdType: u8 | Number/UUID: u64/u128 | Internal ID: u32 |
202    /// +-----------------+-----------------------+------------------+
203    /// A single entry is thus either 1+8+4=13 or 1+16+4=21 bytes in size depending
204    /// on the PointIdType.
205    fn store_mapping<W: Write>(
206        mappings: &CompressedPointMappings,
207        mut writer: W,
208    ) -> OperationResult<()> {
209        let number_of_entries = mappings.total_point_count();
210
211        // Serialize the header (=length).
212        writer.write_u64::<FileEndianess>(number_of_entries as u64)?;
213
214        // Serialize all entries
215        for (internal_id, external_id) in mappings.iter_internal_raw() {
216            Self::write_entry(&mut writer, internal_id, external_id)?;
217        }
218
219        Ok(())
220    }
221
222    fn write_entry<W: Write>(
223        mut writer: W,
224        internal_id: PointOffsetType,
225        external_id: PointIdType,
226    ) -> OperationResult<()> {
227        // Byte to distinguish between Number and UUID
228        writer.write_u8(ExternalIdType::from_point_id(&external_id) as u8)?;
229
230        // Serializing External ID
231        match external_id {
232            PointIdType::NumId(num) => {
233                // The PointID's number
234                writer.write_u64::<FileEndianess>(num)?;
235            }
236            PointIdType::Uuid(uuid) => {
237                // The PointID's UUID
238                writer.write_u128::<FileEndianess>(uuid.to_u128_le())?;
239            }
240        }
241
242        // Serializing Internal ID
243        writer.write_u32::<FileEndianess>(internal_id)?;
244
245        Ok(())
246    }

248    pub fn open(segment_path: &Path) -> OperationResult<Self> {
249        let deleted_raw = open_write_mmap(
250            &Self::deleted_file_path(segment_path),
251            AdviceSetting::Global,
252            true,
253        )?;
254        let deleted_mmap = MmapBitSlice::try_from(deleted_raw, 0)?;
255        let deleted_bitvec = deleted_mmap.to_bitvec();
256        let deleted_wrapper = MmapBitSliceBufferedUpdateWrapper::new(deleted_mmap);
257
258        let internal_to_version_map = open_write_mmap(
259            &Self::version_mapping_file_path(segment_path),
260            AdviceSetting::Global,
261            true,
262        )?;
263        let internal_to_version_mapslice: MmapSlice<SeqNumberType> =
264            unsafe { MmapSlice::try_from(internal_to_version_map)? };
265        let internal_to_version = CompressedVersions::from_slice(&internal_to_version_mapslice);
266        let internal_to_version_wrapper =
267            MmapSliceBufferedUpdateWrapper::new(internal_to_version_mapslice);
268
269        let reader = BufReader::new(File::open(Self::mappings_file_path(segment_path))?);
270        let mappings = Self::load_mapping(reader, Some(deleted_bitvec))?;
271
272        Ok(Self {
273            path: segment_path.to_path_buf(),
274            deleted_wrapper,
275            internal_to_version_wrapper,
276            internal_to_version,
277            mappings,
278        })
279    }

281    pub fn new(
282        path: &Path,
283        internal_to_version: &[SeqNumberType],
284        mappings: CompressedPointMappings,
285    ) -> OperationResult<Self> {
286        // Create mmap file for deleted bitvec
287        let deleted_filepath = Self::deleted_file_path(path);
288        {
289            let deleted_size = bitmap_mmap_size(mappings.total_point_count());
290            create_and_ensure_length(&deleted_filepath, deleted_size)?;
291        }
292
293        debug_assert!(mappings.deleted().len() <= mappings.total_point_count());
294
295        let deleted_mmap = open_write_mmap(&deleted_filepath, AdviceSetting::Global, false)?;
296        let mut deleted_new = MmapBitSlice::try_from(deleted_mmap, 0)?;
297        deleted_new[..mappings.deleted().len()].copy_from_bitslice(mappings.deleted());
298
299        for i in mappings.deleted().len()..mappings.total_point_count() {
300            deleted_new.set(i, true);
301        }
302
303        let deleted_wrapper = MmapBitSliceBufferedUpdateWrapper::new(deleted_new);
304
305        // Create mmap file for internal-to-version list
306        let version_filepath = Self::version_mapping_file_path(path);
307
308        // Amount of points without version
309        let missing_version_count = mappings
310            .total_point_count()
311            .saturating_sub(internal_to_version.len());
312
313        let missing_versions_size = missing_version_count * size_of::<SeqNumberType>();
314        let internal_to_version_size = size_of_val(internal_to_version);
315        let min_size = internal_to_version_size + missing_versions_size;
316        {
317            let version_size = mmap_size::<SeqNumberType>(min_size);
318            create_and_ensure_length(&version_filepath, version_size)?;
319        }
320        let mut internal_to_version_wrapper = unsafe {
321            MmapSlice::try_from(open_write_mmap(
322                &version_filepath,
323                AdviceSetting::Global,
324                false,
325            )?)?
326        };
327
328        internal_to_version_wrapper[..internal_to_version.len()]
329            .copy_from_slice(internal_to_version);
330        let internal_to_version = CompressedVersions::from_slice(&internal_to_version_wrapper);
331
332        debug_assert_eq!(internal_to_version.len(), mappings.total_point_count());
333
334        let internal_to_version_wrapper =
335            MmapSliceBufferedUpdateWrapper::new(internal_to_version_wrapper);
336
337        // Write mappings to disk.
338        let file = File::create(Self::mappings_file_path(path))?;
339        let mut writer = BufWriter::new(file);
340        Self::store_mapping(&mappings, &mut writer)?;
341
342        // Explicitly fsync file contents to ensure durability
343        writer.flush()?;
344        let file = writer.into_inner().unwrap();
345        file.sync_all()?;
346
347        deleted_wrapper.flusher()()?;
348        internal_to_version_wrapper.flusher()()?;
349
350        Ok(Self {
351            path: path.to_path_buf(),
352            deleted_wrapper,
353            internal_to_version_wrapper,
354            internal_to_version,
355            mappings,
356        })
357    }

372/// Returns the required mmap filesize for a given length of a slice of type `T`.
373fn mmap_size<T>(len: usize) -> usize {
374    let item_width = size_of::<T>();
375    len.div_ceil(item_width) * item_width // Make it a multiple of usize-width.
376}
377
378/// Returns the required mmap filesize for a `BitSlice`.
379fn bitmap_mmap_size(number_of_elements: usize) -> usize {
380    mmap_size::<usize>(number_of_elements.div_ceil(u8::BITS as usize))
381}

631            let mut dropped_points = HashSet::new();
632            let mut custom_version = HashMap::new();
633
634            for (index, point) in TEST_POINTS.iter().enumerate() {
635                if index % 2 == 0 {
636                    continue;
637                }
638
639                if index % 3 == 0 {
640                    id_tracker.drop(*point).unwrap();
641                    dropped_points.insert(*point);
642                    continue;
643                }
644
645                if index % 5 == 0 {
646                    let new_version = rng.next_u64();
647                    id_tracker
648                        .set_internal_version(index as PointOffsetType, new_version)
649                        .unwrap();
650                    custom_version.insert(index as PointOffsetType, new_version);
651                }
652            }
653
654            id_tracker.mapping_flusher()().unwrap();
655            id_tracker.versions_flusher()().unwrap();
656
657            (dropped_points, custom_version)
658        };
659
660        let id_tracker = ImmutableIdTracker::open(dir.path()).unwrap();
661        for (index, point) in TEST_POINTS.iter().enumerate() {
662            let internal_id = index as PointOffsetType;
663
664            if dropped_points.contains(point) {
665                assert!(id_tracker.is_deleted_point(internal_id));
666                assert_eq!(id_tracker.external_id(internal_id), None);
667                assert!(id_tracker.mappings.internal_id(point).is_none());
668
669                continue;
670            }
671
672            // Check version
673            let expect_version = custom_version
674                .get(&internal_id)
675                .copied()
676                .unwrap_or(DEFAULT_VERSION);
677
678            assert_eq!(
679                id_tracker.internal_to_version.get(internal_id),
680                Some(expect_version)
681            );
682
683            // Check that unmodified points still haven't changed.
684            assert_eq!(
685                id_tracker.external_id(index as PointOffsetType),
686                Some(*point)
687            );
688        }
689    }

761    #[test]
762    fn test_point_mappings_de_serialization() {
763        let mut rng = StdRng::seed_from_u64(RAND_SEED);
764
765        let mut buf = vec![];
766
767        // Test different sized PointMappings, growing exponentially to also test large ones.
768        // This way we test up to 2^16 entries.
769        for size_exp in (0..16u32).step_by(3) {
770            buf.clear();
771
772            let size = 2usize.pow(size_exp);
773
774            let mappings = CompressedPointMappings::random(&mut rng, size as u32);
775
776            ImmutableIdTracker::store_mapping(&mappings, &mut buf).unwrap();
777
778            // 16 is the min byte size of an entry. The exact number is not that important
779            // we just want to ensure that the written bytes correlate to the amount of entries.
780            assert!(buf.len() >= size * 16);
781
782            let new_mappings = ImmutableIdTracker::load_mapping(&*buf, None).unwrap();
783
784            assert_eq!(new_mappings.total_point_count(), size);
785            assert_eq!(mappings, new_mappings);
786        }
787    }

19pub fn check_is_empty<'a>(values: impl IntoIterator<Item = &'a Value>) -> bool {
20    values.into_iter().all(|x| match x {
21        serde_json::Value::Null => true,
22        serde_json::Value::Array(arr) => arr.is_empty(),
23        _ => false,
24    })
25}
26
27pub fn check_is_null<'a>(values: impl IntoIterator<Item = &'a Value>) -> bool {
28    values.into_iter().any(|x| x.is_null())
29    // { "a": [ { "b": null }, { "b": 1 } ] } => true
30    // { "a": [ { "b": 1 }, { "b": null } ] } => true
31    // { "a": [ { "b": 1 }, { "b": 2 } ] } => false
32}

38// Merge source map into destination map
39pub fn merge_map(
40    dest: &mut serde_json::Map<String, Value>,
41    source: &serde_json::Map<String, Value>,
42) {
43    for (key, value) in source {
44        match value {
45            Value::Null => dest.remove(key),
46            _ => dest.insert(key.to_owned(), value.to_owned()),
47        };
48    }
49}

51pub fn transpose_map_into_named_vector<TVector: Into<VectorInternal>>(
52    map: HashMap<VectorNameBuf, Vec<TVector>>,
53) -> Vec<NamedVectors<'static>> {
54    let mut result = Vec::new();
55    for (key, values) in map {
56        result.resize_with(values.len(), NamedVectors::default);
57        for (i, value) in values.into_iter().enumerate() {
58            result[i].insert(key.clone(), value.into());
59        }
60    }
61    result
62}

64/// Hashes an iterator of unique items in an order-independent way.
65///
66/// The order of items does not affect the resulting hash.
67/// Assumes that each item is unique. Suitable for hashing, e.g., sets and maps,
68/// but not for vectors or multisets.
69#[inline(always)]
70pub fn unordered_hash_unique<T: Hash, I: Iterator<Item = T>, H: Hasher>(state: &mut H, iter: I) {
71    iter.fold(0u64, |res, item| {
72        let mut hasher = DefaultHasher::new();
73        item.hash(&mut hasher);
74        // NOTE: xor is used, duplicates will cancel each other out
75        res ^ hasher.finish()
76    })
77    .hash(state);
78}

80/// Deserializer helper for `Option<Vec<T>>` that allows deserializing both single and an array of values.
81///
82/// Use via `#[serde(with = "MaybeOneOrMany")]` and `#[schemars(with="MaybeOneOrMany<T>")]` field attributes
83pub struct MaybeOneOrMany<T>(pub Option<Vec<T>>);
84
85impl<T: Serialize> MaybeOneOrMany<T> {
86    pub fn serialize<S>(value: &Option<Vec<T>>, serializer: S) -> Result<S::Ok, S::Error>
87    where
88        S: serde::Serializer,
89    {
90        value.serialize(serializer)
91    }
92}
93
94impl<'de, T: Deserialize<'de>> MaybeOneOrMany<T> {
95    pub fn deserialize<D>(deserializer: D) -> Result<Option<Vec<T>>, D::Error>
96    where
97        D: serde::Deserializer<'de>,
98    {
99        use serde_untagged::UntaggedEnumVisitor;
100
101        UntaggedEnumVisitor::new()
102            .unit(|| Ok(None))
103            .seq(|x| x.deserialize().map(Some))
104            .map(|x| x.deserialize().map(|x| vec![x]).map(Some))
105            .deserialize(deserializer)
106    }
107}

109impl<T: JsonSchema> JsonSchema for MaybeOneOrMany<T> {
110    fn json_schema(generator: &mut schemars::r#gen::SchemaGenerator) -> schemars::schema::Schema {
111        use schemars::schema::SchemaObject;
112
113        #[derive(JsonSchema)]
114        #[serde(untagged)]
115        enum OneOrMany<T> {
116            _One(T),
117            _Many(Vec<T>),
118            _None(()),
119        }
120
121        let schema: SchemaObject = <OneOrMany<T>>::json_schema(generator).into();
122        schema.into()
123    }
124
125    fn schema_name() -> String {
126        <Vec<T>>::schema_name()
127    }
128
129    fn is_referenceable() -> bool {
130        false
131    }
132}

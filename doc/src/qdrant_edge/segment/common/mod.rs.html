45/// Check that the given vector name and elements are compatible with the given segment config.
46///
47/// Returns an error if incompatible.
48pub fn check_vector(
49    vector_name: &VectorName,
50    query_vector: &QueryVector,
51    segment_config: &SegmentConfig,
52) -> OperationResult<()> {
53    let vector_config = get_vector_config_or_error(vector_name, segment_config);
54    if vector_config.is_ok() {
55        check_query_vector(query_vector, vector_config?)
56    } else {
57        let sparse_vector_config = get_sparse_vector_config_or_error(vector_name, segment_config)?;
58        check_query_sparse_vector(query_vector, sparse_vector_config)
59    }
60}
61
62fn check_query_vector(
63    query_vector: &QueryVector,
64    vector_config: &VectorDataConfig,
65) -> OperationResult<()> {
66    match query_vector {
67        QueryVector::Nearest(vector) => {
68            check_vector_against_config(VectorRef::from(vector), vector_config)?
69        }
70        QueryVector::RecommendBestScore(reco_query)
71        | QueryVector::RecommendSumScores(reco_query) => {
72            reco_query.flat_iter().try_for_each(|vector| {
73                check_vector_against_config(VectorRef::from(vector), vector_config)
74            })?
75        }
76        QueryVector::Discover(discover_query) => {
77            discover_query.flat_iter().try_for_each(|vector| {
78                check_vector_against_config(VectorRef::from(vector), vector_config)
79            })?
80        }
81        QueryVector::Context(context_query) => {
82            context_query.flat_iter().try_for_each(|vector| {
83                check_vector_against_config(VectorRef::from(vector), vector_config)
84            })?
85        }
86        QueryVector::FeedbackNaive(feedback_query) => {
87            feedback_query.flat_iter().try_for_each(|vector| {
88                check_vector_against_config(VectorRef::from(vector), vector_config)
89            })?
90        }
91    }
92
93    Ok(())
94}

130/// Check that the given vector name and elements are compatible with the given segment config.
131///
132/// Returns an error if incompatible.
133pub fn check_query_vectors(
134    vector_name: &VectorName,
135    query_vectors: &[&QueryVector],
136    segment_config: &SegmentConfig,
137) -> OperationResult<()> {
138    let vector_config = get_vector_config_or_error(vector_name, segment_config);
139    if let Ok(vector_config) = vector_config {
140        query_vectors
141            .iter()
142            .try_for_each(|qv| check_query_vector(qv, vector_config))?;
143    } else {
144        let sparse_vector_config = get_sparse_vector_config_or_error(vector_name, segment_config)?;
145        query_vectors
146            .iter()
147            .try_for_each(|qv| check_query_sparse_vector(qv, sparse_vector_config))?;
148    }
149    Ok(())
150}

152/// Check that the given named vectors are compatible with the given segment config.
153///
154/// Returns an error if incompatible.
155pub fn check_named_vectors(
156    vectors: &NamedVectors,
157    segment_config: &SegmentConfig,
158) -> OperationResult<()> {
159    for (vector_name, vector_data) in vectors.iter() {
160        check_vector(vector_name, &vector_data.into(), segment_config)?;
161    }
162    Ok(())

165/// Get the vector config for the given name, or return a name error.
166///
167/// Returns an error if incompatible.
168fn get_vector_config_or_error<'a>(
169    vector_name: &VectorName,
170    segment_config: &'a SegmentConfig,
171) -> OperationResult<&'a VectorDataConfig> {
172    segment_config
173        .vector_data
174        .get(vector_name)
175        .ok_or_else(|| OperationError::vector_name_not_exists(vector_name))
176}

178/// Get the sparse vector config for the given name, or return a name error.

181fn get_sparse_vector_config_or_error<'a>(

191/// Check if the given dense vector data is compatible with the given configuration.
192///
193/// Returns an error if incompatible.
194fn check_vector_against_config(
195    vector: VectorRef,
196    vector_config: &VectorDataConfig,
197) -> OperationResult<()> {
198    match vector {
199        VectorRef::Dense(vector) => {
200            // Check dimensionality
201            let dim = vector_config.size;
202            if vector.len() != dim {
203                return Err(OperationError::WrongVectorDimension {
204                    expected_dim: dim,
205                    received_dim: vector.len(),
206                });
207            }
208            Ok(())
209        }
210        VectorRef::Sparse(_) => Err(OperationError::WrongSparse),
211        VectorRef::MultiDense(multi_vector) => {
212            // Check dimensionality
213            let dim = vector_config.size;
214            for vector in multi_vector.multi_vectors() {
215                if vector.len() != dim {
216                    return Err(OperationError::WrongVectorDimension {
217                        expected_dim: dim,
218                        received_dim: vector.len(),
219                    });
220                }
221            }
222            Ok(())
223        }
224    }
225}

227fn check_sparse_vector_against_config(

230) -> OperationResult<()> {
231    match vector {
232        VectorRef::Dense(_) => Err(OperationError::WrongSparse),
233        VectorRef::Sparse(_vector) => Ok(()), // TODO(sparse) check vector by config
234        VectorRef::MultiDense(_) => Err(OperationError::WrongMulti),
235    }

238pub fn check_stopped(is_stopped: &AtomicBool) -> OperationResult<()> {
239    if is_stopped.load(std::sync::atomic::Ordering::Relaxed) {
240        return Err(OperationError::Cancelled {
241            description: "Operation is stopped externally".to_string(),
242        });
243    }
244    Ok(())
245}

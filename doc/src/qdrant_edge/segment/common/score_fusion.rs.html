10pub struct ScoreFusion {
11    /// Defines how to combine the scores of the same point in different lists
12    pub method: Aggregation,
13    /// Defines how to normalize the scores in each list
14    pub norm: Normalization,
15    /// Multipliers for each list of scores
16    pub weights: Vec<f32>,
17    /// Final ordering of the results
18    pub order: Order,
19}

21impl ScoreFusion {
22    /// Params for the distribution-based score fusion
23    pub fn dbsf() -> Self {
24        Self {
25            method: Aggregation::Sum,
26            norm: Normalization::Distr,
27            weights: vec![],
28            order: Order::LargeBetter,
29        }
30    }
31}

33/// Defines how to combine the scores of the same point in different lists
34pub enum Aggregation {
35    /// Sums the scores
36    Sum,
37}
38
39pub enum Normalization {
40    /// Uses the minimum and maximum scores as extremes
41    MinMax,
42    /// Uses the 3rd standard deviation as extremes
43    Distr,
44}
45
46pub fn score_fusion(
47    all_results: impl IntoIterator<Item = Vec<ScoredPoint>>,
48    params: ScoreFusion,
49) -> Vec<ScoredPoint> {
50    let ScoreFusion {
51        method,
52        norm,
53        weights,
54        order,
55    } = params;
56
57    let weights = weights.into_iter().chain(iter::repeat(1.0));
58
59    all_results
60        .into_iter()
61        // normalize
62        .map(|points| match norm {
63            Normalization::MinMax => min_max_norm(points),
64            Normalization::Distr => distr_norm(points),
65        })
66        // weight each list of points
67        .zip(weights)
68        .flat_map(|(points, weight)| {
69            points.into_iter().map(move |p| ScoredPoint {
70                score: p.score * weight,
71                ..p
72            })
73        })
74        // combine to deduplicate
75        .fold(
76            AHashMap::<PointIdType, ScoredPoint>::new(),
77            |mut acc, point| {
78                acc.entry(point.id)
79                    .and_modify(|entry| match method {
80                        Aggregation::Sum => entry.score += point.score,
81                    })
82                    .or_insert(point);
83
84                acc
85            },
86        )
87        // sort and return
88        .into_values()
89        .sorted_by(|a, b| match order {
90            Order::SmallBetter => a.cmp(b),
91            Order::LargeBetter => b.cmp(a),
92        })
93        .collect()
94}

96/// Normalizes the scores of the given points between 0.0 and 1.0, using the given minimum and maximum scores as extremes.
97fn norm(mut points: Vec<ScoredPoint>, min: ScoreType, max: ScoreType) -> Vec<ScoredPoint> {
98    // Protect against division by zero
99    if min == max {
100        points.iter_mut().for_each(|p| p.score = 0.5);
101        return points;
102    }
103
104    points.iter_mut().for_each(|p| {
105        p.score = (p.score - min) / (max - min);
106    });
107
108    points
109}
110
111pub fn min_max_norm(points: Vec<ScoredPoint>) -> Vec<ScoredPoint> {
112    let (min, max) = match points.iter().map(|p| OrderedFloat(p.score)).minmax() {
113        MinMaxResult::NoElements | MinMaxResult::OneElement(_) => return points,
114        MinMaxResult::MinMax(min, max) => (min.0, max.0),
115    };
116
117    norm(points, min, max)
118}

120/// Welford's method for stable one-pass mean and variance calculation.
121/// <https://jonisalonen.com/2013/deriving-welfords-method-for-computing-variance/>
122///
123/// # Panics
124///
125/// Panics if the given vector of points has less than 2 elements.
126fn welfords_mean_variance(points: &[ScoredPoint]) -> (f32, f32) {
127    debug_assert!(
128        points.len() > 1,
129        "Not enough points to calculate mean and variance"
130    );
131
132    let mut mean = 0.0;
133    let mut aggregate = 0.0;
134    for (p, k) in points.iter().zip(1usize..) {
135        let old_delta = p.score - mean;
136        mean += old_delta / (k as f32);
137
138        let delta = p.score - mean;
139        aggregate += old_delta * delta;
140    }
141
142    let sample_variance = aggregate / (points.len() as f32 - 1.0);
143
144    (mean, sample_variance)
145}
146
147/// Estimates the mean and variance of the given points and normalizes them between 0.0 and 1.0, using the 3rd
148/// standard deviation as extremes.
149pub fn distr_norm(mut points: Vec<ScoredPoint>) -> Vec<ScoredPoint> {
150    if points.len() < 2 {
151        if points.len() == 1 {
152            points[0].score = 0.5;
153        }
154        return points;
155    }
156
157    let (mean, variance) = welfords_mean_variance(&points);
158
159    let std_dev = variance.sqrt();
160    let min = mean - 3.0 * std_dev;
161    let max = mean + 3.0 * std_dev;
162
163    norm(points, min, max)
164}

184    fn assert_close(a: f32, b: f32) {
185        // Choose the more relaxed tolerance, absolute or relative based on the values.
186        let abs_tolerance = 1e-5f32;
187        let rel_tolerance = 1e-4f32;
188
189        let diff = (a - b).abs();
190        let max_val = a.abs().max(b.abs());
191        let tolerance = abs_tolerance.max(rel_tolerance * max_val);
192
193        assert!(
194            diff <= tolerance,
195            "{a} is not close to {b}: difference {diff} exceeds tolerance {tolerance}"
196        );
197    }

203            let naive_variance = scores.iter().map(|p| (p - naive_mean).powi(2)).sum::<f32>()

42    pub fn open(dir: &Path) -> Result<Self> {
43        let mut pages = Self::new(dir.to_path_buf());
44
45        let page_files: HashSet<_> = S::list_files(&dir.join("page_"))?.into_iter().collect();
46
47        for page_id in 0.. {
48            let page_path = pages.page_path(page_id);
49            if !page_files.contains(&page_path) {
50                break;
51            }
52            pages.attach_page(&page_path)?;
53        }
54
55        Ok(pages)
56    }

80    /// Computes the page ranges and relative offsets for reading or writing a value described by
81    /// the given [`ValuePointer`].
82    ///
83    /// A value may span across two consecutive pages if it starts near the end of a page. This
84    /// method returns the list of `(page_index, range)` pairs that together cover the full extent
85    /// of the value, along with the corresponding byte offsets into the value buffer so that each
86    /// section can be read into or written from the correct position.
87    ///
88    /// Returns a tuple of:
89    /// - `SmallVec<[(FileIndex, ElementsRange); 2]>` — the per-page file ranges to access.
90    /// - `SmallVec<[usize; 2]>` — the offset within the value buffer that each page range
91    ///   corresponds to.
92    fn get_page_value_ranges(
93        pointer: ValuePointer,
94        config: &StorageConfig,
95    ) -> (PageRanges, BufferOffsets) {
96        let ValuePointer {
97            page_id,
98            block_offset,
99            length,
100        } = pointer;
101
102        let page_len = config.page_size_bytes as u64;
103        let block_size_bytes = config.block_size_bytes as u64;
104        let total_length = u64::from(length);
105
106        let value_start = u64::from(block_offset) * block_size_bytes;
107        assert!(value_start < page_len);
108
109        // Do not expect payload to span more than 2 pages, but can be easily extended if needed
110        let mut page_ranges = PageRanges::new();
111        // Store relative offsets to fill the raw_value buffer after fetching in batch
112        let mut buffer_offsets = BufferOffsets::new();
113
114        let mut length_so_far: u64 = 0;
115        let mut page_idx = page_id as FileIndex;
116        let mut start = value_start;
117
118        while length_so_far < total_length {
119            let remaining = total_length - length_so_far;
120            let available_on_page = page_len - start;
121            let section_len = remaining.min(available_on_page);
122
123            page_ranges.push((
124                page_idx,
125                ElementsRange {
126                    start,
127                    length: section_len,
128                },
129            ));
130            buffer_offsets.push(length_so_far as usize);
131
132            length_so_far += section_len;
133            page_idx += 1;
134            start = 0;
135        }
136
137        (page_ranges, buffer_offsets)
138    }

140    pub fn read_from_pages<const READ_SEQUENTIAL: bool>(
141        &self,
142        pointer: ValuePointer,
143        config: &StorageConfig,
144    ) -> Result<Vec<u8>> {
145        // Avoid initializing buffer with zeros, as it will be overwritten by file access;
146        let mut raw_value = vec![MaybeUninit::<u8>::uninit(); pointer.length as usize];
147
148        let (read_ranges, buffer_offsets) = Self::get_page_value_ranges(pointer, config);
149
150        S::read_multi::<READ_SEQUENTIAL>(self.pages.as_slice(), read_ranges, |idx, _, slice| {
151            let offset = buffer_offsets[idx];
152            raw_value[offset..offset + slice.len()].write_copy_of_slice(slice);
153            Ok(())
154        })?;
155
156        Ok(unsafe { assume_init_vec(raw_value) })
157    }

176    /// This method reloads the pages storage from "disk", so that
177    /// it should make newly written data is readable.
178    ///
179    /// Important assumptions:
180    ///
181    /// - Should only be called on read-only instances of the Pages.
182    /// - Only appending new data is supported, for modifications of existing data there are no consistency guarantees.
183    /// - Partial writes are possible, it is up to the caller to read only fully written data.
184    pub fn live_reload(&mut self) -> Result<()> {
185        let num_pages = self.pages.len();
186        let next_page_id = num_pages as PageId;
187
188        if num_pages > 0 {
189            // Re-attach the last page, which should have the latest data.
190            self.pages.pop();
191            let page_path = self.page_path((num_pages - 1) as PageId);
192            self.attach_page(&page_path)?;
193        }
194
195        for page_id in next_page_id.. {
196            let page_path = self.page_path(page_id);
197            if !S::exists(&page_path)? {
198                break;
199            }
200            self.attach_page(&page_path)?;
201        }
202
203        Ok(())
204    }
205}

207impl<S: UniversalWrite<u8>> Pages<S> {
208    pub fn write_to_pages(
209        &mut self,
210        pointer: ValuePointer,
211        value: &[u8],
212        config: &StorageConfig,
213    ) -> Result<()> {
214        // Compute write ranges
215        let (page_ranges, buffer_offsets) = Self::get_page_value_ranges(pointer, config);
216
217        let writes = page_ranges
218            .iter()
219            .zip(&buffer_offsets)
220            .map(|(&(file_idx, range), &start)| {
221                (
222                    file_idx,
223                    range.start,
224                    &value[start..start + range.length as usize],
225                )
226            });
227
228        // Execute writes (mutable borrow of self.pages)
229        S::write_multi(self.pages.as_mut_slice(), writes)?;
230
231        Ok(())
232    }

234    pub fn flusher(&self) -> Flusher {
235        let mut flushers = Vec::with_capacity(self.pages.len());
236        for page in &self.pages {
237            flushers.push(page.flusher());
238        }
239        Box::new(move || {
240            for flusher in flushers {
241                flusher()?;
242            }
243            Ok(())
244        })
245    }

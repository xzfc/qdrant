3/// Expect JSON values to have roughly 3–5 fields with mostly small values.

45#[derive(Debug, Clone, Serialize, Deserialize)]
46pub(crate) struct StorageConfig {
47    /// Size of a page in bytes
48    ///
49    /// Default is 32MB

52    /// Size of a block in bytes

69impl TryFrom<StorageOptions> for StorageConfig {
70    type Error = &'static str;
71
72    fn try_from(options: StorageOptions) -> Result<Self, Self::Error> {
73        let page_size_bytes = options.page_size_bytes.unwrap_or(DEFAULT_PAGE_SIZE_BYTES);
74        let block_size_bytes = options.block_size_bytes.unwrap_or(DEFAULT_BLOCK_SIZE_BYTES);
75        let region_size_blocks = options
76            .region_size_blocks
77            .map(|x| x as usize)
78            .unwrap_or(DEFAULT_REGION_SIZE_BLOCKS);
79
80        if block_size_bytes == 0 {
81            return Err("Block size must be greater than 0");
82        }
83
84        if region_size_blocks == 0 {
85            return Err("Region size must be greater than 0");
86        }
87
88        if page_size_bytes == 0 {
89            return Err("Page size must be greater than 0");
90        }
91
92        let region_size_bytes = block_size_bytes * region_size_blocks;
93
94        if page_size_bytes < region_size_bytes {
95            return Err("Page size must be greater than or equal to (block size * region size)");
96        }
97
98        if !page_size_bytes.is_multiple_of(region_size_bytes) {
99            return Err("Page size must be a multiple of (block size * region size)");
100        }
101
102        Ok(Self {
103            page_size_bytes,
104            block_size_bytes,
105            region_size_blocks,
106            compression: options.compression.unwrap_or_default(),
107        })
108    }
109}

111impl From<&StorageConfig> for StorageOptions {
112    fn from(config: &StorageConfig) -> Self {
113        Self {
114            page_size_bytes: Some(config.page_size_bytes),
115            block_size_bytes: Some(config.block_size_bytes),
116            region_size_blocks: Some(config.region_size_blocks as u16),
117            compression: Some(config.compression),
118        }
119    }

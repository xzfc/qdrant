34/// Read-write storage for values of type `V`.
35///
36/// Uses `Arc<RwLock<...>>` for pages and tracker to support concurrent flushing.
37/// Assumes sequential IDs to the values (0, 1, 2, 3, ...)
38#[derive(Debug)]
39pub struct Gridstore<V> {
40    pub(super) config: StorageConfig,
41    pub(super) tracker: Arc<RwLock<Tracker>>,
42    pub(super) pages: Arc<RwLock<Pages<MmapUniversal<u8>>>>,
43    /// Bitmask to represent which "blocks" of data in the pages are used and which are free.
44    ///
45    /// 0 is free, 1 is used.
46    pub(super) bitmask: Arc<RwLock<Bitmask>>,
47    pub(super) base_path: PathBuf,
48    pub(super) _value_type: std::marker::PhantomData<V>,
49    /// Lock to prevent concurrent flushes and used for waiting for ongoing flushes to finish.
50    is_alive_flush_lock: IsAliveLock,
51}

55    fn with_view<R>(&self, f: impl FnOnce(GridstoreView<'_, V, MmapUniversal<u8>>) -> R) -> R {

61    /// List all files belonging to this storage (tracker, pages, bitmask, config).
62    pub fn files(&self) -> Vec<PathBuf> {
63        let tracker = self.tracker.read();
64        let pages = self.pages.read();
65        let num_pages = pages.num_pages();
66
67        let mut paths = Vec::with_capacity(num_pages + 2);
68        for tracker_file in tracker.files() {
69            paths.push(tracker_file);
70        }
71        for page_id in 0..num_pages as PageId {
72            paths.push(pages.page_path(page_id));
73        }
74        paths.push(self.base_path.join(CONFIG_FILENAME));
75        for bitmask_file in self.bitmask.read().files() {
76            paths.push(bitmask_file);
77        }
78        paths
79    }

85    /// Opens an existing storage, or initializes a new one.
86    /// Depends on the existence of the config file at the `base_path`.
87    ///
88    /// In case of opening, it ignores the `create_options` parameter.
89    pub fn open_or_create(base_path: PathBuf, create_options: StorageOptions) -> Result<Self> {
90        let config_path = base_path.join(CONFIG_FILENAME);
91        if config_path.exists() {
92            Self::open(base_path)
93        } else {
94            fs::create_dir_all(&base_path).map_err(|err| {
95                GridstoreError::service_error(format!(
96                    "Failed to create gridstore storage directory: {err}"
97                ))
98            })?;
99            Self::new(base_path, create_options)
100        }
101    }

103    /// Initializes a new storage with a single empty page.
104    ///
105    /// `base_path` is the directory where the storage files will be stored.
106    /// It should exist already.
107    pub fn new(base_path: PathBuf, options: StorageOptions) -> Result<Self> {
108        let config = StorageConfig::try_from(options).map_err(GridstoreError::service_error)?;
109        let config_path = base_path.join(CONFIG_FILENAME);
110
111        let bitmask = Bitmask::create(&base_path, config.clone())?;
112
113        let storage = Self {
114            tracker: Arc::new(RwLock::new(Tracker::new(&base_path, None)?)),
115            pages: Arc::new(RwLock::new(Pages::new(base_path.clone()))),
116            base_path,
117            config,
118            _value_type: std::marker::PhantomData,
119            bitmask: Arc::new(RwLock::new(bitmask)),
120            is_alive_flush_lock: IsAliveLock::new(),
121        };
122
123        let new_page_id = storage.next_page_id();
124        let path = page_path(&storage.base_path, new_page_id);
125        create_and_ensure_length(&path, storage.config.page_size_bytes)?;
126        storage.pages.write().attach_page(&path)?;
127
128        atomic_save_json(&config_path, &storage.config)
129            .map_err(|err| GridstoreError::service_error(err.to_string()))?;
130
131        Ok(storage)
132    }

134    /// Open an existing storage at the given path.
135    ///
136    /// Uses the bitmask to infer page count for consistency with the write path.
137    pub fn open(base_path: PathBuf) -> Result<Self> {
138        let (config, tracker) = reader::read_config_and_tracker(&base_path)?;
139        let bitmask = Bitmask::open(&base_path, config.clone())?;
140        let num_pages = bitmask.infer_num_pages();
141
142        let pages = Pages::open(&base_path)?;
143        let loaded_pages = pages.num_pages();
144
145        if loaded_pages != num_pages {
146            return Err(GridstoreError::service_error(format!(
147                "Inconsistent number of gridstore pages at {base_path:?}: expected {num_pages}, but found {loaded_pages}",
148            )));
149        }
150
151        Ok(Self {
152            config,
153            tracker: Arc::new(RwLock::new(tracker)),
154            pages: Arc::new(RwLock::new(pages)),
155            bitmask: Arc::new(RwLock::new(bitmask)),
156            base_path,
157            _value_type: std::marker::PhantomData,
158            is_alive_flush_lock: IsAliveLock::new(),
159        })
160    }

162    /// Create a new page and return its id.
163    #[allow(clippy::needless_pass_by_ref_mut)]
164    fn create_new_page(&mut self) -> Result<u32> {
165        let new_page_id = self.next_page_id();
166        let path = page_path(&self.base_path, new_page_id);
167        create_and_ensure_length(&path, self.config.page_size_bytes)?;
168        self.pages.write().attach_page(&path)?;
169
170        self.bitmask.write().cover_new_page()?;
171
172        Ok(new_page_id)
173    }

175    fn find_or_create_available_blocks(
176        &mut self,
177        num_blocks: u32,
178    ) -> Result<(PageId, BlockOffset)> {
179        debug_assert!(num_blocks > 0, "num_blocks must be greater than 0");
180
181        let bitmask_guard = self.bitmask.read();
182        if let Some((page_id, block_offset)) = bitmask_guard.find_available_blocks(num_blocks) {
183            return Ok((page_id, block_offset));
184        }
185        let trailing_free_blocks = bitmask_guard.trailing_free_blocks();
186
187        drop(bitmask_guard);
188
189        let missing_blocks = num_blocks.saturating_sub(trailing_free_blocks) as usize;
190
191        let num_pages =
192            (missing_blocks * self.config.block_size_bytes).div_ceil(self.config.page_size_bytes);
193        for _ in 0..num_pages {
194            self.create_new_page()?;
195        }
196
197        let available = self
198            .bitmask
199            .read()
200            .find_available_blocks(num_blocks)
201            .unwrap();
202
203        Ok(available)
204    }

220    /// Put a value in the storage.
221    ///
222    /// Returns true if the value existed previously and was updated, false if it was newly inserted.
223    pub fn put_value(
224        &mut self,
225        point_offset: PointOffset,
226        value: &V,
227        hw_counter: HwMetricRefCounter,
228    ) -> Result<bool> {
229        // This function needs to NOT corrupt data in case of a crash.
230        //
231        // Since we cannot know deterministically when a write is persisted without flushing explicitly,
232        // and we don't want to flush on every write, we decided to follow this approach:
233        // ┌─────────┐           ┌───────┐┌────┐┌────┐   ┌───────┐                            ┌─────┐
234        // │put_value│           │Tracker││Page││Gaps│   │Bitmask│                            │flush│
235        // └────┬────┘           └───┬───┘└─┬──┘└─┬──┘   └───┬───┘                            └──┬──┘
236        //      │                    │      │     │          │                                   │
237        //      │            Find a spot to write │          │                                   │
238        //      │───────────────────────────────────────────>│                                   │
239        //      │                    │      │     │          │                                   │
240        //      │              Page id + offset   │          │                                   │
241        //      │<───────────────────────────────────────────│                                   │
242        //      │                    │      │     │          │                                   │
243        //      │    Write data to page     │     │          │                                   │
244        //      │──────────────────────────>│     │          │                                   │
245        //      │                    │      │     │          │                                   │
246        //      │                Mark as used     │          │                                   │
247        //      │───────────────────────────────────────────>│                                   │
248        //      │                    │      │     │          │                                   │
249        //      │                    │      │     │Update gap│                                   │
250        //      │                    │      │     │<─────────│                                   │
251        //      │                    │      │     │          │                                   │
252        //      │Set pointer (in-ram)│      │     │          │                                   │
253        //      │───────────────────>│      │     │          │                                   │
254        //      │                    │      │     │          │                                   │
255        //      │                    │      │     │          │               flush               │
256        //      │                    │      │     │          │<──────────────────────────────────│
257        //      │                    │      │     │          │                                   │
258        //      │                    │      │     │  flush   │                                   │
259        //      │                    │      │     │<─────────│                                   │
260        //      │                    │      │     │          │                                   │
261        //      │                    │      │     │          │      flush                        │
262        //      │                    │      │<───────────────────────────────────────────────────│
263        //      │                    │      │     │          │                                   │
264        //      │                    │      │     │   Write from memory AND flush                │
265        //      │                    │<──────────────────────────────────────────────────────────│
266        //      │                    │      │     │          │                                   │
267        //      │                    │      │     │          │Mark all old as available and flush│
268        //      │                    │      │     │          │<──────────────────────────────────│
269        // ┌────┴────┐           ┌───┴───┐┌─┴──┐┌─┴──┐   ┌───┴───┐                            ┌──┴──┐
270        // │put_value│           │Tracker││Page││Gaps│   │Bitmask│                            │flush│
271        // └─────────┘           └───────┘└────┘└────┘   └───────┘                            └─────┘
272        //
273        // In case of crashing somewhere in the middle of this operation, the worst
274        // that should happen is that we mark more cells as used than they actually are,
275        // so will never reuse such space, but data will not be corrupted.
276
277        let value_bytes = value.to_bytes();
278        let comp_value = self.with_view(|view| view.compress(value_bytes));
279        let value_size = comp_value.len();
280
281        hw_counter.incr_delta(value_size);
282
283        let required_blocks = Self::blocks_for_value(value_size, self.config.block_size_bytes);
284        let (start_page_id, block_offset) =
285            self.find_or_create_available_blocks(required_blocks)?;
286
287        self.write_into_pages(&comp_value, start_page_id, block_offset)?;
288
289        self.bitmask
290            .write()
291            .mark_blocks(start_page_id, block_offset, required_blocks, true);
292
293        let mut tracker_guard = self.tracker.write();
294        let is_update = tracker_guard.has_pointer(point_offset)?;
295        tracker_guard.set(
296            point_offset,
297            ValuePointer::new(start_page_id, block_offset, value_size as u32),
298        );
299
300        Ok(is_update)
301    }

303    /// Delete a value from the storage.
304    ///
305    /// Returns None if the point_offset, page, or value was not found.
306    /// Returns the deleted value otherwise.
307    pub fn delete_value(&mut self, point_offset: PointOffset) -> Result<Option<V>> {
308        let Some(pointer) = self.tracker.write().unset(point_offset)? else {
309            return Ok(None);
310        };
311
312        let raw = self.with_view(|view| view.read_from_pages::<false>(pointer))?;
313        let decompressed = self.with_view(|view| view.decompress(raw));
314        let value = V::from_bytes(&decompressed);
315
316        Ok(Some(value))
317    }

319    /// Clear the storage, going back to the initial state.
320    ///
321    /// Completely wipes the storage, and recreates it with a single empty page.
322    pub fn clear(&mut self) -> Result<()> {
323        let create_options = StorageOptions::from(&self.config);
324        let base_path = self.base_path.clone();
325
326        self.is_alive_flush_lock.blocking_mark_dead();
327
328        self.pages.write().clear();
329        fs::remove_dir_all(&base_path).map_err(|err| {
330            GridstoreError::service_error(format!(
331                "Failed to remove gridstore storage directory: {err}"
332            ))
333        })?;
334
335        fs::create_dir_all(&base_path).map_err(|err| {
336            GridstoreError::service_error(format!(
337                "Failed to create gridstore storage directory: {err}"
338            ))
339        })?;
340        *self = Self::new(base_path, create_options)?;
341
342        Ok(())
343    }

345    /// Wipe the storage, drop all pages and delete the base directory.
346    ///
347    /// Takes ownership because this function leaves Gridstore in an inconsistent state which does
348    /// not allow further usage. Use [`clear`](Self::clear) instead to clear and reuse the storage.
349    pub fn wipe(self) -> Result<()> {
350        let base_path = self.base_path.clone();
351
352        self.is_alive_flush_lock.blocking_mark_dead();
353
354        drop(self);
355
356        fs::remove_dir_all(base_path).map_err(|err| {
357            GridstoreError::service_error(format!(
358                "Failed to remove gridstore storage directory: {err}"
359            ))
360        })
361    }

385    pub fn iter<F, E>(
386        &self,
387        mut callback: F,
388        hw_counter: HwMetricRefCounter,
389    ) -> std::result::Result<(), E>
390    where
391        F: FnMut(PointOffset, V) -> std::result::Result<bool, E>,
392        E: From<GridstoreError>,
393    {
394        const BATCH_SIZE: usize = 128;
395
396        let mut from_offset = 0;
397        // Iterate in batches to allow releasing read locks, see:
398        // <https://github.com/qdrant/qdrant/pull/7983>
399        while let ControlFlow::Continue(next_offset) = self.with_view(|view| {
400            view.iter(
401                from_offset,
402                PointOffset::MAX,
403                BATCH_SIZE,
404                &mut callback,
405                hw_counter,
406            )
407        })? {
408            from_offset = next_offset;
409        }
410
411        Ok(())
412    }

425    /// Create flusher that durably persists all pending changes when invoked.
426    pub fn flusher(&self) -> Flusher {
427        let pending_updates = self.tracker.read().pending_updates.clone();
428
429        let pages = Arc::downgrade(&self.pages);
430        let tracker = Arc::downgrade(&self.tracker);
431        let bitmask = Arc::downgrade(&self.bitmask);
432        let block_size_bytes = self.config.block_size_bytes;
433
434        let is_alive_flush_lock = self.is_alive_flush_lock.handle();
435
436        Box::new(move || {
437            let (Some(is_alive_flush_guard), Some(pages), Some(tracker), Some(bitmask)) = (
438                is_alive_flush_lock.lock_if_alive(),
439                pages.upgrade(),
440                tracker.upgrade(),
441                bitmask.upgrade(),
442            ) else {
443                log::trace!("Gridstore was cleared, cancelling flush");
444                return Err(GridstoreError::FlushCancelled);
445            };
446
447            let bitmask_flusher = bitmask.read().flusher();
448            bitmask_flusher()?;
449
450            let pages_flusher = pages.read().flusher();
451            pages_flusher()?;
452
453            let old_pointers = Self::flush_tracker(&tracker, pending_updates)?;
454
455            if old_pointers.is_empty() {
456                return Ok(());
457            }
458
459            Self::flush_free_blocks(&bitmask, old_pointers, block_size_bytes)?;
460
461            drop(is_alive_flush_guard);
462
463            Ok(())
464        })
465    }

467    /// Write pending updates to the tracker and flush it.
468    fn flush_tracker(
469        tracker: &Arc<RwLock<Tracker>>,
470        pending_updates: AHashMap<PointOffset, PointerUpdates>,
471    ) -> crate::gridstore::Result<Vec<ValuePointer>> {
472        let (old_pointers, tracker_flusher) = {
473            let mut guard = tracker.write();
474            let old_pointers = guard.write_pending(pending_updates)?;
475            let flusher = guard.flusher();
476            (old_pointers, flusher)
477        };
478        tracker_flusher()?;
479        Ok(old_pointers)
480    }
481
482    /// Update all free blocks in the bitmask for old pointers and flush it.
483    fn flush_free_blocks(
484        bitmask: &Arc<RwLock<Bitmask>>,
485        old_pointers: Vec<ValuePointer>,
486        block_size_bytes: usize,
487    ) -> crate::gridstore::Result<()> {
488        let bitmask_flusher = {
489            let mut guard = bitmask.write();
490            for (page_id, pointer_group) in
491                &old_pointers.into_iter().chunk_by(|pointer| pointer.page_id)
492            {
493                let local_ranges = pointer_group.map(|pointer| {
494                    let start = pointer.block_offset;
495                    let end = pointer.block_offset
496                        + Self::blocks_for_value(pointer.length as usize, block_size_bytes);
497                    start as usize..end as usize
498                });
499                guard.mark_blocks_batch(page_id, local_ranges, false);
500            }
501            guard.flusher()
502        };
503        bitmask_flusher()?;
504        Ok(())
505    }

51pub fn random_word(rng: &mut impl Rng) -> String {
52    let len = rng.random_range(1..10);
53    let mut word = String::with_capacity(len);
54    for _ in 0..len {
55        word.push(rng.random_range(b'a'..=b'z') as char);
56    }
57    word
58}

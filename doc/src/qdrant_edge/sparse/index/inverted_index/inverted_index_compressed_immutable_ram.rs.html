17#[derive(Debug, Clone, PartialEq)]
18pub struct InvertedIndexCompressedImmutableRam<W: Weight> {
19    pub(super) postings: Vec<CompressedPostingList<W>>,
20    pub(super) vector_count: usize,
21    pub(super) total_sparse_size: usize,
22}
23
24impl<W: Weight> InvertedIndexCompressedImmutableRam<W> {
25    #[allow(dead_code)]
26    pub(super) fn into_postings(self) -> Vec<CompressedPostingList<W>> {
27        self.postings
28    }
29}

31impl<W: Weight> InvertedIndex for InvertedIndexCompressedImmutableRam<W> {
32    type Iter<'a> = CompressedPostingListIterator<'a, W>;
33
34    type Version = <InvertedIndexCompressedMmap<W> as InvertedIndex>::Version;
35
36    fn is_on_disk(&self) -> bool {
37        false
38    }
39
40    fn open(path: &Path) -> std::io::Result<Self> {
41        let mmap_inverted_index = InvertedIndexCompressedMmap::load(path)?;
42        let mut inverted_index = InvertedIndexCompressedImmutableRam {
43            postings: Vec::with_capacity(mmap_inverted_index.file_header.posting_count),
44            vector_count: mmap_inverted_index.file_header.vector_count,
45            total_sparse_size: mmap_inverted_index.total_sparse_vectors_size(),
46        };
47
48        let hw_counter = HardwareCounterCell::disposable();
49
50        for i in 0..mmap_inverted_index.file_header.posting_count as DimId {
51            let posting_list = mmap_inverted_index.get(i, &hw_counter).ok_or_else(|| {
52                std::io::Error::new(
53                    std::io::ErrorKind::InvalidData,
54                    format!("Posting list {i} not found"),
55                )
56            })?;
57            inverted_index.postings.push(posting_list.to_owned());
58        }
59
60        Ok(inverted_index)
61    }
62
63    fn save(&self, path: &Path) -> std::io::Result<()> {
64        InvertedIndexCompressedMmap::convert_and_save(self, path)?;
65        Ok(())
66    }
67
68    fn get<'a>(
69        &'a self,
70        id: DimOffset,
71        hw_counter: &'a HardwareCounterCell, // Ignored for in-ram index
72    ) -> Option<Self::Iter<'a>> {
73        self.postings
74            .get(id as usize)
75            .map(|posting_list| posting_list.iter(hw_counter))
76    }
77
78    fn len(&self) -> usize {
79        self.postings.len()
80    }
81
82    fn posting_list_len(&self, id: &DimOffset, hw_counter: &HardwareCounterCell) -> Option<usize> {
83        self.get(*id, hw_counter)
84            .map(|posting_list| posting_list.len_to_end())
85    }
86
87    fn files(path: &Path) -> Vec<std::path::PathBuf> {
88        InvertedIndexCompressedMmap::<W>::files(path)
89    }
90
91    fn immutable_files(path: &Path) -> Vec<std::path::PathBuf> {
92        // `InvertedIndexCompressedImmutableRam` is always immutable
93        InvertedIndexCompressedMmap::<W>::immutable_files(path)
94    }
95
96    fn remove(&mut self, _id: PointOffsetType, _old_vector: RemappedSparseVector) {
97        panic!("Cannot remove from a read-only RAM inverted index")
98    }
99
100    fn upsert(
101        &mut self,
102        _id: PointOffsetType,
103        _vector: RemappedSparseVector,
104        _old_vector: Option<RemappedSparseVector>,
105    ) {
106        panic!("Cannot upsert into a read-only RAM inverted index")
107    }
108
109    fn from_ram_index<P: AsRef<Path>>(
110        ram_index: Cow<InvertedIndexRam>,
111        _path: P,
112    ) -> std::io::Result<Self> {
113        let mut postings = Vec::with_capacity(ram_index.postings.len());
114        for old_posting_list in &ram_index.postings {
115            let mut new_posting_list = CompressedPostingBuilder::new();
116            for elem in &old_posting_list.elements {
117                new_posting_list.add(elem.record_id, elem.weight);
118            }
119            postings.push(new_posting_list.build());
120        }
121
122        let hw_counter = HardwareCounterCell::disposable();
123
124        let total_sparse_size = postings
125            .iter()
126            .map(|p| p.view(&hw_counter).store_size().total)
127            .sum();
128
129        Ok(InvertedIndexCompressedImmutableRam {
130            postings,
131            vector_count: ram_index.vector_count,
132            total_sparse_size,
133        })
134    }
135
136    fn vector_count(&self) -> usize {
137        self.vector_count
138    }
139
140    fn total_sparse_vectors_size(&self) -> usize {
141        self.total_sparse_size
142    }
143
144    fn max_index(&self) -> Option<DimOffset> {
145        self.postings
146            .len()
147            .checked_sub(1)
148            .map(|len| len as DimOffset)
149    }
150}

154    use tempfile::Builder;

161    #[test]
162    fn test_save_load_tiny() {
163        let mut builder = InvertedIndexBuilder::new();
164        builder.add(1, vec![(1, 10.0), (2, 10.0), (3, 10.0)].try_into().unwrap());
165        builder.add(2, vec![(1, 20.0), (2, 20.0), (3, 20.0)].try_into().unwrap());
166        builder.add(3, vec![(1, 30.0), (2, 30.0), (3, 30.0)].try_into().unwrap());
167        let inverted_index_ram = builder.build();
168
169        check_save_load::<f32>(&inverted_index_ram);
170        check_save_load::<half::f16>(&inverted_index_ram);
171        check_save_load::<u8>(&inverted_index_ram);
172        check_save_load::<QuantizedU8>(&inverted_index_ram);
173    }
174
175    #[test]
176    fn test_save_load_large() {
177        let mut rnd_gen = rand::rng();
178        let mut builder = InvertedIndexBuilder::new();
179        // Enough elements to put some of them into chunks
180        for i in 0..1024 {
181            builder.add(i, random_sparse_vector(&mut rnd_gen, 3).into_remapped());
182        }
183        let inverted_index_ram = builder.build();
184
185        check_save_load::<f32>(&inverted_index_ram);
186        check_save_load::<half::f16>(&inverted_index_ram);
187        check_save_load::<u8>(&inverted_index_ram);
188        check_save_load::<QuantizedU8>(&inverted_index_ram);
189    }
190
191    fn check_save_load<W: Weight>(inverted_index_ram: &InvertedIndexRam) {
192        let tmp_dir_path = Builder::new().prefix("test_index_dir").tempdir().unwrap();
193        let inverted_index_immutable_ram =
194            InvertedIndexCompressedImmutableRam::<W>::from_ram_index(
195                Cow::Borrowed(inverted_index_ram),
196                tmp_dir_path.path(),
197            )
198            .unwrap();
199        inverted_index_immutable_ram
200            .save(tmp_dir_path.path())
201            .unwrap();
202
203        let loaded_inverted_index =
204            InvertedIndexCompressedImmutableRam::<W>::open(tmp_dir_path.path()).unwrap();
205        assert_eq!(inverted_index_immutable_ram, loaded_inverted_index);
206    }

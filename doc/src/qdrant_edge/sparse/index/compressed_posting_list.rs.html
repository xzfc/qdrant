113/// Defines possible results of the search for the chunk by the ID.
114enum IdChunkPosition {
115    /// The Id is smaller than any data in chunks and therefore
116    /// not in the posting list.
117    Before,
118    /// Id if possibly in the chunk, but it is not guaranteed.
119    Chunk(usize),
120    /// The Id is greater than any data in chunks, but may be in the remainder
121    After,
122}

187    fn decompress_chunk(
188        &self,
189        chunk_index: usize,
190        decompressed_chunk: &mut [PointOffsetType; CHUNK_SIZE],
191    ) {
192        let chunk = &self.chunks[chunk_index];
193        let chunk_size = Self::get_chunk_size(self.chunks, self.id_data, chunk_index);
194        self.hw_counter.vector_io_read().incr_delta(chunk_size);
195        let chunk_bits = chunk_size * u8::BITS as usize / CHUNK_SIZE;
196        BitPackerImpl::new().decompress_strictly_sorted(
197            chunk.initial.checked_sub(1),
198            &self.id_data[chunk.offset as usize..chunk.offset as usize + chunk_size],
199            decompressed_chunk,
200            chunk_bits as u8,
201        );
202    }

204    fn chunk_id_by_position(&self, position: usize) -> Option<usize> {
205        let chunk_index = position / CHUNK_SIZE;
206        if chunk_index < self.chunks.len() {
207            Some(chunk_index)
208        } else {
209            None
210        }
211    }

213    /// Finds the chunk index by the point id.
214    /// It doesn't guarantee that the ID is inside the chunk,
215    /// but if the ID exists, it would be in the chunk.
216    fn chunk_id_by_id(&self, id: PointOffsetType) -> IdChunkPosition {
217        let chunk_index_result = self.chunks.binary_search_by(|c| c.initial.cmp(&id));
218        match chunk_index_result {
219            Ok(chunk_id) => {
220                // Found chunk with the first element exactly equal to the id.
221                IdChunkPosition::Chunk(chunk_id)
222            }
223            Err(position) => {
224                // ┌────── position 0, before any chunk
225                // │       Means first chunk is already greater than required
226                // ▼
227                //     ┌─────────┬────────┬─────────────────┐
228                //     │Chunk-1  │Chunk-2 │....             │
229                //     └─────────┴────────┴─────────────────┘   ▲
230                //                   ▲                          │
231                //                   │                          │
232                //   It might────────┘
233                //   be inside position-1, if             Position == length
234                //   position < length                    it might be either in the chunk (position-1)
235                //                                        Or be inside the remainder
236                if position == self.chunks.len() {
237                    if let Some(first_remainder) = self.remainders.first() {
238                        // If first element of remainder is greater than id,
239                        // then Id might still be in the last chunk, if it exists
240                        if id < first_remainder.record_id {
241                            if position > 0 {
242                                IdChunkPosition::Chunk(position - 1)
243                            } else {
244                                IdChunkPosition::Before
245                            }
246                        } else {
247                            IdChunkPosition::After
248                        }
249                    } else {
250                        // There are no remainder, so we don't know the last id of the last chunk
251                        // Therefore, it is still possible that the id is in the last chunk
252                        IdChunkPosition::Chunk(position - 1)
253                    }
254                } else if position == 0 {
255                    // The id is smaller than the first element of the first chunk
256                    IdChunkPosition::Before
257                } else {
258                    // The id is between two chunks
259                    IdChunkPosition::Chunk(position - 1)
260                }
261            }
262        }
263    }

265    /// Get byte size of the compressed chunk.
266    fn get_chunk_size(
267        chunks: &[CompressedPostingChunk<W>],
268        data: &[u8],
269        chunk_index: usize,
270    ) -> usize {
271        if chunk_index + 1 < chunks.len() {
272            chunks[chunk_index + 1].offset as usize - chunks[chunk_index].offset as usize
273        } else {
274            // Last chunk
275            data.len() - chunks[chunk_index].offset as usize
276        }
277    }

309    /// Warning: This function panics if the index is out of bounds.
310    #[inline]
311    fn get_weight(&self, pos: usize) -> W {
312        self.hw_counter.vector_io_read().incr_delta(size_of::<W>());
313        let chunk = &self.chunks[pos / CHUNK_SIZE];
314        chunk.weights[pos % CHUNK_SIZE]
315    }
316
317    #[inline]
318    fn weights_range(&self, pos: usize, count: usize) -> &[W] {
319        debug_assert!(count <= CHUNK_SIZE);
320        self.hw_counter
321            .vector_io_read()
322            .incr_delta(size_of::<W>() * count);
323
324        let chunk = &self.chunks[pos / CHUNK_SIZE];
325        let start = pos % CHUNK_SIZE;
326        chunk.weights[start..start + count].as_ref()
327    }

338impl CompressedPostingBuilder {

340    pub fn new() -> Self {
341        CompressedPostingBuilder {
342            elements: Vec::new(),
343        }
344    }

346    /// Add a new record to the posting list.
347    pub fn add(&mut self, record_id: PointOffsetType, weight: DimWeight) {
348        self.elements.push(PostingElement { record_id, weight });
349    }
350
351    pub fn build<W: Weight>(mut self) -> CompressedPostingList<W> {
352        self.elements.sort_unstable_by_key(|e| e.record_id);
353
354        let quantization_params =
355            W::quantization_params_for(self.elements.iter().map(|e| e.weight));
356
357        // Check for duplicates
358        #[cfg(debug_assertions)]
359        if let Some(e) = self.elements.iter().duplicates_by(|e| e.record_id).next() {
360            panic!("Duplicate id {} in posting list", e.record_id);
361        }
362
363        let mut this_chunk = Vec::with_capacity(CHUNK_SIZE);
364
365        let bitpacker = BitPackerImpl::new();
366        let mut chunks = Vec::with_capacity(self.elements.len() / CHUNK_SIZE);
367        let mut data_size = 0;
368        let mut remainders = Vec::with_capacity(self.elements.len() % CHUNK_SIZE);
369        for chunk in self.elements.chunks(CHUNK_SIZE) {
370            if chunk.len() == CHUNK_SIZE {
371                this_chunk.clear();
372                this_chunk.extend(chunk.iter().map(|e| e.record_id));
373
374                let initial = this_chunk[0];
375                let chunk_bits =
376                    bitpacker.num_bits_strictly_sorted(initial.checked_sub(1), &this_chunk);
377                let chunk_size = BitPackerImpl::compressed_block_size(chunk_bits);
378                chunks.push(CompressedPostingChunk {
379                    initial,
380                    offset: data_size as u32,
381                    weights: chunk
382                        .iter()
383                        .map(|e| Weight::from_f32(quantization_params, e.weight))
384                        .collect::<Vec<_>>()
385                        .try_into()
386                        .expect("Invalid chunk size"),
387                });
388                data_size += chunk_size;
389            } else {
390                for e in chunk {
391                    remainders.push(GenericPostingElement {
392                        record_id: e.record_id,
393                        weight: Weight::from_f32(quantization_params, e.weight),
394                    });
395                }
396            }
397        }
398
399        let mut id_data = vec![0u8; data_size];
400        for (chunk_index, chunk_data) in self.elements.chunks_exact(CHUNK_SIZE).enumerate() {
401            this_chunk.clear();
402            this_chunk.extend(chunk_data.iter().map(|e| e.record_id));
403
404            let chunk = &chunks[chunk_index];
405            let chunk_size =
406                CompressedPostingListView::get_chunk_size(&chunks, &id_data, chunk_index);
407            let chunk_bits = chunk_size * u8::BITS as usize / CHUNK_SIZE;
408            bitpacker.compress_strictly_sorted(
409                chunk.initial.checked_sub(1),
410                &this_chunk,
411                &mut id_data[chunk.offset as usize..chunk.offset as usize + chunk_size],
412                chunk_bits as u8,
413            );
414        }
415
416        CompressedPostingList {
417            id_data,
418            chunks,
419            remainders,
420            last_id: self.elements.last().map(|e| e.record_id),
421            quantization_params,
422        }
423    }
424}

427pub struct CompressedPostingListIterator<'a, W: Weight> {
428    list: CompressedPostingListView<'a, W>,
429
430    /// If true, then `decompressed_chunk` contains the unpacked chunk for the current position.
431    unpacked: bool,
432
433    decompressed_chunk: [PointOffsetType; CHUNK_SIZE],
434
435    /// Offset inside the posting list along with optional current element.
436    /// Defined as a tuple to ensure that we won't forget to update the element
437    pos: (usize, Option<PointOffsetType>),
438}
439
440impl<'a, W: Weight> CompressedPostingListIterator<'a, W> {
441    #[inline]
442    fn new(list: &CompressedPostingListView<'a, W>) -> Self {
443        Self {
444            list: list.clone(),
445            unpacked: false,
446            decompressed_chunk: [0; CHUNK_SIZE],
447            pos: (0, None),
448        }
449    }
450
451    #[inline]
452    fn next_from(&mut self, peek: PostingElementEx) -> PostingElement {
453        if self.pos.0 / CHUNK_SIZE < self.list.chunks.len() {
454            self.pos = (self.pos.0 + 1, None);
455            if self.pos.0.is_multiple_of(CHUNK_SIZE) {
456                self.unpacked = false;
457            }
458        } else {
459            self.pos = (self.pos.0 + 1, None);
460        }
461
462        peek.into()
463    }
464
465    #[inline]
466    fn next(&mut self) -> Option<PostingElement> {
467        let result = self.peek()?;
468
469        Some(self.next_from(result))
470    }
471}

473impl<W: Weight> PostingListIter for CompressedPostingListIterator<'_, W> {
474    #[inline]
475    fn peek(&mut self) -> Option<PostingElementEx> {
476        let pos = self.pos.0;
477        if pos / CHUNK_SIZE < self.list.chunks_len() {
478            if !self.unpacked {
479                self.list
480                    .decompress_chunk(pos / CHUNK_SIZE, &mut self.decompressed_chunk);
481                self.unpacked = true;
482            }
483
484            return Some(PostingElementEx {
485                record_id: self.decompressed_chunk[pos % CHUNK_SIZE],
486                weight: self.list.get_weight(pos).to_f32(self.list.multiplier),
487                max_next_weight: Default::default(),
488            });
489        }
490
491        self.list
492            .get_remainder_id(pos - self.list.chunks_len() * CHUNK_SIZE)
493            .map(|e| PostingElementEx {
494                record_id: e.record_id,
495                weight: e.weight.to_f32(self.list.multiplier),
496                max_next_weight: Default::default(),
497            })
498    }
499
500    #[inline]
501    fn last_id(&self) -> Option<PointOffsetType> {
502        self.list.last_id
503    }
504
505    fn element_size(&self) -> usize {
506        size_of::<W>()
507    }
508
509    fn skip_to(&mut self, record_id: PointOffsetType) -> Option<PostingElementEx> {
510        // 1. Define which chunk we need to unpack (maybe it is current)
511        // 2. If current, change the position to the element and do peek
512
513        // Shortcut peeking into memory
514        if let Some(current_record_id) = self.pos.1.as_ref()
515            && record_id < *current_record_id
516        {
517            // We are already ahead
518            return None;
519        }
520
521        // If None, we are already reading remainder
522        let current_chunk_id_opt = self.list.chunk_id_by_position(self.pos.0);
523        // Required chunk id
524        let required_chunk_id = self.list.chunk_id_by_id(record_id);
525
526        match (required_chunk_id, current_chunk_id_opt) {
527            (IdChunkPosition::Chunk(chunk_id), Some(current_chunk_id)) => {
528                match chunk_id.cmp(&current_chunk_id) {
529                    Ordering::Less => {
530                        // Chunk is already skipped
531                        // Return None, don't change the position
532                        return None;
533                    }
534                    Ordering::Equal => {
535                        let min_pos = chunk_id * CHUNK_SIZE;
536                        self.pos = (std::cmp::max(self.pos.0, min_pos), None);
537                    }
538                    Ordering::Greater => {
539                        // Chunk is ahead, move to it
540                        self.pos = (chunk_id * CHUNK_SIZE, None);
541                        self.unpacked = false;
542                    }
543                }
544            }
545            (IdChunkPosition::Chunk(_), None) => {
546                // We are already in the remainder, and we can't go back
547                return None;
548            }
549            (IdChunkPosition::Before, _) => {
550                // Don't change anything, as current `pos` is by definition higher
551                return None;
552            }
553            (IdChunkPosition::After, _) => {
554                // Go to after the chunks
555                let min_pos = self.list.chunks_len() * CHUNK_SIZE;
556                self.pos = (std::cmp::max(self.pos.0, min_pos), None);
557                self.unpacked = false;
558            }
559        };
560
561        while let Some(current_element) = self.peek() {
562            // Save the current element to avoid further peeking
563            self.pos = (self.pos.0, Some(current_element.record_id));
564            match current_element.record_id.cmp(&record_id) {
565                Ordering::Equal => return Some(current_element),
566                Ordering::Greater => return None,
567                Ordering::Less => {
568                    // Go to the next element
569                    self.next_from(current_element);
570                }
571            }
572        }
573        None
574    }
575
576    #[inline]
577    fn skip_to_end(&mut self) {
578        self.pos = (
579            self.list.chunks_len() * CHUNK_SIZE + self.list.remainder_len(),
580            None,
581        );
582    }
583
584    #[inline]
585    fn len_to_end(&self) -> usize {
586        self.list.len() - self.pos.0
587    }
588
589    #[inline]
590    fn current_index(&self) -> usize {
591        self.pos.0
592    }
593
594    #[inline]
595    fn for_each_till_id<Ctx: ?Sized>(
596        &mut self,
597        id: PointOffsetType,
598        ctx: &mut Ctx,
599        mut f: impl FnMut(&mut Ctx, PointOffsetType, DimWeight),
600    ) {
601        let mut pos = self.pos.0;
602
603        // Iterate over compressed chunks
604        let mut weights_buf = [0.0; CHUNK_SIZE];
605
606        let mut need_unpack = !self.unpacked;
607        while pos / CHUNK_SIZE < self.list.chunks_len() {
608            if need_unpack {
609                self.list
610                    .decompress_chunk(pos / CHUNK_SIZE, &mut self.decompressed_chunk);
611            }
612            need_unpack = true;
613
614            let start = pos % CHUNK_SIZE;
615            let count = count_le_sorted(id, &self.decompressed_chunk[start..]);
616
617            let weights = self.list.weights_range(pos, count);
618            let weights =
619                W::into_f32_slice(self.list.multiplier, weights, &mut weights_buf[..count]);
620
621            for (idx, weight) in
622                std::iter::zip(&self.decompressed_chunk[start..start + count], weights)
623            {
624                f(ctx, *idx, *weight);
625            }
626            pos += count;
627            if start + count != CHUNK_SIZE {
628                self.unpacked = true;
629                self.pos = (pos, None);
630                return;
631            }
632        }
633
634        // Iterate over remainders
635        for e in self
636            .list
637            .iter_remainder_from(pos - self.list.chunks_len() * CHUNK_SIZE)
638        {
639            if e.record_id > id {
640                self.pos = (pos, None);
641                return;
642            }
643            f(ctx, e.record_id, e.weight.to_f32(self.list.multiplier));
644            pos += 1;
645        }
646        self.pos = (pos, None);
647    }

650        false

669/// Find the amount of elements in the sorted array that are less or equal to `val`. In other words,
670/// the first index `i` such that `data[i] > val`, or `data.len()` if all elements are less or equal
671/// to `val`.
672fn count_le_sorted<T: Copy + Eq + Ord>(val: T, data: &[T]) -> usize {
673    if data.last().is_none_or(|&x| x < val) {
674        // Happy case
675        return data.len();
676    }

678    data.binary_search(&val).map_or_else(|x| x, |x| x + 1)
679}

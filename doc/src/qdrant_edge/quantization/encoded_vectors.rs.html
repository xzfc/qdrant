84impl DistanceType {
85    pub fn distance(&self, a: &[f32], b: &[f32]) -> f32 {
86        match self {
87            DistanceType::Dot => a.iter().zip(b).map(|(a, b)| a * b).sum(),
88            DistanceType::L1 => a.iter().zip(b).map(|(a, b)| (a - b).abs()).sum(),
89            DistanceType::L2 => a.iter().zip(b).map(|(a, b)| (a - b) * (a - b)).sum(),
90        }
91    }
92}

94pub(crate) fn validate_vector_parameters<'a>(
95    data: impl Iterator<Item = impl AsRef<[f32]> + 'a> + Clone,
96    vector_parameters: &VectorParameters,
97) -> Result<(), EncodingError> {
98    let mut count = 0;
99    for vector in data {
100        let vector = vector.as_ref();
101        if vector.len() != vector_parameters.dim {
102            return Err(EncodingError::ArgumentsError(format!(
103                "Vector length {} does not match vector parameters dim {}",
104                vector.len(),
105                vector_parameters.dim
106            )));
107        }
108        count += 1;
109    }
110    if let Some(vectors_count) = vector_parameters.deprecated_count
111        && count != vectors_count
112    {
113        return Err(EncodingError::ArgumentsError(format!(
114            "Vector count {count} does not match vector parameters count {vectors_count}"
115        )));
116    }
117    Ok(())
